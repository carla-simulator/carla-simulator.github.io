32  // A deleter cannot throw exceptions; and unlike std::unique_ptr, the deleter
33  // of (std|boost)::shared_ptr is invoked even if the managed pointer is null.
34  struct GarbageCollector {
35  void operator()(::carla::client::Actor *ptr) const noexcept {
36  if ((ptr != nullptr) && ptr->IsAlive()) {
37  try {
38  ptr->Destroy();
39  delete ptr;
40  } catch (const ::rpc::timeout &timeout) {
41  log_error(timeout.what());
42  log_error(
43  "timeout while trying to garbage collect Actor",
44  ptr->GetDisplayId(),
45  "actor hasn't been removed from the simulation");
46  } catch (const std::exception &e) {
47  log_critical(
48  "exception thrown while trying to garbage collect Actor",
49  ptr->GetDisplayId(),
50  e.what());
51  throw; // calls terminate.
52  } catch (...) {
53  log_critical(
54  "unknown exception thrown while trying to garbage collect an Actor :",
55  ptr->GetDisplayId());
56  throw; // calls terminate.
57  }
58  }
59  }
60  };

62  template <typename ActorT>
63  static auto MakeActorImpl(ActorInitializer init, GarbageCollectionPolicy gc) {
64  if (gc == GarbageCollectionPolicy::Enabled) {
65  return SharedPtr<ActorT>{new ActorT(std::move(init)), GarbageCollector()};
66  }
67  DEBUG_ASSERT(gc == GarbageCollectionPolicy::Disabled);

71  SharedPtr<Actor> ActorFactory::MakeActor(

76  if (description.description.id == "sensor.other.lane_invasion") {

81  return MakeActorImpl<ServerSideSensor>(std::move(init), gc);
82  } else if (StringUtil::StartsWith(description.description.id, "vehicle.")) {
83  return MakeActorImpl<Vehicle>(std::move(init), gc);
84  } else if (StringUtil::StartsWith(description.description.id, "walker.")) {
85  return MakeActorImpl<Walker>(std::move(init), gc);
86  } else if (StringUtil::StartsWith(description.description.id, "traffic.traffic_light")) {
87  return MakeActorImpl<TrafficLight>(std::move(init), gc);
88  } else if (StringUtil::StartsWith(description.description.id, "traffic.")) {
89  return MakeActorImpl<TrafficSign>(std::move(init), gc);
90  } else if (description.description.id == "controller.ai.walker") {
91  return MakeActorImpl<WalkerAIController>(std::move(init), gc);
92  }
93  return MakeActorImpl<Actor>(std::move(init), gc);
