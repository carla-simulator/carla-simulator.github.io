84 template <typename TSensor>
85 void FPixelReader::SendPixelsInRenderThread(TSensor &Sensor)
86 {
87  check(Sensor.CaptureRenderTarget != nullptr);
88 
89  if (!Sensor.HasActorBegunPlay() || Sensor.IsPendingKill())
90  {
91  return;
92  }
93 
94  // Enqueue a command in the render-thread that will write the image buffer to
95  // the data stream. The stream is created in the capture thus executed in the
96  // game-thread.
97  ENQUEUE_RENDER_COMMAND(FWritePixels_SendPixelsInRenderThread)
98  (
99  [&Sensor, Stream=Sensor.GetDataStream(Sensor)](auto &InRHICmdList) mutable
100  {
101  /// @todo Can we make sure the sensor is not going to be destroyed?
102  if (!Sensor.IsPendingKill())
103  {
104  auto Buffer = Stream.PopBufferFromPool();
105  WritePixelsToBuffer(
106  *Sensor.CaptureRenderTarget,
107  Buffer,
108  carla::sensor::SensorRegistry::get<TSensor *>::type::header_offset,
109  InRHICmdList);
110 
111  if(Buffer.data())
112  {
113  SCOPE_CYCLE_COUNTER(STAT_CarlaSensorStreamSend);
114  Stream.Send(Sensor, std::move(Buffer));
115  }
116  }
117  }
118  );
119 }
