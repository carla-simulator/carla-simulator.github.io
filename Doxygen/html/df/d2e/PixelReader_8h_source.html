92 template <typename TSensor, typename TPixel>
93 void FPixelReader::SendPixelsInRenderThread(TSensor &Sensor, bool use16BitFormat, std::function<TArray<TPixel>(void *, uint32)> Conversor)
94 {
95  TRACE_CPUPROFILER_EVENT_SCOPE(FPixelReader::SendPixelsInRenderThread);
96  check(Sensor.CaptureRenderTarget != nullptr);
97 
98  if (!Sensor.HasActorBegunPlay() || Sensor.IsPendingKill())
99  {
100  return;
101  }
102 
103  /// Blocks until the render thread has finished all it's tasks.
104  Sensor.EnqueueRenderSceneImmediate();
105 
106  // Enqueue a command in the render-thread that will write the image buffer to
107  // the data stream. The stream is created in the capture thus executed in the
108  // game-thread.
109  ENQUEUE_RENDER_COMMAND(FWritePixels_SendPixelsInRenderThread)
110  (
111  [&Sensor, use16BitFormat, Conversor = std::move(Conversor)](auto &InRHICmdList) mutable
112  {
113  TRACE_CPUPROFILER_EVENT_SCOPE_STR("FWritePixels_SendPixelsInRenderThread");
114 
115  /// @todo Can we make sure the sensor is not going to be destroyed?
116  if (!Sensor.IsPendingKill())
117  {
118  FPixelReader::Payload FuncForSending =
119  [&Sensor, Frame = FCarlaEngine::GetFrameCounter(), Conversor = std::move(Conversor)](void *LockedData, uint32 Size, uint32 Offset, uint32 ExpectedRowBytes)
120  {
121  if (Sensor.IsPendingKill()) return;
122 
123  TArray<TPixel> Converted;
124 
125  // optional conversion of data
126  if (Conversor)
127  {
128  TRACE_CPUPROFILER_EVENT_SCOPE_STR("Data conversion");
129  Converted = Conversor(LockedData, Size);
130  LockedData = reinterpret_cast<void *>(Converted.GetData());
131  Size = Converted.Num() * Converted.GetTypeSize();
132  }
133 
134  auto Stream = Sensor.GetDataStream(Sensor);
135  Stream.SetFrameNumber(Frame);
136  auto Buffer = Stream.PopBufferFromPool();
137 
138  uint32 CurrentRowBytes = ExpectedRowBytes;
139 
140 #ifdef _WIN32
141  // DirectX uses additional bytes to align each row to 256 boundry,
142  // so we need to remove that extra data
143  if (IsD3DPlatform(GMaxRHIShaderPlatform, false))
144  {
145  CurrentRowBytes = Align(ExpectedRowBytes, D3D12_TEXTURE_DATA_PITCH_ALIGNMENT);
146  if (ExpectedRowBytes != CurrentRowBytes)
147  {
148  TRACE_CPUPROFILER_EVENT_SCOPE_STR("Buffer Copy (windows, row by row)");
149  Buffer.reset(Offset + Size);
150  auto DstRow = Buffer.begin() + Offset;
151  const uint8 *SrcRow = reinterpret_cast<uint8 *>(LockedData);
152  uint32 i = 0;
153  while (i < Size)
154  {
155  FMemory::Memcpy(DstRow, SrcRow, ExpectedRowBytes);
156  DstRow += ExpectedRowBytes;
157  SrcRow += CurrentRowBytes;
158  i += ExpectedRowBytes;
159  }
160  }
161  }
162 #endif // _WIN32
163 
164  if (ExpectedRowBytes == CurrentRowBytes)
165  {
166  check(ExpectedRowBytes == CurrentRowBytes);
167  TRACE_CPUPROFILER_EVENT_SCOPE_STR("Buffer Copy");
168  Buffer.copy_from(Offset, boost::asio::buffer(LockedData, Size));
169  }
170 
171  {
172  // send
173  TRACE_CPUPROFILER_EVENT_SCOPE_STR("Sending buffer");
174  if(Buffer.data())
175  {
176  SCOPE_CYCLE_COUNTER(STAT_CarlaSensorStreamSend);
177  TRACE_CPUPROFILER_EVENT_SCOPE_STR("Stream Send");
178  Stream.Send(Sensor, std::move(Buffer));
179  }
180  }
181  };
182 
183  WritePixelsToBuffer(
184  *Sensor.CaptureRenderTarget,
185  carla::sensor::SensorRegistry::get<TSensor *>::type::header_offset,
186  InRHICmdList,
187  std::move(FuncForSending));
188  }
189  }
190  );
191 
192  // Blocks until the render thread has finished all it's tasks
193  Sensor.WaitForRenderThreadToFinish();
194 }
