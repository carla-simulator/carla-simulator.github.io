327 void ALargeMapManager::GenerateMap(FString InAssetsPath)
328 {
329  TRACE_CPUPROFILER_EVENT_SCOPE(ALargeMapManager::GenerateMap);
330  LM_LOG(Warning, "Generating Map %s ...", *InAssetsPath);
331  ClearWorldAndTiles();
332 
333  AssetsPath = InAssetsPath;
334 
335  /// Retrive all the assets in the path
336  TArray<FAssetData> AssetsData;
337  UObjectLibrary* ObjectLibrary = UObjectLibrary::CreateLibrary(UWorld::StaticClass(), true, true);
338  ObjectLibrary->LoadAssetDataFromPath(InAssetsPath);
339  ObjectLibrary->GetAssetDataList(AssetsData);
340 
341  /// Generate tiles based on mesh positions
342  UWorld* World = GetWorld();
343  MapTiles.Reset();
344  for (const FAssetData& AssetData : AssetsData)
345  {
346  #if WITH_EDITOR
347  // LM_LOG(Warning, "Loading asset name: %s", *(AssetData.AssetName.ToString()));
348  // LM_LOG(Warning, "Asset class: %s", *(AssetData.AssetClass.ToString()));
349  #endif
350  FString TileName = AssetData.AssetName.ToString();
351  if (!TileName.Contains("_Tile_"))
352  {
353  continue;
354  }
355  FString TileName_X = "";
356  FString TileName_Y = "";
357  size_t i = TileName.Len()-1;
358  for (; i > 0; i--) {
359  TCHAR character = TileName[i];
360  if (character == '_') {
361  break;
362  }
363  TileName_Y = FString::Chr(character) + TileName_Y;
364  }
365  i--;
366  for (; i > 0; i--) {
367  TCHAR character = TileName[i];
368  if (character == '_') {
369  break;
370  }
371  TileName_X = FString::Chr(character) + TileName_X;
372  }
373  FIntVector TileVectorID = FIntVector(FCString::Atoi(*TileName_X), FCString::Atoi(*TileName_Y), 0);
374  #if WITH_EDITOR
375  // LM_LOG(Warning, "Tile: %d, %d", TileVectorID.X, TileVectorID.Y);
376  #endif
377  TileID TileId = GetTileID(TileVectorID);
378  LoadCarlaMapTile(InAssetsPath + "/" + AssetData.AssetName.ToString(), TileId);
379  }
380  ObjectLibrary->ConditionalBeginDestroy();
381  GEngine->ForceGarbageCollection(true);
382 
383  ActorsToConsider.Reset();
384 
385 #if WITH_EDITOR
386  LM_LOG(Warning, "GenerateMap num Tiles generated %d", MapTiles.Num());
387  DumpTilesTable();
388 #endif // WITH_EDITOR
389 }

576 FCarlaMapTile& ALargeMapManager::LoadCarlaMapTile(FString TileMapPath, TileID TileId) {
577  TRACE_CPUPROFILER_EVENT_SCOPE(ALargeMapManager::LoadCarlaMapTile);
578  // Need to generate a new Tile
579  FCarlaMapTile NewTile;
580  // 1 - Calculate the Tile position
581  FIntVector VTileID = GetTileVectorID(TileId);
582  NewTile.Location = GetTileLocation(TileId);
583  NewTile.Name = TileMapPath;
584 
585  // 3 - Generate the StreamLevel
586  FVector TileLocation = NewTile.Location;
587  FString TileName = NewTile.Name;
588  UWorld* World = GetWorld();
589  UWorldComposition* WorldComposition = World->WorldComposition;
590 
591  FString FullName = TileMapPath;
592  FString PackageFileName = FullName;
593  FString LongLevelPackageName = FPackageName::FilenameToLongPackageName(PackageFileName);
594  FString UniqueLevelPackageName = LongLevelPackageName;
595 
596  ULevelStreamingDynamic* StreamingLevel = NewObject<ULevelStreamingDynamic>(World, *TileName);
597  check(StreamingLevel);
598 
599  StreamingLevel->SetWorldAssetByPackageName(*UniqueLevelPackageName);
600 
601 #if WITH_EDITOR
602  if (World->IsPlayInEditor())
603  {
604  FWorldContext WorldContext = GEngine->GetWorldContextFromWorldChecked(World);
605  StreamingLevel->RenameForPIE(WorldContext.PIEInstance);
606  }
607  StreamingLevel->SetShouldBeVisibleInEditor(true);
608  StreamingLevel->LevelColor = FColor::MakeRandomColor();
609 #endif // WITH_EDITOR
610 
611  StreamingLevel->SetShouldBeLoaded(false);
612  StreamingLevel->SetShouldBeVisible(false);
613  StreamingLevel->bShouldBlockOnLoad = ShouldTilesBlockOnLoad;
614  StreamingLevel->bInitiallyLoaded = false;
615  StreamingLevel->bInitiallyVisible = false;
616  StreamingLevel->LevelTransform = FTransform(TileLocation);
617  StreamingLevel->PackageNameToLoad = *FullName;
618 
619  if (!FPackageName::DoesPackageExist(FullName, NULL, &PackageFileName))
620  {
621  LM_LOG(Error, "Level does not exist in package with FullName variable -> %s", *FullName);
622  }
623 
624  if (!FPackageName::DoesPackageExist(LongLevelPackageName, NULL, &PackageFileName))
625  {
626  LM_LOG(Error, "Level does not exist in package with LongLevelPackageName variable -> %s", *LongLevelPackageName);
627  }
628 
629  //Actual map package to load
630  StreamingLevel->PackageNameToLoad = *LongLevelPackageName;
631 
632  NewTile.StreamingLevel = StreamingLevel;
633 
634  // 4 - Add it to the map
635  return MapTiles.Add(TileId, NewTile);
636 }

638 void ALargeMapManager::UpdateTilesState()
639 {
640  TRACE_CPUPROFILER_EVENT_SCOPE(ALargeMapManager::UpdateTilesState);
641  TSet<TileID> TilesToConsider;
642 
643  // Loop over ActorsToConsider to update the state of the map tiles
644  // if the actor is not valid will be removed
645  for (AActor* Actor : ActorsToConsider)
646  {
647  if (IsValid(Actor))
648  {
649  GetTilesToConsider(Actor, TilesToConsider);
650  }

873 void ALargeMapManager::GetTilesToConsider(const AActor* ActorToConsider,
874  TSet<TileID>& OutTilesToConsider)
875 {

881  // Calculate Current Tile

888  FIntVector LowerTileId = GetTileVectorID(LowerPos);
