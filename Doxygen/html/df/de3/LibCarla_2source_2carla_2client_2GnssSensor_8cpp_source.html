69  void GnssSensor::Listen(CallbackFunctionType callback) {
70  auto episode = GetEpisode().Lock();
71  SharedPtr<Map> map = episode->GetCurrentMap();
72  DEBUG_ASSERT(map != nullptr);
73 
74  const size_t callback_id = episode->RegisterOnTickEvent(GnssCallback(
75  GetId(),
76  map->GetGeoReference(),
77  std::move(callback)));
78 
79  const size_t previous = _callback_id.exchange(callback_id);
80  if (previous != 0u) {
81  episode->RemoveOnTickEvent(previous);
82  }
83  }
