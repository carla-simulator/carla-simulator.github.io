489  Lane *MapBuilder::GetEdgeLanePointer(RoadId road_id, bool from_start, LaneId lane_id) {

509  // return a list of pointers to all lanes from a lane (using road and junction
510  // info)
511  std::vector<Lane *> MapBuilder::GetLaneNext(
512  RoadId road_id,
513  SectionId section_id,
514  LaneId lane_id) {
515  std::vector<Lane *> result;
516 
517  if (!_map_data.ContainsRoad(road_id)) {
518  return result;
519  }
520  Road &road = _map_data.GetRoad(road_id);
521 
522  // get the section
523  LaneSection &section = road._lane_sections.GetById(section_id);
524 
525  // get the lane
526  Lane *lane = section.GetLane(lane_id);
527  DEBUG_ASSERT(lane != nullptr);
528 
529  // successor and predecessor (road and lane)
530  LaneId next;
531  RoadId next_road;
532  if (lane_id <= 0) {
533  next_road = road.GetSuccessor();
534  next = lane->GetSuccessor();
535  } else {
536  next_road = road.GetPredecessor();
537  next = lane->GetPredecessor();
538  }
539 
540  // check to see if next is a road or a junction
541  bool next_is_junction = !_map_data.ContainsRoad(next_road);
542  double s = section.GetDistance();
543 
544  // check if we are in a lane section in the middle
545  if ((lane_id > 0 && s > 0) ||
546  (lane_id <= 0 && road._lane_sections.upper_bound(s) != road._lane_sections.end())) {
547  // check if lane has a next link (if not, it deads in the middle section)
548  if (next != 0 || (lane_id == 0 && next == 0)) {
549  // change to next / prev section
550  if (lane_id <= 0) {
551  result.push_back(road.GetNextLane(s, next));
552  } else {
553  result.push_back(road.GetPrevLane(s, next));
554  }
555  }
556  } else if (!next_is_junction) {
557  // change to another road / junction
558  if (next != 0 || (lane_id == 0 && next == 0)) {
559  // single road
560  result.push_back(GetEdgeLanePointer(next_road, (next <= 0), next));
561  }
562  } else {
563  // several roads (junction)
564 
565  /// @todo Is it correct to use a road id as section id? (NS: I just added
566  /// this cast to avoid compiler warnings).
567  auto next_road_as_junction = static_cast<JuncId>(next_road);
568  auto options = GetJunctionLanes(next_road_as_junction, road_id, lane_id);
569  for (auto opt : options) {
570  result.push_back(GetEdgeLanePointer(opt.first, (opt.second <= 0), opt.second));
571  }
572  }
573 
574  return result;
575  }

577  std::vector<std::pair<RoadId, LaneId>> MapBuilder::GetJunctionLanes(
578  JuncId junction_id,
579  RoadId road_id,
580  LaneId lane_id) {
581  std::vector<std::pair<RoadId, LaneId>> result;
582 
583  // get the junction
584  Junction *junction = _map_data.GetJunction(junction_id);
585  if (junction == nullptr) {
586  return result;
587  }
588 
589  // check all connections
590  for (auto con : junction->_connections) {
591  // only connections for our road
592  if (con.second.incoming_road == road_id) {
593  // for center lane it is always next lane id 0, we don't need to search
594  // because it is not in the junction
595  if (lane_id == 0) {
596  result.push_back(std::make_pair(con.second.connecting_road, 0));
597  } else {
598  // check all lane links
599  for (auto link : con.second.lane_links) {
600  // is our lane id ?
601  if (link.from == lane_id) {
602  // add as option
603  result.push_back(std::make_pair(con.second.connecting_road, link.to));
604  }
605  }
606  }
607  }
608  }
609 
610  return result;
611  }

613  // assign pointers to the next lanes
614  void MapBuilder::CreatePointersBetweenRoadSegments(void) {
615  // process each lane to define its nexts
616  for (auto &road : _map_data._roads) {
617  for (auto &section : road.second._lane_sections) {
618  for (auto &lane : section.second._lanes) {
619 
620  // assign the next lane pointers
621  lane.second._next_lanes = GetLaneNext(road.first, section.second._id, lane.first);
622 
623  // add to each lane found, this as its predecessor
624  for (auto next_lane : lane.second._next_lanes) {
625  // add as previous
626  DEBUG_ASSERT(next_lane != nullptr);
627  next_lane->_prev_lanes.push_back(&lane.second);
628  }
629 
630  }
631  }
632  }
633 
634  // process each lane to define its nexts
635  for (auto &road : _map_data._roads) {
636  for (auto &section : road.second._lane_sections) {
637  for (auto &lane : section.second._lanes) {
638 
639  // add next roads
640  for (auto next_lane : lane.second._next_lanes) {
641  DEBUG_ASSERT(next_lane != nullptr);
642  // avoid same road
643  if (next_lane->GetRoad() != &road.second) {
644  if (std::find(road.second._nexts.begin(), road.second._nexts.end(),
645  next_lane->GetRoad()) == road.second._nexts.end()) {
646  road.second._nexts.push_back(next_lane->GetRoad());
647  }
648  }
649  }
650 
651  // add prev roads
652  for (auto prev_lane : lane.second._prev_lanes) {
653  DEBUG_ASSERT(prev_lane != nullptr);
654  // avoid same road
655  if (prev_lane->GetRoad() != &road.second) {
656  if (std::find(road.second._prevs.begin(), road.second._prevs.end(),
657  prev_lane->GetRoad()) == road.second._prevs.end()) {
658  road.second._prevs.push_back(prev_lane->GetRoad());
659  }
660  }
661  }
662 
663  }
664  }
665  }
666  }
