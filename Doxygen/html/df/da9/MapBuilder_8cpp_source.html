36  boost::optional<Map> MapBuilder::Build() {
37 
38  CreatePointersBetweenRoadSegments();
39 
40  for (auto &&info : _temp_road_info_container) {
41  DEBUG_ASSERT(info.first != nullptr);
42  info.first->_info = InformationSet(std::move(info.second));
43  }
44 
45  for (auto &&info : _temp_lane_info_container) {
46  DEBUG_ASSERT(info.first != nullptr);
47  info.first->_info = InformationSet(std::move(info.second));
48  }
49 
50  // compute transform requires the roads to have the RoadInfo
51  SolveSignalReferencesAndTransforms();
52 
53  SolveControllerAndJuntionReferences();
54 
55  // remove temporal already used information
56  _temp_road_info_container.clear();
57  _temp_lane_info_container.clear();
58 
59  // _map_data is a memeber of MapBuilder so you must especify if
60  // you want to keep it (will return copy -> Map(const Map &))
61  // or move it (will return move -> Map(Map &&))
62  Map map(std::move(_map_data));
63  CreateJunctionBoundingBoxes(map);
64  ComputeJunctionRoadConflicts(map);
65  CheckSignalsOnRoads(map);
66 
67  return map;
68  }

566  Lane *MapBuilder::GetEdgeLanePointer(RoadId road_id, bool from_start, LaneId lane_id) {

586  // return a list of pointers to all lanes from a lane (using road and junction
587  // info)
588  std::vector<Lane *> MapBuilder::GetLaneNext(
589  RoadId road_id,
590  SectionId section_id,
591  LaneId lane_id) {
592  std::vector<Lane *> result;
593 
594  if (!_map_data.ContainsRoad(road_id)) {
595  return result;
596  }
597  Road &road = _map_data.GetRoad(road_id);
598 
599  // get the section
600  LaneSection &section = road._lane_sections.GetById(section_id);
601 
602  // get the lane
603  Lane *lane = section.GetLane(lane_id);
604  DEBUG_ASSERT(lane != nullptr);
605 
606  // successor and predecessor (road and lane)
607  LaneId next;
608  RoadId next_road;
609  if (lane_id <= 0) {
610  next_road = road.GetSuccessor();
611  next = lane->GetSuccessor();
612  } else {
613  next_road = road.GetPredecessor();
614  next = lane->GetPredecessor();
615  }
616 
617  // check to see if next is a road or a junction
618  bool next_is_junction = !_map_data.ContainsRoad(next_road);
619  double s = section.GetDistance();
620 
621  // check if we are in a lane section in the middle
622  if ((lane_id > 0 && s > 0) ||
623  (lane_id <= 0 && road._lane_sections.upper_bound(s) != road._lane_sections.end())) {
624  // check if lane has a next link (if not, it deads in the middle section)
625  if (next != 0 || (lane_id == 0 && next == 0)) {
626  // change to next / prev section
627  if (lane_id <= 0) {
628  result.push_back(road.GetNextLane(s, next));
629  } else {
630  result.push_back(road.GetPrevLane(s, next));
631  }
632  }
633  } else if (!next_is_junction) {
634  // change to another road / junction
635  if (next != 0 || (lane_id == 0 && next == 0)) {
636  // single road
637  result.push_back(GetEdgeLanePointer(next_road, (next <= 0), next));
638  }
639  } else {
640  // several roads (junction)
641 
642  /// @todo Is it correct to use a road id as section id? (NS: I just added
643  /// this cast to avoid compiler warnings).
644  auto next_road_as_junction = static_cast<JuncId>(next_road);
645  auto options = GetJunctionLanes(next_road_as_junction, road_id, lane_id);
646  for (auto opt : options) {
647  result.push_back(GetEdgeLanePointer(opt.first, (opt.second <= 0), opt.second));
648  }
649  }
650 
651  return result;
652  }

654  std::vector<std::pair<RoadId, LaneId>> MapBuilder::GetJunctionLanes(
655  JuncId junction_id,
656  RoadId road_id,
657  LaneId lane_id) {
658  std::vector<std::pair<RoadId, LaneId>> result;
659 
660  // get the junction
661  Junction *junction = _map_data.GetJunction(junction_id);
662  if (junction == nullptr) {
663  return result;
664  }
665 
666  // check all connections
667  for (auto con : junction->_connections) {
668  // only connections for our road
669  if (con.second.incoming_road == road_id) {
670  // for center lane it is always next lane id 0, we don't need to search
671  // because it is not in the junction
672  if (lane_id == 0) {
673  result.push_back(std::make_pair(con.second.connecting_road, 0));
674  } else {
675  // check all lane links
676  for (auto link : con.second.lane_links) {
677  // is our lane id ?
678  if (link.from == lane_id) {
679  // add as option
680  result.push_back(std::make_pair(con.second.connecting_road, link.to));
681  }
682  }
683  }
684  }
685  }
686 
687  return result;
688  }

690  // assign pointers to the next lanes
691  void MapBuilder::CreatePointersBetweenRoadSegments(void) {
692  // process each lane to define its nexts
693  for (auto &road : _map_data._roads) {
694  for (auto &section : road.second._lane_sections) {
695  for (auto &lane : section.second._lanes) {
696 
697  // assign the next lane pointers
698  lane.second._next_lanes = GetLaneNext(road.first, section.second._id, lane.first);
699 
700  // add to each lane found, this as its predecessor
701  for (auto next_lane : lane.second._next_lanes) {
702  // add as previous

783  void MapBuilder::CreateJunctionBoundingBoxes(Map &map) {
784  for (auto &junctionpair : map._data.GetJunctions()) {
785  auto* junction = map.GetJunction(junctionpair.first);
786  auto waypoints = map.GetJunctionWaypoints(junction->GetId(), Lane::LaneType::Any);
787  const int number_intervals = 10;
788 
789  float minx = std::numeric_limits<float>::max();
790  float miny = std::numeric_limits<float>::max();
791  float minz = std::numeric_limits<float>::max();
792  float maxx = -std::numeric_limits<float>::max();
793  float maxy = -std::numeric_limits<float>::max();
794  float maxz = -std::numeric_limits<float>::max();
795 
796  auto get_min_max = [&](geom::Location position) {
797  if (position.x < minx) {
798  minx = position.x;
799  }
800  if (position.y < miny) {
801  miny = position.y;
802  }
803  if (position.z < minz) {
804  minz = position.z;
805  }
806 
807  if (position.x > maxx) {
808  maxx = position.x;
809  }
810  if (position.y > maxy) {
811  maxy = position.y;
812  }
813  if (position.z > maxz) {
814  maxz = position.z;
815  }
816  };
817 
818  for (auto &waypoint_p : waypoints) {
819  auto &waypoint_start = waypoint_p.first;
820  auto &waypoint_end = waypoint_p.second;
821  double interval = (waypoint_end.s - waypoint_start.s) / static_cast<double>(number_intervals);
822  auto next_wp = waypoint_end;
823  auto location = map.ComputeTransform(next_wp).location;
824 
825  get_min_max(location);
826 
827  next_wp = waypoint_start;
828  location = map.ComputeTransform(next_wp).location;
829 
830  get_min_max(location);
831 
832  for (int i = 0; i < number_intervals; ++i) {
833  if (interval < std::numeric_limits<double>::epsilon())
834  break;
835  next_wp = map.GetNext(next_wp, interval).back();
836 
837  location = map.ComputeTransform(next_wp).location;
838  get_min_max(location);
839  }
840  }
841  carla::geom::Location location(0.5f * (maxx + minx), 0.5f * (maxy + miny), 0.5f * (maxz + minz));
842  carla::geom::Vector3D extent(0.5f * (maxx - minx), 0.5f * (maxy - miny), 0.5f * (maxz - minz));
843 
844  junction->_bounding_box = carla::geom::BoundingBox(location, extent);
845  }
846  }
