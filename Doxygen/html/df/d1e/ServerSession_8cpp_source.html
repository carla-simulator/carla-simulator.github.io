35  void ServerSession::Open(
36  callback_function_type on_opened,
37  callback_function_type on_closed) {
38  DEBUG_ASSERT(on_opened && on_closed);
39  _on_closed = std::move(on_closed);
40  StartTimer();
41  auto self = shared_from_this(); // To keep myself alive.
42  _strand.post([=]() {
43 
44  auto handle_query = [this, self, callback=std::move(on_opened)](
45  const boost::system::error_code &ec,
46  size_t DEBUG_ONLY(bytes_received)) {
47  if (!ec) {
48  DEBUG_ASSERT_EQ(bytes_received, sizeof(_stream_id));
49  log_debug("session", _session_id, "for stream", _stream_id, " started");
50  _strand.context().post([=]() { callback(self); });
51  } else {
52  log_error("session", _session_id, ": error retrieving stream id :", ec.message());
53  CloseNow();
54  }
55  };
56 
57  // Read the stream id.
58  _deadline.expires_from_now(_timeout);
59  boost::asio::async_read(
60  _socket,
61  boost::asio::buffer(&_stream_id, sizeof(_stream_id)),
62  _strand.wrap(handle_query));
63  });
64  }

66  void ServerSession::Write(std::shared_ptr<const Message> message) {
67  DEBUG_ASSERT(message != nullptr);
68  DEBUG_ASSERT(!message->empty());
69  auto self = shared_from_this();
70  _strand.post([=]() {
71  if (!_socket.is_open()) {
72  return;
73  }
74  if (_is_writing) {
75  log_debug("session", _session_id, ": connection too slow: message discarded");
76  return;
77  }
78  _is_writing = true;
79 
80  auto handle_sent = [this, self, message](const boost::system::error_code &ec, size_t DEBUG_ONLY(bytes)) {
81  _is_writing = false;
82  if (ec) {
83  log_info("session", _session_id, ": error sending data :", ec.message());
84  CloseNow();
85  } else {
86  DEBUG_ONLY(log_debug("session", _session_id, ": successfully sent", bytes, "bytes"));
87  DEBUG_ASSERT_EQ(bytes, sizeof(message_size_type) + message->size());
88  }
89  };
90 
91  log_debug("session", _session_id, ": sending message of", message->size(), "bytes");
92 
93  _deadline.expires_from_now(_timeout);
94  boost::asio::async_write(
95  _socket,
96  message->GetBufferSequence(),
97  _strand.wrap(handle_sent));
98  });
99  }

105  void ServerSession::StartTimer() {
106  if (_deadline.expires_at() <= boost::asio::deadline_timer::traits_type::now()) {
107  log_debug("session", _session_id, "timed out");
108  Close();
109  } else {
110  _deadline.async_wait([this, self=shared_from_this()](boost::system::error_code ec) {
111  if (!ec) {
112  StartTimer();
113  } else {
114  log_debug("session", _session_id, "timed out error:", ec.message());
115  }
116  });
117  }
118  }

120  void ServerSession::CloseNow() {
121  DEBUG_ASSERT(_strand.running_in_this_thread());
122  _deadline.cancel();
123  if (_socket.is_open()) {
124  _socket.close();
125  }
126  _strand.context().post([self=shared_from_this()]() {
127  DEBUG_ASSERT(self->_on_closed);
128  self->_on_closed(self);
129  });
130  log_debug("session", _session_id, "closed");
131  }
