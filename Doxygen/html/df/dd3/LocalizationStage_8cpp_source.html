64  void LocalizationStage::Action() {
65 
66  // Selecting output frames based on selector keys.
67  const auto current_planner_frame = planner_frame_selector ? planner_frame_a : planner_frame_b;
68  const auto current_collision_frame = collision_frame_selector ? collision_frame_a : collision_frame_b;
69  const auto current_traffic_light_frame =
70  traffic_light_frame_selector ? traffic_light_frame_a : traffic_light_frame_b;
71 
72  // Looping over registered actors.
73  for (uint i = 0u; i < actor_list.size(); ++i) {
74 
75  const Actor vehicle = actor_list.at(i);
76  const ActorId actor_id = vehicle->GetId();
77  const cg::Location vehicle_location = vehicle->GetLocation();
78  const float vehicle_velocity = vehicle->GetVelocity().Length();
79 
80  //TODO: Improve search so it doesn't do it every loop..
81  auto search = idle_time.find(actor_id);
82  if (search == idle_time.end()) {
83  idle_time[actor_id] = chr::system_clock::now();
84  }
85 
86  const float horizon_size = std::max(
87  WAYPOINT_TIME_HORIZON * std::sqrt(vehicle_velocity * 10.0f),
88  MINIMUM_HORIZON_LENGTH);
89 
90  Buffer &waypoint_buffer = buffer_list->at(i);
91 
92  // Purge passed waypoints.
93  if (!waypoint_buffer.empty()) {
94  float dot_product = DeviationDotProduct(vehicle, waypoint_buffer.front()->GetLocation(), true);
95 
96  while (dot_product <= 0.0f && !waypoint_buffer.empty()) {
97 
98  PopWaypoint(waypoint_buffer, actor_id);
99  if (!waypoint_buffer.empty()) {
100  dot_product = DeviationDotProduct(vehicle, waypoint_buffer.front()->GetLocation(), true);
101  }
102  }
103  }
104 
105  // Initializing buffer if it is empty.
106  if (waypoint_buffer.empty()) {
107  SimpleWaypointPtr closest_waypoint = local_map.GetWaypointInVicinity(vehicle_location);
108  if (closest_waypoint == nullptr) {
109  closest_waypoint = local_map.GetWaypoint(vehicle_location);
110  }
111  PushWaypoint(waypoint_buffer, actor_id, closest_waypoint);
112  }
113 
114  // Assign a lane change.
115  const SimpleWaypointPtr front_waypoint = waypoint_buffer.front();
116  const ChangeLaneInfo lane_change_info = parameters.GetForceLaneChange(vehicle);
117  const bool force_lane_change = lane_change_info.change_lane;
118  const bool lane_change_direction = lane_change_info.direction;
119 
120  if ((parameters.GetAutoLaneChange(vehicle) || force_lane_change) &&
121  !front_waypoint->CheckJunction()) {
122 
123  SimpleWaypointPtr change_over_point = AssignLaneChange(
124  vehicle, force_lane_change, lane_change_direction);
125 
126  if (change_over_point != nullptr) {
127  auto number_of_pops = waypoint_buffer.size();
128  for (uint j = 0u; j < number_of_pops; ++j) {
129  PopWaypoint(waypoint_buffer, actor_id);
130  }
131 
132  PushWaypoint(waypoint_buffer, actor_id, change_over_point);
133  }
134  }
135 
136  // Populating the buffer.
137  while (waypoint_buffer.back()->DistanceSquared(waypoint_buffer.front())
138  <= std::pow(horizon_size, 2)) {
139 
140  std::vector<SimpleWaypointPtr> next_waypoints = waypoint_buffer.back()->GetNextWaypoint();
141  uint selection_index = 0u;
142  // Pseudo-randomized path selection if found more than one choice.
143  if (next_waypoints.size() > 1) {
144  selection_index = static_cast<uint>(rand()) % next_waypoints.size();
145  }
146 
147  PushWaypoint(waypoint_buffer, actor_id, next_waypoints.at(selection_index));
148  }
149 
150  // Generating output.
151  const float target_point_distance = std::max(std::ceil(vehicle_velocity * TARGET_WAYPOINT_TIME_HORIZON),
152  TARGET_WAYPOINT_HORIZON_LENGTH);
153  SimpleWaypointPtr target_waypoint = waypoint_buffer.front();
154  for (uint j = 0u;
155  (j < waypoint_buffer.size()) &&
156  (waypoint_buffer.front()->DistanceSquared(target_waypoint)
157  < std::pow(target_point_distance, 2));
158  ++j) {
159  target_waypoint = waypoint_buffer.at(j);
160  }
161  const cg::Location target_location = target_waypoint->GetLocation();
162  float dot_product = DeviationDotProduct(vehicle, target_location);
163  float cross_product = DeviationCrossProduct(vehicle, target_location);
164  dot_product = 1.0f - dot_product;
165  if (cross_product < 0.0f) {
166  dot_product *= -1.0f;
167  }
168 
169  float distance = 0.0f; // TODO: use in PID
170 
171  // Filtering out false junctions on highways.
172  // On highways, if there is only one possible path and the section is
173  // marked as intersection, ignore it.
174  const auto vehicle_reference = boost::static_pointer_cast<cc::Vehicle>(vehicle);
175  const float speed_limit = vehicle_reference->GetSpeedLimit();
176  const float look_ahead_distance = std::max(2.0f * vehicle_velocity, MINIMUM_JUNCTION_LOOK_AHEAD);
177 
178  SimpleWaypointPtr look_ahead_point = waypoint_buffer.front();
179  uint look_ahead_index = 0u;
180  for (uint j = 0u;
181  (waypoint_buffer.front()->DistanceSquared(look_ahead_point)
182  < std::pow(look_ahead_distance, 2)) &&
183  (j < waypoint_buffer.size());
184  ++j) {
185  look_ahead_point = waypoint_buffer.at(j);
186  look_ahead_index = j;
187  }
188 
189  bool approaching_junction = false;
190  if (waypoint_buffer.front()->CheckJunction() || (look_ahead_point->CheckJunction() && !(waypoint_buffer.front()->CheckJunction()))) {
191  if (speed_limit > HIGHWAY_SPEED) {
192  for (uint j = 0u; (j < look_ahead_index) && !approaching_junction; ++j) {
193  SimpleWaypointPtr swp = waypoint_buffer.at(j);
194  if (swp->GetNextWaypoint().size() > 1) {
195  approaching_junction = true;
196  }
197  }
198  } else {
199  approaching_junction = true;
200  }
201  }
202 
203  // Clean up tracking list by remove vehicles that are too far away.
204  const ActorIdSet current_tracking_list = track_traffic.GetOverlappingVehicles(actor_id);
205  for (ActorId tracking_id: current_tracking_list) {
206  if (!waypoint_buffer.empty()) {
207 
208  const cg::Location tracking_location = actor_list.at(
209  vehicle_id_to_index.at(tracking_id))->GetLocation();
210 
211  const cg::Location buffer_front_loc = waypoint_buffer.front()->GetLocation();
212  const cg::Location buffer_mid_lock = waypoint_buffer.at(
213  static_cast<unsigned long>(std::floor(waypoint_buffer.size()/2)))->GetLocation();
214  const cg::Location buffer_back_loc = waypoint_buffer.back()->GetLocation();
215 
216  const double squared_buffer_length = std::pow(
217  buffer_front_loc.Distance(buffer_mid_lock) +
218  buffer_mid_lock.Distance(buffer_back_loc), 2);
219 
220  if (cg::Math::DistanceSquared(vehicle_location, tracking_location) > squared_buffer_length) {
221  track_traffic.RemoveOverlappingVehicle(actor_id, tracking_id);
222  track_traffic.RemoveOverlappingVehicle(tracking_id, actor_id);
223  }
224  }
225  }
226 
227  // Editing output frames.
228  LocalizationToPlannerData &planner_message = current_planner_frame->at(i);
229  planner_message.actor = vehicle;
230  planner_message.deviation = dot_product;
231  planner_message.distance = distance;
232  planner_message.approaching_true_junction = approaching_junction;
233 
234  // Reading current messenger state of the collision stage before modifying it's frame.
235  if ((collision_messenger->GetState() != collision_messenger_state) &&
236  !collision_frame_ready) {
237 
238  LocalizationToCollisionData &collision_message = current_collision_frame->at(i);
239  collision_message.actor = vehicle;
240  collision_message.buffer = waypoint_buffer;
241  collision_message.overlapping_actors = track_traffic.GetOverlappingVehicles(actor_id);
242  }
243 
244  LocalizationToTrafficLightData &traffic_light_message = current_traffic_light_frame->at(i);
245  traffic_light_message.actor = vehicle;
246  traffic_light_message.closest_waypoint = waypoint_buffer.front();
247  traffic_light_message.junction_look_ahead_waypoint = waypoint_buffer.at(look_ahead_index);
248  }
249 
250  if ((collision_messenger->GetState() != collision_messenger_state)
251  && !collision_frame_ready) {
252 
253  collision_frame_ready = true;
254  }
255 
256  }

258  void LocalizationStage::DataReceiver() {
259 
260  // Building a list of registered actors and
261  // connecting the vehicle ids to their position indices on data arrays.
262 
263  if (registered_actors_state != registered_actors.GetState()) {
264 
265  actor_list = registered_actors.GetList();
266 
267  uint index = 0u;
268  for (auto &actor: actor_list) {
269 
270  vehicle_id_to_index.insert({actor->GetId(), index});
271  ++index;
272  }
273 
274  registered_actors_state = registered_actors.GetState();
275  }
276 
277  // Allocating new containers for the changed number of registered vehicles.
278  if (number_of_vehicles != actor_list.size()) {
279 
280  number_of_vehicles = static_cast<uint>(actor_list.size());
281  // Allocating the buffer lists.
282  buffer_list = std::make_shared<BufferList>(number_of_vehicles);
283  // Allocating output frames to be shared with the motion planner stage.
284  planner_frame_a = std::make_shared<LocalizationToPlannerFrame>(number_of_vehicles);
285  planner_frame_b = std::make_shared<LocalizationToPlannerFrame>(number_of_vehicles);
286  // Allocating output frames to be shared with the collision stage.
287  collision_frame_a = std::make_shared<LocalizationToCollisionFrame>(number_of_vehicles);
288  collision_frame_b = std::make_shared<LocalizationToCollisionFrame>(number_of_vehicles);
289  // Allocating output frames to be shared with the traffic light stage
290  traffic_light_frame_a = std::make_shared<LocalizationToTrafficLightFrame>(number_of_vehicles);
291  traffic_light_frame_b = std::make_shared<LocalizationToTrafficLightFrame>(number_of_vehicles);
292  }

296  void LocalizationStage::DataSender() {
297 
298  // Since send/receive calls on messenger objects can block if the other
299  // end hasn't received/sent data, choose to block on only those stages
300  // which takes the most priority (which needs the highest rate of data feed)
301  // to run the system well.
302 
303  const DataPacket<std::shared_ptr<LocalizationToPlannerFrame>> planner_data_packet = {
304  planner_messenger_state,
305  planner_frame_selector ? planner_frame_a : planner_frame_b
306  };
307  planner_frame_selector = !planner_frame_selector;
308  planner_messenger_state = planner_messenger->SendData(planner_data_packet);
309 
310  // Send data to collision stage only if it has finished
311  // processing, received the previous message and started processing it.
312  int collision_messenger_current_state = collision_messenger->GetState();
313  if ((collision_messenger_current_state != collision_messenger_state) &&
314  collision_frame_ready) {
315 
316  const DataPacket<std::shared_ptr<LocalizationToCollisionFrame>> collision_data_packet = {
317  collision_messenger_state,
318  collision_frame_selector ? collision_frame_a : collision_frame_b
319  };
320 
321  collision_messenger_state = collision_messenger->SendData(collision_data_packet);
322  collision_frame_selector = !collision_frame_selector;
323  collision_frame_ready = false;
324  }
325 
326  // Send data to traffic light stage only if it has finished
327  // processing, received the previous message and started processing it.
328  const int traffic_light_messenger_current_state = traffic_light_messenger->GetState();
329  if (traffic_light_messenger_current_state != traffic_light_messenger_state) {
330  const DataPacket<std::shared_ptr<LocalizationToTrafficLightFrame>> traffic_light_data_packet = {
331  traffic_light_messenger_state,
332  traffic_light_frame_selector ? traffic_light_frame_a : traffic_light_frame_b
333  };
334 
335  traffic_light_messenger_state = traffic_light_messenger->SendData(traffic_light_data_packet);
336  traffic_light_frame_selector = !traffic_light_frame_selector;
337  }
338  }

348 
349  const uint64_t waypoint_id = waypoint->GetId();
350  buffer.push_back(waypoint);
351  track_traffic.UpdatePassingVehicle(waypoint_id, actor_id);
352 
353  const ActorIdSet current_actors = track_traffic.GetPassingVehicles(actor_id);
354  const ActorIdSet new_overlapping_actors = track_traffic.GetPassingVehicles(waypoint_id);
355  ActorIdSet actor_set_difference;
356 
357  std::set_difference(
358  new_overlapping_actors.begin(), new_overlapping_actors.end(),
359  current_actors.begin(), current_actors.end(),
360  std::inserter(actor_set_difference, actor_set_difference.end())
361  );
362 
363  for (auto new_actor_id: actor_set_difference) {
364 
365  track_traffic.UpdateOverlappingVehicle(actor_id, new_actor_id);
366  track_traffic.UpdateOverlappingVehicle(new_actor_id, actor_id);
367  }

370  void LocalizationStage::PopWaypoint(Buffer& buffer, ActorId actor_id) {
371 
372  const uint64_t removed_waypoint_id = buffer.front()->GetId();
373  buffer.pop_front();
374  track_traffic.RemovePassingVehicle(removed_waypoint_id, actor_id);
375 
376  if (!buffer.empty()) {
377 
378  const ActorIdSet current_actors = track_traffic.GetPassingVehicles(removed_waypoint_id);
379  const ActorIdSet new_overlapping_actors = track_traffic.GetPassingVehicles(buffer.front()->GetId());
380  ActorIdSet actor_set_difference;
381 
382  std::set_difference(
383  current_actors.begin(), current_actors.end(),
384  new_overlapping_actors.begin(), new_overlapping_actors.end(),
385  std::inserter(actor_set_difference, actor_set_difference.end())
386  );
387 
388  for (auto& old_actor_id: actor_set_difference) {
389 
390  track_traffic.RemoveOverlappingVehicle(actor_id, old_actor_id);
391  track_traffic.RemoveOverlappingVehicle(old_actor_id, actor_id);
392  }
393  } else {
394 
395  const ActorIdSet currently_tracked_vehicles = track_traffic.GetOverlappingVehicles(actor_id);
396  for (auto& tracked_id: currently_tracked_vehicles) {
397  track_traffic.RemoveOverlappingVehicle(actor_id, tracked_id);
398  track_traffic.RemoveOverlappingVehicle(tracked_id, actor_id);
399  }

403  SimpleWaypointPtr LocalizationStage::AssignLaneChange(Actor vehicle, bool force, bool direction) {
404 
405  const ActorId actor_id = vehicle->GetId();
406  const cg::Location vehicle_location = vehicle->GetLocation();
407  const float vehicle_velocity = vehicle->GetVelocity().Length();
408 
409  const Buffer& waypoint_buffer = buffer_list->at(vehicle_id_to_index.at(actor_id));
410  const SimpleWaypointPtr& current_waypoint = waypoint_buffer.front();
411 
412  bool need_to_change_lane = false;
413  const auto left_waypoint = current_waypoint->GetLeftWaypoint();
414  const auto right_waypoint = current_waypoint->GetRightWaypoint();
415 
416  if (!force) {
417 
418  const auto blocking_vehicles = track_traffic.GetOverlappingVehicles(actor_id);
419 
420  bool abort_lane_change = false;
421  for (auto i = blocking_vehicles.begin();
422  i != blocking_vehicles.end() && !abort_lane_change;
423  ++i) {
424 
425  const ActorId &other_vehicle_id = *i;
426  const Buffer& other_buffer = buffer_list->at(vehicle_id_to_index.at(other_vehicle_id));
427  const SimpleWaypointPtr& other_current_waypoint = other_buffer.front();
428  const cg::Location other_location = other_current_waypoint->GetLocation();
429 
430  bool distant_lane_availability = false;
431  const auto other_neighbouring_lanes = {
432  other_current_waypoint->GetLeftWaypoint(),
433  other_current_waypoint->GetRightWaypoint()};
434 
435  for (auto& candidate_lane_wp: other_neighbouring_lanes) {
436  if (candidate_lane_wp != nullptr &&
437  track_traffic.GetPassingVehicles(candidate_lane_wp->GetId()).size() == 0 &&
438  vehicle_velocity < HIGHWAY_SPEED) {
439  distant_lane_availability = true;
440  }
441  }
442 
443  const cg::Vector3D reference_heading = current_waypoint->GetForwardVector();
444  const cg::Vector3D other_heading = other_current_waypoint->GetForwardVector();
445 
446  if (other_vehicle_id != actor_id &&
447  !current_waypoint->CheckJunction() &&
448  !other_current_waypoint->CheckJunction() &&
449  cg::Math::Dot(reference_heading, other_heading) > MAXIMUM_LANE_OBSTACLE_CURVATURE) {
450 
451  const float squared_vehicle_distance = cg::Math::DistanceSquared(other_location, vehicle_location);
452  const float deviation_dot = DeviationDotProduct(vehicle, other_location);
453 
454  if (deviation_dot > 0.0f) {
455 
456  if (distant_lane_availability &&
457  squared_vehicle_distance > std::pow(MINIMUM_LANE_CHANGE_DISTANCE, 2)) {
458 
459  need_to_change_lane = true;
460  } else if (squared_vehicle_distance < std::pow(MINIMUM_LANE_CHANGE_DISTANCE, 2)) {
461 
462  need_to_change_lane = false;
463  abort_lane_change = true;
464  }
465 
466  }
467  }
468  }
469 
470  } else {
471 
472  need_to_change_lane = true;
473  }
474 
475  const float change_over_distance = std::max(2.0f*vehicle_velocity, 10.0f);
476  bool possible_to_lane_change = false;
477  SimpleWaypointPtr change_over_point = nullptr;
478 
479  if (need_to_change_lane) {
480 
481  std::vector<SimpleWaypointPtr> candidate_points;
482  if (force) {
483  if (direction) {
484  candidate_points.push_back(left_waypoint);
485  } else {
486  candidate_points.push_back(right_waypoint);
487  }
488  } else {
489  candidate_points.push_back(left_waypoint);
490  candidate_points.push_back(right_waypoint);
491  }
492 
493  for (auto target_lane_wp: candidate_points) {
494  if (!force &&
495  !possible_to_lane_change &&
496  target_lane_wp != nullptr &&
497  track_traffic.GetPassingVehicles(target_lane_wp->GetId()).size() == 0) {
498 
499  possible_to_lane_change = true;
500  change_over_point = target_lane_wp;
501  } else if (force) {
502 
503  possible_to_lane_change = true;
504  change_over_point = target_lane_wp;
505  }
506  }
507  }
508 
509  if (need_to_change_lane && possible_to_lane_change) {
510  const auto starting_point = change_over_point;
511  while (change_over_point->DistanceSquared(starting_point) < std::pow(change_over_distance, 2) &&
512  !change_over_point->CheckJunction()) {
513  change_over_point = change_over_point->GetNextWaypoint()[0];
514  }
515  return change_over_point;
516  } else {
517  return nullptr;
518  }
519  }
