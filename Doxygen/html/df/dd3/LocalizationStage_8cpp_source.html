31 void LocalizationStage::Update(const unsigned long index) {
32 
33  const ActorId actor_id = vehicle_id_list.at(index);
34  const cg::Location vehicle_location = simulation_state.GetLocation(actor_id);
35  const cg::Vector3D heading_vector = simulation_state.GetHeading(actor_id);
36  const cg::Vector3D vehicle_velocity_vector = simulation_state.GetVelocity(actor_id);
37  const float vehicle_speed = vehicle_velocity_vector.Length();
38 
39  // Speed dependent waypoint horizon length.
40  float horizon_length = std::min(vehicle_speed * HORIZON_RATE + MINIMUM_HORIZON_LENGTH, MAXIMUM_HORIZON_LENGTH);
41  const float horizon_square = SQUARE(horizon_length);
42 
43  if (buffer_map.find(actor_id) == buffer_map.end()) {
44  buffer_map.insert({actor_id, Buffer()});
45  }
46 
47  Buffer &waypoint_buffer = buffer_map.at(actor_id);
48 
49  // Clear buffer if vehicle is too far from the first waypoint in the buffer.
50  if (!waypoint_buffer.empty() &&
51  cg::Math::DistanceSquared(waypoint_buffer.front()->GetLocation(),
52  vehicle_location) > SQUARE(MAX_START_DISTANCE)) {
53 
54  auto number_of_pops = waypoint_buffer.size();
55  for (uint64_t j = 0u; j < number_of_pops; ++j) {
56  PopWaypoint(actor_id, track_traffic, waypoint_buffer);
57  }
58  }
59 
60  bool is_at_junction_entrance = false;
61 
62  if (!waypoint_buffer.empty()) {
63  // Purge passed waypoints.
64  float dot_product = DeviationDotProduct(vehicle_location, heading_vector, waypoint_buffer.front()->GetLocation());
65  while (dot_product <= 0.0f && !waypoint_buffer.empty()) {
66 
67  PopWaypoint(actor_id, track_traffic, waypoint_buffer);
68  if (!waypoint_buffer.empty()) {
69  dot_product = DeviationDotProduct(vehicle_location, heading_vector, waypoint_buffer.front()->GetLocation());
70  }
71  }
72 
73  if (!waypoint_buffer.empty()) {
74  // Determine if the vehicle is at the entrance of a junction.
75  SimpleWaypointPtr look_ahead_point = GetTargetWaypoint(waypoint_buffer, JUNCTION_LOOK_AHEAD).first;
76  is_at_junction_entrance = !waypoint_buffer.front()->CheckJunction() && look_ahead_point->CheckJunction();
77  if (is_at_junction_entrance
78  // Exception for roundabout in Town03.
79  && local_map->GetMapName() == "Town03"
80  && vehicle_location.SquaredLength() < SQUARE(30)) {
81  is_at_junction_entrance = false;
82  }
83  }
84 
85  // Purge waypoints too far from the front of the buffer.
86  while (!is_at_junction_entrance
87  && !waypoint_buffer.empty()
88  && waypoint_buffer.back()->DistanceSquared(waypoint_buffer.front()) > horizon_square) {
89  PopWaypoint(actor_id, track_traffic, waypoint_buffer, false);
90  }
91  }
92 
93  // Initializing buffer if it is empty.
94  if (waypoint_buffer.empty()) {
95  SimpleWaypointPtr closest_waypoint = local_map->GetWaypoint(vehicle_location);
96  PushWaypoint(actor_id, track_traffic, waypoint_buffer, closest_waypoint);
97  }
98 
99  // Assign a lane change.
100  const ChangeLaneInfo lane_change_info = parameters.GetForceLaneChange(actor_id);
101  bool force_lane_change = lane_change_info.change_lane;
102  bool lane_change_direction = lane_change_info.direction;
103 
104  if (!force_lane_change) {
105  float perc_keep_right = parameters.GetKeepRightPercentage(actor_id);
106  if (perc_keep_right >= 0.0f && perc_keep_right >= pgen.next()) {
107  force_lane_change = true;
108  lane_change_direction = true;
109  }
110  }
111 
112  const SimpleWaypointPtr front_waypoint = waypoint_buffer.front();
113  const float lane_change_distance = SQUARE(std::max(10.0f * vehicle_speed, INTER_LANE_CHANGE_DISTANCE));
114 
115  bool recently_not_executed_lane_change = last_lane_change_location.find(actor_id) == last_lane_change_location.end();
116  bool done_with_previous_lane_change = true;
117  if (!recently_not_executed_lane_change) {
118  float distance_frm_previous = cg::Math::DistanceSquared(last_lane_change_location.at(actor_id), vehicle_location);
119  done_with_previous_lane_change = distance_frm_previous > lane_change_distance;
120  }
121  bool auto_or_force_lane_change = parameters.GetAutoLaneChange(actor_id) || force_lane_change;
122  bool front_waypoint_not_junction = !front_waypoint->CheckJunction();
123 
124  if (auto_or_force_lane_change
125  && front_waypoint_not_junction
126  && (recently_not_executed_lane_change || done_with_previous_lane_change)) {
127 
128  SimpleWaypointPtr change_over_point = AssignLaneChange(actor_id, vehicle_location, vehicle_speed,
129  force_lane_change, lane_change_direction);
130 
131  if (change_over_point != nullptr) {
132  if (last_lane_change_location.find(actor_id) != last_lane_change_location.end()) {
133  last_lane_change_location.at(actor_id) = vehicle_location;
134  } else {
135  last_lane_change_location.insert({actor_id, vehicle_location});
136  }
137  auto number_of_pops = waypoint_buffer.size();
138  for (uint64_t j = 0u; j < number_of_pops; ++j) {
139  PopWaypoint(actor_id, track_traffic, waypoint_buffer);
140  }
141  PushWaypoint(actor_id, track_traffic, waypoint_buffer, change_over_point);
142  }
143  }
144 
145  // Populating the buffer.
146  while (waypoint_buffer.back()->DistanceSquared(waypoint_buffer.front()) <= horizon_square) {
147 
148  std::vector<SimpleWaypointPtr> next_waypoints = waypoint_buffer.back()->GetNextWaypoint();
149  uint64_t selection_index = 0u;
150  // Pseudo-randomized path selection if found more than one choice.
151  if (next_waypoints.size() > 1) {
152  selection_index = static_cast<uint64_t>(pgen.next()) % next_waypoints.size();
153  }
154  SimpleWaypointPtr next_wp = next_waypoints.at(selection_index);
155  if (next_wp == nullptr) {
156  for (auto &wp : next_waypoints) {
157  if (wp != nullptr) {
158  next_wp = wp;
159  break;
160  }
161  }
162  }
163  PushWaypoint(actor_id, track_traffic, waypoint_buffer, next_wp);
164  }
165 
166  ExtendAndFindSafeSpace(actor_id, is_at_junction_entrance, waypoint_buffer);
167 
168  // Editing output array
169  LocalizationData &output = output_array.at(index);
170  output.is_at_junction_entrance = is_at_junction_entrance;
171 
172  if (is_at_junction_entrance) {
173  const SimpleWaypointPair &safe_space_end_points = vehicles_at_junction_entrance.at(actor_id);
174  output.junction_end_point = safe_space_end_points.first;
175  output.safe_point = safe_space_end_points.second;
176  } else {
177  output.junction_end_point = nullptr;
178  output.safe_point = nullptr;
179  }
180 
181  // Updating geodesic grid position for actor.
182  track_traffic.UpdateGridPosition(actor_id, waypoint_buffer);
183 }

185 void LocalizationStage::ExtendAndFindSafeSpace(const ActorId actor_id,
186  const bool is_at_junction_entrance,
187  Buffer &waypoint_buffer) {
188 
189  SimpleWaypointPtr junction_end_point = nullptr;
190  SimpleWaypointPtr safe_point_after_junction = nullptr;
191 
192  if (is_at_junction_entrance
193  && vehicles_at_junction_entrance.find(actor_id) == vehicles_at_junction_entrance.end()) {
194 
195  bool entered_junction = false;
196  bool past_junction = false;
197  bool safe_point_found = false;
198  SimpleWaypointPtr current_waypoint = nullptr;
199  SimpleWaypointPtr junction_begin_point = nullptr;
200  float safe_distance_squared = SQUARE(SAFE_DISTANCE_AFTER_JUNCTION);
201 
202  // Scanning existing buffer points.
203  for (unsigned long i = 0u; i < waypoint_buffer.size() && !safe_point_found; ++i) {
204  current_waypoint = waypoint_buffer.at(i);
205  if (!entered_junction && current_waypoint->CheckJunction()) {
206  entered_junction = true;
207  junction_begin_point = current_waypoint;
208  }
209  if (entered_junction && !past_junction && !current_waypoint->CheckJunction()) {
210  past_junction = true;
211  junction_end_point = current_waypoint;
212  }
213  if (past_junction && junction_end_point->DistanceSquared(current_waypoint) > safe_distance_squared) {
214  safe_point_found = true;
215  safe_point_after_junction = current_waypoint;
216  }
217  }
218 
219  // Extend buffer if safe point not found.
220  if (!safe_point_found) {
221  while (!past_junction) {
222  current_waypoint = current_waypoint->GetNextWaypoint().front();
223  PushWaypoint(actor_id, track_traffic, waypoint_buffer, current_waypoint);
224  if (!current_waypoint->CheckJunction()) {
225  past_junction = true;
226  junction_end_point = current_waypoint;
227  }
228  }
229 
230  while (!safe_point_found) {
231  std::vector<SimpleWaypointPtr> next_waypoints = current_waypoint->GetNextWaypoint();
232  if ((junction_end_point->DistanceSquared(current_waypoint) > safe_distance_squared)
233  || next_waypoints.size() > 1
234  || current_waypoint->CheckJunction()) {
235 
236  safe_point_found = true;
237  safe_point_after_junction = current_waypoint;
238  } else {
239  current_waypoint = next_waypoints.front();
240  PushWaypoint(actor_id, track_traffic, waypoint_buffer, current_waypoint);
241  }
242  }
243  }
244 
245  if (junction_begin_point->DistanceSquared(junction_end_point) < SQUARE(MIN_JUNCTION_LENGTH)) {
246  junction_end_point = nullptr;
247  safe_point_after_junction = nullptr;
248  }
249 
250  vehicles_at_junction_entrance.insert({actor_id, {junction_end_point, safe_point_after_junction}});
251  }
252  else if (!is_at_junction_entrance
253  && vehicles_at_junction_entrance.find(actor_id) != vehicles_at_junction_entrance.end()) {
254 
255  vehicles_at_junction_entrance.erase(actor_id);
256  }
257 }

269 SimpleWaypointPtr LocalizationStage::AssignLaneChange(const ActorId actor_id,
270  const cg::Location vehicle_location,
271  const float vehicle_speed,
272  bool force, bool direction) {
273 
274  // Waypoint representing the new starting point for the waypoint buffer
275  // due to lane change. Remains nullptr if lane change not viable.
276  SimpleWaypointPtr change_over_point = nullptr;
277 
278  // Retrieve waypoint buffer for current vehicle.
279  const Buffer &waypoint_buffer = buffer_map.at(actor_id);
280 
281  // Check buffer is not empty.
282  if (!waypoint_buffer.empty()) {
283  // Get the left and right waypoints for the current closest waypoint.
284  const SimpleWaypointPtr &current_waypoint = waypoint_buffer.front();
285  const SimpleWaypointPtr left_waypoint = current_waypoint->GetLeftWaypoint();
286  const SimpleWaypointPtr right_waypoint = current_waypoint->GetRightWaypoint();
287 
288  // Retrieve vehicles with overlapping waypoint buffers with current vehicle.
289  const auto blocking_vehicles = track_traffic.GetOverlappingVehicles(actor_id);
290 
291  // Find immediate in-lane obstacle and check if any are too close to initiate lane change.
292  bool obstacle_too_close = false;
293  float minimum_squared_distance = std::numeric_limits<float>::infinity();
294  ActorId obstacle_actor_id = 0u;
295  for (auto i = blocking_vehicles.begin();
296  i != blocking_vehicles.end() && !obstacle_too_close && !force;
297  ++i) {
298  const ActorId &other_actor_id = *i;
299  // Find vehicle in buffer map and check if it's buffer is not empty.
300  if (buffer_map.find(other_actor_id) != buffer_map.end() && !buffer_map.at(other_actor_id).empty()) {
301  const Buffer &other_buffer = buffer_map.at(other_actor_id);
302  const SimpleWaypointPtr &other_current_waypoint = other_buffer.front();
303  const cg::Location other_location = other_current_waypoint->GetLocation();
304 
305  const cg::Vector3D reference_heading = current_waypoint->GetForwardVector();
306  cg::Vector3D reference_to_other = other_location - current_waypoint->GetLocation();
307  const cg::Vector3D other_heading = other_current_waypoint->GetForwardVector();
308 
309  WaypointPtr current_raw_waypoint = current_waypoint->GetWaypoint();
310  WaypointPtr other_current_raw_waypoint = other_current_waypoint->GetWaypoint();
311  // Check both vehicles are not in junction,
312  // Check if the other vehicle is in front of the current vehicle,
313  // Check if the two vehicles have acceptable angular deviation between their headings.
314  if (!current_waypoint->CheckJunction()
315  && !other_current_waypoint->CheckJunction()
316  && other_current_raw_waypoint->GetRoadId() == current_raw_waypoint->GetRoadId()
317  && other_current_raw_waypoint->GetLaneId() == current_raw_waypoint->GetLaneId()
318  && cg::Math::Dot(reference_heading, reference_to_other) > 0.0f
319  && cg::Math::Dot(reference_heading, other_heading) > MAXIMUM_LANE_OBSTACLE_CURVATURE) {
320  float squared_distance = cg::Math::DistanceSquared(vehicle_location, other_location);
321  // Abort if the obstacle is too close.
322  if (squared_distance > SQUARE(MINIMUM_LANE_CHANGE_DISTANCE)) {
323  // Remember if the new vehicle is closer.
324  if (squared_distance < minimum_squared_distance && squared_distance < SQUARE(MAXIMUM_LANE_OBSTACLE_DISTANCE)) {
325  minimum_squared_distance = squared_distance;
326  obstacle_actor_id = other_actor_id;
327  }
328  } else {
329  obstacle_too_close = true;
330  }
331  }
332  }
333  }
334 
335  // If a valid immediate obstacle found.
336  if (!obstacle_too_close && obstacle_actor_id != 0u && !force) {
337  const Buffer &other_buffer = buffer_map.at(obstacle_actor_id);
338  const SimpleWaypointPtr &other_current_waypoint = other_buffer.front();
339  const auto other_neighbouring_lanes = {other_current_waypoint->GetLeftWaypoint(),
340  other_current_waypoint->GetRightWaypoint()};
341 
342  // Flags reflecting whether adjacent lanes are free near the obstacle.
343  bool distant_left_lane_free = false;
344  bool distant_right_lane_free = false;
345 
346  // Check if the neighbouring lanes near the obstructing vehicle are free of other vehicles.
347  bool left_right = true;
348  for (auto &candidate_lane_wp : other_neighbouring_lanes) {
349  if (candidate_lane_wp != nullptr &&
350  track_traffic.GetPassingVehicles(candidate_lane_wp->GetId()).size() == 0) {
351 
352  if (left_right)
353  distant_left_lane_free = true;
354  else
355  distant_right_lane_free = true;
356  }
357  left_right = !left_right;
358  }
359 
360  // Based on what lanes are free near the obstacle,
361  // find the change over point with no vehicles passing through them.
362  if (distant_right_lane_free && right_waypoint != nullptr
363  && track_traffic.GetPassingVehicles(right_waypoint->GetId()).size() == 0) {
364  change_over_point = right_waypoint;
365  } else if (distant_left_lane_free && left_waypoint != nullptr
366  && track_traffic.GetPassingVehicles(left_waypoint->GetId()).size() == 0) {
367  change_over_point = left_waypoint;
368  }
369  } else if (force) {
370  if (direction && right_waypoint != nullptr) {
371  change_over_point = right_waypoint;
372  } else if (!direction && left_waypoint != nullptr) {
373  change_over_point = left_waypoint;
374  }
375  }
376 
377  if (change_over_point != nullptr) {
378  const float change_over_distance = cg::Math::Clamp(1.5f * vehicle_speed, 3.0f, 20.0f);
379  const auto starting_point = change_over_point;
380  while (change_over_point->DistanceSquared(starting_point) < SQUARE(change_over_distance) &&
381  !change_over_point->CheckJunction()) {
382  change_over_point = change_over_point->GetNextWaypoint()[0];
383  }
384  }
385  }
386 
387  return change_over_point;
388 }
