61  void LocalizationStage::Action() {
62 
63  ScanUnregisteredVehicles();
64 
65  // Selecting output frames based on selector keys.
66  const auto current_planner_frame = planner_frame_selector ? planner_frame_a : planner_frame_b;
67  const auto current_collision_frame = collision_frame_selector ? collision_frame_a : collision_frame_b;
68  const auto current_traffic_light_frame =
69  traffic_light_frame_selector ? traffic_light_frame_a : traffic_light_frame_b;
70 
71 
72  // Looping over registered actors.
73  for (uint64_t i = 0u; i < actor_list.size(); ++i) {
74 
75  const Actor vehicle = actor_list.at(i);
76  const ActorId actor_id = vehicle->GetId();
77  const cg::Location vehicle_location = vehicle->GetLocation();
78  const float vehicle_velocity = vehicle->GetVelocity().Length();
79 
80  //TODO: Improve search so it doesn't do it every loop..
81  auto search = idle_time.find(actor_id);
82  if (search == idle_time.end()) {
83  idle_time[actor_id] = chr::system_clock::now();
84  }
85 
86  const float horizon_size = std::max(
87  WAYPOINT_TIME_HORIZON * std::sqrt(vehicle_velocity * 10.0f),
88  MINIMUM_HORIZON_LENGTH);
89 
90  Buffer &waypoint_buffer = buffer_list->at(i);
91 
92  // Purge passed waypoints.
93  if (!waypoint_buffer.empty()) {
94  float dot_product = DeviationDotProduct(vehicle, waypoint_buffer.front()->GetLocation(), true);
95 
96  while (dot_product <= 0.0f && !waypoint_buffer.empty()) {
97 
98  PopWaypoint(waypoint_buffer, actor_id);
99  if (!waypoint_buffer.empty()) {
100  dot_product = DeviationDotProduct(vehicle, waypoint_buffer.front()->GetLocation(), true);
101  }
102  }
103  }
104 
105  // Initializing buffer if it is empty.
106  if (waypoint_buffer.empty()) {
107  SimpleWaypointPtr closest_waypoint = local_map.GetWaypointInVicinity(vehicle_location);
108  if (closest_waypoint == nullptr) {
109  closest_waypoint = local_map.GetWaypoint(vehicle_location);
110  }
111  PushWaypoint(waypoint_buffer, actor_id, closest_waypoint);
112  }
113 
114  // Assign a lane change.
115  const SimpleWaypointPtr front_waypoint = waypoint_buffer.front();
116  const ChangeLaneInfo lane_change_info = parameters.GetForceLaneChange(vehicle);
117  const bool force_lane_change = lane_change_info.change_lane;
118  const bool lane_change_direction = lane_change_info.direction;
119 
120  if ((parameters.GetAutoLaneChange(vehicle) || force_lane_change) &&
121  !front_waypoint->CheckJunction()) {
122 
123  SimpleWaypointPtr change_over_point = AssignLaneChange(
124  vehicle, force_lane_change, lane_change_direction);
125 
126  if (change_over_point != nullptr) {
127  auto number_of_pops = waypoint_buffer.size();
128  for (uint64_t j = 0u; j < number_of_pops; ++j) {
129  PopWaypoint(waypoint_buffer, actor_id);
130  }
131 
132  PushWaypoint(waypoint_buffer, actor_id, change_over_point);
133  }
134  }
135 
136  // Populating the buffer.
137  while (waypoint_buffer.back()->DistanceSquared(waypoint_buffer.front())
138  <= std::pow(horizon_size, 2)) {
139 
140  std::vector<SimpleWaypointPtr> next_waypoints = waypoint_buffer.back()->GetNextWaypoint();
141  uint64_t selection_index = 0u;
142  // Pseudo-randomized path selection if found more than one choice.
143  if (next_waypoints.size() > 1) {
144  selection_index = static_cast<uint64_t>(rand()) % next_waypoints.size();
145  }
146  SimpleWaypointPtr next_wp = next_waypoints.at(selection_index);
147  if (next_wp == nullptr) {
148  for (auto& wp: next_waypoints) {
149  if (wp != nullptr) {
150  next_wp = wp;
151  break;
152  }
153  }
154  }
155  PushWaypoint(waypoint_buffer, actor_id, next_wp);
156  }
157 
158  // Updating geodesic grid position for actor.
159  track_traffic.UpdateGridPosition(actor_id, waypoint_buffer);
160 
161  // Generating output.
162  const float target_point_distance = std::max(std::ceil(vehicle_velocity * TARGET_WAYPOINT_TIME_HORIZON),
163  TARGET_WAYPOINT_HORIZON_LENGTH);
164  SimpleWaypointPtr target_waypoint = waypoint_buffer.front();
165  for (uint64_t j = 0u;
166  (j < waypoint_buffer.size()) &&
167  (waypoint_buffer.front()->DistanceSquared(target_waypoint)
168  < std::pow(target_point_distance, 2));
169  ++j) {
170  target_waypoint = waypoint_buffer.at(j);
171  }
172  const cg::Location target_location = target_waypoint->GetLocation();
173  float dot_product = DeviationDotProduct(vehicle, target_location);
174  float cross_product = DeviationCrossProduct(vehicle, target_location);
175  dot_product = 1.0f - dot_product;
176  if (cross_product < 0.0f) {
177  dot_product *= -1.0f;
178  }
179 
180  float distance = 0.0f; // TODO: use in PID
181 
182  // Filtering out false junctions on highways.
183  // On highways, if there is only one possible path and the section is
184  // marked as intersection, ignore it.
185  const auto vehicle_reference = boost::static_pointer_cast<cc::Vehicle>(vehicle);
186  const float speed_limit = vehicle_reference->GetSpeedLimit();
187  const float look_ahead_distance = std::max(2.0f * vehicle_velocity, MINIMUM_JUNCTION_LOOK_AHEAD);
188 
189  SimpleWaypointPtr look_ahead_point = waypoint_buffer.front();
190  uint64_t look_ahead_index = 0u;
191  for (uint64_t j = 0u;
192  (waypoint_buffer.front()->DistanceSquared(look_ahead_point)
193  < std::pow(look_ahead_distance, 2)) &&
194  (j < waypoint_buffer.size());
195  ++j) {
196  look_ahead_point = waypoint_buffer.at(j);
197  look_ahead_index = j;
198  }
199 
200  bool approaching_junction = false;
201  if (waypoint_buffer.front()->CheckJunction() ||
202  (look_ahead_point->CheckJunction() && !(waypoint_buffer.front()->CheckJunction()))) {
203  if (speed_limit > HIGHWAY_SPEED) {
204  for (uint64_t j = 0u; (j < look_ahead_index) && !approaching_junction; ++j) {
205  SimpleWaypointPtr swp = waypoint_buffer.at(j);
206  if (swp->GetNextWaypoint().size() > 1) {
207  approaching_junction = true;
208  }
209  }
210  } else {
211  approaching_junction = true;
212  }
213  }
214 
215  // Editing output frames.
216  LocalizationToPlannerData &planner_message = current_planner_frame->at(i);
217  planner_message.actor = vehicle;
218  planner_message.deviation = dot_product;
219  planner_message.distance = distance;
220  planner_message.approaching_true_junction = approaching_junction;
221 
222  LocalizationToCollisionData &collision_message = current_collision_frame->at(i);
223  collision_message.actor = vehicle;
224  collision_message.buffer = waypoint_buffer;
225  collision_message.overlapping_actors.clear();
226  ActorIdSet overlapping_actor_set = track_traffic.GetOverlappingVehicles(actor_id);
227  for (ActorId overlapping_actor_id: overlapping_actor_set) {
228  Actor actor_ptr = nullptr;
229  if (vehicle_id_to_index.find(overlapping_actor_id) != vehicle_id_to_index.end()) {
230  actor_ptr = actor_list.at(vehicle_id_to_index.at(overlapping_actor_id));
231  } else if (unregistered_actors.find(overlapping_actor_id) != unregistered_actors.end()) {
232  actor_ptr = unregistered_actors.at(overlapping_actor_id);
233  }
234  collision_message.overlapping_actors.insert({overlapping_actor_id, actor_ptr});
235  }
236  collision_message.closest_waypoint = waypoint_buffer.front();
237  collision_message.junction_look_ahead_waypoint = waypoint_buffer.at(look_ahead_index);
238 
239  LocalizationToTrafficLightData &traffic_light_message = current_traffic_light_frame->at(i);
240  traffic_light_message.actor = vehicle;
241  traffic_light_message.closest_waypoint = waypoint_buffer.front();
242  traffic_light_message.junction_look_ahead_waypoint = waypoint_buffer.at(look_ahead_index);
243  }
244  }

317  void LocalizationStage::ScanUnregisteredVehicles() {
318  ++unregistered_scan_duration;
319  // Periodically check for actors not spawned by TrafficManager.
320  if (unregistered_scan_duration == UNREGISTERED_ACTORS_SCAN_INTERVAL) {
321  unregistered_scan_duration = 0;
322 
323  const auto world_actors = world.GetActors()->Filter("vehicle.*");
324  const auto world_walker = world.GetActors()->Filter("walker.*");
325  // Scanning for vehicles.
326  for (auto actor: *world_actors.get()) {
327  const auto unregistered_id = actor->GetId();
328  if (vehicle_id_to_index.find(unregistered_id) == vehicle_id_to_index.end() &&
329  unregistered_actors.find(unregistered_id) == unregistered_actors.end()) {
330  unregistered_actors.insert({unregistered_id, actor});
331  }
332  }
333  // Scanning for pedestrians.
334  for (auto walker: *world_walker.get()) {
335  const auto unregistered_id = walker->GetId();
336  if (unregistered_actors.find(unregistered_id) == unregistered_actors.end()) {
337  unregistered_actors.insert({unregistered_id, walker});
338  }
339  }
340  }
341 
342  // Regularly update unregistered actors.
343  std::vector<ActorId> actor_ids_to_erase;
344  for (auto& actor_info: unregistered_actors) {
345  if (actor_info.second->IsAlive()) {
346  cg::Location actor_location = actor_info.second->GetLocation();
347  SimpleWaypointPtr nearest_waypoint = nullptr;
348  const auto unregistered_type = actor_info.second->GetTypeId();
349  if (unregistered_type[0] == 'v') {
350  nearest_waypoint = local_map.GetWaypointInVicinity(actor_location);
351  } else if (unregistered_type[0] == 'w') {
352  nearest_waypoint = local_map.GetPedWaypoint(actor_location);
353  }
354  if (nearest_waypoint == nullptr) {
355  nearest_waypoint = local_map.GetWaypoint(actor_location);
356  }
357  track_traffic.UpdateUnregisteredGridPosition(actor_info.first, nearest_waypoint);
358  } else {
359  track_traffic.DeleteActor(actor_info.first);
360  actor_ids_to_erase.push_back(actor_info.first);
361  }
362  }
363  for (auto actor_id: actor_ids_to_erase) {
364  unregistered_actors.erase(actor_id);
365  }
366  }

368  SimpleWaypointPtr LocalizationStage::AssignLaneChange(Actor vehicle, bool force, bool direction) {
369 
370  const ActorId actor_id = vehicle->GetId();
371  const cg::Location vehicle_location = vehicle->GetLocation();
372  const float vehicle_velocity = vehicle->GetVelocity().Length();
373  const float speed_limit = boost::static_pointer_cast<cc::Vehicle>(vehicle)->GetSpeedLimit();
374 
375  const Buffer& waypoint_buffer = buffer_list->at(vehicle_id_to_index.at(actor_id));
376  const SimpleWaypointPtr& current_waypoint = waypoint_buffer.front();
377 
378  bool need_to_change_lane = false;
379  const auto left_waypoint = current_waypoint->GetLeftWaypoint();
380  const auto right_waypoint = current_waypoint->GetRightWaypoint();
381 
382  if (!force && current_waypoint != nullptr) {
383 
384  const auto blocking_vehicles = track_traffic.GetOverlappingVehicles(actor_id);
385 
386  bool abort_lane_change = false;
387  for (auto i = blocking_vehicles.begin();
388  i != blocking_vehicles.end() && !abort_lane_change;
389  ++i) {
390 
391  const ActorId &other_vehicle_id = *i;
392 
393  // This is totally ignoring unregistered actors during lane changes.
394  // Need to fix this. Only a temporary solution.
395  if (vehicle_id_to_index.find(other_vehicle_id) != vehicle_id_to_index.end()) {
396  const Buffer& other_buffer = buffer_list->at(vehicle_id_to_index.at(other_vehicle_id));
397 
398  if (!other_buffer.empty()) {
399  const SimpleWaypointPtr& other_current_waypoint = other_buffer.front();
400  const cg::Location other_location = other_current_waypoint->GetLocation();
401 
402  bool distant_lane_availability = false;
403  const auto other_neighbouring_lanes = {
404  other_current_waypoint->GetLeftWaypoint(),
405  other_current_waypoint->GetRightWaypoint()};
406  if (other_current_waypoint != nullptr) {
407  for (auto& candidate_lane_wp: other_neighbouring_lanes) {
408  if (candidate_lane_wp != nullptr &&
409  track_traffic.GetPassingVehicles(candidate_lane_wp->GetId()).size() == 0 &&
410  speed_limit < HIGHWAY_SPEED) {
411  distant_lane_availability = true;
412  }
413  }
414 
415  const cg::Vector3D reference_heading = current_waypoint->GetForwardVector();
416  const cg::Vector3D other_heading = other_current_waypoint->GetForwardVector();
417 
418  if (other_vehicle_id != actor_id &&
419  !current_waypoint->CheckJunction() &&
420  !other_current_waypoint->CheckJunction() &&
421  cg::Math::Dot(reference_heading, other_heading) > MAXIMUM_LANE_OBSTACLE_CURVATURE) {
422 
423  const float squared_vehicle_distance = cg::Math::DistanceSquared(other_location, vehicle_location);
424  const float deviation_dot = DeviationDotProduct(vehicle, other_location);
425 
426  if (deviation_dot > 0.0f) {
427 
428  if (distant_lane_availability &&
429  squared_vehicle_distance > std::pow(MINIMUM_LANE_CHANGE_DISTANCE, 2)) {
430 
431  need_to_change_lane = true;
432  } else if (squared_vehicle_distance < std::pow(MINIMUM_LANE_CHANGE_DISTANCE, 2)) {
433 
434  need_to_change_lane = false;
435  abort_lane_change = true;
436  }
437 
438  }
439  }
440  }
441  }
442  }
443  }
444 
445  } else {
446 
447  need_to_change_lane = true;
448  }
449 
450  const float change_over_distance = std::max(2.0f*vehicle_velocity, 10.0f);
451  bool possible_to_lane_change = false;
452  SimpleWaypointPtr change_over_point = nullptr;
453 
454  if (need_to_change_lane) {
455 
456  std::vector<SimpleWaypointPtr> candidate_points;
457  if (force) {
458  if (direction && left_waypoint != nullptr) {
459  candidate_points.push_back(left_waypoint);
460  } else if (!direction && right_waypoint != nullptr) {
461  candidate_points.push_back(right_waypoint);
462  }
463  } else {
464  candidate_points.push_back(left_waypoint);
465  candidate_points.push_back(right_waypoint);
466  }
467 
468  for (auto target_lane_wp: candidate_points) {
469  if (!force &&
470  !possible_to_lane_change &&
471  target_lane_wp != nullptr &&
472  track_traffic.GetPassingVehicles(target_lane_wp->GetId()).size() == 0) {
473 
474  possible_to_lane_change = true;
475  change_over_point = target_lane_wp;
476  } else if (force) {
477 
478  possible_to_lane_change = true;
479  change_over_point = target_lane_wp;
480  }
481  }
482  }
483 
484  if (need_to_change_lane && possible_to_lane_change) {
485  const auto starting_point = change_over_point;
486  while (change_over_point->DistanceSquared(starting_point) < std::pow(change_over_distance, 2) &&
487  !change_over_point->CheckJunction()) {
488  change_over_point = change_over_point->GetNextWaypoint()[0];
489  }
490  return change_over_point;
491  } else {
492  return nullptr;
493  }
494  }
