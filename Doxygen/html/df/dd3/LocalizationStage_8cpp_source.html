73  void LocalizationStage::Action() {
74 
75  ScanUnregisteredVehicles();
76 
77  // Selecting output frames based on selector keys.
78  const auto current_planner_frame = planner_frame_selector ? planner_frame_a : planner_frame_b;
79  const auto current_collision_frame = collision_frame_selector ? collision_frame_a : collision_frame_b;
80  const auto current_traffic_light_frame =
81  traffic_light_frame_selector ? traffic_light_frame_a : traffic_light_frame_b;
82 
83  // Selecting current timestamp from the world snapshot.
84  current_timestamp = episode_proxy_ls.Lock()->GetWorldSnapshot().GetTimestamp();
85 
86  // Looping over registered actors.
87  for (uint64_t i = 0u; i < actor_list.size(); ++i) {
88 
89  const Actor vehicle = actor_list.at(i);
90  const ActorId actor_id = vehicle->GetId();
91  const cg::Location vehicle_location = vehicle->GetLocation();
92  const float vehicle_velocity = vehicle->GetVelocity().Length();
93 
94  // Initializing idle times.
95  if (idle_time.find(actor_id) == idle_time.end() && current_timestamp.elapsed_seconds != 0) {
96  idle_time[actor_id] = current_timestamp.elapsed_seconds;
97  }
98 
99  const float horizon_size = std::max(
100  WAYPOINT_TIME_HORIZON * std::sqrt(vehicle_velocity * 10.0f),
101  MINIMUM_HORIZON_LENGTH);
102 
103  if (buffer_list->find(actor_id) == buffer_list->end()) {
104  buffer_list->insert({actor_id, Buffer()});
105  }
106 
107  Buffer &waypoint_buffer = buffer_list->at(actor_id);
108 
109  // Clear buffer if vehicle is too far from the first waypoint in the buffer.
110  if (!waypoint_buffer.empty() &&
111  cg::Math::DistanceSquared(waypoint_buffer.front()->GetLocation(), vehicle_location) > std::pow(30.0f, 2)) {
112 
113  auto number_of_pops = waypoint_buffer.size();
114  for (uint64_t j = 0u; j < number_of_pops; ++j) {
115  PopWaypoint(waypoint_buffer, actor_id);
116  }
117  }
118 
119  // Purge passed waypoints.
120  if (!waypoint_buffer.empty()) {
121  float dot_product = DeviationDotProduct(vehicle, vehicle_location, waypoint_buffer.front()->GetLocation(), true);
122 
123  while (dot_product <= 0.0f && !waypoint_buffer.empty()) {
124 
125  PopWaypoint(waypoint_buffer, actor_id);
126  if (!waypoint_buffer.empty()) {
127  dot_product = DeviationDotProduct(vehicle, vehicle_location, waypoint_buffer.front()->GetLocation(), true);
128  }
129  }
130  }
131 
132  // Initializing buffer if it is empty.
133  if (waypoint_buffer.empty()) {
134  SimpleWaypointPtr closest_waypoint = local_map.GetWaypointInVicinity(vehicle_location);
135  if (closest_waypoint == nullptr) {
136  closest_waypoint = local_map.GetWaypoint(vehicle_location);
137  }
138  PushWaypoint(waypoint_buffer, actor_id, closest_waypoint);
139  }
140 
141  // Assign a lane change.
142 
143  const ChangeLaneInfo lane_change_info = parameters.GetForceLaneChange(vehicle);
144  bool force_lane_change = lane_change_info.change_lane;
145  bool lane_change_direction = lane_change_info.direction;
146 
147  if (!force_lane_change) {
148  float perc_keep_right = parameters.GetKeepRightPercentage(vehicle);
149  if (perc_keep_right >= 0.0f && perc_keep_right >= (rand() % 101)) {
150  force_lane_change = true;
151  lane_change_direction = true;
152  }
153  }
154 
155  const SimpleWaypointPtr front_waypoint = waypoint_buffer.front();
156  const double lane_change_distance = std::pow(std::max(10.0f * vehicle_velocity, INTER_LANE_CHANGE_DISTANCE), 2);
157 
158  if (((parameters.GetAutoLaneChange(vehicle) || force_lane_change) && !front_waypoint->CheckJunction())
159  && (last_lane_change_location.find(actor_id) == last_lane_change_location.end()
160  || cg::Math::DistanceSquared(last_lane_change_location.at(actor_id), vehicle_location)
161  > lane_change_distance )) {
162 
163  SimpleWaypointPtr change_over_point = AssignLaneChange(
164  vehicle, vehicle_location, force_lane_change, lane_change_direction);
165 
166  if (change_over_point != nullptr) {
167  if (last_lane_change_location.find(actor_id) != last_lane_change_location.end()) {
168  last_lane_change_location.at(actor_id) = vehicle_location;
169  } else {
170  last_lane_change_location.insert({actor_id, vehicle_location});
171  }
172  auto number_of_pops = waypoint_buffer.size();
173  for (uint64_t j = 0u; j < number_of_pops; ++j) {
174  PopWaypoint(waypoint_buffer, actor_id);
175  }
176  PushWaypoint(waypoint_buffer, actor_id, change_over_point);
177  }
178  }
179 
180  // Populating the buffer.
181  while (waypoint_buffer.back()->DistanceSquared(waypoint_buffer.front())
182  <= std::pow(horizon_size, 2)) {
183 
184  std::vector<SimpleWaypointPtr> next_waypoints = waypoint_buffer.back()->GetNextWaypoint();
185  uint64_t selection_index = 0u;
186  // Pseudo-randomized path selection if found more than one choice.
187  if (next_waypoints.size() > 1) {
188  selection_index = static_cast<uint64_t>(rand()) % next_waypoints.size();
189  }
190  SimpleWaypointPtr next_wp = next_waypoints.at(selection_index);
191  if (next_wp == nullptr) {
192  for (auto& wp: next_waypoints) {
193  if (wp != nullptr) {
194  next_wp = wp;
195  break;
196  }
197  }
198  }
199  PushWaypoint(waypoint_buffer, actor_id, next_wp);
200  }
201 
202  // Updating geodesic grid position for actor.
203  track_traffic.UpdateGridPosition(actor_id, waypoint_buffer);
204 
205  // Generating output.
206  const float target_point_distance = std::max(std::ceil(vehicle_velocity * TARGET_WAYPOINT_TIME_HORIZON),
207  TARGET_WAYPOINT_HORIZON_LENGTH);
208  SimpleWaypointPtr target_waypoint = waypoint_buffer.front();
209  for (uint64_t j = 0u;
210  (j < waypoint_buffer.size()) &&
211  (waypoint_buffer.front()->DistanceSquared(target_waypoint)
212  < std::pow(target_point_distance, 2));
213  ++j) {
214  target_waypoint = waypoint_buffer.at(j);
215  }
216  const cg::Location target_location = target_waypoint->GetLocation();
217  float dot_product = DeviationDotProduct(vehicle, vehicle_location, target_location);
218  float cross_product = DeviationCrossProduct(vehicle, vehicle_location, target_location);
219  dot_product = 1.0f - dot_product;
220  if (cross_product < 0.0f) {
221  dot_product *= -1.0f;
222  }
223 
224  float distance = 0.0f; // TODO: use in PID
225 
226  // Filtering out false junctions on highways:
227  // on highways, if there is only one possible path and the section is
228  // marked as intersection, ignore it.
229  const auto vehicle_reference = boost::static_pointer_cast<cc::Vehicle>(vehicle);
230  const float speed_limit = vehicle_reference->GetSpeedLimit();
231  const float look_ahead_distance = std::max(2.0f * vehicle_velocity, MINIMUM_JUNCTION_LOOK_AHEAD);
232 
233  SimpleWaypointPtr look_ahead_point = waypoint_buffer.front();
234  uint64_t look_ahead_index = 0u;
235  for (uint64_t j = 0u;
236  (waypoint_buffer.front()->DistanceSquared(look_ahead_point)
237  < std::pow(look_ahead_distance, 2)) &&
238  (j < waypoint_buffer.size());
239  ++j) {
240  look_ahead_point = waypoint_buffer.at(j);
241  look_ahead_index = j;
242  }
243 
244  bool approaching_junction = false;
245  if (look_ahead_point->CheckJunction() && !(waypoint_buffer.front()->CheckJunction())) {
246  if (speed_limit*3.6f > HIGHWAY_SPEED) {
247  for (uint64_t j = 0u; (j < look_ahead_index) && !approaching_junction; ++j) {
248  SimpleWaypointPtr swp = waypoint_buffer.at(j);
249  if (swp->GetNextWaypoint().size() > 1) {
250  approaching_junction = true;
251  }
252  }
253  } else {
254  approaching_junction = true;
255  }
256  }
257 
258  // Reset the variables when no longer approaching a junction.
259  if (!approaching_junction && approached[actor_id]){
260  final_safe_points[actor_id] = nullptr;
261  approached[actor_id] = false;
262  }
263 
264  // Only do once, when the junction has just been seen.
265  else if (approaching_junction && !approached[actor_id]){
266 
267  SimpleWaypointPtr final_point = nullptr;
268  final_point = GetSafeLocationAfterJunction(vehicle_reference, waypoint_buffer);
269  if(final_point != nullptr){
270  final_safe_points[actor_id] = final_point;
271  approaching_junction = false;
272  approached[actor_id] = true;
273  }
274  }
275 
276  // Editing output frames.
277  LocalizationToPlannerData &planner_message = current_planner_frame->at(i);
278  planner_message.actor = vehicle;
279  planner_message.deviation = dot_product;
280  planner_message.distance = distance;
281  planner_message.approaching_true_junction = approaching_junction;
282 
283  LocalizationToCollisionData &collision_message = current_collision_frame->at(i);
284  collision_message.actor = vehicle;
285  collision_message.buffer = waypoint_buffer;
286  collision_message.overlapping_actors.clear();
287  ActorIdSet overlapping_actor_set = track_traffic.GetOverlappingVehicles(actor_id);
288  for (ActorId overlapping_actor_id: overlapping_actor_set) {
289  Actor actor_ptr = nullptr;
290  if (vehicle_id_to_index.find(overlapping_actor_id) != vehicle_id_to_index.end()) {
291  actor_ptr = actor_list.at(vehicle_id_to_index.at(overlapping_actor_id));
292  } else if (unregistered_actors.find(overlapping_actor_id) != unregistered_actors.end()) {
293  actor_ptr = unregistered_actors.at(overlapping_actor_id);
294  }
295  if (actor_ptr!=nullptr) {
296  collision_message.overlapping_actors.insert({overlapping_actor_id, actor_ptr});
297  }
298  collision_message.safe_point_after_junction = final_safe_points[actor_id];
299  }
300  collision_message.closest_waypoint = waypoint_buffer.front();
301  collision_message.junction_look_ahead_waypoint = waypoint_buffer.at(look_ahead_index);
302 
303  LocalizationToTrafficLightData &traffic_light_message = current_traffic_light_frame->at(i);
304  traffic_light_message.actor = vehicle;
305  traffic_light_message.closest_waypoint = waypoint_buffer.front();
306  traffic_light_message.junction_look_ahead_waypoint = waypoint_buffer.at(look_ahead_index);
307 
308  // Updating idle time when necessary.
309  UpdateIdleTime(vehicle);
310  }
311 
312  if (IsVehicleStuck(maximum_idle_time.first)) {
313  TryDestroyVehicle(maximum_idle_time.first);
314  }
315 
316  // Updating maximum idle time to null for the next iteration.
317  maximum_idle_time = std::make_pair(nullptr, current_timestamp.elapsed_seconds);
318  }

320  void LocalizationStage::DataReceiver() {
321  bool is_deleted_actors_present = false;
322  std::set<uint32_t> world_actor_id;
323  std::vector<ActorPtr> actor_list_to_be_deleted;
324 
325  // Filter function to collect the data.
326  auto Filter = [&](auto &actors, auto &wildcard_pattern) {
327  std::vector<carla::client::detail::ActorVariant> filtered;
328  for (auto &&actor : actors) {
329  if (carla::StringUtil::Match(carla::client::detail::ActorVariant(actor).GetTypeId(), wildcard_pattern)) {
330  filtered.push_back(actor);
331  }
332  }
333  return filtered;
334  };
335 
336  // Get all the actors.
337  auto world_actors_list = episode_proxy_ls.Lock()->GetAllTheActorsInTheEpisode();
338 
339  // Filter with vehicle wildcard.
340  auto vehicles = Filter(world_actors_list, "vehicle.*");
341 
342  // Building a set of vehicle ids in the world.
343  for (const auto &actor : vehicles) {
344  world_actor_id.insert(actor.GetId());
345  }
346 
347  // Search for invalid/destroyed vehicles.
348  for (auto &actor : actor_list) {
349  if (world_actor_id.find(actor->GetId()) == world_actor_id.end()) {
350  actor_list_to_be_deleted.emplace_back(actor);
351  track_traffic.DeleteActor(actor->GetId());
352  last_lane_change_location.erase(actor->GetId());
353  }
354  }
355 
356  // Clearing the registered actor list.
357  if(!actor_list_to_be_deleted.empty()) {
358  registered_actors.Remove(actor_list_to_be_deleted);
359  actor_list.clear();
360  actor_list = registered_actors.GetList();
361  is_deleted_actors_present = true;
362  }
363 
364  // Building a list of registered actors and connecting
365  // the vehicle ids to their position indices on data arrays.
366 
367  if (is_deleted_actors_present || (registered_actors_state != registered_actors.GetState())) {
368  actor_list.clear();
369  actor_list_to_be_deleted.clear();
370  actor_list = registered_actors.GetList();
371  uint64_t index = 0u;
372  vehicle_id_to_index.clear();
373  for (auto &actor : actor_list) {
374  vehicle_id_to_index.insert({actor->GetId(), index});
375  ++index;
376  }
377  registered_actors_state = registered_actors.GetState();
378  }
379 
380  // Allocating new containers for the changed number of registered vehicles.
381  if (number_of_vehicles != actor_list.size()) {
382  number_of_vehicles = static_cast<uint64_t>(actor_list.size());
383  // Allocating output frames to be shared with the motion planner stage.
384  planner_frame_a = std::make_shared<LocalizationToPlannerFrame>(number_of_vehicles);
385  planner_frame_b = std::make_shared<LocalizationToPlannerFrame>(number_of_vehicles);
386  // Allocating output frames to be shared with the collision stage.
387  collision_frame_a = std::make_shared<LocalizationToCollisionFrame>(number_of_vehicles);
388  collision_frame_b = std::make_shared<LocalizationToCollisionFrame>(number_of_vehicles);
389  // Allocating output frames to be shared with the traffic light stage
390  traffic_light_frame_a = std::make_shared<LocalizationToTrafficLightFrame>(number_of_vehicles);
391  traffic_light_frame_b = std::make_shared<LocalizationToTrafficLightFrame>(number_of_vehicles);
392  }
393  }

435  void LocalizationStage::ScanUnregisteredVehicles() {
436  ++unregistered_scan_duration;
437  // Periodically check for actors not spawned by TrafficManager.
438  if (unregistered_scan_duration == UNREGISTERED_ACTORS_SCAN_INTERVAL) {
439  unregistered_scan_duration = 0;
440 
441  auto Filter = [&](auto &actors, auto &wildcard_pattern) {
442  std::vector<carla::client::detail::ActorVariant> filtered;
443  for (auto &&actor : actors) {
444  if (carla::StringUtil::Match
445  ( carla::client::detail::ActorVariant(actor).GetTypeId()
446  , wildcard_pattern)) {
447  filtered.push_back(actor);
448  }
449  }
450  return filtered;
451  };
452 
453  /// Get all actors of the world
454  auto world_actors_list = episode_proxy_ls.Lock()->GetAllTheActorsInTheEpisode();
455 
456  /// Filter based on wildcard_pattern
457  const auto world_actors = Filter(world_actors_list, "vehicle.*");
458  const auto world_walker = Filter(world_actors_list, "walker.*");
459 
460  // Scanning for vehicles.
461  for (auto actor: world_actors) {
462  const auto unregistered_id = actor.GetId();
463  if (vehicle_id_to_index.find(unregistered_id) == vehicle_id_to_index.end() &&
464  unregistered_actors.find(unregistered_id) == unregistered_actors.end()) {
465  unregistered_actors.insert({unregistered_id, actor.Get(episode_proxy_ls)});
466  }
467  }
468  // Scanning for pedestrians.
469  for (auto walker: world_walker) {
470  const auto unregistered_id = walker.GetId();
471  if (unregistered_actors.find(unregistered_id) == unregistered_actors.end()) {
472  unregistered_actors.insert({unregistered_id, walker.Get(episode_proxy_ls)});
473  }
474  }
475  }
476 
477  // Regularly update unregistered actors.
478  const auto current_snapshot = episode_proxy_ls.Lock()->GetWorldSnapshot();
479  for (auto it = unregistered_actors.cbegin(); it != unregistered_actors.cend();) {
480  if (registered_actors.Contains(it->first) || !current_snapshot.Contains(it->first)) {
481  track_traffic.DeleteActor(it->first);
482  it = unregistered_actors.erase(it);
483  } else {
484  // Updating data structures.
485  cg::Location location = it->second->GetLocation();
486  const auto type = it->second->GetTypeId();
487 
488  SimpleWaypointPtr nearest_waypoint = nullptr;
489  if (type[0] == 'v') {
490  nearest_waypoint = local_map.GetWaypointInVicinity(location);
491  } else if (type[0] == 'w') {
492  nearest_waypoint = local_map.GetPedWaypoint(location);
493  }
494  if (nearest_waypoint == nullptr) {
495  nearest_waypoint = local_map.GetWaypoint(location);
496  }
497 
498  track_traffic.UpdateUnregisteredGridPosition(it->first, nearest_waypoint);
499 
500  ++it;
501  }
502  }

505  SimpleWaypointPtr LocalizationStage::AssignLaneChange(Actor vehicle, const cg::Location &vehicle_location,
506  bool force, bool direction)
507  {
508 
509  const ActorId actor_id = vehicle->GetId();
510  const float vehicle_velocity = vehicle->GetVelocity().Length();
511 
512  // Waypoint representing the new starting point for the waypoint buffer
513  // due to lane change. Remains nullptr if lane change not viable.
514  SimpleWaypointPtr change_over_point = nullptr;
515 
516  // Retrieve waypoint buffer for current vehicle.
517  const Buffer& waypoint_buffer = buffer_list->at(actor_id);
518 
519  // Check buffer is not empty.
520  if (!waypoint_buffer.empty())
521  {
522  // Get the left and right waypoints for the current closest waypoint.
523  const SimpleWaypointPtr& current_waypoint = waypoint_buffer.front();
524  const SimpleWaypointPtr left_waypoint = current_waypoint->GetLeftWaypoint();
525  const SimpleWaypointPtr right_waypoint = current_waypoint->GetRightWaypoint();
526 
527  // Retrieve vehicles with overlapping waypoint buffers with current vehicle.
528  const auto blocking_vehicles = track_traffic.GetOverlappingVehicles(actor_id);
529 
530  // Find immediate in-lane obstacle and check if any are too close to initiate lane change.
531  bool obstacle_too_close = false;
532  float minimum_squared_distance = std::numeric_limits<float>::infinity();
533  ActorId obstacle_actor_id = 0u;
534  for (auto i = blocking_vehicles.begin();
535  i != blocking_vehicles.end() && !obstacle_too_close && !force;
536  ++i)
537  {
538  const ActorId &other_actor_id = *i;
539  // Find vehicle in registered list and check if it's buffer is not empty.
540  if (vehicle_id_to_index.find(other_actor_id) != vehicle_id_to_index.end()
541  && buffer_list->find(other_actor_id) != buffer_list->end()
542  && !buffer_list->at(other_actor_id).empty())
543  {
544  const Buffer& other_buffer = buffer_list->at(other_actor_id);
545  const SimpleWaypointPtr& other_current_waypoint = other_buffer.front();
546  const cg::Location other_location = other_current_waypoint->GetLocation();
547 
548  const cg::Vector3D reference_heading = current_waypoint->GetForwardVector();
549  cg::Vector3D reference_to_other = other_current_waypoint->GetLocation()
550  - current_waypoint->GetLocation();
551  const cg::Vector3D other_heading = other_current_waypoint->GetForwardVector();
552 
553  // Check both vehicles are not in junction,
554  // Check if the other vehicle is in front of the current vehicle,
555  // Check if the two vehicles have acceptable angular deviation between their headings.
556  if (!current_waypoint->CheckJunction()
557  && !other_current_waypoint->CheckJunction()
558  && other_current_waypoint->GetWaypoint()->GetRoadId() == current_waypoint->GetWaypoint()->GetRoadId()
559  && other_current_waypoint->GetWaypoint()->GetLaneId() == current_waypoint->GetWaypoint()->GetLaneId()
560  && cg::Math::Dot(reference_heading, reference_to_other) > 0.0f
561  && cg::Math::Dot(reference_heading, other_heading) > MAXIMUM_LANE_OBSTACLE_CURVATURE)
562  {
563  float squared_distance = cg::Math::DistanceSquared(vehicle_location, other_location);
564  // Abort if the obstacle is too close.
565  if (squared_distance > std::pow(MINIMUM_LANE_CHANGE_DISTANCE, 2))
566  {
567  // Remember if the new vehicle is closer.
568  if (squared_distance < minimum_squared_distance
569  && squared_distance < std::pow(MAXIMUM_LANE_OBSTACLE_DISTANCE, 2.0f))
570  {
571  minimum_squared_distance = squared_distance;
572  obstacle_actor_id = other_actor_id;
573  }
574  } else {
575  obstacle_too_close = true;
576  }
577  }
578  }
579  }
580 
581  // If a valid immediate obstacle found.
582  if (!obstacle_too_close && obstacle_actor_id != 0u && !force)
583  {
584  const Buffer& other_buffer = buffer_list->at(obstacle_actor_id);
585  const SimpleWaypointPtr& other_current_waypoint = other_buffer.front();
586  const auto other_neighbouring_lanes = {other_current_waypoint->GetLeftWaypoint(),
587  other_current_waypoint->GetRightWaypoint()};
588 
589  // Flags reflecting whether adjacent lanes are free near the obstacle.
590  bool distant_left_lane_free = false;
591  bool distant_right_lane_free = false;
592 
593  // Check if the neighbouring lanes near the obstructing vehicle are free of other vehicles.
594  bool left_right = true;
595  for (auto& candidate_lane_wp: other_neighbouring_lanes) {
596  if (candidate_lane_wp != nullptr &&
597  track_traffic.GetPassingVehicles(candidate_lane_wp->GetId()).size() == 0) {
598 
599  if (left_right) distant_left_lane_free = true;
600  else distant_right_lane_free = true;
601 
602  }
603  left_right = !left_right;
604  }
605 
606  // Based on what lanes are free near the obstacle,
607  // find the change over point with no vehicles passing through them.
608  if (distant_right_lane_free && right_waypoint != nullptr
609  && track_traffic.GetPassingVehicles(right_waypoint->GetId()).size() == 0)
610  {
611  change_over_point = right_waypoint;
612  } else if (distant_left_lane_free && left_waypoint != nullptr
613  && track_traffic.GetPassingVehicles(left_waypoint->GetId()).size() == 0)
614  {
615  change_over_point = left_waypoint;
616  }
617  } else if (force) {
618  if (direction && right_waypoint != nullptr) {
619  change_over_point = right_waypoint;
620  } else if (!direction && left_waypoint != nullptr) {
621  change_over_point = left_waypoint;
622  }
623  }
624 
625  if (change_over_point != nullptr)
626  {
627  const float change_over_distance = cg::Math::Clamp(1.5f*vehicle_velocity, 3.0f, 20.0f);
628  const auto starting_point = change_over_point;
629  while (change_over_point->DistanceSquared(starting_point) < std::pow(change_over_distance, 2) &&
630  !change_over_point->CheckJunction()) {
631  change_over_point = change_over_point->GetNextWaypoint()[0];
632  }
633 
634  // Reset this variable if needed
635  if (approached[actor_id]){
636  approached[actor_id] = false;
637  }
638  }
639  }
640 
641  return change_over_point;
642  }

644  SimpleWaypointPtr LocalizationStage::GetSafeLocationAfterJunction(const Vehicle &vehicle, Buffer &waypoint_buffer){
645 
646  ActorId actor_id = vehicle->GetId();
647  // Get the length of the car
648  float length = vehicle->GetBoundingBox().extent.x;
649  // First Waypoint before the junction
650  const SimpleWaypointPtr initial_point;
651  uint64_t initial_index = 0;
652  // First Waypoint after the junction
653  SimpleWaypointPtr safe_point = nullptr;
654  uint64_t safe_index = 0;
655  // Vehicle position after the junction
656  SimpleWaypointPtr final_point = nullptr;
657  // Safe space after the junction
658  const float safe_distance = 1.5f*length;
659 
660  for (uint64_t j = 0u; j < waypoint_buffer.size(); ++j){
661  if (waypoint_buffer.at(j)->CheckJunction()){
662  initial_index = j;
663  break;
664  }
665  }
666 
667  // Stop if something failed
668  if (initial_index == 0 && !waypoint_buffer.front()->CheckJunction()){
669  return final_point;
670  }
671 
672  // 2) Search for the end of the intersection (if it is in the buffer)
673  for (uint64_t i = initial_index; i < waypoint_buffer.size(); ++i){
674 
675  if (!waypoint_buffer.at(i)->CheckJunction()){
676  safe_point = waypoint_buffer.at(i);
677  safe_index = i;
678  break;
679  }
680  }
681 
682  // If it hasn't been found, extend the buffer
683  if(safe_point == nullptr){
684  while (waypoint_buffer.back()->CheckJunction()) {
685 
686  std::vector<SimpleWaypointPtr> next_waypoints = waypoint_buffer.back()->GetNextWaypoint();
687  uint64_t selection_index = 0u;
688  if (next_waypoints.size() > 1) {
689  selection_index = static_cast<uint64_t>(rand()) % next_waypoints.size();
690  }
691 
692  PushWaypoint(waypoint_buffer, actor_id, next_waypoints.at(selection_index));
693  }
694  // Save the last one
695  safe_point = waypoint_buffer.back();
696  }
697 
698  // Stop if something failed
699  if (safe_index == 0){
700  return final_point;
701  }
702 
703  // 3) Search for final_point (again, if it is in the buffer)
704 
705  for(uint64_t k = safe_index; k < waypoint_buffer.size(); ++k){
706 
707  if(safe_point->Distance(waypoint_buffer.at(k)->GetLocation()) > safe_distance){
708  final_point = waypoint_buffer.at(k);
709  break;
710  }
711  }
712 
713  // If it hasn't been found, extend the buffer
714  if(final_point == nullptr){
715  while (safe_point->Distance(waypoint_buffer.back()->GetLocation()) < safe_distance) {
716 
717  // Record the last point as a safe one and save it
718  std::vector<SimpleWaypointPtr> next_waypoints = waypoint_buffer.back()->GetNextWaypoint();
719  uint64_t selection_index = 0u;
720  // Pseudo-randomized path selection if found more than one choice.
721  if (next_waypoints.size() > 1) {
722  selection_index = static_cast<uint64_t>(rand()) % next_waypoints.size();
723  }
724 
725  PushWaypoint(waypoint_buffer, actor_id, next_waypoints.at(selection_index));
726  }
727  final_point = waypoint_buffer.back();
728  }
729 
730  return final_point;
731  }
