79  void LocalizationStage::Action() {
80 
81  ScanUnregisteredVehicles();
82 
83  UpdateSwarmVelocities();
84 
85  // Selecting output frames based on selector keys.
86  const auto current_planner_frame = planner_frame_selector ? planner_frame_a : planner_frame_b;
87  const auto current_collision_frame = collision_frame_selector ? collision_frame_a : collision_frame_b;
88  const auto current_traffic_light_frame =
89  traffic_light_frame_selector ? traffic_light_frame_a : traffic_light_frame_b;
90 
91  // Selecting current timestamp from the world snapshot.
92  current_timestamp = episode_proxy_ls.Lock()->GetWorldSnapshot().GetTimestamp();
93 
94  // Looping over registered actors.
95  for (uint64_t i = 0u; i < actor_list.size(); ++i) {
96 
97  const Actor vehicle = actor_list.at(i);
98  const ActorId actor_id = vehicle->GetId();
99  const cg::Location vehicle_location = vehicle->GetLocation();
100  const cg::Vector3D vehicle_velocity_vector = GetVelocity(actor_id);
101  const float vehicle_velocity = vehicle_velocity_vector.Length();
102 
103  // Initializing idle times.
104  if (idle_time.find(actor_id) == idle_time.end() && current_timestamp.elapsed_seconds != 0) {
105  idle_time[actor_id] = current_timestamp.elapsed_seconds;
106  }
107 
108  // Speed dependent waypoint horizon length.
109  const float horizon_square = std::min(std::pow(vehicle_velocity * HORIZON_RATE + MINIMUM_HORIZON_LENGTH, 2.0f),
110  std::pow(MAXIMUM_HORIZON_LENGTH, 2.0f));
111 
112  if (buffer_list->find(actor_id) == buffer_list->end()) {
113  buffer_list->insert({actor_id, Buffer()});
114  }
115 
116  Buffer &waypoint_buffer = buffer_list->at(actor_id);
117 
118  // Clear buffer if vehicle is too far from the first waypoint in the buffer.
119  if (!waypoint_buffer.empty() &&
120  cg::Math::DistanceSquared(waypoint_buffer.front()->GetLocation(), vehicle_location) > std::pow(30.0f, 2)) {
121 
122  auto number_of_pops = waypoint_buffer.size();
123  for (uint64_t j = 0u; j < number_of_pops; ++j) {
124  PopWaypoint(waypoint_buffer, actor_id);
125  }
126  }
127 
128  if (!waypoint_buffer.empty()) {
129  // Purge passed waypoints.
130  float dot_product = DeviationDotProduct(vehicle, vehicle_location, waypoint_buffer.front()->GetLocation());
131  while (dot_product <= 0.0f && !waypoint_buffer.empty()) {
132 
133  PopWaypoint(waypoint_buffer, actor_id);
134  if (!waypoint_buffer.empty()) {
135  dot_product = DeviationDotProduct(vehicle, vehicle_location, waypoint_buffer.front()->GetLocation());
136  }
137  }
138 
139  // Purge waypoints too far from the front of the buffer.
140  while (!waypoint_buffer.empty()
141  && waypoint_buffer.back()->DistanceSquared(waypoint_buffer.front()) > horizon_square) {
142  PopWaypoint(waypoint_buffer, actor_id, false);
143  }
144  }
145 
146  // Initializing buffer if it is empty.
147  if (waypoint_buffer.empty()) {
148  SimpleWaypointPtr closest_waypoint = local_map.GetWaypointInVicinity(vehicle_location);
149  if (closest_waypoint == nullptr) {
150  closest_waypoint = local_map.GetWaypoint(vehicle_location);
151  }
152  PushWaypoint(waypoint_buffer, actor_id, closest_waypoint);
153  }
154 
155  // Assign a lane change.
156 
157  const ChangeLaneInfo lane_change_info = parameters.GetForceLaneChange(vehicle);
158  bool force_lane_change = lane_change_info.change_lane;
159  bool lane_change_direction = lane_change_info.direction;
160 
161  if (!force_lane_change) {
162  float perc_keep_right = parameters.GetKeepRightPercentage(vehicle);
163  if (perc_keep_right >= 0.0f && perc_keep_right >= (rand() % 101)) {
164  force_lane_change = true;
165  lane_change_direction = true;
166  }
167  }
168 
169  const SimpleWaypointPtr front_waypoint = waypoint_buffer.front();
170  const double lane_change_distance = std::pow(std::max(10.0f * vehicle_velocity, INTER_LANE_CHANGE_DISTANCE), 2);
171 
172  if (((parameters.GetAutoLaneChange(vehicle) || force_lane_change) && !front_waypoint->CheckJunction())
173  && (last_lane_change_location.find(actor_id) == last_lane_change_location.end()
174  || cg::Math::DistanceSquared(last_lane_change_location.at(actor_id), vehicle_location)
175  > lane_change_distance )) {
176 
177  SimpleWaypointPtr change_over_point = AssignLaneChange(
178  vehicle, vehicle_location, force_lane_change, lane_change_direction);
179 
180  if (change_over_point != nullptr) {
181  if (last_lane_change_location.find(actor_id) != last_lane_change_location.end()) {
182  last_lane_change_location.at(actor_id) = vehicle_location;
183  } else {
184  last_lane_change_location.insert({actor_id, vehicle_location});
185  }
186  auto number_of_pops = waypoint_buffer.size();
187  for (uint64_t j = 0u; j < number_of_pops; ++j) {
188  PopWaypoint(waypoint_buffer, actor_id);
189  }
190  PushWaypoint(waypoint_buffer, actor_id, change_over_point);
191  }
192  }
193 
194  // Populating the buffer.
195  while (waypoint_buffer.back()->DistanceSquared(waypoint_buffer.front()) <= horizon_square) {
196 
197  std::vector<SimpleWaypointPtr> next_waypoints = waypoint_buffer.back()->GetNextWaypoint();
198  uint64_t selection_index = 0u;
199  // Pseudo-randomized path selection if found more than one choice.
200  if (next_waypoints.size() > 1) {
201  selection_index = static_cast<uint64_t>(rand()) % next_waypoints.size();
202  }
203  SimpleWaypointPtr next_wp = next_waypoints.at(selection_index);
204  if (next_wp == nullptr) {
205  for (auto& wp: next_waypoints) {
206  if (wp != nullptr) {
207  next_wp = wp;
208  break;
209  }
210  }
211  }
212  PushWaypoint(waypoint_buffer, actor_id, next_wp);
213  }
214 
215  // Begining point of the waypoint buffer;
216  const SimpleWaypointPtr& updated_front_waypoint = waypoint_buffer.front();
217 
218  // Updating geodesic grid position for actor.
219  track_traffic.UpdateGridPosition(actor_id, waypoint_buffer);
220  // WayPoint Binning Changes
221  // Generating output.
222  const float target_point_distance = std::max(std::ceil(vehicle_velocity * TARGET_WAYPOINT_TIME_HORIZON),
223  TARGET_WAYPOINT_HORIZON_LENGTH);
224 
225  std::pair<SimpleWaypointPtr,uint64_t> target_waypoint_index_pair = track_traffic.GetTargetWaypoint(waypoint_buffer, target_point_distance);
226  SimpleWaypointPtr &target_waypoint = target_waypoint_index_pair.first;
227  const cg::Location target_location = target_waypoint->GetLocation();
228  float dot_product = DeviationDotProduct(vehicle, vehicle_location, target_location);
229  float cross_product = DeviationCrossProduct(vehicle, vehicle_location, target_location);
230  dot_product = 1.0f - dot_product;
231  if (cross_product < 0.0f) {
232  dot_product *= -1.0f;
233  }
234 
235  float distance = 0.0f; // TODO: use in PID
236 
237  // Filtering out false junctions on highways:
238  // on highways, if there is only one possible path and the section is
239  // marked as intersection, ignore it.
240  const auto vehicle_reference = boost::static_pointer_cast<cc::Vehicle>(vehicle);
241  const float speed_limit = vehicle_reference->GetSpeedLimit();
242  const float look_ahead_distance = std::max(2.0f * vehicle_velocity, MINIMUM_JUNCTION_LOOK_AHEAD);
243 
244  std::pair<SimpleWaypointPtr,uint64_t> look_ahead_point_index_pair = track_traffic.GetTargetWaypoint(waypoint_buffer, look_ahead_distance);
245  SimpleWaypointPtr &look_ahead_point = look_ahead_point_index_pair.first;
246  uint64_t &look_ahead_index = look_ahead_point_index_pair.second;
247  bool approaching_junction = false;
248  if (look_ahead_point->CheckJunction() && !(updated_front_waypoint->CheckJunction())) {
249  if (speed_limit*3.6f > HIGHWAY_SPEED) {
250  for (uint64_t j = 0u; (j < look_ahead_index) && !approaching_junction; ++j) {
251  SimpleWaypointPtr swp = waypoint_buffer.at(j);
252  if (swp->GetNextWaypoint().size() > 1) {
253  approaching_junction = true;
254  }
255  }
256  } else {
257  approaching_junction = true;
258  }
259  }
260 
261  // Reset the variables when no longer approaching a junction.
262  if (!approaching_junction && approached[actor_id]){
263  final_safe_points[actor_id] = nullptr;
264  approached[actor_id] = false;
265  }
266 
267  // Only do once, when the junction has just been seen.
268  else if (approaching_junction && !approached[actor_id]){
269 
270  SimpleWaypointPtr final_point = nullptr;
271  final_point = GetSafeLocationAfterJunction(vehicle_reference, waypoint_buffer);
272  if(final_point != nullptr){
273  final_safe_points[actor_id] = final_point;
274  approaching_junction = false;
275  approached[actor_id] = true;
276  }
277  }
278 
279  // Determining possible collision candidates around the ego vehicle.
280  ActorIdSet overlapping_actor_set = track_traffic.GetOverlappingVehicles(actor_id);
281  using ActorInfoMap = std::unordered_map<ActorId, std::pair<Actor, cg::Location>>;
282  ActorInfoMap overlapping_actor_info;
283  std::vector<ActorId> collision_candidate_ids;
284 
285  // Run through vehicles with overlapping paths, obtain actor reference,
286  // and filter them based on distance to ego vehicle.
287  Actor overlapping_actor_ptr = nullptr;
288  for (ActorId overlapping_actor_id: overlapping_actor_set) {
289 
290  // If actor is part of the registered actors.
291  if (vehicle_id_to_index.find(overlapping_actor_id) != vehicle_id_to_index.end()) {
292  overlapping_actor_ptr = actor_list.at(vehicle_id_to_index.at(overlapping_actor_id));
293  }
294  // If actor is part of the unregistered actors.
295  else if (unregistered_actors.find(overlapping_actor_id) != unregistered_actors.end()) {
296  overlapping_actor_ptr = unregistered_actors.at(overlapping_actor_id);
297  }
298  // If actor is within maximum collision avoidance range.
299  if (overlapping_actor_ptr!=nullptr && overlapping_actor_ptr->IsAlive()
300  && cg::Math::DistanceSquared(overlapping_actor_ptr->GetLocation(),
301  vehicle_location) < std::pow(MAX_COLLISION_RADIUS, 2))
302  {
303  overlapping_actor_info.insert({overlapping_actor_id,
304  {overlapping_actor_ptr, overlapping_actor_ptr->GetLocation()}});
305  collision_candidate_ids.push_back(overlapping_actor_id);
306  }
307  }
308 
309  // Sorting collision candidates in accending order of distance to current vehicle.
310  std::sort(collision_candidate_ids.begin(), collision_candidate_ids.end(),
311  [&overlapping_actor_info, &vehicle_location] (const ActorId& a_id_1, const ActorId& a_id_2) {
312  const cg::Location& e_loc = vehicle_location;
313  const cg::Location& loc_1 = overlapping_actor_info.at(a_id_1).second;
314  const cg::Location& loc_2 = overlapping_actor_info.at(a_id_2).second;
315  return (cg::Math::DistanceSquared(e_loc, loc_1) < cg::Math::DistanceSquared(e_loc, loc_2));
316  });
317 
318  // Constructing output vector.
319  std::vector<std::tuple<ActorId, Actor, cg::Vector3D>> collision_candidates;
320  std::for_each(collision_candidate_ids.begin(), collision_candidate_ids.end(),
321  [&overlapping_actor_info, &collision_candidates, this] (const ActorId& a_id) {
322  collision_candidates.push_back({a_id,
323  overlapping_actor_info.at(a_id).first,
324  this->GetVelocity(a_id)});
325  });
326 
327  // Sampling waypoint window for teleportation in hybrid physics mode.
328  std::vector<SimpleWaypointPtr> position_window;
329  if (hybrid_physics_mode) {
330 
331  cg::Vector3D heading = vehicle->GetTransform().GetForwardVector();
332  bool window_begin = false;
333  SimpleWaypointPtr begining_wp;
334  bool window_complete = false;
335 
336  for (uint32_t j = 0u; j < waypoint_buffer.size() && !window_complete; ++j) {
337 
338  SimpleWaypointPtr &swp = waypoint_buffer.at(j);
339  cg::Vector3D relative_position = swp->GetLocation() - vehicle_location;
340 
341  // Sample waypoints in front of the vehicle;
342  if (!window_begin && cg::Math::Dot(relative_position, heading) > 0.0f) {
343  window_begin = true;
344  begining_wp = swp;
345  }
346 
347  if (window_begin && !window_complete) {
348  if (swp->DistanceSquared(begining_wp) > std::pow(POSITION_WINDOW_SIZE, 2)) {
349  // Stop when maximum size is reached.
350  window_complete = true;
351  } else {
352  position_window.push_back(swp);
353  }
354  }
355  }
356  }
357 
358  // Editing output frames.
359  LocalizationToPlannerData &planner_message = current_planner_frame->at(i);
360  planner_message.actor = vehicle;
361  planner_message.deviation = dot_product;
362  planner_message.distance = distance;
363  planner_message.approaching_true_junction = approaching_junction;
364  planner_message.velocity = vehicle_velocity_vector;
365  planner_message.position_window = std::move(position_window);
366  planner_message.physics_enabled = IsPhysicsEnabled(actor_id);
367 
368  LocalizationToCollisionData &collision_message = current_collision_frame->at(i);
369  collision_message.actor = vehicle;
370  collision_message.buffer = waypoint_buffer;
371  collision_message.overlapping_actors = std::move(collision_candidates);
372  collision_message.closest_waypoint = updated_front_waypoint;
373  collision_message.junction_look_ahead_waypoint = waypoint_buffer.at(look_ahead_index);
374  collision_message.safe_point_after_junction = final_safe_points[actor_id];
375  collision_message.velocity = vehicle_velocity_vector;
376 
377  LocalizationToTrafficLightData &traffic_light_message = current_traffic_light_frame->at(i);
378  traffic_light_message.actor = vehicle;
379  traffic_light_message.closest_waypoint = updated_front_waypoint;
380  traffic_light_message.junction_look_ahead_waypoint = waypoint_buffer.at(look_ahead_index);
381 
382  // Updating idle time when necessary.
383  UpdateIdleTime(vehicle);
384  }
385 
386  if (IsVehicleStuck(maximum_idle_time.first)) {
387  TryDestroyVehicle(maximum_idle_time.first);
388  }
389 
390  // Updating maximum idle time to null for the next iteration.
391  maximum_idle_time = std::make_pair(nullptr, current_timestamp.elapsed_seconds);
392  }

406  void LocalizationStage::DataReceiver() {
407 
408  // Wait for external trigger to initiate the pipeline in synchronous mode.
409  if (parameters.GetSynchronousMode()) {
410  std::unique_lock<std::mutex> lock(step_execution_mutex);
411  while (!run_step.load()) {
412  step_execution_trigger.wait_for(lock, 1ms, [this]() {return run_step.load();});
413  }
414  // Set flag to false, unblock RunStep() call and release mutex lock.
415  run_step.store(false);
416  }
417 
418  hybrid_physics_mode = parameters.GetHybridPhysicsMode();
419 
420  bool is_deleted_actors_present = false;
421  std::set<uint32_t> world_actor_id;
422  std::vector<ActorPtr> actor_list_to_be_deleted;
423 
424  // Filter function to collect the data.
425  auto Filter = [&](auto &actors, auto &wildcard_pattern) {
426  std::vector<carla::client::detail::ActorVariant> filtered;
427  for (auto &&actor : actors) {
428  if (carla::StringUtil::Match(carla::client::detail::ActorVariant(actor).GetTypeId(), wildcard_pattern)) {
429  filtered.push_back(actor);
430  }
431  }
432  return filtered;
433  };
434 
435  // TODO: Combine this actor scanning logic with ScanUnregisteredActors()
436  // Get all the actors.
437  auto world_actors_list = episode_proxy_ls.Lock()->GetAllTheActorsInTheEpisode();
438 
439  // Filter with vehicle wildcard.
440  auto vehicles = Filter(world_actors_list, "vehicle.*");
441 
442  // Building a set of vehicle ids in the world.
443  for (const auto &actor : vehicles) {
444  world_actor_id.insert(actor.GetId());
445 
446  // Identify hero vehicles if system is in hybrid physics mode.
447  if (hybrid_physics_mode) {
448  Actor actor_ptr = actor.Get(episode_proxy_ls);
449  ActorId hero_actor_id = actor_ptr->GetId();
450  if (hero_actors.find(hero_actor_id) == hero_actors.end()) {
451  for (auto&& attribute: actor_ptr->GetAttributes()) {
452  if (attribute.GetId() == "role_name" && attribute.GetValue() == "hero") {
453  hero_actors.insert({hero_actor_id, actor_ptr});
454  }
455  }
456  }
457  }
458  }
459 
460  // Invalidate hero actor pointer if it is not alive anymore.
461  ActorIdSet hero_actors_to_delete;
462  if (hybrid_physics_mode && hero_actors.size() != 0u) {
463  for (auto &hero_actor_info: hero_actors) {
464  if(world_actor_id.find(hero_actor_info.first) == world_actor_id.end()) {
465  hero_actors_to_delete.insert(hero_actor_info.first);
466  }
467  }
468  }
469  for (auto &deletion_id: hero_actors_to_delete) {
470  hero_actors.erase(deletion_id);
471  }
472 
473  // Search for invalid/destroyed vehicles.
474  for (auto &actor : actor_list) {
475  ActorId deletion_id = actor->GetId();
476  if (world_actor_id.find(deletion_id) == world_actor_id.end()) {
477  actor_list_to_be_deleted.emplace_back(actor);
478  track_traffic.DeleteActor(deletion_id);
479  last_lane_change_location.erase(deletion_id);
480  kinematic_state_map.erase(deletion_id);
481  }
482  }
483 
484  // Clearing the registered actor list.
485  if(!actor_list_to_be_deleted.empty()) {
486  registered_actors.Remove(actor_list_to_be_deleted);
487  actor_list.clear();
488  actor_list = registered_actors.GetList();
489  is_deleted_actors_present = true;
490  }
491 
492  // Building a list of registered actors and connecting
493  // the vehicle ids to their position indices on data arrays.
494 
495  if (is_deleted_actors_present || (registered_actors_state != registered_actors.GetState())) {
496  actor_list.clear();
497  actor_list_to_be_deleted.clear();
498  actor_list = registered_actors.GetList();
499  uint64_t index = 0u;
500  vehicle_id_to_index.clear();
501  for (auto &actor : actor_list) {
502  vehicle_id_to_index.insert({actor->GetId(), index});
503  ++index;
504  }
505  registered_actors_state = registered_actors.GetState();
506  }
507 
508  // Allocating new containers for the changed number of registered vehicles.
509  if (number_of_vehicles != actor_list.size()) {
510  number_of_vehicles = static_cast<uint64_t>(actor_list.size());
511  // Allocating output frames to be shared with the motion planner stage.
512  planner_frame_a = std::make_shared<LocalizationToPlannerFrame>(number_of_vehicles);
513  planner_frame_b = std::make_shared<LocalizationToPlannerFrame>(number_of_vehicles);
514  // Allocating output frames to be shared with the collision stage.
515  collision_frame_a = std::make_shared<LocalizationToCollisionFrame>(number_of_vehicles);
516  collision_frame_b = std::make_shared<LocalizationToCollisionFrame>(number_of_vehicles);
517  // Allocating output frames to be shared with the traffic light stage
518  traffic_light_frame_a = std::make_shared<LocalizationToTrafficLightFrame>(number_of_vehicles);
519  traffic_light_frame_b = std::make_shared<LocalizationToTrafficLightFrame>(number_of_vehicles);
520  }
521  }

647  SimpleWaypointPtr LocalizationStage::AssignLaneChange(Actor vehicle, const cg::Location &vehicle_location,
648  bool force, bool direction)
649  {
650 
651  const ActorId actor_id = vehicle->GetId();
652  const float vehicle_velocity = GetVelocity(actor_id).Length();
653 
654  // Waypoint representing the new starting point for the waypoint buffer
655  // due to lane change. Remains nullptr if lane change not viable.
656  SimpleWaypointPtr change_over_point = nullptr;
657 
658  // Retrieve waypoint buffer for current vehicle.
659  const Buffer& waypoint_buffer = buffer_list->at(actor_id);
660 
661  // Check buffer is not empty.
662  if (!waypoint_buffer.empty())
663  {
664  // Get the left and right waypoints for the current closest waypoint.
665  const SimpleWaypointPtr& current_waypoint = waypoint_buffer.front();
666  const SimpleWaypointPtr left_waypoint = current_waypoint->GetLeftWaypoint();
667  const SimpleWaypointPtr right_waypoint = current_waypoint->GetRightWaypoint();
668 
669  // Retrieve vehicles with overlapping waypoint buffers with current vehicle.
670  const auto blocking_vehicles = track_traffic.GetOverlappingVehicles(actor_id);
671 
672  // Find immediate in-lane obstacle and check if any are too close to initiate lane change.
673  bool obstacle_too_close = false;
674  float minimum_squared_distance = std::numeric_limits<float>::infinity();
675  ActorId obstacle_actor_id = 0u;
676  for (auto i = blocking_vehicles.begin();
677  i != blocking_vehicles.end() && !obstacle_too_close && !force;
678  ++i)
679  {
680  const ActorId &other_actor_id = *i;
681  // Find vehicle in registered list and check if it's buffer is not empty.
682  if (vehicle_id_to_index.find(other_actor_id) != vehicle_id_to_index.end()
683  && buffer_list->find(other_actor_id) != buffer_list->end()
684  && !buffer_list->at(other_actor_id).empty())
685  {
686  const Buffer& other_buffer = buffer_list->at(other_actor_id);
687  const SimpleWaypointPtr& other_current_waypoint = other_buffer.front();
688  const cg::Location other_location = other_current_waypoint->GetLocation();
689 
690  const cg::Vector3D reference_heading = current_waypoint->GetForwardVector();
691  cg::Vector3D reference_to_other = other_current_waypoint->GetLocation()
692  - current_waypoint->GetLocation();
693  const cg::Vector3D other_heading = other_current_waypoint->GetForwardVector();
694 
695  // Check both vehicles are not in junction,
696  // Check if the other vehicle is in front of the current vehicle,
697  // Check if the two vehicles have acceptable angular deviation between their headings.
698  if (!current_waypoint->CheckJunction()
699  && !other_current_waypoint->CheckJunction()
700  && other_current_waypoint->GetWaypoint()->GetRoadId() == current_waypoint->GetWaypoint()->GetRoadId()
701  && other_current_waypoint->GetWaypoint()->GetLaneId() == current_waypoint->GetWaypoint()->GetLaneId()
702  && cg::Math::Dot(reference_heading, reference_to_other) > 0.0f
703  && cg::Math::Dot(reference_heading, other_heading) > MAXIMUM_LANE_OBSTACLE_CURVATURE)
704  {
705  float squared_distance = cg::Math::DistanceSquared(vehicle_location, other_location);
706  // Abort if the obstacle is too close.
707  if (squared_distance > std::pow(MINIMUM_LANE_CHANGE_DISTANCE, 2))
708  {
709  // Remember if the new vehicle is closer.
710  if (squared_distance < minimum_squared_distance
711  && squared_distance < std::pow(MAXIMUM_LANE_OBSTACLE_DISTANCE, 2.0f))
712  {
713  minimum_squared_distance = squared_distance;
714  obstacle_actor_id = other_actor_id;
715  }
716  } else {
717  obstacle_too_close = true;
718  }
719  }
720  }
721  }
722 
723  // If a valid immediate obstacle found.
724  if (!obstacle_too_close && obstacle_actor_id != 0u && !force)
725  {
726  const Buffer& other_buffer = buffer_list->at(obstacle_actor_id);
727  const SimpleWaypointPtr& other_current_waypoint = other_buffer.front();
728  const auto other_neighbouring_lanes = {other_current_waypoint->GetLeftWaypoint(),
729  other_current_waypoint->GetRightWaypoint()};
730 
731  // Flags reflecting whether adjacent lanes are free near the obstacle.
732  bool distant_left_lane_free = false;
733  bool distant_right_lane_free = false;
734 
735  // Check if the neighbouring lanes near the obstructing vehicle are free of other vehicles.
736  bool left_right = true;
737  for (auto& candidate_lane_wp: other_neighbouring_lanes) {
738  if (candidate_lane_wp != nullptr &&
739  track_traffic.GetPassingVehicles(candidate_lane_wp->GetId()).size() == 0) {
740 
741  if (left_right) distant_left_lane_free = true;
742  else distant_right_lane_free = true;
743 
744  }
745  left_right = !left_right;
746  }
747 
748  // Based on what lanes are free near the obstacle,
749  // find the change over point with no vehicles passing through them.
750  if (distant_right_lane_free && right_waypoint != nullptr
751  && track_traffic.GetPassingVehicles(right_waypoint->GetId()).size() == 0)
752  {
753  change_over_point = right_waypoint;
754  } else if (distant_left_lane_free && left_waypoint != nullptr
755  && track_traffic.GetPassingVehicles(left_waypoint->GetId()).size() == 0)
756  {
757  change_over_point = left_waypoint;
758  }
759  } else if (force) {
760  if (direction && right_waypoint != nullptr) {
761  change_over_point = right_waypoint;
762  } else if (!direction && left_waypoint != nullptr) {
763  change_over_point = left_waypoint;
764  }
765  }
766 
767  if (change_over_point != nullptr)
768  {
769  const float change_over_distance = cg::Math::Clamp(1.5f*vehicle_velocity, 3.0f, 20.0f);
770  const auto starting_point = change_over_point;
771  while (change_over_point->DistanceSquared(starting_point) < std::pow(change_over_distance, 2) &&
772  !change_over_point->CheckJunction()) {
773  change_over_point = change_over_point->GetNextWaypoint()[0];
774  }
775 
776  // Reset this variable if needed
777  if (approached[actor_id]){
778  approached[actor_id] = false;
779  }
780  }
781  }
782 
783  return change_over_point;
784  }

787

790  float length = vehicle->GetBoundingBox().extent.x;
791  // First Waypoint before the junction
792  const SimpleWaypointPtr initial_point;
793  uint64_t initial_index = 0;
794  // First Waypoint after the junction
795  SimpleWaypointPtr safe_point = nullptr;
796  uint64_t safe_index = 0;
797  // Vehicle position after the junction
798  SimpleWaypointPtr final_point = nullptr;
799  // Safe space after the junction
800  const float safe_distance = 1.5f*length;
801 
802  for (uint64_t j = 0u; j < waypoint_buffer.size(); ++j){
803  if (waypoint_buffer.at(j)->CheckJunction()){
804  initial_index = j;
805  break;
806  }
807  }
808 
809  // Stop if something failed
810  if (initial_index == 0 && !waypoint_buffer.front()->CheckJunction()){
811  return final_point;
812  }
813 
814  // 2) Search for the end of the intersection (if it is in the buffer)
815  for (uint64_t i = initial_index; i < waypoint_buffer.size(); ++i){
816 
817  if (!waypoint_buffer.at(i)->CheckJunction()){
818  safe_point = waypoint_buffer.at(i);
819  safe_index = i;
820  break;
821  }
822  }
823 
824  // If it hasn't been found, extend the buffer
825  if(safe_point == nullptr){
826  while (waypoint_buffer.back()->CheckJunction()) {
827 
828  std::vector<SimpleWaypointPtr> next_waypoints = waypoint_buffer.back()->GetNextWaypoint();
829  uint64_t selection_index = 0u;
830  if (next_waypoints.size() > 1) {
831  selection_index = static_cast<uint64_t>(rand()) % next_waypoints.size();
832  }
833 
834  PushWaypoint(waypoint_buffer, actor_id, next_waypoints.at(selection_index));

836  // Save the last one

838  }
839 
840  // Stop if something failed
841  if (safe_index == 0){
842  return final_point;
843  }
844 
845  // 3) Search for final_point (again, if it is in the buffer)
846 
847  for(uint64_t k = safe_index; k < waypoint_buffer.size(); ++k){
848 
849  if(safe_point->Distance(waypoint_buffer.at(k)->GetLocation()) > safe_distance){
850  final_point = waypoint_buffer.at(k);
851  break;
852  }
853  }
854 
855  // If it hasn't been found, extend the buffer
856  if(final_point == nullptr){
857  while (safe_point->Distance(waypoint_buffer.back()->GetLocation()) < safe_distance) {
858 
859  // Record the last point as a safe one and save it
860  std::vector<SimpleWaypointPtr> next_waypoints = waypoint_buffer.back()->GetNextWaypoint();
861  uint64_t selection_index = 0u;
862  // Pseudo-randomized path selection if found more than one choice.
863  if (next_waypoints.size() > 1) {
864  selection_index = static_cast<uint64_t>(rand()) % next_waypoints.size();
865  }
866

868  }

870  }

937  void LocalizationStage::UpdateSwarmVelocities() {
938 
939  // Location of hero vehicle if present.
940  std::vector<cg::Location> hero_locations;
941  if (hybrid_physics_mode && hero_actors.size() != 0u) {
942  for (auto &hero_actor_info: hero_actors) {
943  hero_locations.push_back(hero_actor_info.second->GetLocation());
944  }
945  }
946 
947  // Using (1/20)s time delta for computing velocity.
948  float dt = HYBRID_MODE_DT;
949  // Skipping velocity update if elapsed time is less than 0.05s in asynchronous, hybrid mode.
950  if (!parameters.GetSynchronousMode()) {
951  TimePoint current_instance = chr::system_clock::now();
952  chr::duration<float> elapsed_time = current_instance - previous_update_instance;
953  if (elapsed_time.count() > dt) {
954  previous_update_instance = current_instance;
955  } else if (hybrid_physics_mode) {
956  return;
957  }
958  }
959 
960  // Update velocity for all registered vehicles.
961  for (const Actor &actor: actor_list) {
962 
963  ActorId actor_id = actor->GetId();
964  cg::Location vehicle_location = actor->GetLocation();
965 
966  // Adding entry if not present.
967  if (kinematic_state_map.find(actor_id) == kinematic_state_map.end()) {
968  kinematic_state_map.insert({actor_id, KinematicState{true, vehicle_location, cg::Vector3D()}});
969  }
970 
971  // Check if current actor is in range of hero actor and enable physics in hybrid mode.
972  bool in_range_of_hero_actor = false;
973  if (hybrid_physics_mode
974  && hero_actors.size() != 0u){
975  for (auto &hero_location: hero_locations) {
976  if (cg::Math::DistanceSquared(vehicle_location, hero_location) < std::pow(PHYSICS_RADIUS, 2)) {
977  in_range_of_hero_actor = true;
978  break;
979  }
980  }
981  }
982  bool enable_physics = hybrid_physics_mode? in_range_of_hero_actor: true;
983  kinematic_state_map.at(actor_id).physics_enabled = enable_physics;
984  actor->SetSimulatePhysics(enable_physics);
985 
986  // When we say velocity, we usually mean velocity for a vehicle along it's heading.
987  // Velocity component due to rotation can be removed by taking dot product with heading vector.
988  cg::Vector3D heading = actor->GetTransform().GetForwardVector();
989  if (enable_physics) {
990 
991  kinematic_state_map.at(actor_id).velocity = cg::Math::Dot(actor->GetVelocity(), heading) * heading;
992 
993  } else {
994 
995  cg::Vector3D displacement = (vehicle_location - kinematic_state_map.at(actor_id).location);
996  cg::Vector3D displacement_along_heading = cg::Math::Dot(displacement, heading) * heading;
997  cg::Vector3D velocity = displacement_along_heading/dt;
998  kinematic_state_map.at(actor_id).velocity = velocity;
999  }
1000 
1001  // Updating location after velocity is computed.
1002  kinematic_state_map.at(actor_id).location = vehicle_location;
1003  }
