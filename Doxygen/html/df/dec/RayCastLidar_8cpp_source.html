45 void ARayCastLidar::CreateLasers()
46 {
47  const auto NumberOfLasers = Description.Channels;
48  check(NumberOfLasers > 0u);
49  const float DeltaAngle = NumberOfLasers == 1u ? 0.f :
50  (Description.UpperFovLimit - Description.LowerFovLimit) /
51  static_cast<float>(NumberOfLasers - 1);
52  LaserAngles.Empty(NumberOfLasers);

54  {
55  const float VerticalAngle =
56  Description.UpperFovLimit - static_cast<float>(i) * DeltaAngle;
57  LaserAngles.Emplace(VerticalAngle);
58  }

71 void ARayCastLidar::ReadPoints(const float DeltaTime)
72 {
73  const uint32 ChannelCount = Description.Channels;
74  const uint32 PointsToScanWithOneLaser =
75  FMath::RoundHalfFromZero(
76  Description.PointsPerSecond * DeltaTime / float(ChannelCount));
77 
78  if (PointsToScanWithOneLaser <= 0)
79  {
80  UE_LOG(
81  LogCarla,
82  Warning,
83  TEXT("%s: no points requested this frame, try increasing the number of points per second."),
84  *GetName());
85  return;
86  }
87 
88  check(ChannelCount == LaserAngles.Num());
89 
90  const float CurrentHorizontalAngle = carla::geom::Math::ToDegrees(
91  LidarMeasurement.GetHorizontalAngle());
92  const float AngleDistanceOfTick = Description.RotationFrequency * 360.0f * DeltaTime;
93  const float AngleDistanceOfLaserMeasure = AngleDistanceOfTick / PointsToScanWithOneLaser;
94 
95  LidarMeasurement.Reset(ChannelCount * PointsToScanWithOneLaser);
96 
97  for (auto Channel = 0u; Channel < ChannelCount; ++Channel)
98  {
99  for (auto i = 0u; i < PointsToScanWithOneLaser; ++i)
100  {
101  FVector Point;
102  const float Angle = CurrentHorizontalAngle + AngleDistanceOfLaserMeasure * i;
103  if (ShootLaser(Channel, Angle, Point))
104  {
105  LidarMeasurement.WritePoint(Channel, Point);
106  }
107  }
108  }
109 
110  const float HorizontalAngle = carla::geom::Math::ToRadians(
111  std::fmod(CurrentHorizontalAngle + AngleDistanceOfTick, 360.0f));
112  LidarMeasurement.SetHorizontalAngle(HorizontalAngle);
113 }

115 bool ARayCastLidar::ShootLaser(const uint32 Channel, const float HorizontalAngle, FVector &XYZ) const
116 {
117  const float VerticalAngle = LaserAngles[Channel];
118 
119  FCollisionQueryParams TraceParams = FCollisionQueryParams(FName(TEXT("Laser_Trace")), true, this);
120  TraceParams.bTraceComplex = true;
121  TraceParams.bReturnPhysicalMaterial = false;
122 
123  FHitResult HitInfo(ForceInit);
124 
125  FVector LidarBodyLoc = GetActorLocation();
126  FRotator LidarBodyRot = GetActorRotation();
127  FRotator LaserRot (VerticalAngle, HorizontalAngle, 0); // float InPitch, float InYaw, float InRoll
128  FRotator ResultRot = UKismetMathLibrary::ComposeRotators(
129  LaserRot,
130  LidarBodyRot
131  );
132  const auto Range = Description.Range;
133  FVector EndTrace = Range * UKismetMathLibrary::GetForwardVector(ResultRot) + LidarBodyLoc;
134 
135  GetWorld()->LineTraceSingleByChannel(
136  HitInfo,
137  LidarBodyLoc,
138  EndTrace,
139  ECC_MAX,
140  TraceParams,
141  FCollisionResponseParams::DefaultResponseParam
142  );
143 
144  if (HitInfo.bBlockingHit)
145  {
146  if (Description.ShowDebugPoints)
147  {
148  DrawDebugPoint(
149  GetWorld(),
150  HitInfo.ImpactPoint,
151  10, //size
152  FColor(255,0,255),
153  false, //persistent (never goes away)
154  0.1 //point leaves a trail on moving object
155  );
156  }
157 
158  XYZ = LidarBodyLoc - HitInfo.ImpactPoint;
159  XYZ = UKismetMathLibrary::RotateAngleAxis(
160  XYZ,
161  - LidarBodyRot.Yaw + 90,
162  FVector(0, 0, 1)
163  );
164 
165  return true;
166  } else {
167  return false;
168  }
169 }
