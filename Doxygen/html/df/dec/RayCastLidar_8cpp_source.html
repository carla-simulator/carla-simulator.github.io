105  bool ARayCastLidar::PostprocessDetection(FDetection& Detection) const
106  {
107  if (Description.NoiseStdDev > std::numeric_limits<float>::epsilon()) {
108  const auto ForwardVector = Detection.point.MakeUnitVector();
109  const auto Noise = ForwardVector * RandomEngine->GetNormalDistribution(0.0f, Description.NoiseStdDev);
110  Detection.point += Noise;
111  }
112 
113  const float Intensity = Detection.intensity;
114  if(Intensity > Description.DropOffIntensityLimit)
115  return true;
116  else
117  return RandomEngine->GetUniformFloat() < DropOffAlpha * Intensity + DropOffBeta;
118  }

120  void ARayCastLidar::ComputeAndSaveDetections(const FTransform& SensorTransform) {
121  for (auto idxChannel = 0u; idxChannel < Description.Channels; ++idxChannel)
122  PointsPerChannel[idxChannel] = RecordedHits[idxChannel].size();
123  LidarData.ResetSerPoints(PointsPerChannel);
124 
125  for (auto idxChannel = 0u; idxChannel < Description.Channels; ++idxChannel) {
126  for (auto& hit : RecordedHits[idxChannel]) {
127  FDetection Detection = ComputeDetection(hit, SensorTransform);

129  LidarData.WritePointSync(Detection);
