51 void ARayCastLidar::CreateLasers()
52 {
53  const auto NumberOfLasers = Description.Channels;
54  check(NumberOfLasers > 0u);
55  const float DeltaAngle = NumberOfLasers == 1u ? 0.f :
56  (Description.UpperFovLimit - Description.LowerFovLimit) /
57  static_cast<float>(NumberOfLasers - 1);
58  LaserAngles.Empty(NumberOfLasers);
59  for(auto i = 0u; i < NumberOfLasers; ++i)
60  {
61  const float VerticalAngle =
62  Description.UpperFovLimit - static_cast<float>(i) * DeltaAngle;
63  LaserAngles.Emplace(VerticalAngle);
64  }
65 
66  // Compute drop off model parameters
67  DropOffBeta = 1.0f - Description.DropOffAtZeroIntensity;
68  DropOffAlpha = Description.DropOffAtZeroIntensity / Description.DropOffIntensityLimit;
69  DropOffGenActive = Description.DropOffGenRate > std::numeric_limits<float>::epsilon();
70 }

82 void ARayCastLidar::ReadPoints(const float DeltaTime)
83 {
84  const uint32 ChannelCount = Description.Channels;
85  const uint32 PointsToScanWithOneLaser =
86  FMath::RoundHalfFromZero(
87  Description.PointsPerSecond * DeltaTime / float(ChannelCount));
88 
89  if (PointsToScanWithOneLaser <= 0)
90  {
91  UE_LOG(
92  LogCarla,
93  Warning,
94  TEXT("%s: no points requested this frame, try increasing the number of points per second."),
95  *GetName());
96  return;
97  }
98 
99  check(ChannelCount == LaserAngles.Num());
100 
101  const float CurrentHorizontalAngle = carla::geom::Math::ToDegrees(
102  LidarMeasurement.GetHorizontalAngle());
103  const float AngleDistanceOfTick = Description.RotationFrequency * 360.0f * DeltaTime;
104  const float AngleDistanceOfLaserMeasure = AngleDistanceOfTick / PointsToScanWithOneLaser;
105 
106  LidarMeasurement.Reset(ChannelCount, PointsToScanWithOneLaser);
107 
108 
109  GetWorld()->GetPhysicsScene()->GetPxScene()->lockRead();
110  ParallelFor(ChannelCount, [&](int32 idxChannel) {
111 
112  FCriticalSection Mutex;
113  ParallelFor(PointsToScanWithOneLaser, [&](int32 idxPtsOneLaser) {
114  FVector Point;
115  float Intensity;
116  const float Angle = CurrentHorizontalAngle + AngleDistanceOfLaserMeasure * idxPtsOneLaser;
117  if (ShootLaser(idxChannel, Angle, Point, Intensity)) {
118  Mutex.Lock();
119  LidarMeasurement.WritePointAsync(idxChannel, {Point, Intensity});
120  Mutex.Unlock();
121  }
122  });
123  });
124  GetWorld()->GetPhysicsScene()->GetPxScene()->unlockRead();
125 
126  LidarMeasurement.SaveDetections();
127 
128 
129  const float HorizontalAngle = carla::geom::Math::ToRadians(
130  std::fmod(CurrentHorizontalAngle + AngleDistanceOfTick, 360.0f));
131  LidarMeasurement.SetHorizontalAngle(HorizontalAngle);
132 }

134 float ARayCastLidar::ComputeIntensity(const FVector &LidarBodyLoc, const FHitResult& HitInfo) const
135 {
136  const FVector HitPoint = HitInfo.ImpactPoint - LidarBodyLoc;
137  constexpr float TO_METERS = 1e-2;
138  const float Distance = TO_METERS * HitPoint.Size();
139 
140  const float AttenAtm = Description.AtmospAttenRate;
141  const float AbsAtm = exp(-AttenAtm * Distance);
142 
143  const float IntRec = AbsAtm;
144 
145  return IntRec;
146 }

148 bool ARayCastLidar::ShootLaser(const uint32 Channel, const float HorizontalAngle, FVector &XYZ, float &Intensity) const
149 {
150  if(DropOffGenActive && RandomEngine->GetUniformFloat() < Description.DropOffGenRate)
151  return false;
152 
153  const float VerticalAngle = LaserAngles[Channel];
154 
155  FCollisionQueryParams TraceParams = FCollisionQueryParams(FName(TEXT("Laser_Trace")), true, this);
156  TraceParams.bTraceComplex = true;
157  TraceParams.bReturnPhysicalMaterial = false;
158 
159  FHitResult HitInfo(ForceInit);
160 
161  FTransform actorTransf = GetTransform();
162  FVector LidarBodyLoc = GetActorLocation();
163  FRotator LidarBodyRot = GetActorRotation();
164  FRotator LaserRot (VerticalAngle, HorizontalAngle, 0); // float InPitch, float InYaw, float InRoll
165  FRotator ResultRot = UKismetMathLibrary::ComposeRotators(
166  LaserRot,
167  LidarBodyRot
168  );
169  const auto Range = Description.Range;
170  FVector EndTrace = Range * UKismetMathLibrary::GetForwardVector(ResultRot) + LidarBodyLoc;
171 
172  GetWorld()->LineTraceSingleByChannel(
173  HitInfo,
174  LidarBodyLoc,
175  EndTrace,
176  ECC_GameTraceChannel2,
177  TraceParams,
178  FCollisionResponseParams::DefaultResponseParam
179  );
180 
181  if (HitInfo.bBlockingHit)
182  {
183  if (Description.ShowDebugPoints)
184  {
185  DrawDebugPoint(
186  GetWorld(),
187  HitInfo.ImpactPoint,
188  10, //size
189  FColor(255,0,255),
190  false, //persistent (never goes away)
191  0.1 //point leaves a trail on moving object
192  );
193  }
194 
195  const FVector hp = HitInfo.ImpactPoint;
196  XYZ = actorTransf.Inverse().TransformPosition(hp);
197 
198  Intensity = ComputeIntensity(LidarBodyLoc, HitInfo);
199 
200  if(Intensity > Description.DropOffIntensityLimit)
201  return true;
202  else
203  return RandomEngine->GetUniformFloat() < DropOffAlpha * Intensity + DropOffBeta;
204 
205  } else {
206  return false;
207  }
208 }
