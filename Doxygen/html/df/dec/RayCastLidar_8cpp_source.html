42 void ARayCastLidar::CreateLasers()
43 {
44  const auto NumberOfLasers = Description.Channels;
45  check(NumberOfLasers > 0u);
46  const float DeltaAngle = NumberOfLasers == 1u ? 0.f :
47  (Description.UpperFovLimit - Description.LowerFovLimit) /
48  static_cast<float>(NumberOfLasers - 1);
49  LaserAngles.Empty(NumberOfLasers);
50  for(auto i = 0u; i < NumberOfLasers; ++i)
51  {
52  const float VerticalAngle =
53  Description.UpperFovLimit - static_cast<float>(i) * DeltaAngle;
54  LaserAngles.Emplace(VerticalAngle);
55  }

68 void ARayCastLidar::ReadPoints(const float DeltaTime)
69 {
70  const uint32 ChannelCount = Description.Channels;
71  const uint32 PointsToScanWithOneLaser =
72  FMath::RoundHalfFromZero(
73  Description.PointsPerSecond * DeltaTime / float(ChannelCount));
74 
75  if (PointsToScanWithOneLaser <= 0)
76  {
77  UE_LOG(
78  LogCarla,
79  Warning,
80  TEXT("%s: no points requested this frame, try increasing the number of points per second."),
81  *GetName());
82  return;
83  }
84 
85  check(ChannelCount == LaserAngles.Num());
86 
87  const float CurrentHorizontalAngle = LidarMeasurement.GetHorizontalAngle();
88  const float AngleDistanceOfTick = Description.RotationFrequency * 360.0f * DeltaTime;
89  const float AngleDistanceOfLaserMeasure = AngleDistanceOfTick / PointsToScanWithOneLaser;
90 
91  LidarMeasurement.Reset(ChannelCount * PointsToScanWithOneLaser);
92 
93  for (auto Channel = 0u; Channel < ChannelCount; ++Channel)
94  {
95  for (auto i = 0u; i < PointsToScanWithOneLaser; ++i)
96  {
97  FVector Point;
98  const float Angle = CurrentHorizontalAngle + AngleDistanceOfLaserMeasure * i;
99  if (ShootLaser(Channel, Angle, Point))
100  {
101  LidarMeasurement.WritePoint(Channel, Point);
102  }
103  }
104  }
105 
106  const float HorizontalAngle = std::fmod(CurrentHorizontalAngle + AngleDistanceOfTick, 360.0f);
107  LidarMeasurement.SetHorizontalAngle(HorizontalAngle);
108 }

110 bool ARayCastLidar::ShootLaser(const uint32 Channel, const float HorizontalAngle, FVector &XYZ) const
111 {
112  const float VerticalAngle = LaserAngles[Channel];
113 
114  FCollisionQueryParams TraceParams = FCollisionQueryParams(FName(TEXT("Laser_Trace")), true, this);
115  TraceParams.bTraceComplex = true;
116  TraceParams.bReturnPhysicalMaterial = false;
117 
118  FHitResult HitInfo(ForceInit);
119 
120  FVector LidarBodyLoc = GetActorLocation();
121  FRotator LidarBodyRot = GetActorRotation();
122  FRotator LaserRot (VerticalAngle, HorizontalAngle, 0); // float InPitch, float InYaw, float InRoll
123  FRotator ResultRot = UKismetMathLibrary::ComposeRotators(
124  LaserRot,
125  LidarBodyRot
126  );
127  const auto Range = Description.Range;
128  FVector EndTrace = Range * UKismetMathLibrary::GetForwardVector(ResultRot) + LidarBodyLoc;
129 
130  GetWorld()->LineTraceSingleByChannel(
131  HitInfo,
132  LidarBodyLoc,
133  EndTrace,
134  ECC_MAX,
135  TraceParams,
136  FCollisionResponseParams::DefaultResponseParam
137  );
138 
139  if (HitInfo.bBlockingHit)
140  {
141  if (Description.ShowDebugPoints)
142  {
143  DrawDebugPoint(
144  GetWorld(),
145  HitInfo.ImpactPoint,
146  10, //size
147  FColor(255,0,255),
148  false, //persistent (never goes away)
149  0.1 //point leaves a trail on moving object
150  );
151  }
152 
153  XYZ = LidarBodyLoc - HitInfo.ImpactPoint;
154  XYZ = UKismetMathLibrary::RotateAngleAxis(
155  XYZ,
156  - LidarBodyRot.Yaw + 90,
157  FVector(0, 0, 1)
158  );
159 
160  return true;
161  } else {
162  return false;
163  }
164 }
