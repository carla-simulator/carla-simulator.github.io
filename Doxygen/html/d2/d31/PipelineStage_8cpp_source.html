37  void PipelineStage::ReceiverThreadManager() {
38 
39  while (run_stage.load()) {
40  std::unique_lock<std::mutex> lock(thread_coordination_mutex);
41  // Wait for notification from sender thread and
42  // break waiting if the stage is stopped.
43  while (!run_receiver.load() && run_stage.load()) {
44  wake_receiver_notifier.wait_for(lock, 1ms, [=] {return run_receiver.load();});
45  }
46  lock.unlock();
47  run_receiver.store(false);
48 
49  // Receive data.
50  if (run_stage.load()) {
51  DataReceiver();
52  }
53 
54  // Notify action thread.
55  run_action.store(true);
56  wake_action_notifier.notify_one();
57  }
58  }
