38  void PipelineStage::ReceiverThreadManager() {
39 
40  while (run_stage.load()) {
41  std::unique_lock<std::mutex> lock(thread_coordination_mutex);
42  // Wait for notification from sender thread and
43  // break waiting if the stage is stopped.
44  while (!run_receiver.load() && run_stage.load()) {
45  wake_receiver_notifier.wait_for(lock, 1ms, [=] {return run_receiver.load();});
46  }
47  lock.unlock();
48  run_receiver.store(false);
49 
50  // Receive data.
51  if (run_stage.load()) {
52  DataReceiver();
53  }
54 
55  // Notify action thread.
56  run_action.store(true);
57  wake_action_notifier.notify_one();
58  }
59  }
