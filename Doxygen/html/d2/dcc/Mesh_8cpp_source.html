17  bool Mesh::IsValid() const {
18  // should be at least some one vertex
19  if (_vertices.empty()) {
20  std::cout << "Mesh validation error: there are no vertices in the mesh." << std::endl;
21  return false;
22  }
23 
24  // if there are indices, the amount must be multiple of 3
25  if (!_indexes.empty() && _indexes.size() % 3 != 0) {
26  std::cout << "Mesh validation error: the index amount must be multiple of 3." << std::endl;
27  return false;
28  }
29 
30  if (!_materials.empty() && _materials.back().index_end == 0) {
31  std::cout << "Mesh validation error: last material was not closed." << std::endl;
32  return false;
33  }
34 
35  return true;
36  }

54  void Mesh::AddMaterial(const std::string &material_name) {
55  const size_t open_index = _indexes.size();
56  if (!_materials.empty()) {
57  if (_materials.back().index_end == 0) {
58  std::cout << "last material was not closed, closing it..." << std::endl;
59  EndMaterial();
60  }
61  }
62  if (open_index % 3 != 0) {
63  std::cout << "open_index % 3 != 0" << std::endl;
64  return;
65  }
66  _materials.emplace_back(material_name, open_index, 0);
67  }

69  void Mesh::EndMaterial() {
70  const size_t close_index = _indexes.size();
71  if (_materials.empty() ||
72  _materials.back().index_start == close_index ||
73  _materials.back().index_end != 0) {
74  std::cout << "WARNING: Bad end of material. Material not started." << std::endl;
75  return;
76  }
77  if (_indexes.empty() || close_index % 3 != 0) {
78  std::cout << "WARNING: Bad end of material. Face not started/ended." << std::endl;
79  return;
80  }
81  _materials.back().index_end = close_index;
82  }

84  std::string Mesh::GenerateOBJ() const {
85  if (!IsValid()) {
86  return "";
87  }
88  std::stringstream out;
89  out << std::fixed; // Avoid using scientific notation
90 
91  out << "# List of geometric vertices, with (x, y, z) coordinates." << std::endl;
92  for (auto &v : _vertices) {
93  out << "v " << v.x << " " << v.y << " " << v.z << std::endl;
94  }
95 
96  if (!_uvs.empty()) {
97  out << std::endl << "# List of texture coordinates, in (u, v) coordinates, these will vary between 0 and 1." << std::endl;
98  for (auto &vt : _uvs) {
99  out << "vt " << vt.x << " " << vt.y << std::endl;
100  }
101  }
102 
103  if (!_normals.empty()) {
104  out << std::endl << "# List of vertex normals in (x, y, z) form; normals might not be unit vectors." << std::endl;
105  for (auto &vn : _normals) {
106  out << "vn " << vn.x << " " << vn.y << " " << vn.z << std::endl;
107  }
108  }
109 
110  if (!_indexes.empty()) {
111  out << std::endl << "# Polygonal face element." << std::endl;
112  auto it_m = _materials.begin();
113  auto it = _indexes.begin();
114  size_t index_counter = 0u;
115  while (it != _indexes.end()) {
116  // While exist materials
117  if (it_m != _materials.end()) {
118  // If the current material ends at this index
119  if (it_m->index_end == index_counter) {
120  ++it_m;
121  }
122  // If the current material start at this index
123  if (it_m->index_start == index_counter) {
124  out << "\nusemtl " << it_m->name << std::endl;
125  }
126  }
127 
128  // Add the actual face using the 3 consecutive indices
129  out << "f " << *it; ++it;
130  out << " " << *it; ++it;
131  out << " " << *it << std::endl; ++it;
132 
133  index_counter += 3;
134  }
135  }
136 
137  return out.str();
138  }

155  auto it_m = _materials.begin();
156  auto it = _indexes.begin();
157  size_t index_counter = 0u;
158  while (it != _indexes.end()) {
159  // While exist materials
160  if (it_m != _materials.end()) {
161  // If the current material ends at this index
162  if (it_m->index_end == index_counter) {
163  ++it_m;
164  }
165  // If the current material start at this index
166  if (it_m->index_start == index_counter) {
167  out << "\nusemtl " << it_m->name << std::endl;
168  }
169  }
170  // Add the actual face using the 3 consecutive indices
171  // Changes the face build direction to clockwise since
172  // the space has changed.
173  out << "f " << *it; ++it;
174  const auto i_2 = *it; ++it;
175  const auto i_3 = *it; ++it;

177  index_counter += 3;
178  }
179  }
180
