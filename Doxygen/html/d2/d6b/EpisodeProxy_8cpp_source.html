38  template <typename T>
39  typename EpisodeProxyImpl<T>::SharedPtrType EpisodeProxyImpl<T>::Lock() const {
40  auto ptr = Load(_simulator);
41  if (ptr == nullptr) {
42  throw_exception(std::runtime_error(
43  "trying to operate on a destroyed actor; an actor's function "
44  "was called, but the actor is already destroyed."));
45  }
46  if (_episode_id != ptr->GetCurrentEpisodeId()) {
47  throw_exception(std::runtime_error(
48  "trying to access an expired episode; a new episode was started "
