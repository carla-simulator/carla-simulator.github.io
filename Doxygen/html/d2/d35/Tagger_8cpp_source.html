62 void ATagger::TagActor(const AActor &Actor, bool bTagForSemanticSegmentation)
63 {
64 #ifdef CARLA_TAGGER_EXTRA_LOG
65  UE_LOG(LogCarla, Log, TEXT("Actor: %s"), *Actor.GetName());
66 #endif // CARLA_TAGGER_EXTRA_LOG
67 
68  // Iterate static meshes.
69  TArray<UStaticMeshComponent *> StaticMeshComponents;
70  Actor.GetComponents<UStaticMeshComponent>(StaticMeshComponents);
71  for (UStaticMeshComponent *Component : StaticMeshComponents) {
72  const auto Label = GetLabelByPath(Component->GetStaticMesh());
73  SetStencilValue(*Component, Label, bTagForSemanticSegmentation);
74 #ifdef CARLA_TAGGER_EXTRA_LOG
75  UE_LOG(LogCarla, Log, TEXT(" + StaticMeshComponent: %s"), *Component->GetName());
76  UE_LOG(LogCarla, Log, TEXT(" - Label: \"%s\""), *GetTagAsString(Label));
77 #endif // CARLA_TAGGER_EXTRA_LOG
78  }
79 
80  // Iterate skeletal meshes.
81  TArray<USkeletalMeshComponent *> SkeletalMeshComponents;
82  Actor.GetComponents<USkeletalMeshComponent>(SkeletalMeshComponents);
83  for (USkeletalMeshComponent *Component : SkeletalMeshComponents) {
84  const auto Label = GetLabelByPath(Component->GetPhysicsAsset());
85  SetStencilValue(*Component, Label, bTagForSemanticSegmentation);
86 #ifdef CARLA_TAGGER_EXTRA_LOG
87  UE_LOG(LogCarla, Log, TEXT(" + SkeletalMeshComponent: %s"), *Component->GetName());
88  UE_LOG(LogCarla, Log, TEXT(" - Label: \"%s\""), *GetTagAsString(Label));
89 #endif // CARLA_TAGGER_EXTRA_LOG
90  }
91 }
