56 void ATagger::TagActor(const AActor &Actor, bool bTagForSemanticSegmentation)
57 {
58 #ifdef CARLA_TAGGER_EXTRA_LOG
59  UE_LOG(LogCarla, Log, TEXT("Actor: %s"), *Actor.GetName());
60 #endif // CARLA_TAGGER_EXTRA_LOG
61 
62  // Iterate static meshes.
63  TArray<UStaticMeshComponent *> StaticMeshComponents;
64  Actor.GetComponents<UStaticMeshComponent>(StaticMeshComponents);
65  for (UStaticMeshComponent *Component : StaticMeshComponents) {
66  const auto Label = GetLabelByPath(Component->GetStaticMesh());
67  SetStencilValue(*Component, Label, bTagForSemanticSegmentation);
68 #ifdef CARLA_TAGGER_EXTRA_LOG
69  UE_LOG(LogCarla, Log, TEXT(" + StaticMeshComponent: %s"), *Component->GetName());
70  UE_LOG(LogCarla, Log, TEXT(" - Label: \"%s\""), *GetTagAsString(Label));
71 #endif // CARLA_TAGGER_EXTRA_LOG
72  }
73 
74  // Iterate skeletal meshes.
75  TArray<USkeletalMeshComponent *> SkeletalMeshComponents;
76  Actor.GetComponents<USkeletalMeshComponent>(SkeletalMeshComponents);
77  for (USkeletalMeshComponent *Component : SkeletalMeshComponents) {
78  const auto Label = GetLabelByPath(Component->GetPhysicsAsset());
79  SetStencilValue(*Component, Label, bTagForSemanticSegmentation);
80 #ifdef CARLA_TAGGER_EXTRA_LOG
81  UE_LOG(LogCarla, Log, TEXT(" + SkeletalMeshComponent: %s"), *Component->GetName());
82  UE_LOG(LogCarla, Log, TEXT(" - Label: \"%s\""), *GetTagAsString(Label));
83 #endif // CARLA_TAGGER_EXTRA_LOG
84  }
85 }
