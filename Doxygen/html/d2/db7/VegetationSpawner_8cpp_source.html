35 void AVegetationSpawner::CheckForProcedurals()
36 {
37  const UObject* World = GetWorld();
38  TArray<AActor*> Spawners;
39  TArray<FSpawnedFoliage> FrameFound;
40  UGameplayStatics::GetAllActorsOfClass(World, AActor::StaticClass(), Spawners);
41  for (const AActor* Spawner : Spawners)
42  {
43  const TSet<UActorComponent*>& Components = Spawner->GetComponents();
44  for (UActorComponent* Component : Components)
45  {
46  UInstancedStaticMeshComponent* Mesh = Cast<UInstancedStaticMeshComponent>(Component);
47  if (Mesh == nullptr)
48  continue;
49  if (IsInstancedStaticMeshComponentLoaded(Mesh))
50  continue;
51  const int32 NumOfInstances = Mesh->GetInstanceCount();
52  TArray<int32> SpawnerOverlappingInstances;
53  TArray<FTransform> SpawnerOverlappingTransforms;
54  TArray<AActor*> SpawnedActors;
55  SpawnerOverlappingInstances.Init(0, NumOfInstances);
56  SpawnerOverlappingTransforms.SetNum(NumOfInstances);
57  SpawnedActors.Init(nullptr, NumOfInstances);
58 
59  for (int32 i = 0; i < NumOfInstances; ++i)
60  {
61  FTransform Transform;
62  Mesh->GetInstanceTransform(i, Transform, true);
63  SpawnerOverlappingTransforms[i] = Transform;
64  }
65  FSpawnedFoliage Aux;
66  Aux.InUse = true;
67  Aux.Mesh = Mesh;
68  Aux.Indices = SpawnerOverlappingInstances;
69  Aux.Transforms = SpawnerOverlappingTransforms;
70  Aux.SpawnedActors = SpawnedActors;
71  ProceduralInstances.Add(Aux);
72  FrameFound.Add(Aux);
73  }
74  }
75 
76  for (auto& Element : ProceduralInstances)
77  {
78  if (Element.InUse == false)
79  continue;
80  bool Found = false;
81  for (const FSpawnedFoliage& NewElement : FrameFound)
82  {
83  if (Element.Mesh == NewElement.Mesh)
84  {
85  Found = true;
86  break;
87  }
88  }
89  Element.InUse = Found;
90  }
91 }

118 void AVegetationSpawner::UpdateProceduralInstanceCount()
119 {
120  TArray<int32> ProceduralInstanceCount;
121  for (FSpawnedFoliage& Foliage : ProceduralInstances)
122  {
123  if (!Foliage.InUse)
124  continue;
125  const int32 NumOfInstances = Foliage.Mesh->GetInstanceCount();
126  TArray<int32> SpawnerOverlappingInstances;
127  SpawnerOverlappingInstances.Init(0, NumOfInstances);
128 
129  for (int32 i = 0; i < NumOfInstances; ++i)
130  {
131  const FVector Location = Foliage.Transforms[i].GetLocation();
132  for (const AActor* Vehicle : VehiclesInLevel)
133  {
134  const ACarlaWheeledVehicle* CarlaVehicle = Cast<ACarlaWheeledVehicle>(Vehicle);
135  if (CarlaVehicle == nullptr)
136  continue;
137  if (CarlaVehicle->IsInVehicleRange(Location))
138  {
139  SpawnerOverlappingInstances[i]++;
140  }
141  }
142  }
143  UpdateFoliage(Foliage, SpawnerOverlappingInstances);
144  }

147 void AVegetationSpawner::UpdateFoliage(FSpawnedFoliage& Foliage, TArray<int32>& VehiclesDetection)
148 {
149  auto FoliageIterator = Foliage.Indices.CreateIterator();
150  auto VehicleDetectionIterator = VehiclesDetection.CreateConstIterator();
151  for (int32 i = 0; FoliageIterator && VehicleDetectionIterator;
152  ++i, ++FoliageIterator, ++VehicleDetectionIterator)
153  {
154  if (*FoliageIterator > 0 && *VehicleDetectionIterator == 0)
155  {
156  *FoliageIterator = 0;
157  if (Foliage.SpawnedActors[i])
158  {
159  Foliage.SpawnedActors[i]->Destroy();
160  Foliage.SpawnedActors[i] = nullptr;
161  }
162  Foliage.Mesh->UpdateInstanceTransform(i, Foliage.Transforms[i], true, true, true);
163  }
164  else if (*FoliageIterator == 0 && *VehicleDetectionIterator > 0)
165  {
166  const FString Path = Foliage.Mesh->GetStaticMesh()->GetPathName();
167  if (!IsFoliageTypeEnabled(Path))
168  continue;
169  const FFoliageBlueprintCache FullClassName = GetClassFromPath(Path);
170  if (FullClassName.SpawnedClass == nullptr)
171  continue;
172  *FoliageIterator = *VehicleDetectionIterator;
173  Foliage.Mesh->UpdateInstanceTransform(i, {}, true, true, true);
174  Foliage.SpawnedActors[i] = SpawnFoliage(FullClassName, Foliage.Transforms[i]);
175  }
176  }
177 }

179  FFoliageBlueprintCache AVegetationSpawner::GetClassFromPath(const FString& Path)
180  {
181  FFoliageBlueprintCache Result = FindInCache(Path);
182  if (!Result.Path.IsEmpty())
183  return Result;
184 
185  Result.Path = Path;
186 
187  TArray< FString > ParsedString;
188  Path.ParseIntoArray(ParsedString, TEXT("/"), false);
189  int Position = ParsedString.Num() - 1;
190  const FString Version = ParsedString[Position];
191  --Position;
192  const FString Folder = ParsedString[Position];
193  ++Position;
194  const FString FullVersion = GetVersionFromFString(Version);
195  if (FullVersion.IsEmpty())
196  {
197  Result.BPClassName = Path;
198  FoliageCache.Add(Path, Result);
199  return Result;
200  }
201  FString ClassPath = "BP_" + Folder + FullVersion;
202  FString FullClassPath = "Blueprint'";
203  for (int i = 0; i < Position; ++i)
204  {
205  FullClassPath += ParsedString[i];
206  FullClassPath += '/';
207  }
208  FullClassPath += ClassPath;
209  FullClassPath += ".";
210  FullClassPath += ClassPath;
211  FullClassPath += "'";
212 
213  UObject* LoadedObject = StaticLoadObject(UObject::StaticClass(), nullptr, *FullClassPath);
214  UBlueprint* CastedBlueprint = Cast<UBlueprint>(LoadedObject);
215 
216  Result.GetBPName();
217 
218  if (CastedBlueprint && CastedBlueprint->GeneratedClass->IsChildOf(AActor::StaticClass()))
219  {
220  Result.SpawnedClass = *CastedBlueprint->GeneratedClass;
221  }
222 
223  FoliageCache.Add(Path, Result);
224 
225  return Result;
226  }

272 FString AVegetationSpawner::GetVersionFromFString(const FString& string)
273 {
274  auto IsDigit = [](TCHAR charToTest) {
275  if (charToTest == TCHAR('0')) return true;
276  if (charToTest == TCHAR('1')) return true;
277  if (charToTest == TCHAR('2')) return true;
278  if (charToTest == TCHAR('3')) return true;
279  if (charToTest == TCHAR('4')) return true;
280  if (charToTest == TCHAR('5')) return true;
281  if (charToTest == TCHAR('6')) return true;
282  if (charToTest == TCHAR('7')) return true;
283  if (charToTest == TCHAR('8')) return true;
284  if (charToTest == TCHAR('9')) return true;
285  return false;
286  };
287  int index = string.Find(TEXT("_v"));
288  if (index != -1)
289  {
290  index += 2;
291  FString Version = "_v";
292  while(IsDigit(string[index]))
293  {
294  Version += string[index];
295  ++index;
296  if (index == string.Len())
297  return Version;
298  }
299  return Version;
300  }
301  return FString();
302 }
