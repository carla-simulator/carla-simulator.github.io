34  static void ValidateVersions(Client &client) {
35  const auto vc = client.GetClientVersion();
36  const auto vs = client.GetServerVersion();
37  if (vc != vs) {
38  log_warning(
39  "Version mismatch detected: You are trying to connect to a simulator",
40  "that might be incompatible with this API");
41  log_warning("Client API version =", vc);
42  log_warning("Simulator API version =", vs);
43  }
44  }

70  EpisodeProxy Simulator::LoadEpisode(std::string map_name) {
71  const auto id = GetCurrentEpisode().GetId();
72  _client.LoadEpisode(std::move(map_name));
73  size_t number_of_attempts = _client.GetTimeout().milliseconds() / 10u;
74  for (auto i = 0u; i < number_of_attempts; ++i) {
75  using namespace std::literals::chrono_literals;
76  _episode->WaitForState(10ms);
77  auto episode = GetCurrentEpisode();
78  if (episode.GetId() != id) {
79  return episode;
80  }
81  }
82  throw_exception(std::runtime_error("failed to connect to newly created map"));
83  }

89  EpisodeProxy Simulator::GetCurrentEpisode() {
90  if (_episode == nullptr) {
91  ValidateVersions(_client);
92  _episode = std::make_shared<Episode>(_client);
93  _episode->Listen();
94  if (!GetEpisodeSettings().synchronous_mode) {
95  WaitForTick(_client.GetTimeout());
96  }
97  }
98  return EpisodeProxy{shared_from_this()};
99  }

109  WorldSnapshot Simulator::WaitForTick(time_duration timeout) {
110  DEBUG_ASSERT(_episode != nullptr);
111  auto result = _episode->WaitForState(timeout);
112  if (!result.has_value()) {
113  throw_exception(TimeoutException(_client.GetEndpoint(), timeout));
114  }

140  if (settings.synchronous_mode && !settings.fixed_delta_seconds) {

189  SharedPtr<Actor> Simulator::SpawnActor(
190  const ActorBlueprint &blueprint,
191  const geom::Transform &transform,
192  Actor *parent,
193  rpc::AttachmentType attachment_type,
194  GarbageCollectionPolicy gc) {
195  rpc::Actor actor;

197  actor = _client.SpawnActorWithParent(
198  blueprint.MakeActorDescription(),
199  transform,
200  parent->GetId(),
201  attachment_type);
202  } else {
203  actor = _client.SpawnActor(
204  blueprint.MakeActorDescription(),
205  transform);
206  }
207  DEBUG_ASSERT(_episode != nullptr);
208  _episode->RegisterActor(actor);
209  const auto gca = (gc == GarbageCollectionPolicy::Inherit ? _gc_policy : gc);
210  auto result = ActorFactory::MakeActor(GetCurrentEpisode(), actor, gca);
211  log_debug(
212  result->GetDisplayId(),
213  "created",
214  gca == GarbageCollectionPolicy::Enabled ? "with" : "without",
215  "garbage collection");
216  return result;
217  }

219  bool Simulator::DestroyActor(Actor &actor) {
220  bool success = true;
221  success = _client.DestroyActor(actor.GetId());
222  if (success) {
223  // Remove it's persistent state so it cannot access the client anymore.
224  actor.GetEpisode().Clear();
225  log_debug(actor.GetDisplayId(), "destroyed.");
226  } else {
227  log_debug("failed to destroy", actor.GetDisplayId());
228  }
229  return success;
