79 TArray<AStaticMeshActor *> UPrepareAssetsForCookingCommandlet::SpawnMeshesToWorld(
80  const TArray<FString> &AssetsPaths,
81  bool bUseCarlaMaterials)
82 {
83  TArray<AStaticMeshActor *> SpawnedMeshes;
84 
85  // Load assets specified in AssetsPaths by using an object library
86  // for building map world
87  AssetsObjectLibrary = UObjectLibrary::CreateLibrary(UStaticMesh::StaticClass(), false, GIsEditor);
88  AssetsObjectLibrary->AddToRoot();
89 
90  AssetsObjectLibrary->LoadAssetDataFromPaths(AssetsPaths);
91  AssetsObjectLibrary->LoadAssetsFromAssetData();
92  MapContents.Empty();
93  AssetsObjectLibrary->GetAssetDataList(MapContents);
94 
95  // Create default Transform for all assets to spawn
96  const FTransform zeroTransform = FTransform();
97 
98  UStaticMesh *MeshAsset;
99  AStaticMeshActor *MeshActor;
100 
101  for (auto MapAsset : MapContents)
102  {
103  // Spawn Static Mesh
104  MeshAsset = Cast<UStaticMesh>(MapAsset.GetAsset());
105  if (MeshAsset)
106  {
107  MeshActor = World->SpawnActor<AStaticMeshActor>(AStaticMeshActor::StaticClass(), zeroTransform);
108  UStaticMeshComponent *MeshComponent = MeshActor->GetStaticMeshComponent();
109  MeshComponent->SetStaticMesh(CastChecked<UStaticMesh>(MeshAsset));
110 
111  // Rename asset
112  FString AssetName;
113  MapAsset.AssetName.ToString(AssetName);
114  // Remove the prefix with the FBX name
115  int32 FindIndex = AssetName.Find("_", ESearchCase::IgnoreCase, ESearchDir::FromStart, 0);
116  if(FindIndex >= 0) {
117  AssetName.RemoveAt(0, FindIndex + 1, true);
118  }
119  MeshActor->SetActorLabel(AssetName, true);
120 
121  // set complex collision as simple in asset
122  UBodySetup *BodySetup = MeshAsset->BodySetup;
123  if (BodySetup)
124  {
125  BodySetup->CollisionTraceFlag = CTF_UseComplexAsSimple;
126  MeshAsset->MarkPackageDirty();
127  }
128 
129  // rotate all meshes 180 degrees to fit with OpenDRIVE info
130  // (seems that new version of RoadRunner is doing this)
131  // MeshActor->SetActorRotation(FRotator(0.0f, 180.0f, 0.0f));
132 
133  SpawnedMeshes.Add(MeshActor);
134 
135  if (bUseCarlaMaterials)
136  {
137  // Set Carla Materials depending on RoadRunner's Semantic Segmentation
138  // tag
139  if (AssetName.Contains(SSTags::R_MARKING))
140  {
141  MeshActor->GetStaticMeshComponent()->SetMaterial(0, MarkingNodeMaterial);
142  MeshActor->GetStaticMeshComponent()->SetMaterial(1, MarkingNodeMaterialAux);
143  }
144  else if (AssetName.Contains(SSTags::R_ROAD))
145  {
146  MeshActor->GetStaticMeshComponent()->SetMaterial(0, RoadNodeMaterial);
147  }
148  else if (AssetName.Contains(SSTags::R_TERRAIN))
149  {
150  MeshActor->GetStaticMeshComponent()->SetMaterial(0, TerrainNodeMaterial);
151  }
152  }
153  }
154  }
155 
156  // Clear loaded assets in library
157  AssetsObjectLibrary->ClearLoaded();
158 
159  // Mark package dirty
160  World->MarkPackageDirty();
161 
162  return SpawnedMeshes;
163 }

178 bool UPrepareAssetsForCookingCommandlet::SaveWorld(
179  FAssetData &AssetData,
180  const FString &PackageName,
181  const FString &DestPath,
182  const FString &WorldName)
183 {
184  // Create Package to save
185  UPackage *Package = AssetData.GetPackage();
186  Package->SetFolderName(*DestPath);
187  Package->FullyLoad();
188  Package->MarkPackageDirty();
189  FAssetRegistryModule::AssetCreated(World);
190 
191  // Renaming map
192  World->Rename(*WorldName, World->GetOuter());
193  FString PackagePath = DestPath + "/" + WorldName;
194  FAssetRegistryModule::AssetRenamed(World, *PackagePath);
195  World->MarkPackageDirty();
196  World->GetOuter()->MarkPackageDirty();
197 
198  // Check if OpenDrive file exists
199  FString PathXODR = FPaths::ProjectContentDir() + PackageName + TEXT("/Maps/") + WorldName + TEXT(
200  "/OpenDrive/") + WorldName + TEXT(".xodr");
201 
202  bool bPackageSaved = false;
203  if (FPaths::FileExists(PathXODR))
204  {
205  // We need to spawn OpenDrive assets before saving the map
206  AOpenDriveActor *OpenWorldActor =
207  CastChecked<AOpenDriveActor>(World->SpawnActor(AOpenDriveActor::StaticClass(),
208  new FVector(), NULL));
209 
210  OpenWorldActor->BuildRoutes(WorldName);
211  OpenWorldActor->AddSpawners();
212 
213  SavePackage(PackagePath, Package);
214 
215  // We need to destroy OpenDrive assets once saved the map
216  OpenWorldActor->RemoveRoutes();
217  OpenWorldActor->RemoveSpawners();
218  OpenWorldActor->Destroy();
219  }
220  else
221  {
222  SavePackage(PackagePath, Package);
223  }
224 
225  #if WITH_EDITOR
226  UEditorLoadingAndSavingUtils::SaveMap(World, PackagePath);
227  #endif
228 
229  return bPackageSaved;
230 }

248 FAssetsPaths UPrepareAssetsForCookingCommandlet::GetAssetsPathFromPackage(const FString &PackageName) const
249 {
250  FString PackageJsonFilePath = GetFirstPackagePath(PackageName);
251 
252  FAssetsPaths AssetsPaths;
253 
254  // Get All Maps Path
255  FString MapsFileJsonContent;
256  if (FFileHelper::LoadFileToString(MapsFileJsonContent, *PackageJsonFilePath))
257  {
258  TSharedPtr<FJsonObject> JsonParsed;
259  TSharedRef<TJsonReader<TCHAR>> JsonReader = TJsonReaderFactory<TCHAR>::Create(MapsFileJsonContent);
260  if (FJsonSerializer::Deserialize(JsonReader, JsonParsed))
261  {
262  // Add Maps Path
263  auto MapsJsonArray = JsonParsed->GetArrayField(TEXT("maps"));
264 
265  for (auto &MapJsonValue : MapsJsonArray)
266  {
267  TSharedPtr<FJsonObject> MapJsonObject = MapJsonValue->AsObject();
268 
269  FMapData MapData;
270  MapData.Name = MapJsonObject->GetStringField(TEXT("name"));
271  MapData.Path = MapJsonObject->GetStringField(TEXT("path"));
272  MapData.bUseCarlaMapMaterials = MapJsonObject->GetBoolField(TEXT("use_carla_materials"));
273 
274  AssetsPaths.MapsPaths.Add(std::move(MapData));
275  }
276 
277  // Add Props Path
278  auto PropJsonArray = JsonParsed->GetArrayField(TEXT("props"));
279 
280  for (auto &PropJsonValue : PropJsonArray)
281  {
282  TSharedPtr<FJsonObject> PropJsonObject = PropJsonValue->AsObject();
283 
284  FString PropAssetPath = PropJsonObject->GetStringField(TEXT("path"));
285 
286  AssetsPaths.PropsPaths.Add(std::move(PropAssetPath));
287  }
288  }
289  }
290  return AssetsPaths;
291 }

367 void UPrepareAssetsForCookingCommandlet::PrepareMapsForCooking(
368  const FString &PackageName,
369  const TArray<FMapData> &MapsPaths)
370 {
371  // Load World
372  FAssetData AssetData;
373  LoadWorld(AssetData);
374  World = CastChecked<UWorld>(AssetData.GetAsset());
375 
376  FString BasePath = TEXT("/Game/") + PackageName + TEXT("/Static/");
377 
378  for (const auto &Map : MapsPaths)
379  {
380  FString MapPath = TEXT("/") + Map.Name;
381 
382  FString DefaultPath = TEXT("/Game/") + PackageName + TEXT("/Maps/") + Map.Name;
383  FString RoadsPath = BasePath + SSTags::ROAD + MapPath;
384  FString RoadLinesPath = BasePath + SSTags::ROADLINES + MapPath;
385  FString TerrainPath = BasePath + SSTags::TERRAIN + MapPath;
386 
387  // Spawn assets located in semantic segmentation fodlers
388  TArray<FString> DataPath = {DefaultPath, RoadsPath, RoadLinesPath, TerrainPath};
389 
390  TArray<AStaticMeshActor *> SpawnedActors = SpawnMeshesToWorld(DataPath, Map.bUseCarlaMapMaterials);
391 
392  // Save the World in specified path
393  SaveWorld(AssetData, PackageName, Map.Path, Map.Name);
394 
395  // Remove spawned actors from world to keep equal as BaseMap
396  DestroySpawnedActorsInWorld(SpawnedActors);
397  }
398 }

429 int32 UPrepareAssetsForCookingCommandlet::Main(const FString &Params)
430 {
431  FPackageParams PackageParams = ParseParams(Params);
432 
433  // Get Props and Maps Path

435 
436  if (PackageParams.bOnlyPrepareMaps)

438  PrepareMapsForCooking(PackageParams.Name, AssetsPaths.MapsPaths);

443 
444  if (AssetsPaths.PropsPaths.Num() > 0)
445  {
446  FString MapName("PropsMap");
447  PropsMapPath = TEXT("/Game/") + PackageParams.Name + TEXT("/Maps/") + MapName;
448  PreparePropsForCooking(PackageParams.Name, AssetsPaths.PropsPaths, MapName);
449  }
450 
451  // Save Map Path File for further use
452  GenerateMapPathsFile(AssetsPaths, PropsMapPath);
453 
454  // Saves Package path for further use
455  GeneratePackagePathFile(PackageParams.Name);
456  }
457 
458  #if WITH_EDITOR
459  UEditorLoadingAndSavingUtils::SaveDirtyPackages(true, true);
460  #endif
461 
462  return 0;
