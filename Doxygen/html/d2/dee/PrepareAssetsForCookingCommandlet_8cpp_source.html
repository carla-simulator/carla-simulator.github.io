82 TArray<AStaticMeshActor *> UPrepareAssetsForCookingCommandlet::SpawnMeshesToWorld(
83  const TArray<FString> &AssetsPaths,
84  bool bUseCarlaMaterials)
85 {
86  TArray<AStaticMeshActor *> SpawnedMeshes;
87 
88  // Load assets specified in AssetsPaths by using an object library
89  // for building map world
90  AssetsObjectLibrary = UObjectLibrary::CreateLibrary(UStaticMesh::StaticClass(), false, GIsEditor);
91  AssetsObjectLibrary->AddToRoot();
92 
93  AssetsObjectLibrary->LoadAssetDataFromPaths(AssetsPaths);
94  AssetsObjectLibrary->LoadAssetsFromAssetData();
95  MapContents.Empty();
96  AssetsObjectLibrary->GetAssetDataList(MapContents);
97 
98  // Create default Transform for all assets to spawn
99  const FTransform zeroTransform = FTransform();
100 
101  UStaticMesh *MeshAsset;
102  AStaticMeshActor *MeshActor;
103 
104  for (auto MapAsset : MapContents)
105  {
106  // Spawn Static Mesh
107  MeshAsset = Cast<UStaticMesh>(MapAsset.GetAsset());
108  if (MeshAsset)
109  {
110  MeshActor = World->SpawnActor<AStaticMeshActor>(AStaticMeshActor::StaticClass(), zeroTransform);
111  UStaticMeshComponent *MeshComponent = MeshActor->GetStaticMeshComponent();
112  MeshComponent->SetStaticMesh(CastChecked<UStaticMesh>(MeshAsset));
113 
114  // Rename asset
115  FString AssetName;
116  MapAsset.AssetName.ToString(AssetName);
117  // Remove the prefix with the FBX name
118  int32 FindIndex = AssetName.Find("_", ESearchCase::IgnoreCase, ESearchDir::FromStart, 0);
119  if(FindIndex >= 0) {
120  AssetName.RemoveAt(0, FindIndex + 1, true);
121  }
122  MeshActor->SetActorLabel(AssetName, true);
123 
124  // set complex collision as simple in asset
125  UBodySetup *BodySetup = MeshAsset->BodySetup;
126  if (BodySetup)
127  {
128  BodySetup->CollisionTraceFlag = CTF_UseComplexAsSimple;
129  MeshAsset->MarkPackageDirty();
130  }
131 
132  SpawnedMeshes.Add(MeshActor);
133 
134  if (bUseCarlaMaterials)
135  {
136  // Set Carla Materials depending on RoadRunner's Semantic Segmentation
137  // tag
138  if (AssetName.Contains(SSTags::R_MARKING))
139  {
140  MeshActor->GetStaticMeshComponent()->SetMaterial(0, MarkingNodeMaterial);
141  MeshActor->GetStaticMeshComponent()->SetMaterial(1, MarkingNodeMaterialAux);
142  }
143  else if (AssetName.Contains(SSTags::R_ROAD))
144  {
145  MeshActor->GetStaticMeshComponent()->SetMaterial(0, RoadNodeMaterial);
146  }
147  else if (AssetName.Contains(SSTags::R_TERRAIN))
148  {
149  MeshActor->GetStaticMeshComponent()->SetMaterial(0, TerrainNodeMaterial);
150  }
151  else if (AssetName.Contains(SSTags::R_SIDEWALK))
152  {
153  MeshActor->GetStaticMeshComponent()->SetMaterial(0, SidewalkNodeMaterial);
154  }
155  }
156  }
157  }
158 
159  // Clear loaded assets in library
160  AssetsObjectLibrary->ClearLoaded();
161 
162  // Mark package dirty
163  World->MarkPackageDirty();
164 
165  return SpawnedMeshes;
166 }

181 bool UPrepareAssetsForCookingCommandlet::SaveWorld(
182  FAssetData &AssetData,
183  const FString &PackageName,
184  const FString &DestPath,
185  const FString &WorldName)
186 {
187  // Create Package to save
188  UPackage *Package = AssetData.GetPackage();
189  Package->SetFolderName(*DestPath);
190  Package->FullyLoad();
191  Package->MarkPackageDirty();
192  FAssetRegistryModule::AssetCreated(World);
193 
194  // Renaming map
195  World->Rename(*WorldName, World->GetOuter());
196  FString PackagePath = DestPath + "/" + WorldName;
197  FAssetRegistryModule::AssetRenamed(World, *PackagePath);
198  World->MarkPackageDirty();
199  World->GetOuter()->MarkPackageDirty();
200 
201  // Check if OpenDrive file exists
202  FString PathXODR = FPaths::ProjectContentDir() + PackageName + TEXT("/Maps/") + WorldName + TEXT(
203  "/OpenDrive/") + WorldName + TEXT(".xodr");
204 
205  bool bPackageSaved = false;
206  if (FPaths::FileExists(PathXODR))
207  {
208  // We need to spawn OpenDrive assets before saving the map
209  AOpenDriveActor *OpenWorldActor =
210  CastChecked<AOpenDriveActor>(World->SpawnActor(AOpenDriveActor::StaticClass(),
211  new FVector(), NULL));
212 
213  OpenWorldActor->BuildRoutes(WorldName);
214  OpenWorldActor->AddSpawners();
215 
216  SavePackage(PackagePath, Package);
217 
218  // We need to destroy OpenDrive assets once saved the map
219  OpenWorldActor->RemoveRoutes();
220  OpenWorldActor->RemoveSpawners();
221  OpenWorldActor->Destroy();
222  }
223  else
224  {
225  SavePackage(PackagePath, Package);
226  }
227 
228  return bPackageSaved;
229 }

247 FAssetsPaths UPrepareAssetsForCookingCommandlet::GetAssetsPathFromPackage(const FString &PackageName) const
248 {
249  FString PackageJsonFilePath = GetFirstPackagePath(PackageName);
250 
251  FAssetsPaths AssetsPaths;
252 
253  // Get All Maps Path
254  FString MapsFileJsonContent;
255  if (FFileHelper::LoadFileToString(MapsFileJsonContent, *PackageJsonFilePath))
256  {
257  TSharedPtr<FJsonObject> JsonParsed;
258  TSharedRef<TJsonReader<TCHAR>> JsonReader = TJsonReaderFactory<TCHAR>::Create(MapsFileJsonContent);
259  if (FJsonSerializer::Deserialize(JsonReader, JsonParsed))
260  {
261  // Add Maps Path
262  auto MapsJsonArray = JsonParsed->GetArrayField(TEXT("maps"));
263 
264  for (auto &MapJsonValue : MapsJsonArray)
265  {
266  TSharedPtr<FJsonObject> MapJsonObject = MapJsonValue->AsObject();
267 
268  FMapData MapData;
269  MapData.Name = MapJsonObject->GetStringField(TEXT("name"));
270  MapData.Path = MapJsonObject->GetStringField(TEXT("path"));
271  MapData.bUseCarlaMapMaterials = MapJsonObject->GetBoolField(TEXT("use_carla_materials"));
272 
273  AssetsPaths.MapsPaths.Add(std::move(MapData));
274  }
275 
276  // Add Props Path
277  auto PropJsonArray = JsonParsed->GetArrayField(TEXT("props"));
278 
279  for (auto &PropJsonValue : PropJsonArray)
280  {
281  TSharedPtr<FJsonObject> PropJsonObject = PropJsonValue->AsObject();
282 
283  FString PropAssetPath = PropJsonObject->GetStringField(TEXT("path"));
284 
285  AssetsPaths.PropsPaths.Add(std::move(PropAssetPath));
286  }
287  }
288  }
289  return AssetsPaths;
290 }

367 void UPrepareAssetsForCookingCommandlet::PrepareMapsForCooking(
368  const FString &PackageName,
369  const TArray<FMapData> &MapsPaths)
370 {
371  // Load World
372  FAssetData AssetData;
373  LoadWorld(AssetData);
374  World = CastChecked<UWorld>(AssetData.GetAsset());
375 
376  FString BasePath = TEXT("/Game/") + PackageName + TEXT("/Static/");
377 
378  for (const auto &Map : MapsPaths)
379  {
380  FString MapPath = TEXT("/") + Map.Name;
381 
382  FString DefaultPath = TEXT("/Game/") + PackageName + TEXT("/Maps/") + Map.Name;
383  FString RoadsPath = BasePath + SSTags::ROAD + MapPath;
384  FString RoadLinesPath = BasePath + SSTags::ROADLINES + MapPath;
385  FString TerrainPath = BasePath + SSTags::TERRAIN + MapPath;
386  FString SidewalkPath = BasePath + SSTags::SIDEWALK + MapPath;
387 
388  // Spawn assets located in semantic segmentation fodlers
389  TArray<FString> DataPath = {DefaultPath, RoadsPath, RoadLinesPath, TerrainPath, SidewalkPath};
390 
391  TArray<AStaticMeshActor *> SpawnedActors = SpawnMeshesToWorld(DataPath, Map.bUseCarlaMapMaterials);
392 
393  // Save the World in specified path
394  SaveWorld(AssetData, PackageName, Map.Path, Map.Name);
395 
396  // Remove spawned actors from world to keep equal as BaseMap
397  DestroySpawnedActorsInWorld(SpawnedActors);
398  }

422 
423  FString MapName("PropsMap");
424  SaveWorld(AssetData, PackageName, MapDestPath, MapName);
425 
426  DestroySpawnedActorsInWorld(SpawnedActors);

430 int32 UPrepareAssetsForCookingCommandlet::Main(const FString &Params)
431 {
432  FPackageParams PackageParams = ParseParams(Params);
433 
434  // Get Props and Maps Path

436

444 
445  if (AssetsPaths.PropsPaths.Num() > 0)

447  FString MapName("PropsMap");
448  PropsMapPath = TEXT("/Game/") + PackageParams.Name + TEXT("/Maps/") + MapName;
449  PreparePropsForCooking(PackageParams.Name, AssetsPaths.PropsPaths, MapName);
450  }
451 
452  // Save Map Path File for further use
453  GenerateMapPathsFile(AssetsPaths, PropsMapPath);
454 
455  // Saves Package path for further use
456  GeneratePackagePathFile(PackageParams.Name);
457  }
458 
459  #if WITH_EDITOR
460  UEditorLoadingAndSavingUtils::SaveDirtyPackages(true, true);
461  #endif
462 
463  return 0;
