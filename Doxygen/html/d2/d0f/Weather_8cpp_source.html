30  ActiveBlendables.Add(MakeTuple(PrecipitationPostProcessMaterial, Weather.Precipitation / 100.0f));
31  else
32  ActiveBlendables.Remove(PrecipitationPostProcessMaterial);

34  if (Weather.DustStorm > 0.0f)
35  ActiveBlendables.Add(MakeTuple(DustStormPostProcessMaterial, Weather.DustStorm / 100.0f));
36  else
37  ActiveBlendables.Remove(DustStormPostProcessMaterial);
38 
39  TArray<AActor*> SensorActors;
40  UGameplayStatics::GetAllActorsOfClass(GetWorld(), ASceneCaptureCamera::StaticClass(), SensorActors);
41  for (AActor* SensorActor : SensorActors)

43  ASceneCaptureCamera* Sensor = Cast<ASceneCaptureCamera>(SensorActor);
44  for (auto& ActiveBlendable : ActiveBlendables)
