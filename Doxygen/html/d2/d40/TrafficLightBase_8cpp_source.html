52 void ATrafficLightBase::Tick(float DeltaSeconds)
53 {
54  Super::Tick(DeltaSeconds);
55 
56  if (TimeIsFrozen)
57  {
58  return;
59  }
60 
61  ElapsedTime += DeltaSeconds;
62 
63  float ChangeTime;
64 
65  switch (State)
66  {
67  case ETrafficLightState::Red:
68  ChangeTime = RedTime;
69  break;
70  case ETrafficLightState::Yellow:
71  ChangeTime = YellowTime;
72  break;
73  case ETrafficLightState::Green:
74  ChangeTime = GreenTime;
75  break;
76  default:
77  UE_LOG(LogCarla, Error, TEXT("Invalid traffic light state!"));
78  SetTrafficLightState(ETrafficLightState::Red);
79  return;
80  }
81 
82  if (ElapsedTime > ChangeTime)
83  {
84  // Freeze if part of a group and about to turn red
85  if (GroupTrafficLights.Num() > 0 && State == ETrafficLightState::Yellow)
86  {
87  SetTimeIsFrozen(true);
88  }
89  SwitchTrafficLightState();
90  }
91 }

106 void ATrafficLightBase::SetTrafficLightState(const ETrafficLightState InState)
107 {
108  UTrafficLightComponent* TrafficLightComponent =
109  Cast<UTrafficLightComponent>(FindComponentByClass<UTrafficLightComponent>());
110 
111  if(TrafficLightComponent) {
112  TrafficLightComponent->SetLightState(InState);
113  } else {
114  ElapsedTime = 0.0f;
115  State = InState;
116  SetTrafficSignState(ToTrafficSignState(State));
117  for (auto Controller : Vehicles)
118  {
119  if (Controller != nullptr)
120  {
121  Controller->SetTrafficLightState(State);
122  if (State == ETrafficLightState::Green)
123  {
124  Controller->SetTrafficLight(nullptr);
125  }
126  }
127  }
128  if (State == ETrafficLightState::Green)
129  {
130  Vehicles.Empty();
131  }
132  }
133  OnTrafficLightStateChanged(State);
134 }
