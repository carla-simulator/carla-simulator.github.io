52 void ATrafficLightBase::Tick(float DeltaSeconds)
53 {
54  if (TimeIsFrozen)
55  {
56  return;
57  }
58 
59  ElapsedTime += DeltaSeconds;
60 
61  float ChangeTime;
62 
63  switch (State)
64  {
65  case ETrafficLightState::Red:
66  ChangeTime = RedTime;
67  break;
68  case ETrafficLightState::Yellow:
69  ChangeTime = YellowTime;
70  break;
71  case ETrafficLightState::Green:
72  ChangeTime = GreenTime;
73  break;
74  default:
75  UE_LOG(LogCarla, Error, TEXT("Invalid traffic light state!"));
76  SetTrafficLightState(ETrafficLightState::Red);
77  return;
78  }
79 
80  if (ElapsedTime > ChangeTime)
81  {
82  // Freeze if part of a group and about to turn red
83  if (GroupTrafficLights.Num() > 0 && State == ETrafficLightState::Yellow)
84  {
85  SetTimeIsFrozen(true);
86  }
87  SwitchTrafficLightState();
88  }
89 }

104 void ATrafficLightBase::SetTrafficLightState(const ETrafficLightState InState)
105 {
106  ElapsedTime = 0.0f;
107  State = InState;
108  SetTrafficSignState(ToTrafficSignState(State));
109  for (auto Controller : Vehicles)
110  {
111  if (Controller != nullptr)
112  {
113  Controller->SetTrafficLightState(State);
114  if (State == ETrafficLightState::Green)
115  {
116  Controller->SetTrafficLight(nullptr);
117  }
118  }
119  }
120  if (State == ETrafficLightState::Green)
121  {
122  Vehicles.Empty();
123  }
124  OnTrafficLightStateChanged(State);
125 }

127 void ATrafficLightBase::SwitchTrafficLightState()
128 {
129  switch (State)
130  {
131  case ETrafficLightState::Red:
132  SetTrafficLightState(ETrafficLightState::Green);
133  break;
134  case ETrafficLightState::Yellow:
135  SetTrafficLightState(ETrafficLightState::Red);
136  break;
137  case ETrafficLightState::Green:
138  SetTrafficLightState(ETrafficLightState::Yellow);
139  break;
140  default:
141  UE_LOG(LogCarla, Error, TEXT("Invalid traffic light state!"));
142  SetTrafficLightState(ETrafficLightState::Red);
143  break;
144  }
145 }
