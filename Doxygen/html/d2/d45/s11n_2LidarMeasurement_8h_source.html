18  /// Helper class to store and serialize the data generated by a Lidar.
19  ///
20  /// The header of a Lidar measurement consists of an array of uint32_t's in
21  /// the following layout
22  ///
23  /// {
24  /// Horizontal angle (float),
25  /// Channel count,
26  /// Point count of channel 0,
27  /// ...
28  /// Point count of channel n,
29  /// }
30  ///
31  /// The points are stored in an array of floats
32  ///
33  /// {
34  /// X0, Y0, Z0,
35  /// ...
36  /// Xn, Yn, Zn,
37  /// }
38  ///
39  /// @warning WritePoint should be called sequentially in the order in which
40  /// the points are going to be stored, i.e., starting at channel zero and
41  /// increasing steadily.
42  class LidarMeasurement {
43  static_assert(sizeof(float) == sizeof(uint32_t), "Invalid float size");
44 
45  friend class LidarSerializer;
46  friend class LidarHeaderView;
47 
48  enum Index : size_t {
49  HorizontalAngle,
50  ChannelCount,
51  SIZE
52  };
53 
54  public:
55 
56  explicit LidarMeasurement(uint32_t ChannelCount = 0u)
57  : _header(Index::SIZE + ChannelCount, 0u) {
58  _header[Index::ChannelCount] = ChannelCount;
59  }
60 
61  LidarMeasurement &operator=(LidarMeasurement &&) = default;
62 
63  float GetHorizontalAngle() const {
64  return reinterpret_cast<const float &>(_header[Index::HorizontalAngle]);
65  }
66 
67  void SetHorizontalAngle(float angle) {
68  _header[Index::HorizontalAngle] = reinterpret_cast<const uint32_t &>(angle);
69  }
70 
71  uint32_t GetChannelCount() const {
72  return _header[Index::ChannelCount];
73  }
74 
75  void Reset(uint32_t total_point_count) {
76  std::memset(_header.data() + Index::SIZE, 0, sizeof(uint32_t) * GetChannelCount());
77  _points.clear();
78  _points.reserve(3u * total_point_count);
79  }
80 
81  void WritePoint(uint32_t channel, rpc::Location point) {
82  DEBUG_ASSERT(GetChannelCount() > channel);
83  _header[Index::SIZE + channel] += 1u;
84  _points.emplace_back(point.x);
85  _points.emplace_back(point.y);
86  _points.emplace_back(point.z);
87  }
88 
89  private:
90 
91  std::vector<uint32_t> _header;
92 
93  std::vector<float> _points;
94  };
