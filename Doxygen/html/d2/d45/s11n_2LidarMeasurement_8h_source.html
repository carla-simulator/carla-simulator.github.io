18  /// Helper class to store and serialize the data generated by a Lidar.
19  ///
20  /// The header of a Lidar measurement consists of an array of uint32_t's in
21  /// the following layout
22  ///
23  /// {
24  /// Horizontal angle (float),
25  /// Channel count,
26  /// Point count of channel 0,
27  /// ...
28  /// Point count of channel n,
29  /// }
30  ///
31  /// The points are stored in an array of floats
32  ///
33  /// {
34  /// X0, Y0, Z0, I0
35  /// ...
36  /// Xn, Yn, Zn, In
37  /// }
38  ///
39  /// @warning WritePoint should be called sequentially in the order in which
40  /// the points are going to be stored, i.e., starting at channel zero and
41  /// increasing steadily.
42 
43  class LidarDetection {
44  public:
45  geom::Location point;
46  float intensity;
47 
48  LidarDetection() :
49  point(0.0f, 0.0f, 0.0f), intensity{0.0f} { }
50  LidarDetection(float x, float y, float z, float intensity) :
51  point(x, y, z), intensity{intensity} { }
52  LidarDetection(geom::Location p, float intensity) :
53  point(p), intensity{intensity} { }
54  };

56  class LidarMeasurement {

89  void Reset(uint32_t channels, uint32_t channel_point_count) {
90  std::memset(_header.data() + Index::SIZE, 0, sizeof(uint32_t) * GetChannelCount());
91  _points.clear();
92  _points.reserve(SizeLidarDetection * channels * channel_point_count);
93 
94  _aux_points.resize(channels);
95 
96  for (auto& aux : _aux_points) {
97  aux.clear();
98  aux.reserve(channel_point_count);
99  }
100  }
101 
102  void WritePointAsync(uint32_t channel, LidarDetection detection) {
103  DEBUG_ASSERT(GetChannelCount() > channel);
104  _aux_points[channel].emplace_back(detection);
105  }
106 
107  void SaveDetections() {
108  _points.clear();
109 
110  for (auto idxChannel = 0u; idxChannel < GetChannelCount(); ++idxChannel) {
111  _header[Index::SIZE + idxChannel] = static_cast<uint32_t>(_aux_points.size());
112  for (auto& pt : _aux_points[idxChannel]) {
113  _points.emplace_back(pt.point.x);
114  _points.emplace_back(pt.point.y);
115  _points.emplace_back(pt.point.z);
116  _points.emplace_back(pt.intensity);
117  }
118  }
119  }
120 
121  private:
122  std::vector<uint32_t> _header;
123  std::vector<std::vector<LidarDetection>> _aux_points;
124 
125  std::vector<float> _points;
126 
127  };
