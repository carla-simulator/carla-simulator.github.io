16 // create or reuse an actor for replaying
17 std::pair<int, FActorView>CarlaReplayerHelper::TryToCreateReplayerActor(
18  FVector &Location,
19  FVector &Rotation,
20  FActorDescription &ActorDesc,
21  uint32_t DesiredId)
22 {
23  check(Episode != nullptr);
24 
25  FActorView view_empty;
26 
27  // check type of actor we need
28  if (ActorDesc.Id.StartsWith("traffic."))
29  {
30  AActor *Actor = FindTrafficLightAt(Location);
31  if (Actor != nullptr)
32  {
33  // actor found
34  auto view = Episode->GetActorRegistry().Find(Actor);
35  // reuse that actor
36  return std::pair<int, FActorView>(2, view);
37  }
38  else
39  {
40  // actor not found
41  UE_LOG(LogCarla, Log, TEXT("TrafficLight not found"));
42  return std::pair<int, FActorView>(0, view_empty);
43  }
44  }
45  else if (!ActorDesc.Id.StartsWith("sensor."))
46  {
47  // check if an actor of that type already exist with same id
48  if (Episode->GetActorRegistry().Contains(DesiredId))
49  {
50  auto view = Episode->GetActorRegistry().Find(DesiredId);
51  const FActorDescription *desc = &view.GetActorInfo()->Description;
52  if (desc->Id == ActorDesc.Id)
53  {
54  // we don't need to create, actor of same type already exist
55  return std::pair<int, FActorView>(2, view);
56  }
57  }
58  // create the transform
59  FRotator Rot = FRotator::MakeFromEuler(Rotation);
60  FTransform Trans(Rot, FVector(0, 0, 100000), FVector(1, 1, 1));
61  // create as new actor
62  TPair<EActorSpawnResultStatus, FActorView> Result = Episode->SpawnActorWithInfo(Trans, ActorDesc, DesiredId);
63  if (Result.Key == EActorSpawnResultStatus::Success)
64  {
65  // relocate
66  FTransform Trans2(Rot, Location, FVector(1, 1, 1));
67  Result.Value.GetActor()->SetActorTransform(Trans2, false, nullptr, ETeleportType::TeleportPhysics);
68  return std::pair<int, FActorView>(1, Result.Value);
69  }
70  else
71  {
72  UE_LOG(LogCarla, Log, TEXT("Actor could't be created by replayer"));
73  return std::pair<int, FActorView>(0, Result.Value);
74  }
75  }
76  else
77  {
78  // actor ignored
79  return std::pair<int, FActorView>(0, view_empty);
80  }
81 }

83 AActor *CarlaReplayerHelper::FindTrafficLightAt(FVector Location)
84 {
85  check(Episode != nullptr);
86  auto World = Episode->GetWorld();
87  check(World != nullptr);
88 
89  // get its position (truncated as int's)
90  int x = static_cast<int>(Location.X);
91  int y = static_cast<int>(Location.Y);
92  int z = static_cast<int>(Location.Z);
93 
94  // search an "traffic." actor at that position
95  for (TActorIterator<ATrafficSignBase> It(World); It; ++It)
96  {
97  ATrafficSignBase *Actor = *It;
98  check(Actor != nullptr);
99  FVector vec = Actor->GetTransform().GetTranslation();
100  int x2 = static_cast<int>(vec.X);
101  int y2 = static_cast<int>(vec.Y);
102  int z2 = static_cast<int>(vec.Z);
103  if ((x2 == x) && (y2 == y) && (z2 == z))
104  {
105  // actor found
106  return static_cast<AActor *>(Actor);
107  }
108  }
109  // actor not found
110  return nullptr;
111 }

152 // replay event for creating actor
153 std::pair<int, uint32_t> CarlaReplayerHelper::ProcessReplayerEventAdd(
154  FVector Location,
155  FVector Rotation,
156  CarlaRecorderActorDescription Description,
157  uint32_t DesiredId,
158  bool bIgnoreHero)
159 {
160  check(Episode != nullptr);
161  FActorDescription ActorDesc;
162  bool IsHero = false;
163 
164  // prepare actor description
165  ActorDesc.UId = Description.UId;
166  ActorDesc.Id = Description.Id;
167  for (const auto &Item : Description.Attributes)
168  {
169  FActorAttribute Attr;
170  Attr.Type = static_cast<EActorAttributeType>(Item.Type);
171  Attr.Id = Item.Id;
172  Attr.Value = Item.Value;
173  ActorDesc.Variations.Add(Attr.Id, std::move(Attr));
174  // check for hero
175  if (Item.Id == "role_name" && Item.Value == "hero")
176  IsHero = true;
177  }
178 
179  auto result = TryToCreateReplayerActor(Location, Rotation, ActorDesc, DesiredId);
180 
181  if (result.first != 0)
182  {
183  // disable physics and autopilot on vehicles
184  if (result.second.GetActorType() == FActorView::ActorType::Vehicle)
185  {
186  // ignore hero ?
187  if (!(bIgnoreHero && IsHero))
188  {
189  // disable physics
190  SetActorSimulatePhysics(result.second, false);
191  // disable autopilot
192  SetActorAutopilot(result.second, false, false);
193  }
194  else
195  {
196  // reenable physics just in case
197  SetActorSimulatePhysics(result.second, true);
198  }
199  }
200  }
201 
202  return std::make_pair(result.first, result.second.GetActorId());
203 }

238 // reposition actors
239 bool CarlaReplayerHelper::ProcessReplayerPosition(CarlaRecorderPosition Pos1, CarlaRecorderPosition Pos2, double Per, double DeltaTime)
240 {
241  check(Episode != nullptr);
242  AActor *Actor = Episode->GetActorRegistry().Find(Pos1.DatabaseId).GetActor();
243  FVector Location;
244  FRotator Rotation;
245  if (Actor && !Actor->IsPendingKill())
246  {
247  // check to assign first position or interpolate between both
248  if (Per == 0.0)
249  {
250  // assign position 1
251  Location = FVector(Pos1.Location);
252  Rotation = FRotator::MakeFromEuler(Pos1.Rotation);
253  }
254  else
255  {
256  // interpolate positions
257  Location = FMath::Lerp(FVector(Pos1.Location), FVector(Pos2.Location), Per);
258  Rotation = FMath::Lerp(FRotator::MakeFromEuler(Pos1.Rotation), FRotator::MakeFromEuler(Pos2.Rotation), Per);
259  }
260  // set new transform
261  FTransform Trans(Rotation, Location, FVector(1, 1, 1));
262  Actor->SetActorTransform(Trans, false, nullptr, ETeleportType::None);
263  return true;
264  }
265  return false;
266 }

270 {

277 
278  // check

340 // replay finish
341 bool CarlaReplayerHelper::ProcessReplayerFinish(bool bApplyAutopilot, bool bIgnoreHero, std::unordered_map<uint32_t, bool> &IsHero)
342 {
343  // set autopilot and physics to all AI vehicles
344  auto registry = Episode->GetActorRegistry();
345  for (const FActorView &ActorView : registry)
346  {
347  // enable physics only on vehicles
348  switch (ActorView.GetActorType())
349  {
350 
351  // vehicles
352  case FActorView::ActorType::Vehicle:
353  // check for hero
354  if (!(bIgnoreHero && IsHero[ActorView.GetActorId()]))
355  {
356  SetActorSimulatePhysics(ActorView, true);
357  // autopilot
358  if (bApplyAutopilot)
359  SetActorAutopilot(ActorView, true, true);
360  }
361  break;
362 
363  // walkers
364  case FActorView::ActorType::Walker:
365  // stop walker
366  SetActorVelocity(ActorView, FVector(0, 0, 0));
367  SetWalkerSpeed(ActorView.GetActorId(), 0.0f);
368  break;
369  }
370  }
371  return true;
372 }
