16 // create or reuse an actor for replaying
17 std::pair<int, FActorView>CarlaReplayerHelper::TryToCreateReplayerActor(
18  FVector &Location,
19  FVector &Rotation,
20  FActorDescription &ActorDesc,
21  uint32_t DesiredId)
22 {
23  check(Episode != nullptr);
24 
25  FActorView view_empty;
26 
27  // check type of actor we need
28  if (ActorDesc.Id.StartsWith("traffic."))
29  {
30  AActor *Actor = FindTrafficLightAt(Location);
31  if (Actor != nullptr)
32  {
33  // actor found
34  auto view = Episode->GetActorRegistry().Find(Actor);
35  // reuse that actor
36  return std::pair<int, FActorView>(2, view);
37  }
38  else
39  {
40  // actor not found
41  UE_LOG(LogCarla, Log, TEXT("TrafficLight not found"));
42  return std::pair<int, FActorView>(0, view_empty);
43  }
44  }
45  else if (!ActorDesc.Id.StartsWith("sensor."))
46  {
47  // check if an actor of that type already exist with same id
48  if (Episode->GetActorRegistry().Contains(DesiredId))
49  {
50  auto view = Episode->GetActorRegistry().Find(DesiredId);
51  const FActorDescription *desc = &view.GetActorInfo()->Description;
52  if (desc->Id == ActorDesc.Id)
53  {
54  // we don't need to create, actor of same type already exist
55  return std::pair<int, FActorView>(2, view);
56  }
57  }
58  // create the transform
59  FRotator Rot = FRotator::MakeFromEuler(Rotation);
60  FTransform Trans(Rot, FVector(0, 0, 100000), FVector(1, 1, 1));
61  // create as new actor
62  TPair<EActorSpawnResultStatus, FActorView> Result = Episode->SpawnActorWithInfo(Trans, ActorDesc, DesiredId);
63  if (Result.Key == EActorSpawnResultStatus::Success)
64  {
65  // relocate
66  FTransform Trans2(Rot, Location, FVector(1, 1, 1));
67  Result.Value.GetActor()->SetActorTransform(Trans2, false, nullptr, ETeleportType::TeleportPhysics);
68  return std::pair<int, FActorView>(1, Result.Value);
69  }
70  else
71  {
72  UE_LOG(LogCarla, Log, TEXT("Actor could't be created by replayer"));
73  return std::pair<int, FActorView>(0, Result.Value);
74  }
75  }
76  else
77  {
78  // actor ignored
79  return std::pair<int, FActorView>(0, view_empty);
80  }
81 }

83 AActor *CarlaReplayerHelper::FindTrafficLightAt(FVector Location)
84 {
85  check(Episode != nullptr);
86  auto World = Episode->GetWorld();
87  check(World != nullptr);
88 
89  // get its position (truncated as int's)
90  int x = static_cast<int>(Location.X);
91  int y = static_cast<int>(Location.Y);
92  int z = static_cast<int>(Location.Z);
93 
94  // search an "traffic." actor at that position
95  for (TActorIterator<ATrafficSignBase> It(World); It; ++It)
96  {
97  ATrafficSignBase *Actor = *It;
98  check(Actor != nullptr);
99  FVector vec = Actor->GetTransform().GetTranslation();
100  int x2 = static_cast<int>(vec.X);
101  int y2 = static_cast<int>(vec.Y);
102  int z2 = static_cast<int>(vec.Z);
103  if ((x2 == x) && (y2 == y) && (z2 == z))
104  {
105  // actor found
106  return static_cast<AActor *>(Actor);
107  }
108  }
109  // actor not found
110  return nullptr;
111 }

152 // replay event for creating actor
153 std::pair<int, uint32_t> CarlaReplayerHelper::ProcessReplayerEventAdd(
154  FVector Location,
155  FVector Rotation,
156  CarlaRecorderActorDescription Description,
157  uint32_t DesiredId)
158 {
159  check(Episode != nullptr);
160  FActorDescription ActorDesc;
161 
162  // prepare actor description
163  ActorDesc.UId = Description.UId;
164  ActorDesc.Id = Description.Id;
165  for (const auto &Item : Description.Attributes)
166  {
167  FActorAttribute Attr;
168  Attr.Type = static_cast<EActorAttributeType>(Item.Type);
169  Attr.Id = Item.Id;
170  Attr.Value = Item.Value;
171  ActorDesc.Variations.Add(Attr.Id, std::move(Attr));
172  }
173 
174  auto result = TryToCreateReplayerActor(Location, Rotation, ActorDesc, DesiredId);
175 
176  if (result.first != 0)
177  {
178  // disable physics and autopilot on vehicles
179  if (result.second.GetActorType() == FActorView::ActorType::Vehicle)
180  {
181  // disable physics
182  SetActorSimulatePhysics(result.second, false);
183  // disable autopilot
184  SetActorAutopilot(result.second, false, false);
185  }
186  }
187 
188  return std::make_pair(result.first, result.second.GetActorId());
189 }

224 // reposition actors
225 bool CarlaReplayerHelper::ProcessReplayerPosition(CarlaRecorderPosition Pos1, CarlaRecorderPosition Pos2, double Per, double DeltaTime)
226 {
227  check(Episode != nullptr);
228  AActor *Actor = Episode->GetActorRegistry().Find(Pos1.DatabaseId).GetActor();
229  FVector Location;
230  FRotator Rotation;
231  if (Actor && !Actor->IsPendingKill())
232  {
233  // check to assign first position or interpolate between both
234  if (Per == 0.0)
235  {
236  // assign position 1
237  Location = FVector(Pos1.Location);
238  Rotation = FRotator::MakeFromEuler(Pos1.Rotation);
239  }
240  else
241  {
242  // interpolate positions
243  Location = FMath::Lerp(FVector(Pos1.Location), FVector(Pos2.Location), Per);
244  Rotation = FMath::Lerp(FRotator::MakeFromEuler(Pos1.Rotation), FRotator::MakeFromEuler(Pos2.Rotation), Per);
245  }
246  // set new transform
247  FTransform Trans(Rotation, Location, FVector(1, 1, 1));
248  Actor->SetActorTransform(Trans, false, nullptr, ETeleportType::None);
249  return true;
250  }
251  return false;
252 }

254 // reposition the camera
255 bool CarlaReplayerHelper::SetCameraPosition(uint32_t Id, FVector Offset, FQuat Rotation)
256 {
257  check(Episode != nullptr);
258 
259  // get specator pawn
260  APawn *Spectator = Episode->GetSpectatorPawn();
261  // get the actor to follow
262  AActor *Actor = Episode->FindActor(Id).GetActor();
263 
264  // check
265  if (!Spectator || !Actor)
266  return false;
267 
268  // set the new position
269  FQuat ActorRot = Actor->GetActorTransform().GetRotation();
270  FVector Pos = Actor->GetActorTransform().GetTranslation() + (ActorRot.RotateVector(Offset));
271  Spectator->SetActorLocation(Pos);
272  Spectator->SetActorRotation(ActorRot * Rotation);
273 
274  return true;

326 // replay finish
327 bool CarlaReplayerHelper::ProcessReplayerFinish(bool bApplyAutopilot)
328 {
329  // set autopilot and physics to all AI vehicles
330  auto registry = Episode->GetActorRegistry();
331  for (const FActorView &ActorView : registry)
332  {
333  // enable physics only on vehicles
334  switch (ActorView.GetActorType())
335  {
336 
337  // vehicles
338  case FActorView::ActorType::Vehicle:
339  SetActorSimulatePhysics(ActorView, true);
340  // autopilot
341  if (bApplyAutopilot)
342  SetActorAutopilot(ActorView, true);
343  break;
344 
345  // walkers
346  case FActorView::ActorType::Walker:
347  // stop walker
348  SetActorVelocity(ActorView, FVector(0, 0, 0));
349  SetWalkerSpeed(ActorView.GetActorId(), 0.0f);
350  break;
351  }
352  }
353  return true;
354 }
