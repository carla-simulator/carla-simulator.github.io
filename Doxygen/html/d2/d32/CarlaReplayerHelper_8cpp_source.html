18 // create or reuse an actor for replaying
19 std::pair<int, FActorView>CarlaReplayerHelper::TryToCreateReplayerActor(
20  FVector &Location,
21  FVector &Rotation,
22  FActorDescription &ActorDesc,
23  uint32_t DesiredId)
24 {
25  check(Episode != nullptr);
26 
27  FActorView view_empty;
28 
29  // check type of actor we need
30  if (ActorDesc.Id.StartsWith("traffic."))
31  {
32  AActor *Actor = FindTrafficLightAt(Location);
33  if (Actor != nullptr)
34  {
35  // actor found
36  auto view = Episode->GetActorRegistry().Find(Actor);
37  // reuse that actor
38  return std::pair<int, FActorView>(2, view);
39  }
40  else
41  {
42  // actor not found
43  UE_LOG(LogCarla, Log, TEXT("TrafficLight not found"));
44  return std::pair<int, FActorView>(0, view_empty);
45  }
46  }
47  else if (!ActorDesc.Id.StartsWith("sensor."))
48  {
49  // check if an actor of that type already exist with same id
50  if (Episode->GetActorRegistry().Contains(DesiredId))
51  {
52  auto view = Episode->GetActorRegistry().Find(DesiredId);
53  const FActorDescription *desc = &view.GetActorInfo()->Description;
54  if (desc->Id == ActorDesc.Id)
55  {
56  // we don't need to create, actor of same type already exist
57  // relocate
58  FRotator Rot = FRotator::MakeFromEuler(Rotation);
59  FTransform Trans2(Rot, Location, FVector(1, 1, 1));
60  view.GetActor()->SetActorTransform(Trans2, false, nullptr, ETeleportType::TeleportPhysics);
61  return std::pair<int, FActorView>(2, view);
62  }
63  }
64  // create the transform
65  FRotator Rot = FRotator::MakeFromEuler(Rotation);
66  FTransform Trans(Rot, FVector(0, 0, 100000), FVector(1, 1, 1));
67  // create as new actor
68  TPair<EActorSpawnResultStatus, FActorView> Result = Episode->SpawnActorWithInfo(Trans, ActorDesc, DesiredId);
69  if (Result.Key == EActorSpawnResultStatus::Success)
70  {
71  // relocate
72  FTransform Trans2(Rot, Location, FVector(1, 1, 1));
73  Result.Value.GetActor()->SetActorTransform(Trans2, false, nullptr, ETeleportType::TeleportPhysics);
74  return std::pair<int, FActorView>(1, Result.Value);
75  }
76  else
77  {
78  UE_LOG(LogCarla, Log, TEXT("Actor could't be created by replayer"));
79  return std::pair<int, FActorView>(0, Result.Value);
80  }
81  }
82  else
83  {
84  // actor ignored
85  return std::pair<int, FActorView>(0, view_empty);
86  }
87 }

89 AActor *CarlaReplayerHelper::FindTrafficLightAt(FVector Location)
90 {
91  check(Episode != nullptr);
92  auto World = Episode->GetWorld();
93  check(World != nullptr);
94 
95  // get its position (truncated as int's)
96  int x = static_cast<int>(Location.X);
97  int y = static_cast<int>(Location.Y);
98  int z = static_cast<int>(Location.Z);
99 
100  // search an "traffic." actor at that position
101  for (TActorIterator<ATrafficSignBase> It(World); It; ++It)
102  {
103  ATrafficSignBase *Actor = *It;
104  check(Actor != nullptr);
105  FVector vec = Actor->GetTransform().GetTranslation();
106  int x2 = static_cast<int>(vec.X);
107  int y2 = static_cast<int>(vec.Y);
108  int z2 = static_cast<int>(vec.Z);
109  if ((x2 == x) && (y2 == y) && (z2 == z))
110  {
111  // actor found
112  return static_cast<AActor *>(Actor);
113  }
114  }
115  // actor not found
116  return nullptr;
117 }

158 // replay event for creating actor
159 std::pair<int, uint32_t> CarlaReplayerHelper::ProcessReplayerEventAdd(
160  FVector Location,
161  FVector Rotation,
162  CarlaRecorderActorDescription Description,
163  uint32_t DesiredId,
164  bool bIgnoreHero)
165 {
166  check(Episode != nullptr);
167  FActorDescription ActorDesc;
168  bool IsHero = false;
169 
170  // prepare actor description
171  ActorDesc.UId = Description.UId;
172  ActorDesc.Id = Description.Id;
173  for (const auto &Item : Description.Attributes)
174  {
175  FActorAttribute Attr;
176  Attr.Type = static_cast<EActorAttributeType>(Item.Type);
177  Attr.Id = Item.Id;
178  Attr.Value = Item.Value;
179  ActorDesc.Variations.Add(Attr.Id, std::move(Attr));
180  // check for hero
181  if (Item.Id == "role_name" && Item.Value == "hero")
182  IsHero = true;
183  }
184 
185  auto result = TryToCreateReplayerActor(Location, Rotation, ActorDesc, DesiredId);
186 
187  if (result.first != 0)
188  {
189  // disable physics and autopilot on vehicles
190  if (result.second.GetActorType() == FActorView::ActorType::Vehicle)
191  {
192  // ignore hero ?
193  if (!(bIgnoreHero && IsHero))
194  {
195  // disable physics
196  SetActorSimulatePhysics(result.second, false);
197  // disable autopilot
198  SetActorAutopilot(result.second, false, false);
199  }
200  else
201  {
202  // reenable physics just in case
203  SetActorSimulatePhysics(result.second, true);
204  }
205  }
206  }
207 
208  return std::make_pair(result.first, result.second.GetActorId());
209 }

244 // reposition actors
245 bool CarlaReplayerHelper::ProcessReplayerPosition(CarlaRecorderPosition Pos1, CarlaRecorderPosition Pos2, double Per, double DeltaTime)
246 {
247  check(Episode != nullptr);
248  AActor *Actor = Episode->GetActorRegistry().Find(Pos1.DatabaseId).GetActor();
249  FVector Location;
250  FRotator Rotation;
251  if (Actor && !Actor->IsPendingKill())
252  {
253  // check to assign first position or interpolate between both
254  if (Per == 0.0)
255  {
256  // assign position 1
257  Location = FVector(Pos1.Location);
258  Rotation = FRotator::MakeFromEuler(Pos1.Rotation);
259  }
260  else
261  {
262  // interpolate positions
263  Location = FMath::Lerp(FVector(Pos1.Location), FVector(Pos2.Location), Per);
264  Rotation = FMath::Lerp(FRotator::MakeFromEuler(Pos1.Rotation), FRotator::MakeFromEuler(Pos2.Rotation), Per);
265  }
266  // set new transform
267  FTransform Trans(Rotation, Location, FVector(1, 1, 1));
268  Actor->SetActorTransform(Trans, false, nullptr, ETeleportType::None);
269  return true;
270  }
271  return false;
272 }

386 // replay finish
387 bool CarlaReplayerHelper::ProcessReplayerFinish(bool bApplyAutopilot, bool bIgnoreHero, std::unordered_map<uint32_t, bool> &IsHero)
388 {
389  // set autopilot and physics to all AI vehicles
390  auto registry = Episode->GetActorRegistry();
391  for (const FActorView &ActorView : registry)
392  {
393  // enable physics only on vehicles
394  switch (ActorView.GetActorType())
395  {
396 
397  // vehicles
398  case FActorView::ActorType::Vehicle:
399  // check for hero
400  if (!(bIgnoreHero && IsHero[ActorView.GetActorId()]))
401  {
402  // stop all vehicles
403  SetActorSimulatePhysics(ActorView, true);
404  SetActorVelocity(ActorView, FVector(0, 0, 0));
405  // reset any control assigned
406  auto Veh = Cast<ACarlaWheeledVehicle>(const_cast<AActor *>(ActorView.GetActor()));
407  if (Veh != nullptr)
408  {
409  FVehicleControl Control;
410  Control.Throttle = 0.0f;
411  Control.Steer = 0.0f;
412  Control.Brake = 0.0f;
413  Control.bHandBrake = false;
414  Control.bReverse = false;
415  Control.Gear = 1;
416  Control.bManualGearShift = false;
417  Veh->ApplyVehicleControl(Control, EVehicleInputPriority::User);
418  }
419 
420  }
421  break;
422 
423  // walkers
424  case FActorView::ActorType::Walker:
425  // stop walker
426  SetWalkerSpeed(ActorView.GetActorId(), 0.0f);
427  break;
428  }
429  }
430  return true;
431 }
