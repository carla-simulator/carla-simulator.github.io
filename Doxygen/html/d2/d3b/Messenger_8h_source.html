24  /// This class is the template for messaging functionality between
25  /// different stage classes to send and receive data.
26  /// One object of this type can only facilitate receiving data from
27  /// a sender stage and passing the data onto a receiver stage.
28  /// The class maintains state internally and blocks send or receive
29  /// requests until data is available/successfully passed on.
30  template <typename Data>
31  class Messenger {
32 
33  private:
34 
35  /// Flag used to wake up and join any waiting function calls on this object.
36  std::atomic<bool> stop_messenger;
37  /// State variable that will progress upon every successful communication
38  /// between the sender and receiver.
39  std::atomic<int> state_counter;
40  /// Member used to hold data sent by the sender.
41  Data data;
42  /// Mutex used to manage contention between the sender and receiver.
43  std::mutex data_modification_mutex;
44  /// Variable to conditionally block sender in case waiting for the receiver.
45  std::condition_variable send_condition;
46  /// Variable to conditionally block receiver in case waiting for the sender.
47  std::condition_variable receive_condition;
48 
49  public:
50 
51  Messenger() {
52  state_counter.store(0);
53  stop_messenger.store(false);
54  }
55  ~Messenger() {}
56 
57  /// This method receives data from a sender, stores in a member and
58  /// increments state.
59  int SendData(DataPacket<Data> packet) {
60 
61  std::unique_lock<std::mutex> lock(data_modification_mutex);
62  while (state_counter.load() == packet.id && !stop_messenger.load()) {
63  send_condition.wait_for(lock, 1ms, [=] {return state_counter.load() != packet.id;});
64  }
65  data = packet.data;
66  state_counter.store(state_counter.load() + 1);
67  int present_state = state_counter.load();
68  receive_condition.notify_one();
69 
70  return present_state;
71  }
72 
73  /// This method presents stored data to the receiver and increments state.
74  DataPacket<Data> ReceiveData(int old_state) {
75 
76  std::unique_lock<std::mutex> lock(data_modification_mutex);
77  while (state_counter.load() == old_state && !stop_messenger.load()) {
78  receive_condition.wait_for(lock, 1ms, [=] {return state_counter.load() != old_state;});
79  }
80  state_counter.store(state_counter.load() + 1);
81  DataPacket<Data> packet = {state_counter.load(), data};
82  send_condition.notify_one();
83 
84  return packet;
85  }
86 
87  /// This method returns the current value of the state counter.
88  int GetState() {
89  return state_counter.load();
90  }
91 
92  /// This method unblocks any waiting calls on this object.
93  void Stop() {
94  stop_messenger.store(true);
95  }
96 
97  /// This method restores regular functionality of the messenger.
98  /// This method needs to be called if the messenger has to be
99  /// used again after a call to the Stop() method.
100  void Start() {
101  stop_messenger.store(false);
102  }
103 
104  };
