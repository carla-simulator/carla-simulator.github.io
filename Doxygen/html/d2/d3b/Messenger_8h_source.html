25  /// This class is the template for messaging functionality between
26  /// different stage classes to send and receive data.
27  /// One object of this type can only facilitate receiving data from
28  /// a sender stage and passing the data onto a receiver stage.
29  /// The class maintains state internally and blocks send or receive
30  /// requests until data is available/successfully passed on.
31  template <typename Data>
32  class Messenger {
33 
34  private:
35 
36  /// Flag used to wake up and join any waiting function calls on this object.
37  std::atomic<bool> stop_messenger;
38  /// State variable that will progress upon every successful communication
39  /// between the sender and receiver.
40  std::atomic<int> state_counter;
41  /// Member used to hold data sent by the sender.
42  Data data;
43  /// Mutex used to manage contention between the sender and receiver.
44  std::mutex data_modification_mutex;
45  /// Variable to conditionally block sender in case waiting for the receiver.
46  std::condition_variable send_condition;
47  /// Variable to conditionally block receiver in case waiting for the sender.
48  std::condition_variable receive_condition;
49 
50  public:
51 
52  Messenger() {
53  state_counter.store(0);
54  stop_messenger.store(false);
55  }
56  ~Messenger() {}
57 
58  /// This method receives data from a sender, stores in a member and
59  /// increments state.
60  int SendData(DataPacket<Data> packet) {
61 
62  std::unique_lock<std::mutex> lock(data_modification_mutex);
63  while (state_counter.load() == packet.id && !stop_messenger.load()) {
64  send_condition.wait_for(lock, 1ms, [=] {return state_counter.load() != packet.id;});
65  }
66  data = packet.data;
67  state_counter.store(state_counter.load() + 1);
68  int present_state = state_counter.load();
69  receive_condition.notify_one();
70 
71  return present_state;
72  }
73 
74  /// This method presents stored data to the receiver and increments state.
75  DataPacket<Data> ReceiveData(int old_state) {
76 
77  std::unique_lock<std::mutex> lock(data_modification_mutex);
78  while (state_counter.load() == old_state && !stop_messenger.load()) {
79  receive_condition.wait_for(lock, 1ms, [=] {return state_counter.load() != old_state;});
80  }
81  state_counter.store(state_counter.load() + 1);
82  DataPacket<Data> packet = {state_counter.load(), data};
83  send_condition.notify_one();
84 
85  return packet;
86  }
87 
88  /// This method returns the current value of the state counter.
89  int GetState() {
90  return state_counter.load();
91  }
92 
93  /// This method unblocks any waiting calls on this object.
94  void Stop() {
95  stop_messenger.store(true);
96  }
97 
98  /// This method restores regular functionality of the messenger.
99  /// This method needs to be called if the messenger has to be
100  /// used again after a call to the Stop() method.
101  void Start() {
102  stop_messenger.store(false);
103  }
104 
105  };
