53 void AObstacleDetectionSensor::Tick(float DeltaSeconds)
54 {
55  Super::Tick(DeltaSeconds);
56 
57  const FVector &Start = GetActorLocation();
58  const FVector &End = Start + (GetActorForwardVector() * Distance);
59  UWorld* CurrentWorld = GetWorld();
60 
61  // Struct in which the result of the scan will be saved
62  FHitResult HitOut = FHitResult();
63 
64  // Initialization of Query Parameters
65  FCollisionQueryParams TraceParams(FName(TEXT("ObstacleDetection Trace")), true, this);
66 
67 #if !(UE_BUILD_SHIPPING || UE_BUILD_TEST)
68  // If debug mode enabled, we create a tag that will make the sweep be
69  // displayed.
70  if (bDebugLineTrace)
71  {
72  const FName TraceTag("ObstacleDebugTrace");
73  CurrentWorld->DebugDrawTraceTag = TraceTag;
74  TraceParams.TraceTag = TraceTag;
75  }
76 #endif
77 
78  // Hit against complex meshes
79  TraceParams.bTraceComplex = true;
80 
81  // Ignore trigger boxes
82  TraceParams.bIgnoreTouches = true;
83 
84  // Limit the returned information
85  TraceParams.bReturnPhysicalMaterial = false;
86 
87  // Ignore ourselves
88  TraceParams.AddIgnoredActor(this);
89  if(Super::GetOwner()!=nullptr)
90  TraceParams.AddIgnoredActor(Super::GetOwner());
91 
92  bool isHitReturned;
93  // Choosing a type of sweep is a workaround until everything get properly
94  // organized under correct collision channels and object types.
95  if (bOnlyDynamics)
96  {
97  // If we go only for dynamics, we check the object type AllDynamicObjects
98  FCollisionObjectQueryParams TraceChannel = FCollisionObjectQueryParams(
99  FCollisionObjectQueryParams::AllDynamicObjects);
100  isHitReturned = CurrentWorld->SweepSingleByObjectType(
101  HitOut,
102  Start,
103  End,
104  FQuat(),
105  TraceChannel,
106  FCollisionShape::MakeSphere(HitRadius),
107  TraceParams);
108  }
109  else
110  {
111  // Else, if we go for everything, we get everything that interacts with a
112  // Pawn
113  ECollisionChannel TraceChannel = ECC_WorldStatic;
114  isHitReturned = CurrentWorld->SweepSingleByChannel(
115  HitOut,
116  Start,
117  End,
118  FQuat(),
119  TraceChannel,
120  FCollisionShape::MakeSphere(HitRadius),
121  TraceParams);
122  }
123 
124  if (isHitReturned)
125  {
126  OnObstacleDetectionEvent(this, HitOut.Actor.Get(), HitOut.Distance, HitOut);
127  }
128 }
