43  std::unique_ptr<Mesh> MeshFactory::Generate(
44  const road::Lane &lane, const double s_start, const double s_end) const {
45  RELEASE_ASSERT(road_param.resolution > 0.0);
46  DEBUG_ASSERT(s_start >= 0.0);
47  DEBUG_ASSERT(s_end <= lane.GetDistance() + lane.GetLength());
48  DEBUG_ASSERT(s_end >= EPSILON);
49  DEBUG_ASSERT(s_start < s_end);
50  // The lane with lane_id 0 have no physical representation in OpenDRIVE
51  Mesh out_mesh;
52  if (lane.GetId() == 0) {
53  return std::make_unique<Mesh>(out_mesh);
54  }
55  double s_current = s_start;
56 
57  std::vector<geom::Vector3D> vertices;
58  if (lane.IsStraight()) {
59  // Mesh optimization: If the lane is straight just add vertices at the
60  // begining and at the end of it
61  const auto edges = lane.GetCornerPositions(s_current, road_param.extra_lane_width);
62  vertices.push_back(edges.first);
63  vertices.push_back(edges.second);
64  } else {
65  // Iterate over the lane's 's' and store the vertices based on it's width
66  do {
67  // Get the location of the edges of the current lane at the current waypoint
68  const auto edges = lane.GetCornerPositions(s_current, road_param.extra_lane_width);
69  vertices.push_back(edges.first);
70  vertices.push_back(edges.second);
71 
72  // Update the current waypoint's "s"
73  s_current += road_param.resolution;
74  } while(s_current < s_end);
75  }
76 
77  // This ensures the mesh is constant and have no gaps between roads,
78  // adding geometry at the very end of the lane
79  if (s_end - (s_current - road_param.resolution) > EPSILON) {
80  const auto edges = lane.GetCornerPositions(s_end, road_param.extra_lane_width);
81  vertices.push_back(edges.first);
82  vertices.push_back(edges.second);
83  }
84 
85  // Add the adient material, create the strip and close the material
86  out_mesh.AddMaterial(
87  lane.GetType() == road::Lane::LaneType::Sidewalk ? "sidewalk" : "road");
88  out_mesh.AddTriangleStrip(vertices);
89  out_mesh.EndMaterial();
90  return std::make_unique<Mesh>(out_mesh);
91  }

245  s_current = s_until;

341  }
342  }

355  struct VertexInfo {
356  Mesh::vertex_type * vertex;
357  size_t lane_mesh_idx;
358  bool is_static;

361  // Helper function to compute the weight of neighboring vertices
362  static VertexWeight ComputeVertexWeight(
363  const MeshFactory::RoadParameters &road_param,
364  const VertexInfo &vertex_info,
365  const VertexInfo &neighbor_info) {
366  const float distance3D = geom::Math::Distance(*vertex_info.vertex, *neighbor_info.vertex);
367  // Ignore vertices beyond a certain distance
368  if(distance3D > road_param.max_weight_distance) {
369  return {neighbor_info.vertex, 0};
370  }
371  if(abs(distance3D) < EPSILON) {
372  return {neighbor_info.vertex, 0};
373  }
374  float weight = geom::Math::Clamp<float>(1.0f / distance3D, 0.0f, 100000.0f);
375 
376  // Additional weight to vertices in the same lane
377  if(vertex_info.lane_mesh_idx == neighbor_info.lane_mesh_idx) {
378  weight *= road_param.same_lane_weight_multiplier;
379  // Further additional weight for fixed verices
380  if(neighbor_info.is_static) {
381  weight *= road_param.lane_ends_multiplier;
382  }
383  }
384  return {neighbor_info.vertex, weight};
385  }

387  // Helper function to compute neighborhoord of vertices and their weights
388  std::vector<VertexNeighbors> GetVertexNeighborhoodAndWeights(
389  const MeshFactory::RoadParameters &road_param,
390  std::vector<std::unique_ptr<Mesh>> &lane_meshes) {
391  // Build rtree for neighborhood queries
392  using Rtree = geom::PointCloudRtree<VertexInfo>;
393  using Point = Rtree::BPoint;
394  Rtree rtree;
395  for (size_t lane_mesh_idx = 0; lane_mesh_idx < lane_meshes.size(); ++lane_mesh_idx) {
396  auto& mesh = lane_meshes[lane_mesh_idx];
397  for(size_t i = 0; i < mesh->GetVerticesNum(); ++i) {
398  auto& vertex = mesh->GetVertices()[i];
399  Point point(vertex.x, vertex.y, vertex.z);
400  if (i < 2 || i >= mesh->GetVerticesNum() - 2) {
401  rtree.InsertElement({point, {&vertex, lane_mesh_idx, true}});
402  } else {
403  rtree.InsertElement({point, {&vertex, lane_mesh_idx, false}});
404  }
405  }
406  }
407 
408  // Find neighbors for each vertex and compute their weight
409  std::vector<VertexNeighbors> vertices_neighborhoods;
410  for (size_t lane_mesh_idx = 0; lane_mesh_idx < lane_meshes.size(); ++lane_mesh_idx) {
411  auto& mesh = lane_meshes[lane_mesh_idx];
412  for(size_t i = 0; i < mesh->GetVerticesNum(); ++i) {
413  if (i > 2 && i < mesh->GetVerticesNum() - 2) {
414  auto& vertex = mesh->GetVertices()[i];
415  Point point(vertex.x, vertex.y, vertex.z);
416  auto closest_vertices = rtree.GetNearestNeighbours(point, 20);
417  VertexNeighbors vertex_neighborhood;
418  vertex_neighborhood.vertex = &vertex;
419  for(auto& close_vertex : closest_vertices) {
420  auto &vertex_info = close_vertex.second;
421  if(&vertex == vertex_info.vertex) {
422  continue;
423  }
424  auto vertex_weight = ComputeVertexWeight(
425  road_param, {&vertex, lane_mesh_idx, false}, vertex_info);
426  if(vertex_weight.weight > 0)
427  vertex_neighborhood.neighbors.push_back(vertex_weight);
428  }
429  vertices_neighborhoods.push_back(vertex_neighborhood);
430  }
431  }
432  }
433  return vertices_neighborhoods;
434  }

436  std::unique_ptr<Mesh> MeshFactory::MergeAndSmooth(std::vector<std::unique_ptr<Mesh>> &lane_meshes) const {
437  geom::Mesh out_mesh;
438 
439  auto vertices_neighborhoods = GetVertexNeighborhoodAndWeights(road_param, lane_meshes);
440 
441  // Laplacian function
442  auto Laplacian = [&](const Mesh::vertex_type* vertex, const std::vector<VertexWeight> &neighbors) -> double {
443  double sum = 0;
444  double sum_weight = 0;
445  for(auto &element : neighbors) {
446  sum += (element.vertex->z - vertex->z)*element.weight;
447  sum_weight += element.weight;
448  }
449  if(sum_weight > 0)
450  return sum / sum_weight;
451  else
452  return 0;
453  };
454  // Run iterative algorithm
455  double lambda = 0.5;
456  int iterations = 100;
457  for(int iter = 0; iter < iterations; ++iter) {
458  for (auto& vertex_neighborhood : vertices_neighborhoods) {
459  auto * vertex = vertex_neighborhood.vertex;
460  vertex->z += static_cast<float>(lambda*Laplacian(vertex, vertex_neighborhood.neighbors));
461  }
462  }
463 
464  for(auto &mesh : lane_meshes) {
465  out_mesh += *mesh;
466  }
467 
468  return std::make_unique<Mesh>(out_mesh);
469  }
