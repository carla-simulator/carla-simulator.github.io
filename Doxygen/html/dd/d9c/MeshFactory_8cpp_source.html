62  std::unique_ptr<Mesh> MeshFactory::Generate(
63  const road::Lane &lane, const double s_start, const double s_end) const {
64  RELEASE_ASSERT(road_param.resolution > 0.0);
65  DEBUG_ASSERT(s_start >= 0.0);
66  DEBUG_ASSERT(s_end <= lane.GetDistance() + lane.GetLength());
67  DEBUG_ASSERT(s_end >= EPSILON);
68  DEBUG_ASSERT(s_start < s_end);
69  // The lane with lane_id 0 have no physical representation in OpenDRIVE
70  Mesh out_mesh;
71  if (lane.GetId() == 0) {
72  return std::make_unique<Mesh>(out_mesh);
73  }
74  double s_current = s_start;
75 
76  std::vector<geom::Vector3D> vertices;
77  if (lane.IsStraight()) {
78  // Mesh optimization: If the lane is straight just add vertices at the
79  // begining and at the end of it
80  const auto edges = lane.GetCornerPositions(s_current, road_param.extra_lane_width);
81  vertices.push_back(edges.first);
82  vertices.push_back(edges.second);
83  } else {
84  // Iterate over the lane's 's' and store the vertices based on it's width
85  do {
86  // Get the location of the edges of the current lane at the current waypoint
87  const auto edges = lane.GetCornerPositions(s_current, road_param.extra_lane_width);
88  vertices.push_back(edges.first);
89  vertices.push_back(edges.second);
90 
91  // Update the current waypoint's "s"
92  s_current += road_param.resolution;
93  } while(s_current < s_end);
94  }
95 
96  // This ensures the mesh is constant and have no gaps between roads,
97  // adding geometry at the very end of the lane
98  if (s_end - (s_current - road_param.resolution) > EPSILON) {
99  const auto edges = lane.GetCornerPositions(s_end - MESH_EPSILON, road_param.extra_lane_width);
100  vertices.push_back(edges.first);
101  vertices.push_back(edges.second);
102  }
103 
104  // Add the adient material, create the strip and close the material
105  out_mesh.AddMaterial(
106  lane.GetType() == road::Lane::LaneType::Sidewalk ? "sidewalk" : "road");
107  out_mesh.AddTriangleStrip(vertices);
108  out_mesh.EndMaterial();
109  return std::make_unique<Mesh>(out_mesh);
110  }

112  std::unique_ptr<Mesh> MeshFactory::GenerateTesselated(
113  const road::Lane& lane, const double s_start, const double s_end) const {
114  RELEASE_ASSERT(road_param.resolution > 0.0);
115  DEBUG_ASSERT(s_start >= 0.0);
116  DEBUG_ASSERT(s_end <= lane.GetDistance() + lane.GetLength());
117  DEBUG_ASSERT(s_end >= EPSILON);
118  DEBUG_ASSERT(s_start < s_end);
119  // The lane with lane_id 0 have no physical representation in OpenDRIVE
120  Mesh out_mesh;
121  if (lane.GetId() == 0) {
122  return std::make_unique<Mesh>(out_mesh);
123  }
124  double s_current = s_start;
125 
126  std::vector<geom::Vector3D> vertices;
127  // Ensure minimum vertices in width are two
128  const int vertices_in_width = road_param.vertex_width_resolution >= 2 ? road_param.vertex_width_resolution : 2;
129  const int segments_number = vertices_in_width - 1;
130 
131  // Iterate over the lane's 's' and store the vertices based on it's width
132  do {
133  // Get the location of the edges of the current lane at the current waypoint
134  std::pair<geom::Vector3D, geom::Vector3D> edges = lane.GetCornerPositions(s_current, road_param.extra_lane_width);
135  const geom::Vector3D segments_size = ( edges.second - edges.first ) / segments_number;
136  geom::Vector3D current_vertex = edges.first;
137 
138  for (int i = 0; i < vertices_in_width; ++i) {
139  vertices.push_back(current_vertex);
140  current_vertex = current_vertex + segments_size;
141  }
142  // Update the current waypoint's "s"
143  s_current += road_param.resolution;
144  } while (s_current < s_end);
145 
146  // This ensures the mesh is constant and have no gaps between roads,
147  // adding geometry at the very end of the lane
148 
149  if (s_end - (s_current - road_param.resolution) > EPSILON) {
150  std::pair<carla::geom::Vector3D, carla::geom::Vector3D> edges =
151  lane.GetCornerPositions(s_end - MESH_EPSILON, road_param.extra_lane_width);
152  const geom::Vector3D segments_size = (edges.second - edges.first) / segments_number;
153  geom::Vector3D current_vertex = edges.first;
154 
155  for (int i = 0; i < vertices_in_width; ++i)
156  {
157  vertices.push_back(current_vertex);
158  current_vertex = current_vertex + segments_size;
159  }
160  }
161  out_mesh.AddVertices(vertices);
162 
163  // Add the adient material, create the strip and close the material
164  out_mesh.AddMaterial(
165  lane.GetType() == road::Lane::LaneType::Sidewalk ? "sidewalk" : "road");
166 
167  const size_t number_of_rows = (vertices.size() / vertices_in_width);
168 
169  for (size_t i = 0; i < (number_of_rows - 1); ++i) {
170  for (size_t j = 0; j < vertices_in_width - 1; ++j) {
171  out_mesh.AddIndex( j + ( i * vertices_in_width ) + 1);
172  out_mesh.AddIndex( ( j + 1 ) + ( i * vertices_in_width ) + 1);
173  out_mesh.AddIndex( j + ( ( i + 1 ) * vertices_in_width ) + 1);
174 
175  out_mesh.AddIndex( ( j + 1 ) + ( i * vertices_in_width ) + 1);
176  out_mesh.AddIndex( ( j + 1 ) + ( ( i + 1 ) * vertices_in_width ) + 1);
177  out_mesh.AddIndex( j + ( ( i + 1 ) * vertices_in_width ) + 1);
178  }
179  }
180  out_mesh.EndMaterial();
181  return std::make_unique<Mesh>(out_mesh);
182  }

185  void MeshFactory::GenerateLaneSectionOrdered(
186  const road::LaneSection &lane_section,
187  std::map<carla::road::Lane::LaneType , std::vector<std::unique_ptr<Mesh>>>& result) const {
188 
189  const int vertices_in_width = road_param.vertex_width_resolution >= 2 ? road_param.vertex_width_resolution : 2;
190  std::vector<size_t> redirections;
191  for (auto &&lane_pair : lane_section.GetLanes()) {
192  auto it = std::find(redirections.begin(), redirections.end(), lane_pair.first);
193  if ( it == redirections.end() ) {
194  redirections.push_back(lane_pair.first);
195  it = std::find(redirections.begin(), redirections.end(), lane_pair.first);
196  }
197 
198  size_t PosToAdd = it - redirections.begin();
199 
200  Mesh out_mesh;
201  if(lane_pair.second.GetType() == road::Lane::LaneType::Driving ){
202  out_mesh += *GenerateTesselated(lane_pair.second);
203  }else{
204  out_mesh += *GenerateSidewalk(lane_pair.second);
205  }
206 
207  if( result[lane_pair.second.GetType()].size() <= PosToAdd ){
208  result[lane_pair.second.GetType()].push_back(std::make_unique<Mesh>(out_mesh));
209  } else {
210  uint32_t verticesinwidth = 0;
211  if(lane_pair.second.GetType() == road::Lane::LaneType::Driving) {
212  verticesinwidth = vertices_in_width;
213  }else if(lane_pair.second.GetType() == road::Lane::LaneType::Sidewalk){
214  verticesinwidth = 4;
215  }else{
216  verticesinwidth = 2;
217  }
218  (result[lane_pair.second.GetType()][PosToAdd])->ConcatMesh(out_mesh, verticesinwidth);
219  }
220  }
221  }

316  Mesh out_mesh;
317 
318  const auto min_lane = lane_section.GetLanes().begin()->first == 0 ?
319  1 : lane_section.GetLanes().begin()->first;
320  const auto max_lane = lane_section.GetLanes().rbegin()->first == 0 ?
321  -1 : lane_section.GetLanes().rbegin()->first;
322 
323  for (auto &&lane_pair : lane_section.GetLanes()) {
324  const auto &lane = lane_pair.second;
325  const double s_start = lane.GetDistance() + EPSILON;
326  const double s_end = lane.GetDistance() + lane.GetLength() - EPSILON;
327  if (lane.GetId() == max_lane) {
328  out_mesh += *GenerateLeftWall(lane, s_start, s_end);
329  }
330  if (lane.GetId() == min_lane) {
331  out_mesh += *GenerateRightWall(lane, s_start, s_end);
332  }
333  }
334  return std::make_unique<Mesh>(out_mesh);

452  std::vector<std::unique_ptr<Mesh>> MeshFactory::GenerateWithMaxLen(
453  const road::LaneSection &lane_section) const {
454  std::vector<std::unique_ptr<Mesh>> mesh_uptr_list;
455  if (lane_section.GetLength() < road_param.max_road_len) {
456  mesh_uptr_list.emplace_back(Generate(lane_section));
457  } else {
458  double s_current = lane_section.GetDistance() + EPSILON;
459  const double s_end = lane_section.GetDistance() + lane_section.GetLength() - EPSILON;
460  while(s_current + road_param.max_road_len < s_end) {
461  const auto s_until = s_current + road_param.max_road_len;
462  Mesh lane_section_mesh;
463  for (auto &&lane_pair : lane_section.GetLanes()) {
464  lane_section_mesh += *Generate(lane_pair.second, s_current, s_until);
465  }
466  mesh_uptr_list.emplace_back(std::make_unique<Mesh>(lane_section_mesh));
467  s_current = s_until;
468  }
469  if (s_end - s_current > EPSILON) {
470  Mesh lane_section_mesh;
471  for (auto &&lane_pair : lane_section.GetLanes()) {
472  lane_section_mesh += *Generate(lane_pair.second, s_current, s_end);
473  }
474  mesh_uptr_list.emplace_back(std::make_unique<Mesh>(lane_section_mesh));
475  }
476  }
477  return mesh_uptr_list;

978  struct VertexWeight {
979  Mesh::vertex_type* vertex;
980  double weight;
981  };
982  struct VertexNeighbors {
983  Mesh::vertex_type* vertex;
984  std::vector<VertexWeight> neighbors;
985  };
986  struct VertexInfo {
987  Mesh::vertex_type * vertex;
988  size_t lane_mesh_idx;
989  bool is_static;
990  };
991 
992  // Helper function to compute the weight of neighboring vertices
993  static VertexWeight ComputeVertexWeight(
994  const MeshFactory::RoadParameters &road_param,
995  const VertexInfo &vertex_info,
996  const VertexInfo &neighbor_info) {
997  const float distance3D = geom::Math::Distance(*vertex_info.vertex, *neighbor_info.vertex);
998  // Ignore vertices beyond a certain distance
999  if(distance3D > road_param.max_weight_distance) {
1000  return {neighbor_info.vertex, 0};
1001  }
1002  if(abs(distance3D) < EPSILON) {
1003  return {neighbor_info.vertex, 0};
1004  }
1005  float weight = geom::Math::Clamp<float>(1.0f / distance3D, 0.0f, 100000.0f);
1006 
1007  // Additional weight to vertices in the same lane
1008  if(vertex_info.lane_mesh_idx == neighbor_info.lane_mesh_idx) {
1009  weight *= road_param.same_lane_weight_multiplier;
1010  // Further additional weight for fixed verices
1011  if(neighbor_info.is_static) {
1012  weight *= road_param.lane_ends_multiplier;
1013  }
1014  }
1015  return {neighbor_info.vertex, weight};
1016  }

1018  // Helper function to compute neighborhoord of vertices and their weights
1019  std::vector<VertexNeighbors> GetVertexNeighborhoodAndWeights(
1020  const MeshFactory::RoadParameters &road_param,
1021  std::vector<std::unique_ptr<Mesh>> &lane_meshes) {
1022  // Build rtree for neighborhood queries
1023  using Rtree = geom::PointCloudRtree<VertexInfo>;
1024  using Point = Rtree::BPoint;
1025  Rtree rtree;
1026  for (size_t lane_mesh_idx = 0; lane_mesh_idx < lane_meshes.size(); ++lane_mesh_idx) {
1027  auto& mesh = lane_meshes[lane_mesh_idx];
1028  for(size_t i = 0; i < mesh->GetVerticesNum(); ++i) {
1029  auto& vertex = mesh->GetVertices()[i];
1030  Point point(vertex.x, vertex.y, vertex.z);
1031  if (i < 2 || i >= mesh->GetVerticesNum() - 2) {
1032  rtree.InsertElement({point, {&vertex, lane_mesh_idx, true}});
1033  } else {
1034  rtree.InsertElement({point, {&vertex, lane_mesh_idx, false}});
1035  }
1036  }
1037  }
1038 
1039  // Find neighbors for each vertex and compute their weight
1040  std::vector<VertexNeighbors> vertices_neighborhoods;
1041  for (size_t lane_mesh_idx = 0; lane_mesh_idx < lane_meshes.size(); ++lane_mesh_idx) {
1042  auto& mesh = lane_meshes[lane_mesh_idx];
1043  for(size_t i = 0; i < mesh->GetVerticesNum(); ++i) {
1044  if (i > 2 && i < mesh->GetVerticesNum() - 2) {
1045  auto& vertex = mesh->GetVertices()[i];
1046  Point point(vertex.x, vertex.y, vertex.z);
1047  auto closest_vertices = rtree.GetNearestNeighbours(point, 20);
1048  VertexNeighbors vertex_neighborhood;
1049  vertex_neighborhood.vertex = &vertex;
1050  for(auto& close_vertex : closest_vertices) {
1051  auto &vertex_info = close_vertex.second;
1052  if(&vertex == vertex_info.vertex) {
1053  continue;
1054  }
1055  auto vertex_weight = ComputeVertexWeight(
1056  road_param, {&vertex, lane_mesh_idx, false}, vertex_info);
1057  if(vertex_weight.weight > 0)
1058  vertex_neighborhood.neighbors.push_back(vertex_weight);
1059  }
1060  vertices_neighborhoods.push_back(vertex_neighborhood);
1061  }
1062  }
1063  }
1064  return vertices_neighborhoods;
1065  }

1067  std::unique_ptr<Mesh> MeshFactory::MergeAndSmooth(std::vector<std::unique_ptr<Mesh>> &lane_meshes) const {
1068  geom::Mesh out_mesh;
1069 
1070  auto vertices_neighborhoods = GetVertexNeighborhoodAndWeights(road_param, lane_meshes);
