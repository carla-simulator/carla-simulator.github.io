40  /// If @a Predicate is false, print an error message. If possible the message
41  /// is printed to the editor window.
42  template <typename T, typename ... ARGS>
43  bool OnScreenAssert(bool Predicate, const T &Format, ARGS && ... Args) const
44  {
45  if (!Predicate)
46  {
47  FString Message;
48  for (auto &String : Stack)
49  {
50  Message += String;
51  }
52  Message += TEXT(" ");
53  Message += FString::Printf(Format, std::forward<ARGS>(Args) ...);
54 
55  UE_LOG(LogCarla, Error, TEXT("%s"), *Message);
56 #if WITH_EDITOR
57  if (GEngine)
58  {
59  GEngine->AddOnScreenDebugMessage(42, 15.0f, FColor::Red, Message);
60  }
61 #endif // WITH_EDITOR
62  }
63  return Predicate;
64  }

77  /// Applies @a Validator to each item in @a Array. Pushes a new context to the
78  /// stack for each item.
79  template <typename T, typename F>
80  bool ForEach(const FString &Type, const TArray<T> &Array, F Validator)
81  {
82  bool Result = true;
83  auto Counter = 0u;
84  for (const auto &Item : Array)
85  {
86  auto Scope = Stack.PushScope(GetDisplayId(Type, Counter, Item));
87  Result &= Validator(Item);
88  ++Counter;
89  }
90  return Result;
91  }

93  /// Applies @a IsValid to each item in @a Array. Pushes a new context to the
94  /// stack for each item.
95  template <typename T>
96  bool AreValid(const FString &Type, const TArray<T> &Array)
97  {
98  return ForEach(Type, Array, [this](const auto &Item) { return IsValid(Item); });
99  }

125  bool IsValid(const FActorVariation &Variation)
126  {
127  return
128  IsIdValid(Variation.Id) &&
129  IsValid(Variation.Type) &&
130  OnScreenAssert(Variation.RecommendedValues.Num() > 0, TEXT("Recommended values cannot be empty")) &&
131  ForEach(TEXT("Recommended Value"), Variation.RecommendedValues, [&](auto &Value) {

192 template <typename ... TStrs>
193 static void FillIdAndTags(FActorDefinition &Def, TStrs && ... Strings)
194 {
195  Def.Id = JoinStrings(TEXT("."), std::forward<TStrs>(Strings) ...).ToLower();
196  Def.Tags = JoinStrings(TEXT(","), std::forward<TStrs>(Strings) ...).ToLower();
197  // each actor gets an actor role name attribute (empty by default)
198  FActorVariation ActorRole;
199  ActorRole.Id = TEXT("role_name");
200  ActorRole.Type = EActorAttributeType::String;
201  ActorRole.RecommendedValues = { TEXT("default") };
202  ActorRole.bRestrictToRecommended = false;
203  Def.Variations.Emplace(ActorRole);
204 }

701 FColor UActorBlueprintFunctionLibrary::ActorAttributeToColor(
702  const FActorAttribute &ActorAttribute,
703  const FColor &Default)
704 {
705  if (ActorAttribute.Type != EActorAttributeType::RGBColor)
706  {
707  UE_LOG(LogCarla, Error, TEXT("ActorAttribute '%s' is not a color"), *ActorAttribute.Id);
708  return Default;
709  }
710  TArray<FString> Channels;
711  ActorAttribute.Value.ParseIntoArray(Channels, TEXT(","), false);
712  if (Channels.Num() != 3)
713  {
714  UE_LOG(LogCarla,
715  Error,
716  TEXT("ActorAttribute '%s': invalid color '%s'"),
717  *ActorAttribute.Id,
718  *ActorAttribute.Value);
719  return Default;
720  }
721  TArray<uint8> Colors;
722  for (auto &Str : Channels)
723  {
724  auto Val = FCString::Atoi(*Str);
725  if ((Val < 0) || (Val > std::numeric_limits<uint8>::max()))
726  {
727  UE_LOG(LogCarla,
728  Error,
729  TEXT("ActorAttribute '%s': invalid color '%s'"),
730  *ActorAttribute.Id,
731  *ActorAttribute.Value);
732  return Default;
733  }
734  Colors.Add(Val);
735  }
736  FColor Color;
737  Color.R = Colors[0u];
738  Color.G = Colors[1u];
739  Color.B = Colors[2u];
740  return Color;
741 }
