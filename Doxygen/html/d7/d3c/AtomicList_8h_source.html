19  /// Holds an atomic pointer to a list.
20  ///
21  /// @warning Only Load method is atomic, modifications to the list are locked
22  /// with a mutex.
23  template <typename T>
24  class AtomicList : private NonCopyable {
25  using ListT = std::vector<T>;
26  public:
27 
28  AtomicList() : _list(std::make_shared<ListT>()) {}
29 
30  template <typename ValueT>
31  void Push(ValueT &&value) {
32  std::lock_guard<std::mutex> lock(_mutex);
33  auto new_list = std::make_shared<ListT>(*Load());
34  new_list->emplace_back(std::forward<ValueT>(value));
35  _list = new_list;
36  }
37 
38  void DeleteByIndex(size_t index) {
39  std::lock_guard<std::mutex> lock(_mutex);
40  auto new_list = std::make_shared<ListT>(*Load());
41  auto begin = new_list->begin();
42  std::advance(begin, index);
43  new_list->erase(begin);
44  _list = new_list;
45  }
46 
47  template <typename ValueT>
48  void DeleteByValue(const ValueT &value) {
49  std::lock_guard<std::mutex> lock(_mutex);
50  auto list = std::make_shared<ListT>(*Load());
51  list->erase(std::remove(list->begin(), list->end(), value), list->end());
52  _list = list;
53  }
54 
55  void Clear() {
56  std::lock_guard<std::mutex> lock(_mutex);
57  _list = std::make_shared<ListT>();
58  }
59 
60  /// Returns a pointer to the list.
61  std::shared_ptr<const ListT> Load() const {
62  return _list.load();
63  }
64 
65  private:
66 
67  std::mutex _mutex;
68 
69  AtomicSharedPtr<const ListT> _list;
70  };
