44 void ASensor::Tick(const float DeltaTime)
45 {
46  TRACE_CPUPROFILER_EVENT_SCOPE(ASensor::Tick);
47  Super::Tick(DeltaTime);
48  if (bClientsListening)
49  {
50  if(!Stream.AreClientsListening())
51  {
52  OnLastClientDisconnected();
53  bClientsListening = false;
54  }
55  }
56  else
57  {
58  if(Stream.AreClientsListening())
59  {
60  OnFirstClientConnected();
61  bClientsListening = true;
62  }
63  }
64  if(!bClientsListening)
65  {
66  return;
67  }
68  ReadyToTick = true;
69  PrePhysTick(DeltaTime);
70 }
