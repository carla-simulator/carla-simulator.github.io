66  static road::Lane::LaneType StringToLaneType(std::string &&str) {
67  StringUtil::ToLower(str);
68  if (str == "driving") {
69  return road::Lane::LaneType::Driving;
70  } else if (str == "stop") {
71  return road::Lane::LaneType::Stop;
72  } else if (str == "shoulder") {
73  return road::Lane::LaneType::Shoulder;
74  } else if (str == "biking") {
75  return road::Lane::LaneType::Biking;
76  } else if (str == "sidewalk") {
77  return road::Lane::LaneType::Sidewalk;
78  } else if (str == "border") {
79  return road::Lane::LaneType::Border;
80  } else if (str == "restricted") {
81  return road::Lane::LaneType::Restricted;
82  } else if (str == "parking") {
83  return road::Lane::LaneType::Parking;
84  } else if (str == "bidirectional") {
85  return road::Lane::LaneType::Bidirectional;
86  } else if (str == "median") {
87  return road::Lane::LaneType::Median;
88  } else if (str == "special1") {
89  return road::Lane::LaneType::Special1;
90  } else if (str == "special2") {
91  return road::Lane::LaneType::Special2;
92  } else if (str == "special3") {
93  return road::Lane::LaneType::Special3;
94  } else if (str == "roadworks") {
95  return road::Lane::LaneType::RoadWorks;
96  } else if (str == "tram") {
97  return road::Lane::LaneType::Tram;
98  } else if (str == "rail") {
99  return road::Lane::LaneType::Rail;
100  } else if (str == "entry") {
101  return road::Lane::LaneType::Entry;
102  } else if (str == "exit") {
103  return road::Lane::LaneType::Exit;
104  } else if (str == "offramp") {
105  return road::Lane::LaneType::OffRamp;
106  } else if (str == "onramp") {
107  return road::Lane::LaneType::OnRamp;
108  } else {
109  return road::Lane::LaneType::None;
110  }

117  std::vector<Road> roads;

119  for (pugi::xml_node node_road : xml.child("OpenDRIVE").children("road")) {
120  Road road { 0, "", 0.0, -1, 0, 0, {}, {}, {} };
121 
122  // attributes
123  road.id = node_road.attribute("id").as_uint();
124  road.name = node_road.attribute("name").value();
125  road.length = node_road.attribute("length").as_double();
126  road.junction_id = node_road.attribute("junction").as_int();
127 
128  // link
129  pugi::xml_node link = node_road.child("link");
130  if (link) {
131  if (link.child("predecessor")) {
132  road.predecessor = link.child("predecessor").attribute("elementId").as_uint();
133  }
134  if (link.child("successor")) {
135  road.successor = link.child("successor").attribute("elementId").as_uint();
136  }
137  }
138 
139  // types
140  for (pugi::xml_node node_type : node_road.children("type")) {
141  RoadTypeSpeed type { 0.0, "", 0.0, "" };
142 
143  type.s = node_type.attribute("s").as_double();
144  type.type = node_type.attribute("type").value();
145 
146  // speed type

148  if (speed) {
149  type.max = speed.attribute("max").as_double();

152 
153  // add it

177

179  lane.type = StringToLaneType(node_lane.attribute("type").value());
180  lane.level = node_lane.attribute("level").as_bool();
181 
182  // link

184  if (link2) {
185  if (link2.child("predecessor")) {
186  lane.predecessor = link2.child("predecessor").attribute("id").as_int();
187  }
188  if (link2.child("successor")) {
189  lane.successor = link2.child("successor").attribute("id").as_int();
190  }
191  }
192 
193  // add it
194  section.lanes.emplace_back(lane);
195  }
196 
197  // center lane
198  for (pugi::xml_node node_lane : node_section.child("center").children("lane")) {
199  Lane lane { 0, road::Lane::LaneType::None, false, 0, 0 };
200 
201  lane.id = node_lane.attribute("id").as_int();
202  lane.type = StringToLaneType(node_lane.attribute("type").value());
203  lane.level = node_lane.attribute("level").as_bool();
204 
205  // link (probably it never exists)
206  pugi::xml_node link2 = node_lane.child("link");
207  if (link2) {
208  if (link2.child("predecessor")) {
209  lane.predecessor = link2.child("predecessor").attribute("id").as_int();
210  }
211  if (link2.child("successor")) {
212  lane.successor = link2.child("successor").attribute("id").as_int();

220  // right lane

222  Lane lane { 0, road::Lane::LaneType::None, false, 0, 0 };
223 
224  lane.id = node_lane.attribute("id").as_int();
225  lane.type = StringToLaneType(node_lane.attribute("type").value());
226  lane.level = node_lane.attribute("level").as_bool();
227 
228  // link
