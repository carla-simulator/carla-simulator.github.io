56  uint64_t World::ApplySettings(const rpc::EpisodeSettings &settings, time_duration timeout) {
57  rpc::EpisodeSettings new_settings = settings;
58  uint64_t id = _episode.Lock()->SetEpisodeSettings(settings);
59 
60  time_duration local_timeout = timeout.milliseconds() == 0 ?
61  _episode.Lock()->GetNetworkingTimeout() : timeout;
62 
63  if (settings.fixed_delta_seconds.has_value()) {
64  using namespace std::literals::chrono_literals;
65 
66  const auto number_of_attemps = 30u;
67  uint64_t tics_correct = 0;
68  for (auto i = 0u; i < number_of_attemps; i++) {
69  const auto curr_snapshot = GetSnapshot();
70 
71  const double error = abs(new_settings.fixed_delta_seconds.get() - curr_snapshot.GetTimestamp().delta_seconds);
72  if (error < std::numeric_limits<float>::epsilon())
73  tics_correct++;
74 
75  if (tics_correct >= 2)
76  return id;
77 
78  Tick(local_timeout);
79  }
80 
81  log_warning("World::ApplySettings: After", number_of_attemps, " attemps, the settings were not correctly set. Please check that everything is consistent.");
82  }
83  return id;
84  }

173  for (size_t i = 0; i < actors->size(); i++) {

177  if(sign && (sign->GetSignId() == landmark_id)) {
178  return actor;

262  std::vector<SharedPtr<Actor>> World::GetTrafficLightsFromWaypoint(
263  const Waypoint& waypoint, double distance) const {
264  std::vector<SharedPtr<Actor>> Result;
265  std::vector<SharedPtr<Landmark>> landmarks =
266  waypoint.GetAllLandmarksInDistance(distance);
267  std::set<std::string> added_signals;
268  for (auto& landmark : landmarks) {
269  if (road::SignalType::IsTrafficLight(landmark->GetType())) {
270  SharedPtr<Actor> traffic_light = GetTrafficLight(*(landmark.get()));
271  if (traffic_light) {
272  if(added_signals.count(landmark->GetId()) == 0) {
273  Result.emplace_back(traffic_light);
274  added_signals.insert(landmark->GetId());
275  }
276  }
277  }
278  }
279  return Result;
280  }

282  std::vector<SharedPtr<Actor>> World::GetTrafficLightsInJunction(
283  const road::JuncId junc_id) const {
284  std::vector<SharedPtr<Actor>> Result;
285  SharedPtr<Map> map = GetMap();
286  const road::Junction* junction = map->GetMap().GetJunction(junc_id);
287  for (const road::ContId& cont_id : junction->GetControllers()) {
288  const std::unique_ptr<road::Controller>& controller =
289  map->GetMap().GetControllers().at(cont_id);
290  for (road::SignId sign_id : controller->GetSignals()) {
291  SharedPtr<Actor> traffic_light = GetTrafficLightFromOpenDRIVE(sign_id);
292  if (traffic_light) {
293  Result.emplace_back(traffic_light);
294  }
295  }
296  }
297  return Result;
298  }
