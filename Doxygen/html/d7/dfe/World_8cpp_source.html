117  SharedPtr<Actor> World::GetTrafficSign(const Landmark& landmark) const {
118  SharedPtr<ActorList> actors = GetActors();
119  SharedPtr<TrafficSign> result;
120  std::string landmark_id = landmark.GetId();
121  for (size_t i = 0; i < actors->size(); i++) {
122  SharedPtr<Actor> actor = actors->at(i);
123  if (StringUtil::Match(actor->GetTypeId(), "*traffic.*")) {
124  TrafficSign* sign = static_cast<TrafficSign*>(actor.get());
125  if(sign && (sign->GetSignId() == landmark_id)) {
126  return actor;
127  }
128  }
129  }
130  return nullptr;
131  }
