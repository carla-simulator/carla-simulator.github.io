95 carla::geom::Vector3D AInertialMeasurementUnit::ComputeAccelerometer(
96  const float DeltaTime)
97 {
98  // Used to convert from UE4's cm to meters
99  constexpr float TO_METERS = 1e-2;
100  // Earth's gravitational acceleration is approximately 9.81 m/s^2
101  constexpr float GRAVITY = 9.81f;
102 
103  // 2nd derivative of the polynomic (quadratic) interpolation
104  // using the point in current time and two previous steps:
105  // d2[i] = -2.0*(y1/(h1*h2)-y2/((h2+h1)*h2)-y0/(h1*(h2+h1)))
106  const FVector CurrentLocation = GetActorLocation();
107 
108  const FVector Y2 = PrevLocation[0];
109  const FVector Y1 = PrevLocation[1];
110  const FVector Y0 = CurrentLocation;
111  const float H1 = DeltaTime;
112  const float H2 = PrevDeltaTime;
113 
114  const float H1AndH2 = H2 + H1;
115  const FVector A = Y1 / ( H1 * H2 );
116  const FVector B = Y2 / ( H2 * (H1AndH2) );
117  const FVector C = Y0 / ( H1 * (H1AndH2) );
118  FVector FVectorAccelerometer = TO_METERS * -2.0f * ( A - B - C );
119 
120  // Update the previous locations
121  PrevLocation[0] = PrevLocation[1];
122  PrevLocation[1] = CurrentLocation;
123  PrevDeltaTime = DeltaTime;
124 
125  FQuat ImuRotation =
126  GetRootComponent()->GetComponentTransform().GetRotation();
127  FVectorAccelerometer = ImuRotation.UnrotateVector(FVectorAccelerometer);
128 
129  // Add gravitational acceleration
130  FVectorAccelerometer.Z += GRAVITY;
131 
132  // Cast from FVector to our Vector3D to correctly send the data in m/s^2
133  // and apply the desired noise function, in this case a normal distribution
134  const carla::geom::Vector3D Accelerometer =
135  ComputeAccelerometerNoise(FVectorAccelerometer);
136 
137  return Accelerometer;
138 }

141 {

159 float AInertialMeasurementUnit::ComputeCompass()
160 {
161  // Magnetometer: orientation with respect to the North in rad
162  const FVector ForwVect = GetActorForwardVector().GetSafeNormal2D();
163  float Compass = std::acos(FVector::DotProduct(CarlaNorthVector, ForwVect));
164 
165  // Keep the angle between [0, 2pi)
166  if (FVector::CrossProduct(CarlaNorthVector, ForwVect).Z < 0.0f)
167  {
168  Compass = carla::geom::Math::Pi2<float>() - Compass;
169  }
170 
171  return Compass;
172 }
