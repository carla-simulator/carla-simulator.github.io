13 static FString GetVersionFromFString(const FString& String)
14 {
15  TRACE_CPUPROFILER_EVENT_SCOPE(GetVersionFromFString);
16  auto IsDigit = [](TCHAR charToCheck) {
17  if (charToCheck == TCHAR('0')) return true;
18  if (charToCheck == TCHAR('1')) return true;
19  if (charToCheck == TCHAR('2')) return true;
20  if (charToCheck == TCHAR('3')) return true;
21  if (charToCheck == TCHAR('4')) return true;
22  if (charToCheck == TCHAR('5')) return true;
23  if (charToCheck == TCHAR('6')) return true;
24  if (charToCheck == TCHAR('7')) return true;
25  if (charToCheck == TCHAR('8')) return true;
26  if (charToCheck == TCHAR('9')) return true;
27  return false;
28  };
29  int index = String.Find(TEXT("_v"));
30  if (index != -1)
31  {
32  index += 2;
33  FString Version = "_v";
34  while(IsDigit(String[index]))
35  {
36  Version += String[index];
37  ++index;
38  if (index == String.Len())
39  return Version;
40  }
41  return Version;
42  }
43  return FString();
44 }

140 bool FFoliageBlueprint::SetBPClassName(const FString& Path)
141 {
142  TRACE_CPUPROFILER_EVENT_SCOPE(FoliageBlueprintCache::SetBPClassName);
143  if (Path.IsEmpty())
144  return false;
145  TArray< FString > ParsedString;
146  Path.ParseIntoArray(ParsedString, TEXT("/"), false);
147  int Position = ParsedString.Num() - 1;
148  const FString FullVersion = GetVersionFromFString(ParsedString[Position]);
149  const FString Folder = ParsedString[--Position];
150  const FString BPClassName = "BP_" + Folder + FullVersion;
151  BPFullClassName = "Blueprint'";
152  for (int i = 0; i <= Position; ++i)
153  {
154  BPFullClassName += ParsedString[i];
155  BPFullClassName += '/';
156  }
157  BPFullClassName += BPClassName;
158  BPFullClassName += ".";
159  BPFullClassName += BPClassName;
160  BPFullClassName += "_C'";
