67  std::vector<SharedPtr<Waypoint>> Waypoint::GetNextUntilLaneEnd(double distance) const {
68  std::vector<SharedPtr<Waypoint>> result;
69  std::vector<SharedPtr<Waypoint>> next = GetNext(distance);
70 
71  while (next.size() == 1 && next.front()->GetRoadId() == GetRoadId()) {
72  result.emplace_back(next.front());
73  next = result.back()->GetNext(distance);
74  }
75  double current_s = GetDistance();
76  if(result.size()) {
77  current_s = result.back()->GetDistance();
78  }
79  double remaining_length;
80  double road_length = _parent->GetMap().GetLane(_waypoint).GetRoad()->GetLength();
81  if(_waypoint.lane_id < 0) {
82  remaining_length = road_length - current_s;
83  } else {
84  remaining_length = current_s;
85  }
86  remaining_length -= std::numeric_limits<double>::epsilon();
87  result.emplace_back(result.back()->GetNext(remaining_length).front());
88

165  road::element::LaneMarking::LaneChange Waypoint::GetLaneChange() const {
166  using lane_change_type = road::element::LaneMarking::LaneChange;
167 
168  const auto lane_change_right_info = _mark_record.first;
169  lane_change_type c_right;
170  if (lane_change_right_info != nullptr) {
171  const auto lane_change_right = lane_change_right_info->GetLaneChange();
172  c_right = static_cast<lane_change_type>(lane_change_right);
173  } else {
174  c_right = lane_change_type::Both;
175  }
176 
177  const auto lane_change_left_info = _mark_record.second;
178  lane_change_type c_left;
179  if (lane_change_left_info != nullptr) {
180  const auto lane_change_left = lane_change_left_info->GetLaneChange();
181  c_left = static_cast<lane_change_type>(lane_change_left);
182  } else {
183  c_left = lane_change_type::Both;
184  }
185 
186  if (_waypoint.lane_id > 0) {
187  // if road goes backward
188  if (c_right == lane_change_type::Right) {
189  c_right = lane_change_type::Left;
190  } else if (c_right == lane_change_type::Left) {
191  c_right = lane_change_type::Right;
192  }
193  }
194 
195  if (((_waypoint.lane_id > 0) ? _waypoint.lane_id - 1 : _waypoint.lane_id + 1) > 0) {
196  // if road goes backward
197  if (c_left == lane_change_type::Right) {
198  c_left = lane_change_type::Left;
199  } else if (c_left == lane_change_type::Left) {
200  c_left = lane_change_type::Right;
201  }
202  }
203 
204  return (c_right & lane_change_type::Right) | (c_left & lane_change_type::Left);
205  }
