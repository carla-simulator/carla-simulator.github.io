45  DirectedPoint GeometryArc::PosFromDist(double dist) const {
46  dist = geom::Math::Clamp(dist, 0.0, _length);
47  DEBUG_ASSERT(_length > 0.0);
48  DEBUG_ASSERT(std::fabs(_curvature) > 1e-15);
49  const double radius = 1.0 / _curvature;
50  constexpr double pi_half = geom::Math::Pi<double>() / 2.0;
51  DirectedPoint p(_start_position, _heading);
52  p.location.x += static_cast<float>(radius * std::cos(p.tangent + pi_half));
53  p.location.y += static_cast<float>(radius * std::sin(p.tangent + pi_half));
54  p.tangent += dist * _curvature;
55  p.location.x -= static_cast<float>(radius * std::cos(p.tangent + pi_half));
56  p.location.y -= static_cast<float>(radius * std::sin(p.tangent + pi_half));
57  return p;

134  void GeometryPoly3::PreComputeSpline() {
135  const double delta_u = 0.01; // interval between values of u
136  double current_s = 0;
137  double current_u = 0;
138  double last_u = 0;
139  double last_v = _poly.Evaluate(current_u);
140  double last_s = 0;
141  RtreeValue last_val{last_u, last_v, last_s, _poly.Tangent(current_u)};
142  while (current_s < _length + delta_u) {
143  current_u += delta_u;
144  double current_v = _poly.Evaluate(current_u);
145  double du = current_u - last_u;
146  double dv = current_v - last_v;
147  double ds = sqrt(du * du + dv * dv);
148  current_s += ds;
149  double current_t = _poly.Tangent(current_u);
150  RtreeValue current_val{current_u, current_v, current_s, current_t};
151 
152  Rtree::BPoint p1(static_cast<float>(last_s));
153  Rtree::BPoint p2(static_cast<float>(current_s));
154  _rtree.InsertElement(Rtree::BSegment(p1, p2), last_val, current_val);
155 
156  last_u = current_u;
157  last_v = current_v;
158  last_s = current_s;
159  last_val = current_val;
160 
161  }
162  }
