55  DirectedPoint GeometrySpiral::PosFromDist(double dist) const {
56  // not working yet with negative values
57  dist = geom::Math::Clamp(dist, 0.0, _length);
58  DEBUG_ASSERT(_length > 0.0);
59  DEBUG_ASSERT(std::fabs(_curve_end) > 1e-15);
60  const double radius = 1.0 / _curve_end;
61  const double extra_norm = 1.0 / std::sqrt(geom::Math::Pi<double>() / 2.0);
62  const double norm = 1.0 / std::sqrt(2.0 * radius * _length);
63  const double length = dist * norm;
64  double S, C;
65  fresnl(length * extra_norm, &S, &C);
66  S /= (norm * extra_norm);
67  C /= (norm * extra_norm);
68  DirectedPoint p(_start_position, _heading);
69  const double cos_a = std::cos(p.tangent);
70  const double sin_a = std::sin(p.tangent);
71  p.location.x += static_cast<float>(C * cos_a - S * sin_a);
72  p.location.y += static_cast<float>(S * cos_a + C * sin_a);
73  p.tangent += length * length;
74 
75  return p;
76  }
