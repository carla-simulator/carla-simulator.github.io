36 void AGnssSensor::PostPhysTick(UWorld *World, ELevelTick TickType, float DeltaSeconds)
37 {
38  TRACE_CPUPROFILER_EVENT_SCOPE(AGnssSensor::PostPhysTick);
39 
40  FVector ActorLocation = GetActorLocation();
41  ALargeMapManager * LargeMap = UCarlaStatics::GetLargeMapManager(GetWorld());
42  if (LargeMap)
43  {
44  ActorLocation = LargeMap->LocalToGlobalLocation(ActorLocation);
45  }
46  carla::geom::Location Location = ActorLocation;
47  carla::geom::GeoLocation CurrentLocation = CurrentGeoReference.Transform(Location);
48 
49  // Compute the noise for the sensor
50  const float LatError = RandomEngine->GetNormalDistribution(0.0f, LatitudeDeviation);
51  const float LonError = RandomEngine->GetNormalDistribution(0.0f, LongitudeDeviation);
52  const float AltError = RandomEngine->GetNormalDistribution(0.0f, AltitudeDeviation);
53 
54  // Apply the noise to the sensor
55  double Latitude = CurrentLocation.latitude + LatitudeBias + LatError;
56  double Longitude = CurrentLocation.longitude + LongitudeBias + LonError;
57  double Altitude = CurrentLocation.altitude + AltitudeBias + AltError;
58 
59  {
60  TRACE_CPUPROFILER_EVENT_SCOPE_STR("AGnssSensor Stream Send");
61  if (IsStreamReady())
62  {
63  auto Stream = GetDataStream(*this);
64  Stream.Send(*this, carla::geom::GeoLocation{Latitude, Longitude, Altitude});
65  }
66  }
67 }
