141  VehiclePhysicsControl(const FVehiclePhysicsControl &Control) {
142  // Engine Setup
143  torque_curve = std::vector<carla::geom::Vector2D>();
144  TArray<FRichCurveKey> TorqueCurveKeys = Control.TorqueCurve.GetCopyOfKeys();
145  for (int32 KeyIdx = 0; KeyIdx < TorqueCurveKeys.Num(); KeyIdx++) {
146  geom::Vector2D point(TorqueCurveKeys[KeyIdx].Time, TorqueCurveKeys[KeyIdx].Value);
147  torque_curve.push_back(point);
148  }
149  max_rpm = Control.MaxRPM;
150  moi = Control.MOI;
151  damping_rate_full_throttle = Control.DampingRateFullThrottle;
152  damping_rate_zero_throttle_clutch_engaged = Control.DampingRateZeroThrottleClutchEngaged;
153  damping_rate_zero_throttle_clutch_disengaged = Control.DampingRateZeroThrottleClutchDisengaged;
154 
155  // Transmission Setup
156  use_gear_autobox = Control.bUseGearAutoBox;
157  gear_switch_time = Control.GearSwitchTime;
158  clutch_strength = Control.ClutchStrength;
159  final_ratio = Control.FinalRatio;
160  forward_gears = std::vector<GearPhysicsControl>();
161  for (const auto &Gear : Control.ForwardGears) {
162  forward_gears.push_back(GearPhysicsControl(Gear));
163  }
164 
165  // Vehicle Setup
166  mass = Control.Mass;
167  drag_coefficient = Control.DragCoefficient;
168 
169  steering_curve = std::vector<carla::geom::Vector2D>();
170  TArray<FRichCurveKey> SteeringCurveKeys = Control.SteeringCurve.GetCopyOfKeys();
171  for (int32 KeyIdx = 0; KeyIdx < SteeringCurveKeys.Num(); KeyIdx++) {
172  geom::Vector2D point(SteeringCurveKeys[KeyIdx].Time, SteeringCurveKeys[KeyIdx].Value);
173  steering_curve.push_back(point);
174  }
175 
176  center_of_mass = Control.CenterOfMass;
177 
178  // Wheels Setup
179  wheels = std::vector<WheelPhysicsControl>();
180  for (const auto &Wheel : Control.Wheels) {
181  wheels.push_back(WheelPhysicsControl(Wheel));
182  }
183  }
184 
185  operator FVehiclePhysicsControl() const {
186  FVehiclePhysicsControl Control;
187 
188  // Engine Setup
189  FRichCurve TorqueCurve;
190  for (const auto &point : torque_curve) {
191  TorqueCurve.AddKey(point.x, point.y);
192  }
193  Control.TorqueCurve = TorqueCurve;
194  Control.MaxRPM = max_rpm;
195  Control.MOI = moi;
196  Control.DampingRateFullThrottle = damping_rate_full_throttle;
197  Control.DampingRateZeroThrottleClutchEngaged = damping_rate_zero_throttle_clutch_engaged;
198  Control.DampingRateZeroThrottleClutchDisengaged = damping_rate_zero_throttle_clutch_disengaged;
199 
200  // Transmission Setup
201  Control.bUseGearAutoBox = use_gear_autobox;
202  Control.GearSwitchTime = gear_switch_time;
203  Control.ClutchStrength = clutch_strength;
204  Control.FinalRatio = final_ratio;
205  TArray<FGearPhysicsControl> ForwardGears;
206  for (const auto &gear : forward_gears) {
207  ForwardGears.Add(FGearPhysicsControl(gear));
208  }
209  Control.ForwardGears = ForwardGears;
210 
211 
212  // Vehicle Setup
213  Control.Mass = mass;
214  Control.DragCoefficient = drag_coefficient;
215 
216  // Transmission Setup
217  FRichCurve SteeringCurve;
218  for (const auto &point : steering_curve) {
219  SteeringCurve.AddKey(point.x, point.y);
220  }
221  Control.SteeringCurve = SteeringCurve;
222 
223  Control.CenterOfMass = center_of_mass;
224 
225  // Wheels Setup
226  TArray<FWheelPhysicsControl> Wheels;
227  for (const auto &wheel : wheels) {
228  Wheels.Add(FWheelPhysicsControl(wheel));
229  }
230  Control.Wheels = Wheels;
231 
232  return Control;
233  }
