32  /// Here RPCPort is the traffic manager local instance RPC server port where
33  /// it can listen to remote traffic managers and apply the changes to
34  /// local instance through a TrafficManagerBase pointer.
35  TrafficManagerServer(
36  uint16_t &RPCPort,
37  carla::traffic_manager::TrafficManagerBase* tm)
38  : _RPCPort(RPCPort) {
39 
40  uint16_t counter = 0;
41  while(counter < MIN_TRY_COUNT) {
42  try {
43 
44  /// Create server instance.
45  server = new ::rpc::server(RPCPort);
46 
47  } catch(std::exception& e) {
48  /// Update port number and try again.
49  std::this_thread::sleep_for(500ms);
50  }
51 
52  /// If server created.
53  if(server != nullptr) {
54  break;
55  }
56  counter ++;
57  }
58 
59  /// If server still not created throw a runtime exception.
60  if(server == nullptr) {
61 
62  carla::throw_exception(std::runtime_error(
63  "trying to create rpc server for traffic manager; "
64  "but the system failed to create because of bind error."));
65  } else {
66  /// If the server creation was successful we are
67  /// binding a lambda function to the name "register_vehicle".
68  server->bind("register_vehicle", [=](std :: vector <carla::rpc::Actor> _actor_list) {
69  std::vector<ActorPtr> actor_list;
70  for (auto &&actor : _actor_list) {
71  actor_list.emplace_back(carla::client::detail::ActorVariant(actor).Get(tm->GetEpisodeProxy()));
72  }
73  tm->RegisterVehicles(actor_list);
74  });
75 
76 
77  /// Binding a lambda function to the name "unregister_vehicle".
78  server->bind("unregister_vehicle", [=](std :: vector <carla::rpc::Actor> _actor_list) {
79  std::vector<ActorPtr> actor_list;

84  });
