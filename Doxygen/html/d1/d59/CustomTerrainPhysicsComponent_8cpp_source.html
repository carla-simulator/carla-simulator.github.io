128  {
129  FDVector ParticlePosition = TileOrigin + FDVector(i*ParticleSize,j*ParticleSize, 0.f);
130  float Height = 0;
131  for(uint32_t k = 0; k < NumParticles_Z; k++)
132  {
133  ParticlePosition.Z = Height - k*ParticleSize;
134  Particles[k*NumParticles_X*NumParticles_Y + j*NumParticles_X + i] =
135  {ParticlePosition, ParticleSize/2.f};

137  }

299 void FSparseHighDetailMap::LoadTilesAtPosition(FDVector Position, float RadiusX , float RadiusY )
300 {
301  // Translate UE4 Position to our coords
302 
303  TRACE_CPUPROFILER_EVENT_SCOPE(FSparseHighDetailMap::LoadTilesAtPosition);
304  std::vector<uint64_t> KeysToRemove;
305  std::vector<uint64_t> KeysToLoad;
306  float MinX = std::max(Position.X - RadiusX, 0.0);
307  float MinY = std::max(Position.Y - RadiusX, 0.0);
308  float MaxX = std::min(Position.X + RadiusX, Extension.X - 1.0 );
309  float MaxY = std::min(Position.Y - RadiusY, Extension.Y - 1.0 );
310 
311  {
312  TRACE_CPUPROFILER_EVENT_SCOPE(FSparseHighDetailMap::LoadTilesAtPosition::Process);
313 
314  for(auto it : Map)
315  {
316  KeysToRemove.push_back(it.first);
317  }
318  {
319  FScopeLock ScopeLock(&Lock_Map);
320  for(float X = MinX; X < MaxX; X+=TileSize )
321  {
322  for(float Y = MinY; Y < MaxY; Y+=TileSize )
323  {
324  uint64_t CurrentTileID = GetTileId(FDVector(X,Y,0));
325  if( Map.find(CurrentTileID) != Map.end() )
326  {
327  KeysToRemove.erase(std::remove_if( KeysToRemove.begin(), KeysToRemove.end(), [CurrentTileID](const uint64_t& x) {
328  return x == CurrentTileID;
329  }), KeysToRemove.end());
330  }
331  else
332  {
333  KeysToLoad.push_back(CurrentTileID);
334  Map.emplace(CurrentTileID, FDenseTile() );
335  }
336  }
337  }
338  }
339  }
340 
341  {
342  TRACE_CPUPROFILER_EVENT_SCOPE(FSparseHighDetailMap::LoadTilesAtPosition::Load);
343  for(auto it : KeysToLoad)
344  {
345  InitializeRegion(it);
346  }
347  }
348 
349  {
350  TRACE_CPUPROFILER_EVENT_SCOPE(FSparseHighDetailMap::LoadTilesAtPosition::Save);
351  for(auto it : KeysToRemove)
352  {
353  std::string FileToSavePath = std::string(TCHAR_TO_UTF8(*( SavePath + Map[it].TilePosition.ToString() + ".tile")));
354  std::ofstream OutputStream(FileToSavePath.c_str());
355  WriteFVector(OutputStream, Map[it].TilePosition.ToFVector());
356  WriteStdVector<FParticle> (OutputStream, Map[it].Particles);
357  OutputStream.close();
358  }
359  }
360  {
361  FScopeLock ScopeLock(&Lock_Map);
362  for(auto it : KeysToRemove)
363  {
364  Map.erase(it);
365  }
366  }
367 }
