18  float DeviationCrossProduct(Actor actor, const cg::Location &target_location) {
19 
20  cg::Vector3D heading_vector = actor->GetTransform().GetForwardVector();
21  heading_vector.z = 0.0f;
22  heading_vector = heading_vector.MakeUnitVector();
23  cg::Location next_vector = target_location - actor->GetLocation();
24  next_vector.z = 0.0f;
25  if (next_vector.Length() > 2.0f * std::numeric_limits<float>::epsilon()) {
26  next_vector = next_vector.MakeUnitVector();
27  const float cross_z = heading_vector.x * next_vector.y - heading_vector.y * next_vector.x;
28  return cross_z;
29  } else {
30  return 0.0f;
31  }
32  }

34  float DeviationDotProduct(Actor actor, const cg::Location &target_location, bool rear_offset) {
35 
36  cg::Vector3D heading_vector = actor->GetTransform().GetForwardVector();
37  heading_vector.z = 0.0f;
38  heading_vector = heading_vector.MakeUnitVector();
39  cg::Location next_vector;
40 
41  if (!rear_offset) {
42  next_vector = target_location - actor->GetLocation();
43  } else {
44  const auto vehicle_ptr = boost::static_pointer_cast<cc::Vehicle>(actor);
45  const float vehicle_half_length = vehicle_ptr->GetBoundingBox().extent.x;
46  next_vector = target_location - (cg::Location(-1* vehicle_half_length * heading_vector)
47  + vehicle_ptr->GetLocation());
48  }
49 
50  next_vector.z = 0.0f;
51  if (next_vector.Length() > 2.0f * std::numeric_limits<float>::epsilon()) {
52  next_vector = next_vector.MakeUnitVector();
53  const float dot_product = cg::Math::Dot(next_vector, heading_vector);
54  return dot_product;
55  } else {
56  return 0.0f;
57  }
58  }
