19  float DeviationCrossProduct(Actor actor, const cg::Location &vehicle_location, const cg::Location &target_location) {
20 
21  cg::Vector3D heading_vector = actor->GetTransform().GetForwardVector();
22  heading_vector.z = 0.0f;
23  heading_vector = heading_vector.MakeUnitVector();
24  cg::Location next_vector = target_location - vehicle_location;
25  next_vector.z = 0.0f;
26 
27  if (next_vector.Length() > 2.0f * std::numeric_limits<float>::epsilon()) {
28  next_vector = next_vector.MakeUnitVector();
29  const float cross_z = heading_vector.x * next_vector.y - heading_vector.y * next_vector.x;
30  return cross_z;
31  } else {
32  return 0.0f;
33  }
34  }

36  float DeviationDotProduct(Actor actor, const cg::Location &vehicle_location, const cg::Location &target_location, bool rear_offset) {
37 
38  cg::Vector3D heading_vector = actor->GetTransform().GetForwardVector();
39  heading_vector.z = 0.0f;
40  heading_vector = heading_vector.MakeUnitVector();
41  cg::Location next_vector;
42 
43  if (!rear_offset) {
44  next_vector = target_location - vehicle_location;
45  } else {
46  const auto vehicle_ptr = boost::static_pointer_cast<cc::Vehicle>(actor);
47  const float vehicle_half_length = vehicle_ptr->GetBoundingBox().extent.x;
48  next_vector = target_location - (cg::Location(-1* vehicle_half_length * heading_vector)
49  + vehicle_location);
50  }
51 
52  next_vector.z = 0.0f;
53 
54  if (next_vector.Length() > 2.0f * std::numeric_limits<float>::epsilon()) {
55  next_vector = next_vector.MakeUnitVector();
56  const float dot_product = cg::Math::Dot(next_vector, heading_vector);
57  return dot_product;
58  } else {
59  return 0.0f;
60  }

237  std::pair<SimpleWaypointPtr,uint64_t> TrackTraffic::GetTargetWaypoint(const Buffer& waypoint_buffer,const float& target_point_distance) {
238 
239  SimpleWaypointPtr target_waypoint = waypoint_buffer.front();
240  const SimpleWaypointPtr& buffer_front = waypoint_buffer.front();
241  uint64_t startPosn = static_cast<uint64_t>(std::fabs(target_point_distance/SAMPLING_RESOLUTION));
242  uint64_t index = 0;
243  /// Condition to determine forward or backward scanning of WayPoint Buffer.
244 
245  if (startPosn < waypoint_buffer.size()) {
246  bool mScanForward = false;
247  double target_point_dist_power = std::pow(target_point_distance, 2);
248  if(buffer_front->DistanceSquared(target_waypoint) < target_point_dist_power) {
249  mScanForward = true;
250  }
251 
252  if(mScanForward) {
253  for (index = startPosn ;
254  (index < waypoint_buffer.size()) &&
255  (buffer_front->DistanceSquared(target_waypoint)
256  < target_point_dist_power);
257  ++index) {
258  target_waypoint = waypoint_buffer.at(index);
259  }
260  }
261  else {
262  for (index = startPosn ;
263  (buffer_front->DistanceSquared(target_waypoint)
264  > target_point_dist_power);
265  --index) {
266  target_waypoint = waypoint_buffer.at(index);
267  }
268  }
269  }
270  else {
271  target_waypoint = waypoint_buffer.back();
272  }
273  return std::make_pair(target_waypoint,index);
274  }
