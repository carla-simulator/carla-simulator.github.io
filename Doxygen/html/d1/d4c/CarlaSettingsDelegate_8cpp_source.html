68 void UCarlaSettingsDelegate::ApplyQualityLevelPostRestart()
69 {
70  CheckCarlaSettings(nullptr);
71  UWorld *InWorld = CarlaSettings->GetWorld();
72 
73  const EQualityLevel QualityLevel = CarlaSettings->GetQualityLevel();
74  if (AppliedLowPostResetQualityLevel == QualityLevel)
75  {
76  return;
77  }
78 
79  switch (QualityLevel)
80  {
81  case EQualityLevel::Low:
82  {

84  LaunchLowQualityCommands(InWorld);

86  SetAllLights(InWorld, CarlaSettings->LowLightFadeDistance, false, true);

92  // Disable all post process volumes
93  SetPostProcessEffectsEnabled(InWorld, false);
94  break;
95  }
96  default:
97  UE_LOG(LogCarla, Warning, TEXT("Unknown quality level, falling back to default."));
98  case EQualityLevel::Epic:
99  {
100  LaunchEpicQualityCommands(InWorld);
101  SetAllLights(InWorld, 0.0f, true, false);
102  SetAllRoads(InWorld, 0, CarlaSettings->EpicRoadMaterials);
103  SetAllActorsDrawDistance(InWorld, 0);
104  SetPostProcessEffectsEnabled(InWorld, true);
105  break;
106  }
107  }
108  AppliedLowPostResetQualityLevel = QualityLevel;
109 }
