85  void Client::Connect() {
86  auto self = shared_from_this();
87  _strand.post([this, self]() {
88  if (_done) {
89  return;
90  }
91 
92  using boost::system::error_code;
93 
94  if (_socket.is_open()) {
95  _socket.close();
96  }
97 
98  DEBUG_ASSERT(_token.is_valid());
99  DEBUG_ASSERT(_token.protocol_is_tcp());
100  const auto ep = _token.to_tcp_endpoint();
101 
102  auto handle_connect = [this, self, ep](error_code ec) {
103  if (!ec) {
104  if (_done) {
105  return;
106  }
107  log_debug("streaming client: connected to", ep);
108  // Send the stream id to subscribe to the stream.
109  const auto &stream_id = _token.get_stream_id();
110  log_debug("streaming client: sending stream id", stream_id);
111  boost::asio::async_write(
112  _socket,
113  boost::asio::buffer(&stream_id, sizeof(stream_id)),
114  _strand.wrap([=](error_code ec, size_t DEBUG_ONLY(bytes)) {
115  if (!ec) {
116  DEBUG_ASSERT_EQ(bytes, sizeof(stream_id));
117  // If succeeded start reading data.
118  ReadData();
119  } else {
120  // Else try again.
121  log_info("streaming client: failed to send stream id:", ec.message());
122  Connect();
123  }
124  }));
125  } else {
126  log_info("streaming client: connection failed:", ec.message());
127  Reconnect();
128  }
129  };
130 
131  log_debug("streaming client: connecting to", ep);
132  _socket.async_connect(ep, _strand.wrap(handle_connect));
133  });
134  }

157  void Client::ReadData() {
158  auto self = shared_from_this();
159  _strand.post([this, self]() {
160  if (_done) {
161  return;
162  }
163 
164  log_debug("streaming client: Client::ReadData");
165 
166  auto message = std::make_shared<IncomingMessage>(_buffer_pool->Pop());
167 
168  auto handle_read_data = [this, self, message](boost::system::error_code ec, size_t DEBUG_ONLY(bytes)) {
169  DEBUG_ONLY(log_debug("streaming client: Client::ReadData.handle_read_data", bytes, "bytes"));
170  if (!ec) {
171  DEBUG_ASSERT_EQ(bytes, message->size());
172  DEBUG_ASSERT_NE(bytes, 0u);
173  // Move the buffer to the callback function and start reading the next
174  // piece of data.
175  log_debug("streaming client: success reading data, calling the callback");
176  _strand.context().post([self, message]() { self->_callback(message->pop()); });
177  ReadData();
178  } else {
179  // As usual, if anything fails start over from the very top.
180  log_info("streaming client: failed to read data:", ec.message());
181  Connect();
182  }
183  };
184 
185  auto handle_read_header = [this, self, message, handle_read_data](
186  boost::system::error_code ec,
187  size_t DEBUG_ONLY(bytes)) {
188  DEBUG_ONLY(log_debug("streaming client: Client::ReadData.handle_read_header", bytes, "bytes"));
189  if (!ec && (message->size() > 0u)) {
190  DEBUG_ASSERT_EQ(bytes, sizeof(message_size_type));
191  if (_done) {
192  return;
193  }
194  // Now that we know the size of the coming buffer, we can allocate our
195  // buffer and start putting data into it.
196  boost::asio::async_read(
197  _socket,
198  message->buffer(),
199  _strand.wrap(handle_read_data));
200  } else {
201  log_info("streaming client: failed to read header:", ec.message());
202  DEBUG_ONLY(log_debug("size = ", message->size()));
203  DEBUG_ONLY(log_debug("bytes = ", bytes));
204  Connect();
205  }
206  };
207 
208  // Read the size of the buffer that is coming.
209  boost::asio::async_read(
210  _socket,
211  message->size_as_buffer(),
212  _strand.wrap(handle_read_header));
213  });
214  }
