164 void ACarlaGameModeBase::BeginPlay()
165 {
166  Super::BeginPlay();
167 
168  UWorld* World = GetWorld();
169  check(World != nullptr);
170 
171  LoadMapLayer(GameInstance->GetCurrentMapLayer());
172  ReadyToRegisterObjects = true;
173 
174  if (true) { /// @todo If semantic segmentation enabled.
175  ATagger::TagActorsInLevel(*World, true);
176  TaggerDelegate->SetSemanticSegmentationEnabled();
177  }
178 
179  // HACK: fix transparency see-through issues
180  // The problem: transparent objects are visible through walls.
181  // This is due to a weird interaction between the SkyAtmosphere component,
182  // the shadows of a directional light (the sun)
183  // and the custom depth set to 3 used for semantic segmentation
184  // The solution: Spawn a Decal.
185  // It just works!
186  World->SpawnActor<ADecalActor>(
187  FVector(0,0,-1000000), FRotator(0,0,0), FActorSpawnParameters());
188 
189  ATrafficLightManager* Manager = GetTrafficLightManager();
190  Manager->InitializeTrafficLights();
191 
192  Episode->InitializeAtBeginPlay();
193  GameInstance->NotifyBeginEpisode(*Episode);
194 
195  if (Episode->Weather != nullptr)
196  {
197  Episode->Weather->ApplyWeather(carla::rpc::WeatherParameters::Default);
198  }
199 
200  /// @todo Recorder should not tick here, FCarlaEngine should do it.
201  // check if replayer is waiting to autostart
202  if (Recorder)
203  {
204  Recorder->GetReplayer()->CheckPlayAfterMapLoaded();
205  }
206 
207  if(ReadyToRegisterObjects && PendingLevelsToLoad == 0)
208  {
209  RegisterEnvironmentObjects();
210  }
211 
212  if (LMManager) {
213  LMManager->RegisterInitialObjects();
214  }
215 
216  // Manually run begin play on lights as it may not run on sublevels
217  TArray<AActor*> FoundActors;
218  UGameplayStatics::GetAllActorsOfClass(World, AActor::StaticClass(), FoundActors);
219  for(AActor* Actor : FoundActors)
220  {
221  TArray<UCarlaLight*> Lights;
222  Actor->GetComponents(Lights, false);
223  for(UCarlaLight* Light : Lights)
224  {
225  Light->RegisterLight();
226  }
227  }
228  EnableOverlapEvents();
229 }

375 void ACarlaGameModeBase::DebugShowSignals(bool enable)
376 {
377 
378  auto World = GetWorld();
379  check(World != nullptr);
380 
381  if(!Map)
382  {
383  return;
384  }
385 
386  if(!enable)
387  {
388  UKismetSystemLibrary::FlushDebugStrings(World);
389  UKismetSystemLibrary::FlushPersistentDebugLines(World);
390  return;
391  }
392 
393  //const std::unordered_map<carla::road::SignId, std::unique_ptr<carla::road::Signal>>
394  const auto& Signals = Map->GetSignals();
395  const auto& Controllers = Map->GetControllers();
396 
397  for(const auto& Signal : Signals) {
398  const auto& ODSignal = Signal.second;
399  const FTransform Transform = ODSignal->GetTransform();
400  const FVector Location = Transform.GetLocation();
401  const FQuat Rotation = Transform.GetRotation();
402  const FVector Up = Rotation.GetUpVector();
403  DrawDebugSphere(
404  World,
405  Location,
406  50.0f,
407  10,
408  FColor(0, 255, 0),
409  true
410  );
411  }
412 
413  TArray<const cre::RoadInfoSignal*> References;
414  auto waypoints = Map->GenerateWaypointsOnRoadEntries();
415  std::unordered_set<cr::RoadId> ExploredRoads;
416  for (auto & waypoint : waypoints)
417  {
418  // Check if we already explored this road
419  if (ExploredRoads.count(waypoint.road_id) > 0)
420  {
421  continue;
422  }
423  ExploredRoads.insert(waypoint.road_id);
424 
425  // Multiple times for same road (performance impact, not in behavior)
426  auto SignalReferences = Map->GetLane(waypoint).
427  GetRoad()->GetInfos<cre::RoadInfoSignal>();
428  for (auto *SignalReference : SignalReferences)
429  {
430  References.Add(SignalReference);
431  }
432  }
433  for (auto& Reference : References)
434  {
435  auto RoadId = Reference->GetRoadId();
436  const auto* SignalReference = Reference;
437  const FTransform SignalTransform = SignalReference->GetSignal()->GetTransform();
438  for(auto &validity : SignalReference->GetValidities())
439  {
440  for(auto lane : carla::geom::Math::GenerateRange(validity._from_lane, validity._to_lane))
441  {
442  if(lane == 0)
443  continue;
444 
445  auto signal_waypoint = Map->GetWaypoint(
446  RoadId, lane, SignalReference->GetS()).get();
447 
448  if(Map->GetLane(signal_waypoint).GetType() != cr::Lane::LaneType::Driving)
449  continue;
450 
451  FTransform ReferenceTransform = Map->ComputeTransform(signal_waypoint);
452 
453  DrawDebugSphere(
454  World,
455  ReferenceTransform.GetLocation(),
456  50.0f,
457  10,
458  FColor(0, 0, 255),
459  true
460  );
461 
462  DrawDebugLine(
463  World,
464  ReferenceTransform.GetLocation(),
465  SignalTransform.GetLocation(),
466  FColor(0, 0, 255),
467  true
468  );
469  }
470  }
471  }

554 void ACarlaGameModeBase::ConvertMapLayerMaskToMapNames(int32 MapLayer, TArray<FName>& OutLevelNames)
555 {
556  UWorld* World = GetWorld();
557  const TArray <ULevelStreaming*> Levels = World->GetStreamingLevels();
558  TArray<FString> LayersToLoad;
559 
560  // Get all the requested layers
561  int32 LayerMask = 1;
562  int32 AllLayersMask = static_cast<crp::MapLayerType>(crp::MapLayer::All);
563 
564  while(LayerMask > 0)
565  {
566  // Convert enum to FString
567  FString LayerName = UTF8_TO_TCHAR(MapLayerToString(static_cast<crp::MapLayer>(LayerMask)).c_str());
568  bool included = static_cast<crp::MapLayerType>(MapLayer) & LayerMask;
569  if(included)
570  {
571  LayersToLoad.Emplace(LayerName);
572  }
573  LayerMask = (LayerMask << 1) & AllLayersMask;
574  }
575 
576  // Get all the requested level maps
577  for(ULevelStreaming* Level : Levels)
578  {
579  TArray<FString> StringArray;
580  FString FullSubMapName = Level->GetWorldAssetPackageFName().ToString();
581  // Discard full path, we just need the umap name
582  FullSubMapName.ParseIntoArray(StringArray, TEXT("/"), false);
583  FString SubMapName = StringArray[StringArray.Num() - 1];
584  for(FString LayerName : LayersToLoad)
585  {
586  if(SubMapName.Contains(LayerName))
587  {
588  OutLevelNames.Emplace(FName(*SubMapName));
589  break;
590  }
591  }
592  }
593
