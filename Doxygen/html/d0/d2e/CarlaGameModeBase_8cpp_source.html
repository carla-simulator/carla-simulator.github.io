33 void ACarlaGameModeBase::InitGame(
34  const FString &MapName,
35  const FString &Options,
36  FString &ErrorMessage)
37 {
38  Super::InitGame(MapName, Options, ErrorMessage);
39 
40  checkf(
41  Episode != nullptr,
42  TEXT("Missing episode, can't continue without an episode!"));
43 
44 #if WITH_EDITOR
45  {
46  // When playing in editor the map name gets an extra prefix, here we
47  // remove it.
48  FString CorrectedMapName = MapName;
49  constexpr auto PIEPrefix = TEXT("UEDPIE_0_");
50  CorrectedMapName.RemoveFromStart(PIEPrefix);
51  UE_LOG(LogCarla, Log, TEXT("Corrected map name from %s to %s"), *MapName, *CorrectedMapName);
52  Episode->MapName = CorrectedMapName;
53  }
54 #else
55  Episode->MapName = MapName;
56 #endif // WITH_EDITOR
57 
58  auto World = GetWorld();
59  check(World != nullptr);
60 
61  GameInstance = Cast<UCarlaGameInstance>(GetGameInstance());
62  checkf(
63  GameInstance != nullptr,
64  TEXT("GameInstance is not a UCarlaGameInstance, did you forget to set "
65  "it in the project settings?"));
66 
67  if (TaggerDelegate != nullptr) {
68  TaggerDelegate->RegisterSpawnHandler(World);
69  } else {
70  UE_LOG(LogCarla, Error, TEXT("Missing TaggerDelegate!"));
71  }
72 
73  if(CarlaSettingsDelegate != nullptr) {
74  CarlaSettingsDelegate->ApplyQualityLevelPostRestart();
75  CarlaSettingsDelegate->RegisterSpawnHandler(World);
76  } else {
77  UE_LOG(LogCarla, Error, TEXT("Missing CarlaSettingsDelegate!"));
78  }
79 
80  if (WeatherClass != nullptr) {
81  Episode->Weather = World->SpawnActor<AWeather>(WeatherClass);
82  } else {
83  UE_LOG(LogCarla, Error, TEXT("Missing weather class!"));
84  }
85 
86  GameInstance->NotifyInitGame();
87 
88  SpawnActorFactories();
89 
90  // make connection between Episode and Recorder
91  Recorder->SetEpisode(Episode);
92  Episode->SetRecorder(Recorder);
93 }
