38 void ACarlaGameModeBase::InitGame(
39  const FString &MapName,
40  const FString &Options,
41  FString &ErrorMessage)
42 {
43  Super::InitGame(MapName, Options, ErrorMessage);
44 
45  checkf(
46  Episode != nullptr,
47  TEXT("Missing episode, can't continue without an episode!"));
48 
49 #if WITH_EDITOR
50  {
51  // When playing in editor the map name gets an extra prefix, here we
52  // remove it.
53  FString CorrectedMapName = MapName;
54  constexpr auto PIEPrefix = TEXT("UEDPIE_0_");
55  CorrectedMapName.RemoveFromStart(PIEPrefix);
56  UE_LOG(LogCarla, Log, TEXT("Corrected map name from %s to %s"), *MapName, *CorrectedMapName);
57  Episode->MapName = CorrectedMapName;
58  }
59 #else
60  Episode->MapName = MapName;
61 #endif // WITH_EDITOR
62 
63  auto World = GetWorld();
64  check(World != nullptr);
65 
66  GameInstance = Cast<UCarlaGameInstance>(GetGameInstance());
67  checkf(
68  GameInstance != nullptr,
69  TEXT("GameInstance is not a UCarlaGameInstance, did you forget to set "
70  "it in the project settings?"));
71 
72  if (TaggerDelegate != nullptr) {
73  TaggerDelegate->RegisterSpawnHandler(World);
74  } else {
75  UE_LOG(LogCarla, Error, TEXT("Missing TaggerDelegate!"));
76  }
77 
78  if(CarlaSettingsDelegate != nullptr) {
79  CarlaSettingsDelegate->ApplyQualityLevelPostRestart();
80  CarlaSettingsDelegate->RegisterSpawnHandler(World);
81  } else {
82  UE_LOG(LogCarla, Error, TEXT("Missing CarlaSettingsDelegate!"));
83  }
84 
85  if (WeatherClass != nullptr) {
86  Episode->Weather = World->SpawnActor<AWeather>(WeatherClass);
87  } else {
88  UE_LOG(LogCarla, Error, TEXT("Missing weather class!"));
89  }
90 
91  GameInstance->NotifyInitGame();
92 
93  SpawnActorFactories();
94 
95  // make connection between Episode and Recorder
96  Recorder->SetEpisode(Episode);
97  Episode->SetRecorder(Recorder);
98 
99  ParseOpenDrive(MapName);
100 }

196 void ACarlaGameModeBase::DebugShowSignals(bool enable)
197 {
198  auto World = GetWorld();
199  check(World != nullptr);
200 
201  if(!Map)
202  {
203  return;
204  }
205 
206  if(!enable)
207  {
208  UKismetSystemLibrary::FlushDebugStrings(World);
209  UKismetSystemLibrary::FlushPersistentDebugLines(World);
210  return;
211  }
212 
213  //const std::unordered_map<carla::road::SignId, std::unique_ptr<carla::road::Signal>>
214  const auto& Signals = Map->GetSignals();
215  const auto& Controllers = Map->GetControllers();
216 
217  for(const auto& Signal : Signals) {
218  const auto& ODSignal = Signal.second;
219  const FTransform Transform = ODSignal->GetTransform();
220  const FVector Location = Transform.GetLocation();
221  const FQuat Rotation = Transform.GetRotation();
222  const FVector Up = Rotation.GetUpVector();

236  UKismetSystemLibrary::DrawDebugString (
237  World,
238  Location + Up * 250.0f,
239  Text,
240  nullptr,
241  FLinearColor(0, 255, 0, 255)
242  );
243 
244  FString Text2 = FString(ODSignal->GetType().c_str());
245  Text2 += FString(" - ");
246  Text2 += FString(ODSignal->GetSubtype().c_str());
247 
248  UKismetSystemLibrary::DrawDebugString (
249  World,
250  Location + Up * 175.0f,
251  Text,
252  nullptr,
253  FLinearColor(0, 255, 0, 255),
254  10000.0f
255  );
256  }
257 
258  auto waypoints = Map->GenerateWaypointsOnRoadEntries();
259  for (auto & waypoint : waypoints)
260  {
261  auto SignalReferences = Map->GetLane(waypoint).GetRoad()->GetInfos<carla::road::element::RoadInfoSignal>();
262  for (auto *SignalReference : SignalReferences)
263  {
264  double current_s = waypoint.s;
265  double signal_s = SignalReference->GetS();
266 
267  double delta_s = signal_s - current_s;
268  FTransform ReferenceTransform;
269  if (delta_s == 0)
270  {
271  ReferenceTransform = Map->ComputeTransform(waypoint);
272  }
273  else if (waypoint.lane_id < 0)
274  {
275  auto signal_waypoint = Map->GetNext(waypoint, FMath::Abs(delta_s)).front();
276  ReferenceTransform = Map->ComputeTransform(signal_waypoint);
277  }
278  else if(waypoint.lane_id > 0)
279  {
280  auto signal_waypoint = Map->GetNext(waypoint, FMath::Abs(delta_s)).front();
281  ReferenceTransform = Map->ComputeTransform(signal_waypoint);
282  }
283  else
284  {
285  continue;
286  }
287  DrawDebugSphere(
288  World,
289  ReferenceTransform.GetLocation(),
290  50.0f,
291  10,
292  FColor(0, 255, 0),
293  true
294  );
295 
296  DrawDebugLine(
297  World,
298  ReferenceTransform.GetLocation(),
299  FTransform(SignalReference->GetSignal()->GetTransform()).GetLocation(),
300  FColor(0, 255, 0),

302  );
303  }
