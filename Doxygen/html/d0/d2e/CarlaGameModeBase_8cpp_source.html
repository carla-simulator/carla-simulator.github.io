36 void ACarlaGameModeBase::InitGame(
37  const FString &MapName,
38  const FString &Options,
39  FString &ErrorMessage)
40 {
41  Super::InitGame(MapName, Options, ErrorMessage);
42 
43  checkf(
44  Episode != nullptr,
45  TEXT("Missing episode, can't continue without an episode!"));
46 
47 #if WITH_EDITOR
48  {
49  // When playing in editor the map name gets an extra prefix, here we
50  // remove it.
51  FString CorrectedMapName = MapName;
52  constexpr auto PIEPrefix = TEXT("UEDPIE_0_");
53  CorrectedMapName.RemoveFromStart(PIEPrefix);
54  UE_LOG(LogCarla, Log, TEXT("Corrected map name from %s to %s"), *MapName, *CorrectedMapName);
55  Episode->MapName = CorrectedMapName;
56  }
57 #else
58  Episode->MapName = MapName;
59 #endif // WITH_EDITOR
60 
61  auto World = GetWorld();
62  check(World != nullptr);
63 
64  GameInstance = Cast<UCarlaGameInstance>(GetGameInstance());
65  checkf(
66  GameInstance != nullptr,
67  TEXT("GameInstance is not a UCarlaGameInstance, did you forget to set "
68  "it in the project settings?"));
69 
70  if (TaggerDelegate != nullptr) {
71  TaggerDelegate->RegisterSpawnHandler(World);
72  } else {
73  UE_LOG(LogCarla, Error, TEXT("Missing TaggerDelegate!"));
74  }
75 
76  if(CarlaSettingsDelegate != nullptr) {
77  CarlaSettingsDelegate->ApplyQualityLevelPostRestart();
78  CarlaSettingsDelegate->RegisterSpawnHandler(World);
79  } else {
80  UE_LOG(LogCarla, Error, TEXT("Missing CarlaSettingsDelegate!"));
81  }
82 
83  if (WeatherClass != nullptr) {
84  Episode->Weather = World->SpawnActor<AWeather>(WeatherClass);
85  } else {
86  UE_LOG(LogCarla, Error, TEXT("Missing weather class!"));
87  }
88 
89  GameInstance->NotifyInitGame();
90 
91  SpawnActorFactories();
92 
93  // make connection between Episode and Recorder
94  Recorder->SetEpisode(Episode);
95  Episode->SetRecorder(Recorder);
96 
97  std::string opendrive_xml = carla::rpc::FromFString(UOpenDrive::LoadXODR(MapName));
98 
99  boost::optional<carla::road::Map> map = carla::opendrive::OpenDriveParser::Load(opendrive_xml);
100  if (!map.has_value()) {
101  UE_LOG(LogCarla, Error, TEXT("Invalid Map"));
102  } else {
103  Episode->MapGeoReference = map->GetGeoReference();
104  }
105 
106 }

118 void ACarlaGameModeBase::BeginPlay()
119 {
120  Super::BeginPlay();
121 
122  if (true) { /// @todo If semantic segmentation enabled.
123  check(GetWorld() != nullptr);

125  TaggerDelegate->SetSemanticSegmentationEnabled();

127 
128  Episode->InitializeAtBeginPlay();
