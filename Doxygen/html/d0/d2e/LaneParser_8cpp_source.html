180  void LaneParser::Parse(
181  const pugi::xml_document &xml,
182  carla::road::MapBuilder &map_builder) {
183 
184  pugi::xml_node open_drive_node = xml.child("OpenDRIVE");
185 
186  // Lanes
187  for (pugi::xml_node road_node : open_drive_node.children("road")) {
188  road::RoadId road_id = road_node.attribute("id").as_uint();
189 
190  for (pugi::xml_node lanes_node : road_node.children("lanes")) {
191 
192  for (pugi::xml_node lane_section_node : lanes_node.children("laneSection")) {
193  double s = lane_section_node.attribute("s").as_double();
194  pugi::xml_node left_node = lane_section_node.child("left");
195  if (left_node) {
196  ParseLanes(road_id, s, left_node, map_builder);
197  }
198 
199  pugi::xml_node center_node = lane_section_node.child("center");
200  if (center_node) {
201  ParseLanes(road_id, s, center_node, map_builder);
202  }
203 
204  pugi::xml_node right_node = lane_section_node.child("right");
205  if (right_node) {
206  ParseLanes(road_id, s, right_node, map_builder);
207  }
208  }
209  }
210  }
211  }
