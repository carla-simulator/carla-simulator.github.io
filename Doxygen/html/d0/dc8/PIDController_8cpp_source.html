31  // Initializing present state.
32  StateEntry PIDController::StateUpdate(
33  StateEntry previous_state,
34  float current_velocity,
35  float target_velocity,
36  float angular_deviation,
37  float distance,
38  TimeInstance current_time) {
39 
40  traffic_manager::StateEntry current_state = {
41  angular_deviation, distance,
42  (current_velocity - target_velocity) / target_velocity,
43  current_time,
44  0.0f,
45  0.0f,
46  0.0f
47  };
48 
49  // Calculating integrals.
50  current_state.deviation_integral = angular_deviation * dt + previous_state.deviation_integral;
51  current_state.distance_integral = distance * dt + previous_state.distance_integral;
52  current_state.velocity_integral = dt * current_state.velocity + previous_state.velocity_integral;
53 
54  // Clamp velocity integral to avoid accumulating over-compensation
55  // with time for vehicles that take a long time to reach target velocity.
56  current_state.velocity_integral = CLAMP(current_state.velocity_integral,
57  VELOCITY_INTEGRAL_MAX, VELOCITY_INTEGRAL_MIN);
58 
59  return current_state;
60  }

62  ActuationSignal PIDController::RunStep(
63  StateEntry present_state,
64  StateEntry previous_state,
65  const std::vector<float> &longitudinal_parameters,
66  const std::vector<float> &lateral_parameters) const {
67 
68  // Longitudinal PID calculation.
69  const float expr_v =
70  longitudinal_parameters[0] * present_state.velocity +
71  longitudinal_parameters[1] * present_state.velocity_integral +
72  longitudinal_parameters[2] * (present_state.velocity -
73  previous_state.velocity) / dt;
74 
75  float throttle;
76  float brake;
77 
78  if (expr_v < 0.0f) {
79  throttle = std::min(std::abs(expr_v), MAX_THROTTLE);
80  brake = 0.0f;
81  } else {
82  throttle = 0.0f;
83  brake = std::min(expr_v, MAX_BRAKE);
84  }
85 
86  // Lateral PID calculation.
87  float steer =
88  lateral_parameters[0] * present_state.deviation +
89  lateral_parameters[1] * present_state.deviation_integral +
90  lateral_parameters[2] * (present_state.deviation -
91  previous_state.deviation) / dt;
92 
93  steer = std::max(-0.8f, std::min(steer, 0.8f));
94 
95  return ActuationSignal{throttle, brake, steer};
96  }
