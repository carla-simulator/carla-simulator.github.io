31  traffic_manager::StateEntry current_state = {

34  current_time,

40  // Calculating dt for 'D' and 'I' controller components.
41  const chr::duration<float> duration = current_state.time_instance - previous_state.time_instance;
42  const float dt = duration.count();
43 
44  // Calculating integrals.
45  current_state.deviation_integral = angular_deviation * dt + previous_state.deviation_integral;
46  current_state.distance_integral = distance * dt + previous_state.distance_integral;
47  current_state.velocity_integral = dt * current_state.velocity + previous_state.velocity_integral;
48

52  ActuationSignal PIDController::RunStep(
53  StateEntry present_state,
54  StateEntry previous_state,
55  const std::vector<float> &longitudinal_parameters,
56  const std::vector<float> &lateral_parameters) const {
57 
58  // Calculating dt for updating the integral component.
59  const chr::duration<float> duration = present_state.time_instance - previous_state.time_instance;
60  const float dt = duration.count();
61 
62  // Longitudinal PID calculation.
63  const float expr_v =
64  longitudinal_parameters[0] * present_state.velocity +
65  longitudinal_parameters[1] * present_state.velocity_integral +
66  longitudinal_parameters[2] * (present_state.velocity -
67  previous_state.velocity) / dt;
68 
69  float throttle;
70  float brake;
71 
72  if (expr_v < 0.0f) {
73  throttle = std::min(std::abs(expr_v), MAX_THROTTLE);
74  brake = 0.0f;
75  } else {
76  throttle = 0.0f;
77  brake = std::min(expr_v, MAX_BRAKE);
78  }
79 
80  // Lateral PID calculation.
81  float steer =
82  lateral_parameters[0] * present_state.deviation +
83  lateral_parameters[1] * present_state.deviation_integral +
84  lateral_parameters[2] * (present_state.deviation -
85  previous_state.deviation) / dt;
86 
87  steer = std::max(-1.0f, std::min(steer, 1.0f));
88 
89  return ActuationSignal{throttle, brake, steer};
90  }
