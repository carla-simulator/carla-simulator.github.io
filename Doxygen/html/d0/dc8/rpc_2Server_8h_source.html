27  /// An RPC server in which functions can be bind to run synchronously or
28  /// asynchronously.

93  /// Wraps @a functor into a function type with equivalent signature. The
94  /// wrap function returned. When called, posts @a functor into the
95  /// io_context; if the client called this method synchronously, waits for
96  /// the posted task to finish, otherwise returns immediately.
97  ///
98  /// This way, no matter from which thread the wrap function is called, the
99  /// @a functor provided is always called from the context of the io_context.
100  /// I.e., we can use the io_context to run tasks on a specific thread (e.g.
101  /// game thread).
102  template <typename FuncT>
103  static auto WrapSyncCall(boost::asio::io_context &io, FuncT &&functor) {
104  return [&io, functor=std::forward<FuncT>(functor)](Metadata metadata, Args... args) -> R {
105  auto task = std::packaged_task<R()>([functor=std::move(functor), args...]() {
106  return functor(args...);
107  });
108  if (metadata.IsResponseIgnored()) {
109  // Post task and ignore result.
110  io.post(MoveHandler(task));
111  return R();
112  } else {
113  // Post task and wait for result.
114  auto result = task.get_future();
115  io.post(MoveHandler(task));
116  return result.get();
117  }
118  };
119  }
120 
121  /// Wraps @a functor into a function type with equivalent signature that
122  /// handles the metadata sent by the client. If the client called this
123  /// method asynchronously, the result is ignored.
124  template <typename FuncT>
125  static auto WrapAsyncCall(FuncT &&functor) {
126  return [functor=std::forward<FuncT>(functor)](::carla::rpc::Metadata metadata, Args... args) -> R {
127  if (metadata.IsResponseIgnored()) {
128  functor(args...);
129  return R();
130  } else {
131  return functor(args...);
132  }
133  };
134  }
135  };
