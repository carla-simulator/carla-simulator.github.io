182  std::chrono::duration<double, std::milli>(std::chrono::system_clock::now().time_since_epoch()).count();
183 #if DEBUG_TIMING

186  auto t_end = std::chrono::high_resolution_clock::now();

192 
193 #if DEBUG_TIMING

195  std::cout << "-> ME " << std::chrono::duration<double, std::milli>(t_end - t_start).count()
196  << " before MapMatching" << std::endl;

198 
199  // allow the vehicle to be at least 2.0 m away form the route to not lose

202 
203  if (::ad::map::point::isValid(_carla_rss_state.ego_match_object.enuPosition.centerPoint, false)) {
204  // check for bigger position jumps of the ego vehicle
205  auto const travelled_distance = ::ad::map::point::distance(
206  _carla_rss_state.ego_match_object.enuPosition.centerPoint, ego_match_object.enuPosition.centerPoint);
207  if (travelled_distance > ::ad::physics::Distance(10.)) {
208  _logger->warn("Jump in ego vehicle position detected {} -> {}! Force reroute!",

218 
219 #if DEBUG_TIMING

230  << " after route update " << std::endl;
231 #endif
232

234  timestamp, time_since_epoch_check_start_ms, *carla_ego_vehicle, _carla_rss_state.ego_match_object,
235  _carla_rss_state.ego_route, _carla_rss_state.ego_dynamics_on_route);

237  CreateWorldModel(timestamp, *actors, *carla_ego_vehicle, _carla_rss_state);
238 
239 #if DEBUG_TIMING
240  t_end = std::chrono::high_resolution_clock::now();
241  std::cout << "-> WM " << std::chrono::duration<double, std::milli>(t_end - t_start).count()

246 
247 #if DEBUG_TIMING
248  t_end = std::chrono::high_resolution_clock::now();

255 #if DEBUG_TIMING

259 #endif
260 
261  StoreVisualizationResults(_carla_rss_state);
262 
263 #if DEBUG_TIMING
264  t_end = std::chrono::high_resolution_clock::now();
265  std::cout << "-> VI " << std::chrono::duration<double, std::milli>(t_end - t_start).count() << " end store viz"
266  << std::endl;
267 #endif
268 
269  _carla_rss_state.ego_dynamics_on_route.time_since_epoch_check_end_ms =

271 
272  // store result
273  output_response = _carla_rss_state.proper_response;
274  output_acceleration_restriction = _carla_rss_state.acceleration_restriction;
275  output_rss_state_snapshot = _carla_rss_state.rss_state_snapshot;

278  _logger->debug("===== ROUTE NOT SAFE =====");
279  } else {
280  _logger->debug("===== ROUTE SAFE =====");
281  }
282 
283 #if DEBUG_TIMING
284  t_end = std::chrono::high_resolution_clock::now();
285  std::cout << "-> EC " << std::chrono::duration<double, std::milli>(t_end - t_start).count() << " end check objects"
286  << std::endl;
287 #endif
288 
289  return true;
290  } catch (...) {
291  _logger->error("Exception -> Check failed");
292  return false;
293  }

326 void RssCheck::UpdateRoute(CarlaRssState &carla_rss_state) {
327  _logger->debug("Update route start: {}", carla_rss_state.ego_route);
328 
329  // remove the parts of the route already taken, try to prepend route sections
330  // (i.e. when driving backwards)
331  // try to ensure that the back of the vehicle is still within the route to
332  // support orientation calculation
333  ::ad::map::point::ParaPointList all_lane_matches;
334  for (auto reference_point :
335  {::ad::map::match::ObjectReferencePoints::RearRight, ::ad::map::match::ObjectReferencePoints::RearLeft}) {
336  auto const &reference_position =
337  carla_rss_state.ego_match_object.mapMatchedBoundingBox.referencePointPositions[size_t(reference_point)];
338  auto const para_points = ::ad::map::match::getParaPoints(reference_position);
339  all_lane_matches.insert(all_lane_matches.end(), para_points.begin(), para_points.end());
340  }
341 
342  auto shorten_route_result = ::ad::map::route::shortenRoute(
343  all_lane_matches, carla_rss_state.ego_route,
344  ::ad::map::route::ShortenRouteMode::DontCutIntersectionAndPrependIfSucceededBeforeRoute);
345  if (shorten_route_result == ::ad::map::route::ShortenRouteResult::SucceededIntersectionNotCut) {
346  shorten_route_result = ::ad::map::route::ShortenRouteResult::Succeeded;
347  }
348 
349  bool routing_target_check_finished = false;
350  while ((!_routing_targets.empty()) && (!routing_target_check_finished)) {
351  auto const next_target = _routing_targets.front();
352  auto const &distance_to_next_target =
353  ::ad::map::point::distance(next_target, carla_rss_state.ego_match_object.enuPosition.centerPoint);
354  if (distance_to_next_target < ::ad::physics::Distance(3.)) {
355  _routing_targets.erase(_routing_targets.begin());
356  _logger->debug("Next target reached: {}; remaining targets: {}", next_target, _routing_targets);
357  } else {
358  routing_target_check_finished = true;
359  }
360  }
361 
362  bool reroute_required = false;
363  if (!_routing_targets_to_append.empty()) {
364  reroute_required = true;
365  _routing_targets.insert(_routing_targets.end(), _routing_targets_to_append.begin(),
366  _routing_targets_to_append.end());
367  _logger->debug("Appending new routing targets: {}; resulting targets: {}", _routing_targets_to_append,
368  _routing_targets);
369  _routing_targets_to_append.clear();
370  }
371 
372  ::ad::physics::Distance const route_target_length(50.);
373 
374  if ((!reroute_required) && (shorten_route_result == ::ad::map::route::ShortenRouteResult::Succeeded)) {
375  std::vector<::ad::map::route::FullRoute> additional_routes;
376  auto const route_valid =
377  ::ad::map::route::extendRouteToDistance(carla_rss_state.ego_route, route_target_length, additional_routes);
378 
379  if (route_valid) {
380  if (additional_routes.size() > 0u) {
381  // take a random extension to the route
382  std::size_t route_index = static_cast<std::size_t>(std::rand()) % (additional_routes.size() + 1);
383  if (route_index < additional_routes.size()) {
384  // we decided for one of the additional routes
385  _logger->debug("Additional Routes: {}->{}", additional_routes.size(), route_index);
386  carla_rss_state.ego_route = additional_routes[route_index];
387  } else {
388  // we decided for the extension within route, nothing to be done
389  _logger->debug("Additional Routes: expand current");
390  }
391  }
392  } else {
393  reroute_required = true;
394  }
395  } else {
396  // on all other results we recreate the route
397  reroute_required = true;
398  }
399 
400  // create the route if required
401  if (reroute_required) {
402  // try to create routes
403  std::vector<::ad::map::route::FullRoute> all_new_routes;
404  for (const auto &position :
405  carla_rss_state.ego_match_object.mapMatchedBoundingBox
406  .referencePointPositions[int32_t(::ad::map::match::ObjectReferencePoints::Center)]) {
407  auto start_point = position.lanePoint.paraPoint;
408  auto projected_start_point = start_point;
409  if (!::ad::map::lane::isHeadingInLaneDirection(start_point,
410  carla_rss_state.ego_match_object.enuPosition.heading)) {
411  _logger->debug("EgoVehicle heading in opposite lane direction");
412  if (::ad::map::lane::projectPositionToLaneInHeadingDirection(
413  start_point, carla_rss_state.ego_match_object.enuPosition.heading, projected_start_point)) {
414  _logger->debug("Projected to lane {}", projected_start_point.laneId);
415  }
416  }
417  _logger->debug("Route start_point: {}, projected_start_point: {}", start_point, projected_start_point);
418  auto routing_start_point = ::ad::map::route::planning::createRoutingPoint(
419  projected_start_point, carla_rss_state.ego_match_object.enuPosition.heading);
420  if (!_routing_targets.empty() && ::ad::map::point::isValid(_routing_targets)) {
421  auto new_route = ::ad::map::route::planning::planRoute(routing_start_point, _routing_targets,
422  ::ad::map::route::RouteCreationMode::AllRoutableLanes);
423  all_new_routes.push_back(new_route);
424  } else {
425  auto new_routes = ::ad::map::route::planning::predictRoutesOnDistance(
426  routing_start_point, route_target_length, ::ad::map::route::RouteCreationMode::AllRoutableLanes);
427 
428  for (const auto &new_route : new_routes) {
429  // extend route with all lanes
430  all_new_routes.push_back(new_route);
431  }
432  }
433  }
434 
435  _logger->debug("New routes: {}", all_new_routes.size());
436 
437  if (!all_new_routes.empty()) {
438  // take a random route
439  std::size_t route_index = static_cast<std::size_t>(std::rand()) % (all_new_routes.size());
440  carla_rss_state.ego_route = all_new_routes[route_index];
441  }
442  }
443 
444  _logger->debug("Update route result: {}", carla_rss_state.ego_route);
445 }

447 EgoDynamicsOnRoute RssCheck::CalculateEgoDynamicsOnRoute(carla::client::Timestamp const &current_timestamp,
448  double const &time_since_epoch_check_start_ms,
449  carla::client::Vehicle const &carla_vehicle,
450  ::ad::map::match::Object match_object,
451  ::ad::map::route::FullRoute const &route,
452  EgoDynamicsOnRoute const &last_dynamics) const {
453  EgoDynamicsOnRoute new_dynamics;
454  new_dynamics.timestamp = current_timestamp;
455  new_dynamics.time_since_epoch_check_start_ms = time_since_epoch_check_start_ms;
456  new_dynamics.ego_speed = GetSpeed(carla_vehicle);
457  new_dynamics.ego_center = match_object.enuPosition.centerPoint;
458  new_dynamics.ego_heading = match_object.enuPosition.heading;
459 
460  auto object_route =
461  ::ad::map::route::getRouteSection(match_object, route, ::ad::map::route::RouteSectionCreationMode::AllRouteLanes);
462  auto border = ::ad::map::route::getENUBorderOfRoute(object_route);
463  StoreDebugVisualization(object_route, border);
464  new_dynamics.route_heading = ::ad::map::lane::getENUHeading(border, match_object.enuPosition.centerPoint);
465 
466  auto const object_center = ::ad::map::route::findCenterWaypoint(match_object, object_route);
467  if (object_center.isValid()) {
468  auto lane_center_point = object_center.queryPosition;
469  auto lane_center_point_enu = ::ad::map::lane::getENULanePoint(lane_center_point);
470  if (std::fabs(new_dynamics.route_heading) > ::ad::map::point::ENUHeading(M_PI)) {
471  // if the actual center point is already outside, try to use this extended
472  // object center for the route heading calculation
473  new_dynamics.route_heading = ::ad::map::lane::getENUHeading(border, lane_center_point_enu);
474  }
475 
476  if (object_center.laneSegmentIterator->laneInterval.wrongWay) {
477  // driving on the wrong lane, so we have to project to nominal route
478  // direction
479  ::ad::map::lane::projectPositionToLaneInHeadingDirection(lane_center_point, new_dynamics.route_heading,
480  lane_center_point);
481  lane_center_point_enu = ::ad::map::lane::getENULanePoint(lane_center_point);
482  }
483  new_dynamics.route_nominal_center = lane_center_point_enu;
484 
485  } else {
486  // the ego vehicle is completely outside the route, so we can't update the
487  // values
488  new_dynamics.route_nominal_center = last_dynamics.route_nominal_center;
489  new_dynamics.route_heading = last_dynamics.route_heading;
490  }
491 
492  new_dynamics.heading_diff =
493  ::ad::map::point::normalizeENUHeading(new_dynamics.route_heading - new_dynamics.ego_heading);
494  new_dynamics.route_speed_lon =
495  std::fabs(std::cos(static_cast<double>(new_dynamics.heading_diff))) * new_dynamics.ego_speed;
496  new_dynamics.route_speed_lat = std::sin(static_cast<double>(new_dynamics.heading_diff)) * new_dynamics.ego_speed;
497 
498  bool keep_last_acceleration = true;
499  if (last_dynamics.timestamp.elapsed_seconds > 0.) {
500  ::ad::physics::Duration const delta_time(current_timestamp.elapsed_seconds -
501  last_dynamics.timestamp.elapsed_seconds);
502  if (delta_time > ::ad::physics::Duration(0.0001)) {
503  try {
504  new_dynamics.route_accel_lat = (new_dynamics.route_speed_lat - last_dynamics.route_speed_lat) / delta_time;
505  new_dynamics.avg_route_accel_lat =
506  ((last_dynamics.avg_route_accel_lat * 2.) + new_dynamics.route_accel_lat) / 3.;
507  new_dynamics.route_accel_lon = (new_dynamics.route_speed_lon - last_dynamics.route_speed_lon) / delta_time;
508  new_dynamics.avg_route_accel_lon =
509  ((last_dynamics.avg_route_accel_lon * 2.) + new_dynamics.route_accel_lon) / 3.;
510 
511  if (new_dynamics.avg_route_accel_lat == ::ad::physics::Acceleration(0.)) {
512  // prevent from underrun
513  new_dynamics.avg_route_accel_lat = ::ad::physics::Acceleration(0.);
514  }
515  if (new_dynamics.avg_route_accel_lon == ::ad::physics::Acceleration(0.)) {
516  // prevent from underrun
517  new_dynamics.avg_route_accel_lon = ::ad::physics::Acceleration(0.);
518  }
519  keep_last_acceleration = false;
520  } catch (...) {
521  }
522  }
523  }
524 
525  if (keep_last_acceleration) {
526  new_dynamics.route_accel_lat = last_dynamics.route_accel_lat;
527  new_dynamics.avg_route_accel_lat = last_dynamics.avg_route_accel_lat;
528  new_dynamics.route_accel_lon = last_dynamics.route_accel_lon;
529  new_dynamics.avg_route_accel_lon = last_dynamics.avg_route_accel_lon;
530  }
531 
532  // check if the center point (and only the center point) is still found on the
533  // route
534  ::ad::map::point::ParaPointList in_lane_matches;
535  for (auto &match_position : match_object.mapMatchedBoundingBox
536  .referencePointPositions[int32_t(::ad::map::match::ObjectReferencePoints::Center)]) {
537  if (match_position.type == ::ad::map::match::MapMatchedPositionType::LANE_IN) {
538  in_lane_matches.push_back(match_position.lanePoint.paraPoint);
539  }
540  }
541  auto const object_in_lane_center = ::ad::map::route::findNearestWaypoint(in_lane_matches, route);
542  new_dynamics.ego_center_within_route = object_in_lane_center.isValid();
543  // evaluated by AnalyseResults
544  new_dynamics.crossing_border = false;
545 
546  // calculate the ego stopping distance, to be able to reduce the effort
547  ad::rss::map::RssObjectConversion object_conversion(::ad::rss::world::ObjectId(0u),
548  ::ad::rss::world::ObjectType::EgoVehicle, match_object,
549  new_dynamics.ego_speed, GetEgoVehicleDynamics());
550  if (!object_conversion.calculateMinStoppingDistance(new_dynamics.min_stopping_distance)) {
551  _logger->error(
552  "CalculateEgoDynamicsOnRoute: calculation of min stopping distance "
553  "failed. Setting to 100. ({} {} {})",
554  match_object, new_dynamics.ego_speed, GetEgoVehicleDynamics());
555  new_dynamics.min_stopping_distance = ::ad::physics::Distance(100.);
556  }
557 
558  _logger->trace("CalculateEgoDynamicsOnRoute: route-section {} -> dynamics: {}", object_route, new_dynamics);
559  return new_dynamics;
560 }

562 ::ad::map::landmark::LandmarkIdSet RssCheck::GetGreenTrafficLightsOnRoute(
563  std::vector<SharedPtr<carla::client::TrafficLight>> const &traffic_lights,
564  ::ad::map::route::FullRoute const &route) const {
565  ::ad::map::landmark::LandmarkIdSet green_traffic_lights;
566 
567  auto next_intersection = ::ad::map::intersection::Intersection::getNextIntersectionOnRoute(route);
568  if (next_intersection &&
569  (next_intersection->intersectionType() == ::ad::map::intersection::IntersectionType::TrafficLight)) {
570  // try to guess the the relevant traffic light with the rule: nearest
571  // traffic light in respect to the incoming lane.
572  // @todo: when OpenDrive maps have the traffic lights incorporated, we only
573  // have to fill all green traffic lights into the green_traffic_lights list
574  auto incoming_lanes = next_intersection->incomingLanesOnRoute();
575  // since our route spans the whole street, we have to filter out the
576  // incoming lanes with wrong way flag
577  auto incoming_lanes_iter = incoming_lanes.begin();
578  while (incoming_lanes_iter != incoming_lanes.end()) {
579  auto find_waypoint = ::ad::map::route::findWaypoint(*incoming_lanes_iter, route);
580  if (find_waypoint.isValid() && find_waypoint.laneSegmentIterator->laneInterval.wrongWay) {
581  incoming_lanes_iter = incoming_lanes.erase(incoming_lanes_iter);
582  } else {
583  incoming_lanes_iter++;
584  }
585  }
586 
587  ::ad::map::match::AdMapMatching traffic_light_map_matching;
588  bool found_relevant_traffic_light = false;
589  for (const auto &traffic_light : traffic_lights) {
590  auto traffic_light_state = traffic_light->GetState();
591  carla::geom::BoundingBox trigger_bounding_box = traffic_light->GetTriggerVolume();
592 
593  auto traffic_light_transform = traffic_light->GetTransform();
594  auto trigger_box_location = trigger_bounding_box.location;
595  traffic_light_transform.TransformPoint(trigger_box_location);
596 
597  ::ad::map::point::ENUPoint trigger_box_position;
598  trigger_box_position.x = ::ad::map::point::ENUCoordinate(trigger_box_location.x);
599  trigger_box_position.y = ::ad::map::point::ENUCoordinate(-1 * trigger_box_location.y);
600  trigger_box_position.z = ::ad::map::point::ENUCoordinate(0.);
601 
602  _logger->trace("traffic light[{}] Position: {}", traffic_light->GetId(), trigger_box_position);
603  auto traffic_light_map_matched_positions = traffic_light_map_matching.getMapMatchedPositions(
604  trigger_box_position, ::ad::physics::Distance(0.25), ::ad::physics::Probability(0.1));
605 
606  _logger->trace("traffic light[{}] Map Matched Position: {}", traffic_light->GetId(),
607  traffic_light_map_matched_positions);
608 
609  for (auto matched_position : traffic_light_map_matched_positions) {
610  if (incoming_lanes.find(matched_position.lanePoint.paraPoint.laneId) != incoming_lanes.end()) {
611  if (found_relevant_traffic_light &&
612  (green_traffic_lights.empty() && (traffic_light_state == carla::rpc::TrafficLightState::Green))) {
613  _logger->warn("found another relevant traffic light on lane {}; {} state {}",
614  matched_position.lanePoint.paraPoint.laneId, traffic_light->GetId(),
615  (traffic_light_state == carla::rpc::TrafficLightState::Green) ? "green" : "not green");
616  } else {
617  _logger->debug("found relevant traffic light on lane {}; {} state {}",
618  matched_position.lanePoint.paraPoint.laneId, traffic_light->GetId(),
619  (traffic_light_state == carla::rpc::TrafficLightState::Green) ? "green" : "not green");
620  }
621 
622  found_relevant_traffic_light = true;
623 
624  // found matching traffic light
625  if (traffic_light_state == carla::rpc::TrafficLightState::Green) {
626  // @todo: currently there is only this workaround because of missign
627  // OpenDrive map support for actual traffic light ids
628  green_traffic_lights.insert(::ad::map::landmark::LandmarkId::getMax());
629  } else {
630  // if the light is not green, we don't have priority
631  green_traffic_lights.clear();
632  }
633  break;
634  }
635  }
636  }
637  }
638  return green_traffic_lights;
639 }

677 void RssCheck::CreateWorldModel(carla::client::Timestamp const &timestamp, carla::client::ActorList const &actors,
678  carla::client::Vehicle const &carla_ego_vehicle, CarlaRssState &carla_rss_state) const {
679  // only loop once over the actors since always the respective objects are created
680  std::vector<SharedPtr<carla::client::TrafficLight>> traffic_lights;
681  std::vector<SharedPtr<carla::client::Vehicle>> other_vehicles;
682  for (const auto &actor : actors) {
683  const auto traffic_light = boost::dynamic_pointer_cast<carla::client::TrafficLight>(actor);
684  if (traffic_light != nullptr) {
685  traffic_lights.push_back(traffic_light);
686  continue;
687  }
688 
689  const auto vehicle = boost::dynamic_pointer_cast<carla::client::Vehicle>(actor);
690  if (vehicle != nullptr) {
691  if (vehicle->GetId() == carla_ego_vehicle.GetId()) {
692  continue;
693  }
694  auto const relevant_distance =
695  std::max(static_cast<double>(carla_rss_state.ego_dynamics_on_route.min_stopping_distance), 100.);
696  if (vehicle->GetTransform().location.Distance(carla_ego_vehicle.GetTransform().location) < relevant_distance) {
697  other_vehicles.push_back(vehicle);
698  }
699  }
700  }
701 
702  ::ad::map::landmark::LandmarkIdSet green_traffic_lights =
703  GetGreenTrafficLightsOnRoute(traffic_lights, carla_rss_state.ego_route);
704 
705  auto ego_vehicle_dynamics = GetEgoVehicleDynamics();
706  auto const abs_avg_route_accel_lat = std::fabs(carla_rss_state.ego_dynamics_on_route.avg_route_accel_lat);
707  if (abs_avg_route_accel_lat > ego_vehicle_dynamics.alphaLat.accelMax) {
708  _logger->info(
709  "!! Route lateral dynamics exceed expectations: route:{} expected:{} "
710  "!!",
711  abs_avg_route_accel_lat, ego_vehicle_dynamics.alphaLat.accelMax);
712  ego_vehicle_dynamics.alphaLat.accelMax = std::min(::ad::physics::Acceleration(20.), abs_avg_route_accel_lat);
713  }
714 
715  ::ad::rss::map::RssSceneCreation scene_creation(timestamp.frame, ego_vehicle_dynamics);
716 
717 #ifdef RSS_USE_TBB
718  tbb::parallel_for_each(
719  other_vehicles.begin(), other_vehicles.end(),
720  RssObjectChecker(*this, scene_creation, carla_ego_vehicle, carla_rss_state, green_traffic_lights));
721 #else
722  for (auto const vehicle : other_vehicles) {
723  auto checker = RssObjectChecker(*this, scene_creation, carla_ego_vehicle, carla_rss_state, green_traffic_lights);
724  checker(vehicle);
725  }
726 #endif
727 
728  if (_road_boundaries_mode != RoadBoundariesMode::Off) {
729  // add artifical objects on the road boundaries for "stay-on-road" feature
730  // use 'smart' dynamics
731  ego_vehicle_dynamics.alphaLat.accelMax = ::ad::physics::Acceleration(0.);
732  scene_creation.appendRoadBoundaries(carla_ego_vehicle.GetId(), carla_rss_state.ego_match_object,
733  carla_rss_state.ego_dynamics_on_route.ego_speed, ego_vehicle_dynamics,
734  carla_rss_state.ego_route,
735  // since the route always expanded, route isn't required to expand any
736  // more
737  ::ad::rss::map::RssSceneCreation::AppendRoadBoundariesMode::RouteOnly);
738  }
739 
740  carla_rss_state.world_model = scene_creation.getWorldModel();
741 }

760 void RssCheck::AnalyseCheckResults(CarlaRssState &carla_rss_state) const {
761  carla_rss_state.dangerous_state = false;
762  carla_rss_state.dangerous_vehicle = false;
763  carla_rss_state.dangerous_opposite_state = false;
764  bool left_border_is_dangerous = false;
765  bool right_border_is_dangerous = false;
766  bool vehicle_triggered_left_response = false;
767  bool vehicle_triggered_right_response = false;
768  bool vehicle_triggered_longitudinal_response = false;
769  for (auto const state : carla_rss_state.rss_state_snapshot.individualResponses) {
770  if (::ad::rss::state::isDangerous(state)) {
771  carla_rss_state.dangerous_state = true;
772  _logger->trace("DangerousState: {}", state);
773  auto dangerous_sitation_iter = std::find_if(carla_rss_state.situation_snapshot.situations.begin(),
774  carla_rss_state.situation_snapshot.situations.end(),
775  [&state](::ad::rss::situation::Situation const &situation) {
776  return situation.situationId == state.situationId;
777  });
778  if (dangerous_sitation_iter != carla_rss_state.situation_snapshot.situations.end()) {
779  _logger->trace("Situation: {}", *dangerous_sitation_iter);
780  if (dangerous_sitation_iter->objectId == ::ad::rss::map::RssSceneCreator::getRightBorderObjectId()) {
781  right_border_is_dangerous = true;
782  } else if (dangerous_sitation_iter->objectId == ::ad::rss::map::RssSceneCreator::getLeftBorderObjectId()) {
783  left_border_is_dangerous = true;
784  } else {
785  carla_rss_state.dangerous_vehicle = true;
786  if (state.longitudinalState.response != ::ad::rss::state::LongitudinalResponse::None) {
787  vehicle_triggered_longitudinal_response = true;
788  }
789  if (state.lateralStateLeft.response != ::ad::rss::state::LateralResponse::None) {
790  vehicle_triggered_left_response = true;
791  }
792  if (state.lateralStateRight.response != ::ad::rss::state::LateralResponse::None) {
793  vehicle_triggered_right_response = true;
794  }
795  }
796  if (dangerous_sitation_iter->situationType == ::ad::rss::situation::SituationType::OppositeDirection) {
797  carla_rss_state.dangerous_opposite_state = true;
798  }
799  }
800  }
801  }
802 
803  // border are restricting potentially too much, fix this
804  if (!vehicle_triggered_longitudinal_response &&
805  (carla_rss_state.proper_response.longitudinalResponse != ::ad::rss::state::LongitudinalResponse::None)) {
806  _logger->debug("!! longitudinalResponse only triggered by borders: ignore !!");
807  carla_rss_state.proper_response.longitudinalResponse = ::ad::rss::state::LongitudinalResponse::None;
808  carla_rss_state.acceleration_restriction.longitudinalRange.maximum = GetEgoVehicleDynamics().alphaLon.accelMax;
809  }
810  if (!vehicle_triggered_left_response && !left_border_is_dangerous &&
811  (carla_rss_state.proper_response.lateralResponseLeft != ::ad::rss::state::LateralResponse::None)) {
812  _logger->debug("!! lateralResponseLeft only triggered by right border: ignore !!");
813  carla_rss_state.proper_response.lateralResponseLeft = ::ad::rss::state::LateralResponse::None;
814  carla_rss_state.acceleration_restriction.lateralLeftRange.maximum = GetEgoVehicleDynamics().alphaLat.accelMax;
815  carla_rss_state.ego_dynamics_on_route.crossing_border = true;
816  }
817  if (!vehicle_triggered_right_response && !right_border_is_dangerous &&
818  (carla_rss_state.proper_response.lateralResponseRight != ::ad::rss::state::LateralResponse::None)) {
819  _logger->debug("!! lateralResponseRight only triggered by left border: ignore !!");
820  carla_rss_state.proper_response.lateralResponseRight = ::ad::rss::state::LateralResponse::None;
821  carla_rss_state.acceleration_restriction.lateralRightRange.maximum = GetEgoVehicleDynamics().alphaLat.accelMax;
822  carla_rss_state.ego_dynamics_on_route.crossing_border = true;
823  }
824 
825  _logger->debug("RouteResponse: {}, AccelerationRestriction: {}", carla_rss_state.proper_response,
826  carla_rss_state.acceleration_restriction);
827 }
