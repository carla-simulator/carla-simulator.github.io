31  inline float calculateAngleDelta(const float angle1, const float angle2) {
32  float delta = angle1 - angle2;
33  delta -= std::floor((delta + 180.f) / 360.f) * 360.f;
34  return delta;

77  bool RssCheck::checkObjects(cc::Timestamp const & timestamp, cc::World &world,
78  carla::SharedPtr<cc::ActorList> const & vehicles,
79  carla::SharedPtr<cc::Actor> const & carlaEgoActor,
80  carla::SharedPtr<cc::Map> const & clientMap,
81  ::ad_rss::state::ProperResponse &response,
82  ::ad_rss::world::AccelerationRestriction &accelerationRestriction,
83  ::ad_rss::world::Velocity &rssEgoVelocity, bool visualizeResults) {
84  const auto carlaEgoVehicle = boost::dynamic_pointer_cast<cc::Vehicle>(carlaEgoActor);
85 
86  ::ad_rss::world::WorldModel worldModel;
87  worldModel.timeIndex = timestamp.frame;
88 
89  ::ad_rss::world::Object egoVehicle;
90  ::ad_rss::world::RssDynamics egoDynamics;
91  initVehicle(egoVehicle);
92  initEgoVehicleDynamics(egoDynamics);
93  egoVehicle.objectId = carlaEgoVehicle->GetId();
94  egoVehicle.objectType = ::ad_rss::world::ObjectType::EgoVehicle;
95 
96  cg::Location egoVehicleLocation = carlaEgoVehicle->GetLocation();
97  auto egoClientWaypoint = clientMap->GetWaypoint(egoVehicleLocation, false);
98 
99  if (egoClientWaypoint != nullptr) {
100  // ego vehicle is located on a lane marked for driving
101  auto egoWaypointLoc = egoClientWaypoint->GetTransform();
102  auto egoLaneId = egoClientWaypoint->GetLaneId();
103  auto egoVelocity = carlaEgoVehicle->GetVelocity();
104  auto egoVehicleTransform = carlaEgoVehicle->GetTransform();
105 
106  // calculate driving direction
107  auto yawDiff = calculateAngleDelta(egoWaypointLoc.rotation.yaw, egoVehicleTransform.rotation.yaw);
108 
109  bool drivingInLaneDirection = true;
110  if (std::abs(yawDiff) > 45.f) {
111  drivingInLaneDirection = false;
112  }
113 
114  // calculate road direction
115  bool drivingInRoadDirection = (egoLaneId > 0) ^ drivingInLaneDirection;
116 
117  auto &carlaRoadMap = clientMap->GetMap();
118  auto egoWaypoint = carlaRoadMap.GetWaypoint(egoVehicleLocation);
119  auto &egoLane = carlaRoadMap.GetLane(*egoWaypoint);
120  auto egoRoad = egoLane.GetRoad();
121 
122  ::ad_rss::world::RoadArea roadArea;
123  ::ad_rss::world::RoadSegment roadSegment;
124 
125  // generate road area
126  for (auto &laneSection : egoRoad->GetLaneSections()) {
127  for (const auto &pair : laneSection.GetLanes()) {
128  const auto &lane = pair.second;
129  if ((static_cast<uint32_t>(lane.GetType()) &
130  static_cast<uint32_t>(carla::road::Lane::LaneType::Driving)) > 0) {
131  ::ad_rss::world::LaneSegment laneSegment;
132  // assumption: only one segment per road
133 
134  auto laneLength = lane.GetLength();
135  // evaluate width at lane start
136  double pos = laneSection.GetDistance();
137  const auto lane_width_info = lane.GetInfo<carla::road::element::RoadInfoLaneWidth>(pos);
138  double laneWidth = 0.0;
139  if (lane_width_info != nullptr) {
140  laneWidth = lane_width_info->GetPolynomial().Evaluate(pos);
141  }
142 
143  convertAndSetLaneSegmentId(lane, laneSegment);
144  if ((lane.GetId() < 0) ^ drivingInRoadDirection) {
145  laneSegment.drivingDirection = ::ad_rss::world::LaneDrivingDirection::Negative;
146  } else {
147  laneSegment.drivingDirection = ::ad_rss::world::LaneDrivingDirection::Positive;
148  }
149 
150  // lane segment is assumed to be strait and evenly wide, so minimum
151  // and maximum length and width are set to the same values
152  laneSegment.length.minimum = ::ad_rss::physics::Distance(laneLength);
153  laneSegment.length.maximum = laneSegment.length.minimum;
154  laneSegment.width.minimum = ::ad_rss::physics::Distance(laneWidth);
155  laneSegment.width.maximum = laneSegment.width.minimum;
156 
157  roadSegment.push_back(laneSegment);
158  }
159  }
160  std::sort(
161  roadSegment.begin(), roadSegment.end(),
162  [&drivingInRoadDirection](::ad_rss::world::LaneSegment const &f,
163  ::ad_rss::world::LaneSegment const &s) {
164  return (f.id <= s.id) ^ drivingInRoadDirection;
165  });
166  roadArea.push_back(roadSegment);
167  }
168 
169  calculateLatLonVelocities(egoWaypointLoc, egoVelocity, egoVehicle, !drivingInLaneDirection);
170  rssEgoVelocity = egoVehicle.velocity;
171 
172  auto egoBounds = getVehicleBounds(*carlaEgoVehicle);
173 
174  calculateOccupiedRegions(egoBounds, egoVehicle.occupiedRegions, carlaRoadMap, drivingInRoadDirection,
175  egoVehicleTransform);
176 
177  for (const auto &actor : *vehicles) {
178  const auto vehicle = boost::dynamic_pointer_cast<cc::Vehicle>(actor);
179  if (vehicle == nullptr) {
180  continue;
181  }
182  if (vehicle->GetId() == carlaEgoVehicle->GetId()) {
183  continue;
184  }
185  ::ad_rss::world::Object otherVehicle;
186  ::ad_rss::world::RssDynamics otherDynamics;
187  initVehicle(otherVehicle);
188  initOtherVehicleDynamics(otherDynamics);
189  otherVehicle.objectId = vehicle->GetId();
190  otherVehicle.objectType = ::ad_rss::world::ObjectType::OtherVehicle;
191 
192  auto otherVehicleLoc = vehicle->GetLocation();
193  auto otherVehicleClientWaypoint = clientMap->GetWaypoint(otherVehicleLoc);
194 
195  // same road
196  if (egoClientWaypoint->GetRoadId() == otherVehicleClientWaypoint->GetRoadId()) {
197  auto otherVehicleTransform = vehicle->GetTransform();
198  auto otherBounds = getVehicleBounds(*vehicle);
199  auto otherWaypointLoc = otherVehicleClientWaypoint->GetTransform();
200  auto otherVelocity = vehicle->GetVelocity();
201  calculateLatLonVelocities(otherWaypointLoc, otherVelocity, otherVehicle, false);
202 
203  ::ad_rss::world::Scene scene;
204 
205  calculateOccupiedRegions(otherBounds, otherVehicle.occupiedRegions, carlaRoadMap, drivingInRoadDirection,
206  otherVehicleTransform);
207 
208  auto vehicleDirectionDiff =
209  calculateAngleDelta(otherVehicleTransform.rotation.yaw, egoVehicleTransform.rotation.yaw);
210 
211  bool drivingInSameDirection = true;
212  if (std::abs(vehicleDirectionDiff) > 90.f) {
213  drivingInSameDirection = false;
214  }
215 
216  if (drivingInSameDirection) {
217  scene.situationType = ad_rss::situation::SituationType::SameDirection;
218  } else {
219  scene.situationType = ad_rss::situation::SituationType::OppositeDirection;
220  }
221 
222  scene.object = otherVehicle;
223  scene.objectRssDynamics = otherDynamics;
224  scene.egoVehicle = egoVehicle;
225  scene.egoVehicleRoad = roadArea;
226  worldModel.scenes.push_back(scene);
227  }
228  }
229  worldModel.egoVehicleRssDynamics = egoDynamics;
230 
231  ::ad_rss::situation::SituationSnapshot situationSnapshot;
232  bool result = _rssSituationExtraction->extractSituations(worldModel, situationSnapshot);
233 
234  ::ad_rss::state::RssStateSnapshot stateSnapshot;
235  if (result) {
236  result = _rssSituationChecking->checkSituations(situationSnapshot, stateSnapshot);
237  }
238  if (result) {
239  result = _rssResponseResolving->provideProperResponse(stateSnapshot, response);
240  }
241  if (result) {
242  result = ::ad_rss::core::RssResponseTransformation::transformProperResponse(worldModel, response,
243  accelerationRestriction);
244  }
245 
246  if (result && visualizeResults) {
247  visualizeRssResults(stateSnapshot, egoVehicleLocation, egoVehicleTransform, world);
248  }
249  return result;
250  }
251  return false;
252  }

267  void RssCheck::calculateLatLonVelocities(const cg::Transform &laneLocation,
268  const cg::Vector3D &velocity, ::ad_rss::world::Object &rssObject,
269  bool inverseDirection) const {
270  double roadDirSine = std::sin(laneLocation.rotation.yaw * toRadians);
271  double roadDirCosine = std::cos(laneLocation.rotation.yaw * toRadians);
272  double factor = 1.0;
273  if (inverseDirection) {
274  factor = -1.0;
275  }
276  rssObject.velocity.speedLon =
277  ::ad_rss::physics::Speed(std::abs(roadDirCosine * velocity.x + roadDirSine * velocity.y));
278  rssObject.velocity.speedLat =
279  ::ad_rss::physics::Speed(factor * (-1.0 * roadDirSine * velocity.x + roadDirCosine * velocity.y));
280  }

288 // return front right, back right, back left, front left bounds
289  std::array<cg::Location,
290  4u> RssCheck::getVehicleBounds(const cc::Vehicle &vehicle) const {
291  const auto &box = vehicle.GetBoundingBox();
292  const auto &transform = vehicle.GetTransform();
293  const auto location = transform.location + box.location;
294  const auto yaw = transform.rotation.yaw * toRadians;
295  const float cosine = std::cos(yaw);
296  const float sine = std::sin(yaw);
297  cg::Location frontExtent{cosine *box.extent.x, sine *box.extent.x, 0.0f};
298  cg::Location rightExtent{-sine * box.extent.y, cosine *box.extent.y, 0.0f};
299  return {location + frontExtent + rightExtent, location - frontExtent + rightExtent,
300  location - frontExtent - rightExtent, location + frontExtent - rightExtent};
301  }

303  uint64_t RssCheck::calculateLaneSegmentId(const carla::road::Lane &lane) const {
304  uint64_t laneSegmentId = lane.GetRoad()->GetId();
305  laneSegmentId = laneSegmentId << 32;
306  uint32_t sectionAndLane = lane.GetLaneSection()->GetId();
307  sectionAndLane <<= 16;
308  int16_t laneId = static_cast<int16_t>(lane.GetId());
309  sectionAndLane |= (0xffff & (laneId + 256));
310  laneSegmentId |= sectionAndLane;
311  return laneSegmentId;
312  }

314  void RssCheck::calculateOccupiedRegions(const std::array<cg::Location, 4u> &bounds,
315  ::ad_rss::world::OccupiedRegionVector &occupiedRegions,
316  const carla::road::Map &carlaRoadMap, const bool drivingInRoadDirection,
317  const cg::Transform &vehicleTransform) const {
318  bool regionValid = true;
319  for (const auto &bound : bounds) {
320  auto boundWaypoint = carlaRoadMap.GetWaypoint(bound);
321 
322  if (!boundWaypoint.has_value()) {
323  // vehicle not entirely on road --> stop evaluation
324  regionValid = false;
325  break;
326  }
327  auto &lane = carlaRoadMap.GetLane(*boundWaypoint);
328  auto road = lane.GetRoad();
329  uint64_t laneSegmentId = calculateLaneSegmentId(carlaRoadMap.GetLane(*boundWaypoint));
330 
331  auto length = road->GetLength();
332  auto lonPos = boundWaypoint->s;
333  if (!drivingInRoadDirection) {
334  lonPos = length - lonPos;
335  }
336  double lon = std::max(0.0, std::min(lonPos / length, 1.0));
337 
338  auto waypointLocation = carlaRoadMap.ComputeTransform(*boundWaypoint);
339  double headingSine = std::sin(waypointLocation.rotation.yaw * toRadians);
340  double headingCosine = std::cos(waypointLocation.rotation.yaw * toRadians);
341 
342  auto yawDelta = calculateAngleDelta(waypointLocation.rotation.yaw, vehicleTransform.rotation.yaw);
343  bool inLaneDirection = true;
344  if (std::abs(yawDelta) > 45.f) {
345  inLaneDirection = false;
346  }
347 
348  auto width = carlaRoadMap.GetLaneWidth(*boundWaypoint);
349  double latOffset = (headingSine * (waypointLocation.location.x - bound.x) -
350  headingCosine * (waypointLocation.location.y - bound.y));
351  if (!inLaneDirection) {
352  latOffset *= -1.0;
353  }
354 
355  double lat = std::max(0.0, std::min(0.5 + latOffset / width, 1.0));
356 
357  // find or create occupied region
358  auto regionIt = std::find_if(
359  occupiedRegions.begin(), occupiedRegions.end(),
360  [laneSegmentId](const ::ad_rss::world::OccupiedRegion &region) {
361  return region.segmentId == laneSegmentId;
362  });
363  if (regionIt == occupiedRegions.end()) {
364  ::ad_rss::world::OccupiedRegion newRegion;
365  newRegion.segmentId = laneSegmentId;
366  newRegion.lonRange.minimum = ::ad_rss::physics::ParametricValue(lon);
367  newRegion.lonRange.maximum = newRegion.lonRange.minimum;
368  newRegion.latRange.minimum = ::ad_rss::physics::ParametricValue(lat);
369  newRegion.latRange.maximum = newRegion.latRange.minimum;
370  occupiedRegions.push_back(newRegion);
371  } else {
372  ::ad_rss::physics::ParametricValue const lonParam(lon);
373  ::ad_rss::physics::ParametricValue const latParam(lat);
374 
375  regionIt->lonRange.minimum = std::min(regionIt->lonRange.minimum, lonParam);
376  regionIt->lonRange.maximum = std::max(regionIt->lonRange.maximum, lonParam);
377  regionIt->latRange.minimum = std::min(regionIt->latRange.minimum, latParam);
378  regionIt->latRange.maximum = std::max(regionIt->latRange.maximum, latParam);
379  }
380  }
381  // expand regions if more than one, ordered from right to left
382  ::ad_rss::physics::ParametricValue lonMinParam(1.0);
383  ::ad_rss::physics::ParametricValue lonMaxParam(0.0);
384 
385  for (auto &region : occupiedRegions) {
386  lonMinParam = std::min(lonMinParam, region.lonRange.minimum);
387  lonMaxParam = std::max(lonMaxParam, region.lonRange.maximum);
388  if (region != occupiedRegions.front()) {
389  // not the most right one, so extend lat maximum
390  region.latRange.maximum = ::ad_rss::physics::ParametricValue(1.0);
391  }
392  if (region != occupiedRegions.back()) {
393  // not the most left one, so extend lat minimum
394  region.latRange.minimum = ::ad_rss::physics::ParametricValue(0.0);
395  }
396  }
397  for (auto &region : occupiedRegions) {
398  region.lonRange.minimum = lonMinParam;
399  region.lonRange.maximum = lonMaxParam;
400  }
401  }

404  const cg::Location &egoVehicleLocation,
405  const cg::Transform &egoVehicleTransform,
406  cc::World &world) const {

410  carla::rpc::ActorId vehicleId = static_cast<carla::rpc::ActorId>(state.objectId);

412  world.GetActors(std::vector<carla::rpc::ActorId>{vehicleId});

417  const float sine = std::sin(yaw * toRadians);

421  cg::Location point = vehicle->GetLocation();

423  csd::Color lonColor{0u, 255u, 0u};

432  lonColor.r = 255u;
433  if (dangerous) {
434  lonColor.g = 0u;
435  } else {
436  lonColor.g = 255u;
437  }
438  }

440  latLColor.r = 255u;
441  if (dangerous) {
442  latLColor.g = 0u;

444  latLColor.g = 255u;

448  latRColor.r = 255u;
449  if (dangerous) {
450  latRColor.g = 0u;
451  } else {
452  latRColor.g = 255u;

456  dh.DrawLine(egoPoint - lineOffset, point - lineOffset, 0.1f, latLColor, 0.02f, false);
457  dh.DrawLine(egoPoint + lineOffset, point + lineOffset, 0.1f, latRColor, 0.02f, false);
458  point.z += 3.f;
459  dh.DrawPoint(point, 0.2f, indicatorColor, 0.02f, false);
460  }
461  }
462  }
