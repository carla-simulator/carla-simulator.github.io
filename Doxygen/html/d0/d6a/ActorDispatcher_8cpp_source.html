42 TPair<EActorSpawnResultStatus, FActorView> UActorDispatcher::SpawnActor(
43  const FTransform &Transform,
44  FActorDescription Description,
45  FActorView::IdType DesiredId)
46 {
47  if ((Description.UId == 0u) || (Description.UId > static_cast<uint32>(SpawnFunctions.Num())))
48  {
49  UE_LOG(LogCarla, Error, TEXT("Invalid ActorDescription '%s' (UId=%d)"), *Description.Id, Description.UId);
50  return MakeTuple(EActorSpawnResultStatus::InvalidDescription, FActorView());
51  }
52 
53  UE_LOG(LogCarla, Log, TEXT("Spawning actor '%s'"), *Description.Id);
54 
55  Description.Class = Classes[Description.UId - 1];
56  auto Result = SpawnFunctions[Description.UId - 1](Transform, Description);
57 
58  if ((Result.Status == EActorSpawnResultStatus::Success) && (Result.Actor == nullptr))
59  {
60  UE_LOG(LogCarla, Warning, TEXT("ActorSpawnResult: Trying to spawn '%s'"), *Description.Id);
61  UE_LOG(LogCarla, Warning, TEXT("ActorSpawnResult: Reported success but did not return an actor"));
62  Result.Status = EActorSpawnResultStatus::UnknownError;
63  }
64 
65  auto View = Result.IsValid() ? RegisterActor(*Result.Actor, std::move(Description), DesiredId) : FActorView();
66 
67  if (!View.IsValid())
68  {
69  UE_LOG(LogCarla, Warning, TEXT("Failed to spawn actor '%s'"), *Description.Id);
70  check(Result.Status != EActorSpawnResultStatus::Success);
71  }
72  else
73  {
74  ATagger::TagActor(*View.GetActor(), true);
75  }
76 
77  return MakeTuple(Result.Status, View);
78 }

80 bool UActorDispatcher::DestroyActor(AActor *Actor)
81 {
82  if (Actor == nullptr) {
83  UE_LOG(LogCarla, Error, TEXT("Trying to destroy nullptr actor"));
84  return false;
85  }
86 
87  // Check if the actor is in the registry.
88  auto View = Registry.Find(Actor);
89  if (!View.IsValid()) {
90  UE_LOG(LogCarla, Warning, TEXT("Trying to destroy actor that is not in the registry"));
91  return false;
92  }
93  const auto &Id = View.GetActorInfo()->Description.Id;
94 
95  // Destroy its controller if present.
96  auto Pawn = Cast<APawn>(Actor);
97  auto Controller = (Pawn != nullptr ? Pawn->GetController() : nullptr);
98  if (Controller != nullptr)
99  {
100  UE_LOG(LogCarla, Log, TEXT("Destroying actor's controller: '%s'"), *Id);
101  auto Success = Controller->Destroy();
102  if (!Success)
103  {
104  UE_LOG(LogCarla, Error, TEXT("Failed to destroy actor's controller: '%s'"), *Id);
105  }
106  }
107 
108  // Destroy the actor.
109  UE_LOG(LogCarla, Log, TEXT("Destroying actor: '%s'"), *Id);
110  if (Actor->Destroy())
111  {
112  return true;
113  }
114  UE_LOG(LogCarla, Error, TEXT("Failed to destroy actor: '%s'"), *Id);
115  return false;
116 }
