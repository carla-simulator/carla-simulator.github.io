42 TPair<EActorSpawnResultStatus, FCarlaActor*> UActorDispatcher::SpawnActor(
43  const FTransform &Transform,
44  FActorDescription Description,
45  FCarlaActor::IdType DesiredId)
46 {
47  if ((Description.UId == 0u) || (Description.UId > static_cast<uint32>(SpawnFunctions.Num())))
48  {
49  UE_LOG(LogCarla, Error, TEXT("Invalid ActorDescription '%s' (UId=%d)"), *Description.Id, Description.UId);
50  return MakeTuple(EActorSpawnResultStatus::InvalidDescription, nullptr);
51  }
52 
53  UE_LOG(LogCarla, Log, TEXT("Spawning actor '%s'"), *Description.Id);
54 
55  Description.Class = Classes[Description.UId - 1];
56  FActorSpawnResult Result = SpawnFunctions[Description.UId - 1](Transform, Description);
57 
58  if ((Result.Status == EActorSpawnResultStatus::Success) && (Result.Actor == nullptr))
59  {
60  UE_LOG(LogCarla, Warning, TEXT("ActorSpawnResult: Trying to spawn '%s'"), *Description.Id);
61  UE_LOG(LogCarla, Warning, TEXT("ActorSpawnResult: Reported success but did not return an actor"));
62  Result.Status = EActorSpawnResultStatus::UnknownError;
63  }
64 
65  FCarlaActor* View = Result.IsValid() ?
66  RegisterActor(*Result.Actor, std::move(Description), DesiredId) : nullptr;
67  if (!View)
68  {
69  UE_LOG(LogCarla, Warning, TEXT("Failed to spawn actor '%s'"), *Description.Id);
70  check(Result.Status != EActorSpawnResultStatus::Success);
71  }
72  else
73  {
74  ATagger::TagActor(*View->GetActor(), true);
75  }
76 
77  return MakeTuple(Result.Status, View);
78 }

111 bool UActorDispatcher::DestroyActor(FCarlaActor::IdType ActorId)
112 {
113  // Check if the actor is in the registry.
114  FCarlaActor* View = Registry.FindCarlaActor(ActorId);
115 
116  // Invalid destruction if is not marked to PendingKill (except is dormant, dormant actors can be destroyed)
117  if (!View)
118  {
119  UE_LOG(LogCarla, Warning, TEXT("Trying to destroy actor that is not in the registry"));
120  return false;
121  }
122 
123  const FString &Id = View->GetActorInfo()->Description.Id;
124 
125  // Destroy its controller if present.
126  AActor* Actor = View->GetActor();
127  if(Actor)
128  {
129  APawn* Pawn = Cast<APawn>(Actor);
130  AController* Controller = (Pawn != nullptr ? Pawn->GetController() : nullptr);
131  if (Controller != nullptr)
132  {
133  UE_LOG(LogCarla, Log, TEXT("Destroying actor's controller: '%s'"), *Id);
134  bool Success = Controller->Destroy();
135  if (!Success)
136  {
137  UE_LOG(LogCarla, Error, TEXT("Failed to destroy actor's controller: '%s'"), *Id);
138  }
139  }
140 
141  // Destroy the actor.
142  UE_LOG(LogCarla, Log, TEXT("UActorDispatcher::Destroying actor: '%s' %x"), *Id, Actor);
143  UE_LOG(LogCarla, Log, TEXT(" %s"), Actor?*Actor->GetName():*FString("None"));
144  if (!Actor || !Actor->Destroy())
145  {
146  UE_LOG(LogCarla, Error, TEXT("Failed to destroy actor: '%s'"), *Id);
147  return false;
148  }
149  }
150 
151  Registry.Deregister(ActorId);
152 
153  return true;
154 }
