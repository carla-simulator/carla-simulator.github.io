40 TPair<EActorSpawnResultStatus, FActorView> UActorDispatcher::SpawnActor(
41  const FTransform &Transform,
42  FActorDescription Description,
43  FActorView::IdType DesiredId)
44 {
45  if ((Description.UId == 0u) || (Description.UId > static_cast<uint32>(SpawnFunctions.Num())))
46  {
47  UE_LOG(LogCarla, Error, TEXT("Invalid ActorDescription '%s' (UId=%d)"), *Description.Id, Description.UId);
48  return MakeTuple(EActorSpawnResultStatus::InvalidDescription, FActorView());
49  }
50 
51  UE_LOG(LogCarla, Log, TEXT("Spawning actor '%s'"), *Description.Id);
52 
53  Description.Class = Classes[Description.UId - 1];
54  auto Result = SpawnFunctions[Description.UId - 1](Transform, Description);
55 
56  if ((Result.Status == EActorSpawnResultStatus::Success) && (Result.Actor == nullptr))
57  {
58  UE_LOG(LogCarla, Warning, TEXT("ActorSpawnResult: Trying to spawn '%s'"), *Description.Id);
59  UE_LOG(LogCarla, Warning, TEXT("ActorSpawnResult: Reported success but did not return an actor"));
60  Result.Status = EActorSpawnResultStatus::UnknownError;
61  }
62 
63  auto View = Result.IsValid() ? RegisterActor(*Result.Actor, std::move(Description), DesiredId) : FActorView();
64 
65  if (!View.IsValid())
66  {
67  UE_LOG(LogCarla, Warning, TEXT("Failed to spawn actor '%s'"), *Description.Id);
68  check(Result.Status != EActorSpawnResultStatus::Success);
69  }
70 
71  return MakeTuple(Result.Status, View);
72 }

74 bool UActorDispatcher::DestroyActor(AActor *Actor)
75 {
76  if (Actor == nullptr) {
77  UE_LOG(LogCarla, Error, TEXT("Trying to destroy nullptr actor"));
78  return false;
79  }
80 
81  // Check if the actor is in the registry.
82  auto View = Registry.Find(Actor);
83  if (!View.IsValid()) {
84  UE_LOG(LogCarla, Warning, TEXT("Trying to destroy actor that is not in the registry"));
85  return false;
86  }
87  const auto &Id = View.GetActorInfo()->Description.Id;
88 
89  // Destroy its controller if present.
90  auto Pawn = Cast<APawn>(Actor);
91  auto Controller = (Pawn != nullptr ? Pawn->GetController() : nullptr);
92  if (Controller != nullptr)
93  {
94  UE_LOG(LogCarla, Log, TEXT("Destroying actor's controller: '%s'"), *Id);
95  auto Success = Controller->Destroy();
96  if (!Success)
97  {
98  UE_LOG(LogCarla, Error, TEXT("Failed to destroy actor's controller: '%s'"), *Id);
99  }
100  }
101 
102  // Destroy the actor.
103  UE_LOG(LogCarla, Log, TEXT("Destroying actor: '%s'"), *Id);
104  if (Actor->Destroy())
105  {
106  return true;
107  }
108  UE_LOG(LogCarla, Error, TEXT("Failed to destroy actor: '%s'"), *Id);
109  return false;
110 }
