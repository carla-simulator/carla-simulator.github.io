54  void RotateVector(Vector3D &in_point) const {
55  // Rotates Rz(yaw) * Ry(pitch) * Rx(roll) = first x, then y, then z.
56  const float cy = std::cos(Math::ToRadians(yaw));
57  const float sy = std::sin(Math::ToRadians(yaw));
58  const float cr = std::cos(Math::ToRadians(roll));
59  const float sr = std::sin(Math::ToRadians(roll));
60  const float cp = std::cos(Math::ToRadians(pitch));
61  const float sp = std::sin(Math::ToRadians(pitch));
62 
63  Vector3D out_point;
64  out_point.x =
65  in_point.x * (cp * cy) +
66  in_point.y * (cy * sp * sr - sy * cr) +
67  in_point.z * (-cy * sp * cr - sy * sr);
68 
69  out_point.y =
70  in_point.x * (cp * sy) +
71  in_point.y * (sy * sp * sr + cy * cr) +
72  in_point.z * (-sy * sp * cr + cy * sr);
73 
74  out_point.z =
75  in_point.x * (sp) +
76  in_point.y * (-cp * sr) +
77  in_point.z * (cp * cr);
78 
79  in_point = out_point;
80  }

82  void InverseRotateVector(Vector3D &in_point) const {
83  // Applies the transposed of the matrix used in RotateVector function,
84  // which is the rotation inverse.
85  const float cy = std::cos(Math::ToRadians(yaw));
86  const float sy = std::sin(Math::ToRadians(yaw));
87  const float cr = std::cos(Math::ToRadians(roll));
88  const float sr = std::sin(Math::ToRadians(roll));
89  const float cp = std::cos(Math::ToRadians(pitch));
90  const float sp = std::sin(Math::ToRadians(pitch));
91 
92  Vector3D out_point;
93  out_point.x =
94  in_point.x * (cp * cy) +
95  in_point.y * (cp * sy) +
96  in_point.z * (sp);
97 
98  out_point.y =
99  in_point.x * (cy * sp * sr - sy * cr) +
100  in_point.y * (sy * sp * sr + cy * cr) +
101  in_point.z * (-cp * sr);
102 
103  out_point.z =
104  in_point.x * (-cy * sp * cr - sy * sr) +
105  in_point.y * (-sy * sp * cr + cy * sr) +
106  in_point.z * (cp * cr);
107 
108  in_point = out_point;
109  }
