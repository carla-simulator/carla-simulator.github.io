54  void MeasureExecutionTime(std::string snippet_name, bool begin_or_end) {
55 
56  if (print_clocks.find(snippet_name) == print_clocks.end()) {
57  print_clocks.insert({snippet_name, chr::system_clock::now()});
58  }
59  if (snippet_clocks.find(snippet_name) == snippet_clocks.end()) {
60  snippet_clocks.insert({snippet_name, TimePoint()});
61  }
62  if (snippet_durations.find(snippet_name) == snippet_durations.end()) {
63  snippet_durations.insert({snippet_name, chr::duration<float>()});
64  }
65 
66  TimePoint current_time = chr::system_clock::now();
67  TimePoint& print_clock = print_clocks.at(snippet_name);
68  TimePoint& snippet_clock = snippet_clocks.at(snippet_name);
69  chr::duration<float>& snippet_duration = snippet_durations.at(snippet_name);
70 
71  if (begin_or_end) {
72  snippet_clock = current_time;
73  } else {
74  chr::duration<float> measured_duration = current_time - snippet_clock;
75  snippet_duration = (measured_duration + snippet_duration) / 2.0f;
76  }
77 
78  chr::duration<float> print_duration = current_time - print_clock;
79  if (print_duration.count() > 1.0f) {
80  std::cout << "Snippet name : " << snippet_name << ", "
81  << "avg. duration : " << 1000 * snippet_duration.count()
82  << " ms" << std::endl;
83  print_clock = current_time;
84  }
