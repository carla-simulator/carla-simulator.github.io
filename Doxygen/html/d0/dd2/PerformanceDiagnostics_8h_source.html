55  void MeasureExecutionTime(std::string snippet_name, bool begin_or_end) {
56 
57  if (print_clocks.find(snippet_name) == print_clocks.end()) {
58  print_clocks.insert({snippet_name, chr::system_clock::now()});
59  }
60  if (snippet_clocks.find(snippet_name) == snippet_clocks.end()) {
61  snippet_clocks.insert({snippet_name, TimePoint()});
62  }
63  if (snippet_durations.find(snippet_name) == snippet_durations.end()) {
64  snippet_durations.insert({snippet_name, chr::duration<float>()});
65  }
66  if (number_of_calls.find(snippet_name) == number_of_calls.end()) {
67  number_of_calls.insert({snippet_name, 0u});
68  }
69 
70  TimePoint current_time = chr::system_clock::now();
71  TimePoint& print_clock = print_clocks.at(snippet_name);
72  TimePoint& snippet_clock = snippet_clocks.at(snippet_name);
73  chr::duration<float>& snippet_duration = snippet_durations.at(snippet_name);
74  unsigned long& call_count = number_of_calls.at(snippet_name);
75 
76  if (begin_or_end) {
77  snippet_clock = current_time;
78  } else {
79  chr::duration<float> measured_duration = current_time - snippet_clock;
80  snippet_duration += measured_duration;
81  ++call_count;
82  }
83 
84  chr::duration<float> print_duration = current_time - print_clock;
85  if (print_duration.count() > 1.0f) {
86  call_count = call_count == 0u ? 1 : call_count;
87  std::cout << "Snippet name : " << snippet_name << ", "
88  << "avg. duration : " << 1000 * snippet_duration.count() / call_count << " ms, "
89  << "total duration : " << snippet_duration.count() << " s, "
90  << "total calls : " << call_count << ", "
91  << std::endl;
92 
93  snippet_duration = 0s;
94  call_count = 0u;
95 
96  print_clock = current_time;
97  }
98  }
