44 void UCarlaBlueprintRegistry::AddToCarlaBlueprintRegistry(const TArray<FPropParameters> &PropParametersArray)
45 {
46  TArray<TSharedPtr<FJsonValue>> ResultPropJsonArray;
47 
48  // Load default props file
49  FString DefaultPropFilePath = CommonAttributes::PATH + CommonAttributes::DEFAULT +
50  PropAttributes::REGISTRY_FORMAT;
51  FString JsonString;
52  FFileHelper::LoadFileToString(JsonString, *DefaultPropFilePath);
53 
54  TSharedPtr<FJsonObject> JsonObject = MakeShareable(new FJsonObject());
55  TSharedRef<TJsonReader<>> JsonReader = TJsonReaderFactory<>::Create(JsonString);
56 
57  // Fill Array of Props and save indexes of each prop in PropIndexes Map
58  TMap<FString, int> PropIndexes;
59  if (FJsonSerializer::Deserialize(JsonReader, JsonObject) && JsonObject.IsValid())
60  {
61  ResultPropJsonArray = JsonObject->GetArrayField(CommonAttributes::DEFINITIONS);
62  for (int32 i = 0; i < ResultPropJsonArray.Num(); ++i)
63  {
64  TSharedPtr<FJsonObject> PropJsonObject = ResultPropJsonArray[i]->AsObject();
65 
66  FString Name = PropJsonObject->GetStringField(PropAttributes::NAME);
67 
68  PropIndexes.Add(Name, i);
69  }
70  }
71 
72  // Add Input Props or Update them if already exists
73  for (auto &PropParameter : PropParametersArray)
74  {
75  TSharedPtr<FJsonObject> PropJsonObject;
76 
77  // Create object or update existing one
78  int *PropIndex = PropIndexes.Find(PropParameter.Name);
79  if (PropIndex)
80  {
81  PropJsonObject = ResultPropJsonArray[*PropIndex]->AsObject();
82  }
83  else
84  {
85  PropJsonObject = MakeShareable(new FJsonObject);
86  }
87 
88  // Fill Prop Json
89  PropJsonObject->SetStringField(PropAttributes::NAME, PropParameter.Name);
90  PropJsonObject->SetStringField(PropAttributes::MESH_PATH, PropParameter.Mesh->GetPathName());
91  PropJsonObject->SetStringField(PropAttributes::SIZE, PropSizeTypeToString(PropParameter.Size));
92 
93  // Add or Update
94  TSharedRef<FJsonValue> PropJsonValue = MakeShareable(new FJsonValueObject(PropJsonObject));
95  if (PropIndex)
96  {
97  ResultPropJsonArray[*PropIndex] = PropJsonValue;
98  }
99  else
100  {
101  ResultPropJsonArray.Add(PropJsonValue);
102 
103  PropIndexes.Add(PropParameter.Name, ResultPropJsonArray.Num() - 1);
104  }
105  }
106 
107  // Update Json Object
108  JsonObject->SetArrayField(CommonAttributes::DEFINITIONS, ResultPropJsonArray);
109 
110  // Serialize file
111  FString OutputString;
112  TSharedRef<TJsonWriter<>> Writer = TJsonWriterFactory<>::Create(&OutputString);
113  FJsonSerializer::Serialize(JsonObject.ToSharedRef(), Writer);
114 
115  // Save file
116  FFileHelper::SaveStringToFile(OutputString, *DefaultPropFilePath);
117 
118 }

120 void UCarlaBlueprintRegistry::LoadPropDefinitions(TArray<FPropParameters> &PropParametersArray)
121 {
122 
123  // Find all Package.json files inside Unreal Content folder
124  const FString WildCard = FString("*").Append(PropAttributes::REGISTRY_FORMAT);
125 
126  TArray<FString> PropFileNames;
127  IFileManager::Get().FindFilesRecursive(PropFileNames,
128  *CommonAttributes::PATH,
129  *WildCard,
130  true,
131  false,
132  false);
133 
134  // Sort and place Default File First if it exists
135  PropFileNames.Sort();
136  FString DefaultFileName;
137  for (int32 i = 0; i < PropFileNames.Num() && DefaultFileName.IsEmpty(); ++i)
138  {
139  if (PropFileNames[i].Contains(CommonAttributes::DEFAULT))
140  {
141  DefaultFileName = PropFileNames[i];
142  PropFileNames.RemoveAt(i);
143  }
144  }
145  if (!DefaultFileName.IsEmpty())
146  {
147  PropFileNames.Insert(DefaultFileName, 0);
148  }
149 
150  // Read all registry files and overwrite default registry values with user
151  // registry files
152  TMap<FString, int> PropIndexes;
153 
154  for (int32 i = 0; i < PropFileNames.Num(); ++i)
155  {
156  FString FileJsonContent;
157  if (FFileHelper::LoadFileToString(FileJsonContent, *PropFileNames[i]))
158  {
159  TSharedPtr<FJsonObject> JsonParsed;
160  TSharedRef<TJsonReader<TCHAR>> JsonReader = TJsonReaderFactory<TCHAR>::Create(FileJsonContent);
161  if (FJsonSerializer::Deserialize(JsonReader, JsonParsed))
162  {
163  auto PropJsonArray = JsonParsed->GetArrayField(CommonAttributes::DEFINITIONS);
164 
165  for (auto &PropJsonValue : PropJsonArray)
166  {
167  // Read Prop Json
168  TSharedPtr<FJsonObject> PropJsonObject = PropJsonValue->AsObject();
169 
170  FString PropName = PropJsonObject->GetStringField(PropAttributes::NAME);
171  FString PropMeshPath = PropJsonObject->GetStringField(PropAttributes::MESH_PATH);
172  FString PropSize = PropJsonObject->GetStringField(PropAttributes::SIZE);
173 
174  // Build Prop Parameter
175  UStaticMesh *PropMesh = LoadObject<UStaticMesh>(nullptr, *PropMeshPath);
176  EPropSize PropSizeType = StringToPropSizeType(PropSize);
177  FPropParameters Params {PropName, PropMesh, PropSizeType};
178 
179  // Add or Update
180  if (PropIndexes.Contains(PropName))
181  {
182  PropParametersArray[PropIndexes[PropName]] = Params;
183  }
184  else
185  {
186  PropParametersArray.Add(Params);
187  PropIndexes.Add(Params.Name, PropParametersArray.Num() - 1);
188  }
189  }
190  }
191  }
192  }
