57 void ACarlaRecorder::Tick(float DeltaSeconds)
58 {
59  Super::Tick(DeltaSeconds);
60 
61  if (!Episode)
62  return;
63 
64  // check if recording
65  if (Enabled)
66  {
67  const FActorRegistry &Registry = Episode->GetActorRegistry();
68 
69  // through all actors in registry
70  for (auto It = Registry.begin(); It != Registry.end(); ++It)
71  {
72  FActorView View = *It;
73 
74  switch (View.GetActorType())
75  {
76  // save the transform of all vehicles
77  case FActorView::ActorType::Vehicle:
78  AddActorPosition(View);
79  AddVehicleAnimation(View);
80  break;
81 
82  // save the transform of all walkers
83  case FActorView::ActorType::Walker:
84  AddActorPosition(View);
85  AddWalkerAnimation(View);
86  break;
87 
88  // save the state of each traffic light
89  case FActorView::ActorType::TrafficLight:
90  AddTrafficLightState(View);
91  break;
92  }
93  }
94 
95  // write all data for this frame
96  Write(DeltaSeconds);
97  }
98  else if (Episode->GetReplayer()->IsEnabled())
99  {
100  // replayer
101  Episode->GetReplayer()->Tick(DeltaSeconds);
102  }
103 }

203 std::string ACarlaRecorder::Start(std::string Name, FString MapName)
204 {
205  // stop replayer if any in course
206  if (Replayer.IsEnabled())
207  Replayer.Stop();
208 
209  // stop recording
210  Stop();
211 
212  // reset collisions Id
213  NextCollisionId = 0;
214 
215  // get the final path + filename
216  std::string Filename = GetRecorderFilename(Name);
217 
218  // binary file
219  File.open(Filename, std::ios::binary);
220  if (!File.is_open())
221  {
222  return "";
223  }
224 
225  // save info
226  Info.Version = 1;
227  Info.Magic = TEXT("CARLA_RECORDER");
228  Info.Date = std::time(0);
229  Info.Mapfile = MapName;
230 
231  // write general info
232  Info.Write(File);
233 
234  Frames.Reset();
235 
236  Enable();
237 
238  // add all existing actors
239  AddExistingActors();
240 
241  return std::string(Filename);
242 }

328 void ACarlaRecorder::AddCollision(AActor *Actor1, AActor *Actor2)
329 {
330  if (Enabled)
331  {
332  CarlaRecorderCollision Collision;
333 
334  // some inits
335  Collision.Id = NextCollisionId++;
336  Collision.IsActor1Hero = false;
337  Collision.IsActor2Hero = false;
338 
339  // check actor 1
340  if (Episode->GetActorRegistry().Find(Actor1).GetActorInfo() != nullptr)
341  {
342  auto *Role = Episode->GetActorRegistry().Find(Actor1).GetActorInfo()->Description.Variations.Find("role_name");
343  if (Role != nullptr)
344  Collision.IsActor1Hero = (Role->Value == "hero");
345  }
346  Collision.DatabaseId1 = Episode->GetActorRegistry().Find(Actor1).GetActorId();
347 
348  // check actor 2
349  if (Episode->GetActorRegistry().Find(Actor2).GetActorInfo() != nullptr)
350  {
351  auto Role = Episode->GetActorRegistry().Find(Actor2).GetActorInfo()->Description.Variations.Find("role_name");
352  if (Role != nullptr)
353  Collision.IsActor2Hero = (Role->Value == "hero");
354  }
355  Collision.DatabaseId2 = Episode->GetActorRegistry().Find(Actor2).GetActorId();
356 
357  Collisions.Add(std::move(Collision));
358  }
359 }

404 void ACarlaRecorder::CreateRecorderEventAdd(
405  uint32_t DatabaseId,
406  uint8_t Type,
407  const FTransform &Transform,
408  FActorDescription ActorDescription)
409 {
410  CarlaRecorderActorDescription Description;
411  Description.UId = ActorDescription.UId;
412  Description.Id = ActorDescription.Id;
413 
414  // attributes
415  Description.Attributes.reserve(ActorDescription.Variations.Num());
416  for (const auto &item : ActorDescription.Variations)
417  {
418  CarlaRecorderActorAttribute Attr;
419  Attr.Type = static_cast<uint8_t>(item.Value.Type);
420  Attr.Id = item.Value.Id;
421  Attr.Value = item.Value.Value;
422  // check for empty attributes
423  if (!Attr.Id.IsEmpty())
424  {
425  Description.Attributes.emplace_back(std::move(Attr));
426  }
427  }
428 
429  // recorder event
430  CarlaRecorderEventAdd RecEvent
431  {
432  DatabaseId,
433  Type,
434  Transform.GetTranslation(),
435  Transform.GetRotation().Euler(),
436  std::move(Description)
437  };
438  AddEvent(std::move(RecEvent));
439 }
