72 void ACarlaRecorder::Ticking(float DeltaSeconds)
73 {
74  TRACE_CPUPROFILER_EVENT_SCOPE(ACarlaRecorder::Ticking);
75  Super::Tick(DeltaSeconds);
76 
77  if (!Episode)
78  return;
79 
80  // check if recording
81  if (Enabled)
82  {
83  PlatformTime.UpdateTime();
84  const FActorRegistry &Registry = Episode->GetActorRegistry();
85 
86  // through all actors in registry
87  for (auto It = Registry.begin(); It != Registry.end(); ++It)
88  {
89  FCarlaActor* View = It.Value().Get();
90 
91  switch (View->GetActorType())
92  {
93  // save the transform for props
94  case FCarlaActor::ActorType::Other:
95  AddActorPosition(View);
96  break;
97 
98  // save the transform of all vehicles
99  case FCarlaActor::ActorType::Vehicle:
100  AddActorPosition(View);
101  AddVehicleAnimation(View);
102  AddVehicleLight(View);
103  if (bAdditionalData)
104  {
105  AddActorKinematics(View);
106  }
107  break;
108 
109  // save the transform of all walkers
110  case FCarlaActor::ActorType::Walker:
111  AddActorPosition(View);
112  AddWalkerAnimation(View);
113  if (bAdditionalData)
114  {
115  AddActorKinematics(View);
116  }
117  break;
118 
119  // save the state of each traffic light
120  case FCarlaActor::ActorType::TrafficLight:
121  AddTrafficLightState(View);
122  break;
123  }
124  }
125 
126  // write all data for this frame
127  Write(DeltaSeconds);
128  }
129  else if (Episode->GetReplayer()->IsEnabled())
130  {
131  // replayer
132  Episode->GetReplayer()->Tick(DeltaSeconds);
133  }
134 }

312 std::string ACarlaRecorder::Start(std::string Name, FString MapName, bool AdditionalData)
313 {
314  // stop replayer if any in course
315  if (Replayer.IsEnabled())
316  Replayer.Stop();
317 
318  // stop recording
319  Stop();
320 
321  // reset collisions Id
322  NextCollisionId = 0;
323 
324  // get the final path + filename
325  std::string Filename = GetRecorderFilename(Name);
326 
327  // binary file
328  File.open(Filename, std::ios::binary);
329  if (!File.is_open())
330  {
331  return "";
332  }
333 
334  // save info
335  Info.Version = 1;
336  Info.Magic = TEXT("CARLA_RECORDER");
337  Info.Date = std::time(0);
338  Info.Mapfile = MapName;
339 
340  // write general info
341  Info.Write(File);
342 
343  Frames.Reset();
344  PlatformTime.SetStartTime();
345 
346  Enable();
347 
348  bAdditionalData = AdditionalData;
349 
350  // add all existing actors
351  AddExistingActors();
352 
353  return std::string(Filename);

590 void ACarlaRecorder::CreateRecorderEventAdd(
591  uint32_t DatabaseId,
592  uint8_t Type,
593  const FTransform &Transform,
594  FActorDescription ActorDescription)
595 {
596  CarlaRecorderActorDescription Description;
597  Description.UId = ActorDescription.UId;
598  Description.Id = ActorDescription.Id;
599 
600  // attributes
601  Description.Attributes.reserve(ActorDescription.Variations.Num());
602  for (const auto &item : ActorDescription.Variations)
603  {
604  CarlaRecorderActorAttribute Attr;
605  Attr.Type = static_cast<uint8_t>(item.Value.Type);
606  Attr.Id = item.Value.Id;
607  Attr.Value = item.Value.Value;
608  // check for empty attributes
609  if (!Attr.Id.IsEmpty())
610  {
611  Description.Attributes.emplace_back(std::move(Attr));
612  }
613  }
614 
615  // recorder event
616  CarlaRecorderEventAdd RecEvent
617  {
618  DatabaseId,
619  Type,
620  Transform.GetTranslation(),
621  Transform.GetRotation().Euler(),
622  std::move(Description)
623  };
624  AddEvent(std::move(RecEvent));
625 
626  FCarlaActor* CarlaActor = Episode->FindCarlaActor(DatabaseId);
627  // Other events related to spawning actors
628  // check if it is a vehicle to get initial physics control
629  ACarlaWheeledVehicle* Vehicle = Cast<ACarlaWheeledVehicle>(CarlaActor->GetActor());
630  if (Vehicle)
631  {
632  AddPhysicsControl(*Vehicle);
633  }
634 
635  ATrafficLightBase* TrafficLight = Cast<ATrafficLightBase>(CarlaActor->GetActor());
636  if (TrafficLight)
637  {
638  AddTrafficLightTime(*TrafficLight);
639  }
640 
641  ATrafficSignBase* TrafficSign = Cast<ATrafficSignBase>(CarlaActor->GetActor());
642  if (TrafficSign)
643  {
644  // Trigger volume in global coordinates
645  AddTriggerVolume(*TrafficSign);
646  }
647  else
648  {
649  // Bounding box in local coordinates
650  AddActorBoundingBox(CarlaActor);
651  }
652 }
