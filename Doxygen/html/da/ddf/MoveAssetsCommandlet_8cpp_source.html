91 void MoveFiles(const TArray<UObject *> &Assets, const FString &DestPath)
92 {
93  check(DestPath.Len() > 0);
94 
95  FAssetToolsModule &AssetToolsModule = FModuleManager::LoadModuleChecked<FAssetToolsModule>("AssetTools");
96  TArray<FAssetRenameData> AssetsAndNames;
97  for (auto AssetIt = Assets.CreateConstIterator(); AssetIt; ++AssetIt)
98  {
99  UObject *Asset = *AssetIt;
100 
101  if (!ensure(Asset))
102  {
103  continue;
104  }
105  AssetsAndNames.Emplace(Asset, DestPath, Asset->GetName());
106  }
107 
108  if (AssetsAndNames.Num() > 0)
109  {
110  AssetToolsModule.Get().RenameAssets(AssetsAndNames);
111  }
112 }

114 void UMoveAssetsCommandlet::MoveAssetsFromMapForSemanticSegmentation(
115  const FString &PackageName,
116  const FString &MapName)
117 {
118  // Prepare a UObjectLibrary for moving assets
119  const FString SrcPath = TEXT("/Game/") + PackageName + TEXT("/Maps/") + MapName;
120  AssetsObjectLibrary->LoadAssetDataFromPath(*SrcPath);
121  AssetsObjectLibrary->LoadAssetsFromAssetData();
122 
123  // Load Assets to move
124  MapContents.Empty();
125  AssetsObjectLibrary->GetAssetDataList(MapContents);
126  AssetsObjectLibrary->ClearLoaded();
127 
128  TArray<FString> DestinationPaths = {SSTags::ROAD, SSTags::ROADLINE, SSTags::TERRAIN, SSTags::GRASS, SSTags::SIDEWALK, SSTags::CURB, SSTags::GUTTER};
129 
130  // Init Map with keys
131  TMap<FString, TArray<UObject *>> AssetDataMap;
132  for (const auto &Paths : DestinationPaths)
133  {
134  AssetDataMap.Add(Paths, {});
135  }
136 
137  for (const auto &MapAsset : MapContents)
138  {
139  // Get AssetName
140  FString AssetName;
141  UStaticMesh *MeshAsset = CastChecked<UStaticMesh>(MapAsset.GetAsset());
142  MapAsset.AssetName.ToString(AssetName);
143 
144  if (SrcPath.Len())
145  {
146 
147  const FString CurrentPackageName = MeshAsset->GetOutermost()->GetName();
148 
149  if (!ensure(CurrentPackageName.StartsWith(SrcPath)))
150  {
151  continue;
152  }
153 
154  // Bind between tags and classify assets according to semantic
155  // segmentation
156  if (AssetName.Contains(SSTags::R_ROAD1) || AssetName.Contains(SSTags::R_ROAD2))
157  {
158  AssetDataMap[SSTags::ROAD].Add(MeshAsset);
159  }
160  else if (AssetName.Contains(SSTags::R_MARKING1) || AssetName.Contains(SSTags::R_MARKING2))
161  {
162  AssetDataMap[SSTags::ROADLINE].Add(MeshAsset);
163  }
164  else if (AssetName.Contains(SSTags::R_TERRAIN))
165  {
166  AssetDataMap[SSTags::TERRAIN].Add(MeshAsset);
167  }
168  else if (AssetName.Contains(SSTags::R_SIDEWALK1) || AssetName.Contains(SSTags::R_SIDEWALK2))
169  {
170  AssetDataMap[SSTags::SIDEWALK].Add(MeshAsset);
171  }
172  else if (AssetName.Contains(SSTags::R_CURB1) || AssetName.Contains(SSTags::R_CURB2))
173  {
174  AssetDataMap[SSTags::CURB].Add(MeshAsset);
175  }
176  else if (AssetName.Contains(SSTags::R_GUTTER1) || AssetName.Contains(SSTags::R_GUTTER2))
177  {
178  AssetDataMap[SSTags::GUTTER].Add(MeshAsset);
179  }
180  else
181  {
182  AssetDataMap[SSTags::TERRAIN].Add(MeshAsset);
183  }
184  }
185  }
186 
187  // Move assets to correspoding semantic segmentation folders
188  for (const auto &Elem : AssetDataMap)
189  {
190  FString DestPath = TEXT("/Game/") + PackageName + TEXT("/Static/") + Elem.Key + "/" + MapName;
191 
192  MoveFiles(Elem.Value, DestPath);
193  }
194 }
