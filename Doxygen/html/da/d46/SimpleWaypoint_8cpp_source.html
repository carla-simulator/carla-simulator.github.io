67  void SimpleWaypoint::SetLeftWaypoint(SimpleWaypointPtr _waypoint) {
68 
69  const cg::Vector3D heading_vector = waypoint->GetTransform().GetForwardVector();
70  const cg::Vector3D relative_vector = GetLocation() - _waypoint->GetLocation();
71  if ((heading_vector.x * relative_vector.y - heading_vector.y * relative_vector.x) > 0.0f) {
72  next_left_waypoint = _waypoint;
73  } else {
74  throw std::invalid_argument("Argument not on the left side!");
75  }
76  }
