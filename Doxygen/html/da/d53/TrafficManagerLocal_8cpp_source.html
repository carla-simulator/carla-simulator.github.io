208 bool TrafficManagerLocal::CheckAllFrozen(TLGroup tl_to_freeze) {
209  for (auto& elem : tl_to_freeze) {

211  return false;
212  }

217 void TrafficManagerLocal::ResetAllTrafficLights() {
218 
219  auto Filter = [&](auto &actors, auto &wildcard_pattern) {
220  std::vector<carla::client::detail::ActorVariant> filtered;
221  for (auto &&actor : actors) {
222  if (carla::StringUtil::Match(carla::client::detail::ActorVariant(actor).GetTypeId(), wildcard_pattern)) {
223  filtered.push_back(actor);
224  }
225  }
226  return filtered;
227  };
228 
229  // Get all actors of the world.
230  auto world_actorsList = episodeProxyTM.Lock()->GetAllTheActorsInTheEpisode();
231 
232  // Filter based on wildcard pattern.
233  const auto world_traffic_lights = Filter(world_actorsList, "*traffic_light*");
234 
235  std::vector<TLGroup> list_of_all_groups;
236  TLGroup tl_to_freeze;
237  std::vector<carla::ActorId> list_of_ids;
238  for (auto tl : world_traffic_lights) {
239  if (!(std::find(list_of_ids.begin(), list_of_ids.end(), tl.GetId()) != list_of_ids.end())) {
240  const TLGroup tl_group = boost::static_pointer_cast<cc::TrafficLight>(tl.Get(episodeProxyTM))->GetGroupTrafficLights();
241  list_of_all_groups.push_back(tl_group);
242  for (uint64_t i=0u; i<tl_group.size(); i++) {
243  list_of_ids.push_back(tl_group.at(i).get()->GetId());
244  if(i!=0u) {
245  tl_to_freeze.push_back(tl_group.at(i));
246  }
247  }
248  }
249  }
250 
251  for (TLGroup& tl_group : list_of_all_groups) {
252  tl_group.front()->SetState(TLS::Green);
253  std::for_each(
254  tl_group.begin()+1, tl_group.end(),
255  [] (auto& tl) {tl->SetState(TLS::Red);});
256  }
257 
258  while (!CheckAllFrozen(tl_to_freeze)) {
259  for (auto& tln : tl_to_freeze) {
260  tln->SetState(TLS::Red);
261  tln->Freeze(true);
262  }
263  }
264 
265 }
