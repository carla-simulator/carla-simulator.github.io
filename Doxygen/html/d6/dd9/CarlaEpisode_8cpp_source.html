49 bool UCarlaEpisode::LoadNewEpisode(const FString &MapString)
50 {
51  FString FinalPath = MapString.IsEmpty() ? GetMapName() : MapString;
52  bool bIsFileFound = false;
53  if (MapString.StartsWith("/Game"))
54  {
55  // Full path
56  if (!MapString.EndsWith(".umap"))
57  {
58  FinalPath += ".umap";
59  }
60  // Some conversions...
61  FinalPath = FinalPath.Replace(TEXT("/Game/"), *FPaths::ProjectContentDir());
62  if (FPaths::FileExists(IFileManager::Get().ConvertToAbsolutePathForExternalAppForRead(*FinalPath)))
63  {
64  bIsFileFound = true;
65  FinalPath = MapString;
66  }
67  }
68  else
69  {
70  if (MapString.Contains("/"))
71  {
72  bIsFileFound = false;
73  }
74  else
75  {
76  // Find the full path under Carla
77  TArray<FString> TempStrArray, PathList;
78  if (!MapString.EndsWith(".umap"))
79  {
80  FinalPath += ".umap";
81  }
82  IFileManager::Get().FindFilesRecursive(PathList, *FPaths::ProjectContentDir(), *FinalPath, true, false, false);
83  if (PathList.Num() > 0)
84  {
85  FinalPath = PathList[0];
86  FinalPath.ParseIntoArray(TempStrArray, TEXT("Content/"), true);
87  FinalPath = TempStrArray[1];
88  FinalPath.ParseIntoArray(TempStrArray, TEXT("."), true);
89  FinalPath = "/Game/" + TempStrArray[0];
90  bIsFileFound = true;
91  }
92  }
93  }
94  if (bIsFileFound)
95  {
96  UE_LOG(LogCarla, Warning, TEXT("Loading a new episode: %s"), *FinalPath);
97  UGameplayStatics::OpenLevel(GetWorld(), *FinalPath, true);
98  ApplySettings(FEpisodeSettings{});
99  }
100  return bIsFileFound;
101 }

157 void UCarlaEpisode::InitializeAtBeginPlay()
158 {
159  auto World = GetWorld();
160  check(World != nullptr);
161  auto PlayerController = UGameplayStatics::GetPlayerController(World, 0);
162  if (PlayerController == nullptr)
163  {
164  UE_LOG(LogCarla, Error, TEXT("Can't find player controller!"));
165  return;
166  }
167  Spectator = PlayerController->GetPawn();
168  if (Spectator != nullptr)
169  {
170  FActorDescription Description;
171  Description.Id = TEXT("spectator");
172  Description.Class = Spectator->GetClass();
173  ActorDispatcher->RegisterActor(*Spectator, Description);
174  }
175  else
176  {
177  UE_LOG(LogCarla, Error, TEXT("Can't find spectator!"));
178  }
179 
180  for (TActorIterator<ATrafficSignBase> It(World); It; ++It)
181  {
182  ATrafficSignBase *Actor = *It;
183  check(Actor != nullptr);
184  FActorDescription Description;
185  Description.Id = UCarlaEpisode_GetTrafficSignId(Actor->GetTrafficSignState());
186  Description.Class = Actor->GetClass();
187  ActorDispatcher->RegisterActor(*Actor, Description);
188  }
189 
190  for (TActorIterator<AStaticMeshActor> It(World); It; ++It)
191  {
192  auto Actor = *It;
193  check(Actor != nullptr);
194  auto MeshComponent = Actor->GetStaticMeshComponent();
195  check(MeshComponent != nullptr);
196  if (MeshComponent->Mobility == EComponentMobility::Movable)
197  {
198  FActorDescription Description;
199  Description.Id = TEXT("static.prop");
200  Description.Class = Actor->GetClass();
201  ActorDispatcher->RegisterActor(*Actor, Description);
202  }
203  }
204 }
