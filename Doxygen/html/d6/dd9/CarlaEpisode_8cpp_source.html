58 bool UCarlaEpisode::LoadNewEpisode(const FString &MapString)
59 {
60  FString FinalPath = MapString.IsEmpty() ? GetMapName() : MapString;
61  bool bIsFileFound = false;
62  if (MapString.StartsWith("/Game"))
63  {
64  // Full path
65  if (!MapString.EndsWith(".umap"))
66  {
67  FinalPath += ".umap";
68  }
69  // Some conversions...
70  FinalPath = FinalPath.Replace(TEXT("/Game/"), *FPaths::ProjectContentDir());
71  if (FPaths::FileExists(IFileManager::Get().ConvertToAbsolutePathForExternalAppForRead(*FinalPath)))
72  {
73  bIsFileFound = true;
74  FinalPath = MapString;
75  }
76  }
77  else
78  {
79  if (MapString.Contains("/"))
80  {
81  bIsFileFound = false;
82  }
83  else
84  {
85  // Find the full path under Carla
86  TArray<FString> TempStrArray, PathList;
87  if (!MapString.EndsWith(".umap"))
88  {
89  FinalPath += ".umap";
90  }
91  IFileManager::Get().FindFilesRecursive(PathList, *FPaths::ProjectContentDir(), *FinalPath, true, false, false);
92  if (PathList.Num() > 0)
93  {
94  FinalPath = PathList[0];
95  FinalPath.ParseIntoArray(TempStrArray, TEXT("Content/"), true);
96  FinalPath = TempStrArray[1];
97  FinalPath.ParseIntoArray(TempStrArray, TEXT("."), true);
98  FinalPath = "/Game/" + TempStrArray[0];
99  bIsFileFound = true;
100  }
101  }
102  }
103  if (bIsFileFound)
104  {
105  UE_LOG(LogCarla, Warning, TEXT("Loading a new episode: %s"), *FinalPath);
106  UGameplayStatics::OpenLevel(GetWorld(), *FinalPath, true);
107  ApplySettings(FEpisodeSettings{});
108  }
109  return bIsFileFound;
110 }

132 bool UCarlaEpisode::LoadNewOpendriveEpisode(
133  const FString &OpenDriveString,
134  const carla::rpc::OpendriveGenerationParameters &Params)
135 {
136  if (OpenDriveString.IsEmpty())
137  {
138  UE_LOG(LogCarla, Error, TEXT("The OpenDrive string is empty."));
139  return false;
140  }
141 
142  // Build the Map from the OpenDRIVE data
143  const auto CarlaMap = carla::opendrive::OpenDriveParser::Load(
144  carla::rpc::FromLongFString(OpenDriveString));
145 
146  // Check the Map is correclty generated
147  if (!CarlaMap.has_value())
148  {
149  UE_LOG(LogCarla, Error, TEXT("The OpenDrive string is invalid or not supported"));
150  return false;
151  }
152 
153  // Generate the OBJ (as string)
154  const auto RoadMesh = CarlaMap->GenerateMesh(Params.vertex_distance);
155  const auto CrosswalksMesh = CarlaMap->GetAllCrosswalkMesh();
156  const auto RecastOBJ = (RoadMesh + CrosswalksMesh).GenerateOBJForRecast();
157 
158  const FString AbsoluteOBJPath = FPaths::ConvertRelativePathToFull(
159  FPaths::ProjectContentDir() + "Carla/Maps/Nav/OpenDriveMap.obj");
160 
161  // Store the OBJ string to a file in order to that RecastBuilder can load it
162  FFileHelper::SaveStringToFile(
163  carla::rpc::ToLongFString(RecastOBJ),
164  *AbsoluteOBJPath,
165  FFileHelper::EEncodingOptions::ForceUTF8,
166  &IFileManager::Get());
167 
168  const FString AbsoluteXODRPath = FPaths::ConvertRelativePathToFull(
169  FPaths::ProjectContentDir() + "Carla/Maps/OpenDrive/OpenDriveMap.xodr");
170 
171  // Copy the OpenDrive as a file in the serverside
172  FFileHelper::SaveStringToFile(
173  OpenDriveString,
174  *AbsoluteXODRPath,
175  FFileHelper::EEncodingOptions::ForceUTF8,
176  &IFileManager::Get());
177 
178  if (!FPaths::FileExists(AbsoluteXODRPath))
179  {
180  UE_LOG(LogCarla, Error, TEXT("ERROR: XODR not copied!"));
181  return false;
182  }
183 
184  UCarlaGameInstance * GameInstance = UCarlaStatics::GetGameInstance(GetWorld());
185  if(GameInstance)
186  {
187  GameInstance->SetOpendriveGenerationParameters(Params);
188  }
189  else
190  {
191  carla::log_warning("Missing game instance");
192  }
193 
194  const FString AbsoluteRecastBuilderPath = BuildRecastBuilderFile();
195 
196  if (FPaths::FileExists(AbsoluteRecastBuilderPath) &&
197  Params.enable_pedestrian_navigation)
198  {
199  /// @todo this can take too long to finish, clients need a method
200  /// to know if the navigation is available or not.
201  FPlatformProcess::CreateProc(
202  *AbsoluteRecastBuilderPath, *AbsoluteOBJPath,
203  true, true, true, nullptr, 0, nullptr, nullptr);
204  }
205  else
206  {
207  UE_LOG(LogCarla, Warning, TEXT("'RecastBuilder' not present under '%s', "
208  "the binaries for pedestrian navigation will not be created."),
209  *AbsoluteRecastBuilderPath);
210  }
211 
212  return true;
213 }

269 void UCarlaEpisode::InitializeAtBeginPlay()
270 {
271  auto World = GetWorld();
272  check(World != nullptr);
273  auto PlayerController = UGameplayStatics::GetPlayerController(World, 0);
274  if (PlayerController == nullptr)
275  {
276  UE_LOG(LogCarla, Error, TEXT("Can't find player controller!"));
277  return;
278  }
279  Spectator = PlayerController->GetPawn();
280  if (Spectator != nullptr)
281  {
282  FActorDescription Description;
283  Description.Id = TEXT("spectator");
284  Description.Class = Spectator->GetClass();
285  ActorDispatcher->RegisterActor(*Spectator, Description);
286  }
287  else
288  {
289  UE_LOG(LogCarla, Error, TEXT("Can't find spectator!"));
290  }
291 
292  for (TActorIterator<ATrafficSignBase> It(World); It; ++It)
293  {
294  ATrafficSignBase *Actor = *It;
295  check(Actor != nullptr);
296  FActorDescription Description;
297  Description.Id = UCarlaEpisode_GetTrafficSignId(Actor->GetTrafficSignState());
298  Description.Class = Actor->GetClass();
299  ActorDispatcher->RegisterActor(*Actor, Description);
300  }
301 
302  for (TActorIterator<AStaticMeshActor> It(World); It; ++It)
303  {
304  auto Actor = *It;
305  check(Actor != nullptr);
306  auto MeshComponent = Actor->GetStaticMeshComponent();
307  check(MeshComponent != nullptr);
308  if (MeshComponent->Mobility == EComponentMobility::Movable)
309  {
310  FActorDescription Description;
311  Description.Id = TEXT("static.prop");
312  Description.Class = Actor->GetClass();
313  ActorDispatcher->RegisterActor(*Actor, Description);
314  }
315  }
316 }
