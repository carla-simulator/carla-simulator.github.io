54 void ATrafficLightManager::RegisterLightComponent(UTrafficLightComponent * TrafficLightComponent)
55 {
56  // Cast to std::string
57  carla::road::SignId SignId(TCHAR_TO_UTF8(*(TrafficLightComponent->GetSignId())));
58 
59  ATrafficLightGroup* TrafficLightGroup;
60  UTrafficLightController* TrafficLightController;
61 
62  const auto &Signal = GetMap()->GetSignals().at(SignId);
63  if(Signal->GetControllers().size())
64  {
65  // Only one controller per signal
66  auto ControllerId = *(Signal->GetControllers().begin());
67 
68  // Get controller
69  const auto &Controller = GetMap()->GetControllers().at(ControllerId);
70  if(Controller->GetJunctions().empty())
71  {
72  UE_LOG(LogCarla, Error, TEXT("No junctions in controller %d"), *(ControllerId.c_str()) );
73  return;
74  }
75  // Get junction of the controller
76  auto JunctionId = *(Controller->GetJunctions().begin());
77 
78  // Search/create TrafficGroup (junction traffic light manager)
79  if(!TrafficGroups.Contains(JunctionId))
80  {
81  auto * NewTrafficLightGroup =
82  GetWorld()->SpawnActor<ATrafficLightGroup>();
83  NewTrafficLightGroup->JunctionId = JunctionId;
84  TrafficGroups.Add(JunctionId, NewTrafficLightGroup);
85  }
86  TrafficLightGroup = TrafficGroups[JunctionId];
87 
88  // Search/create controller in the junction
89  if(!TrafficControllers.Contains(ControllerId.c_str()))
90  {
91  auto *NewTrafficLightController = NewObject<UTrafficLightController>();
92  NewTrafficLightController->SetControllerId(ControllerId.c_str());
93  TrafficLightGroup->GetControllers().Add(NewTrafficLightController);
94  TrafficControllers.Add(ControllerId.c_str(), NewTrafficLightController);
95  }
96  TrafficLightController = TrafficControllers[ControllerId.c_str()];
97  }
98  else
99  {
100  auto * NewTrafficLightGroup =
101  GetWorld()->SpawnActor<ATrafficLightGroup>();
102  NewTrafficLightGroup->JunctionId = LoneTrafficLightsGroupControllerId;
103  TrafficGroups.Add(NewTrafficLightGroup->JunctionId, NewTrafficLightGroup);
104  TrafficLightGroup = NewTrafficLightGroup;
105 
106  auto *NewTrafficLightController = NewObject<UTrafficLightController>();
107  NewTrafficLightController->SetControllerId(FString::FromInt(LoneTrafficLightsGroupControllerId));
108  // Set red time longer than the default 2s
109  NewTrafficLightController->SetRedTime(10);
110  TrafficLightGroup->GetControllers().Add(NewTrafficLightController);
111  TrafficControllers.Add(NewTrafficLightController->GetControllerId(), NewTrafficLightController);
112  TrafficLightController = NewTrafficLightController;
113 
114  --LoneTrafficLightsGroupControllerId;
115  }
116 
117  TrafficLightComponent->TrafficLightGroup = TrafficLightGroup;
118  TrafficLightComponent->TrafficLightController = TrafficLightController;
119 
120  // Add signal to controller
121  TrafficLightController->AddTrafficLight(TrafficLightComponent);
122  TrafficLightController->ResetState();
123 
124  // Add signal to map
125  TrafficSignComponents.Add(TrafficLightComponent->GetSignId(), TrafficLightComponent);
126 
127  TrafficLightGroup->ResetGroup();
128 
129 }

232 void ATrafficLightManager::SpawnTrafficLights()
233 {
234  std::unordered_set<std::string> SignalsToSpawn;
235  for(const auto& ControllerPair : GetMap()->GetControllers())
236  {
237  const auto& Controller = ControllerPair.second;
238  for(const auto& SignalId : Controller->GetSignals())
239  {
240  SignalsToSpawn.insert(SignalId);
241  }
242  }
243  const auto& Signals = GetMap()->GetSignals();
244  for(const auto& SignalPair : Signals)
245  {
246  const auto& SignalId = SignalPair.first;
247  const auto& Signal = SignalPair.second;
248  if(!Signal->GetControllers().size() &&
249  !GetMap()->IsJunction(Signal->GetRoadId()) &&
250  carla::road::SignalType::IsTrafficLight(Signal->GetType()) &&
251  !SignalsToSpawn.count(SignalId))
252  {
253  SignalsToSpawn.insert(SignalId);
254  }
255  }
256  for(auto &SignalId : SignalsToSpawn)
257  {
258  // TODO: should this be an assert?
259  // RELEASE_ASSERT(
260  // Signals.count(SignalId) > 0,
261  // "Reference to inexistent signal. Possible OpenDRIVE error.");
262  if (Signals.count(SignalId) == 0)
263  {
264  UE_LOG(LogCarla, Warning,
265  TEXT("Possible OpenDRIVE error, reference to nonexistent signal id: %s"),
266  *carla::rpc::ToFString(SignalId));
267  continue;
268  }
269  const auto& Signal = Signals.at(SignalId);
270  auto CarlaTransform = Signal->GetTransform();
271  FTransform SpawnTransform(CarlaTransform);
272 
273  FVector SpawnLocation = SpawnTransform.GetLocation();
274  FRotator SpawnRotation(SpawnTransform.GetRotation());
275  // Blueprints are all rotated by 90 degrees
276  SpawnRotation.Yaw += 90;
277  // Remove road inclination
278  SpawnRotation.Roll = 0;
279  SpawnRotation.Pitch = 0;
280 
281  FActorSpawnParameters SpawnParams;
282  SpawnParams.Owner = this;
283  SpawnParams.SpawnCollisionHandlingOverride =
284  ESpawnActorCollisionHandlingMethod::AlwaysSpawn;
285  ATrafficLightBase * TrafficLight = GetWorld()->SpawnActor<ATrafficLightBase>(
286  TrafficLightModel,
287  SpawnLocation,
288  SpawnRotation,
289  SpawnParams);
290 
291  // Hack to prevent mixing ATrafficLightBase and UTrafficLightComponent logic
292  TrafficLight->SetTimeIsFrozen(true);
293 
294  TrafficSigns.Add(TrafficLight);
295 
296  UTrafficLightComponent *TrafficLightComponent =
297  NewObject<UTrafficLightComponent>(TrafficLight);
298  TrafficLightComponent->SetSignId(SignalId.c_str());
299  TrafficLightComponent->RegisterComponent();
300  TrafficLightComponent->AttachToComponent(
301  TrafficLight->GetRootComponent(),
302  FAttachmentTransformRules::KeepRelativeTransform);
303 
304  auto ClosestWaypointToSignal =
305  GetMap()->GetClosestWaypointOnRoad(CarlaTransform.location);
306  if (ClosestWaypointToSignal)
307  {
308  auto SignalDistanceToRoad =
309  (GetMap()->ComputeTransform(ClosestWaypointToSignal.get()).location - CarlaTransform.location).Length();
310  double LaneWidth = GetMap()->GetLaneWidth(ClosestWaypointToSignal.get());
311 
312  if(SignalDistanceToRoad < LaneWidth * 0.5)
313  {
314  carla::log_warning("Traffic light",
315  TCHAR_TO_UTF8(*TrafficLightComponent->GetSignId()),
316  "overlaps a driving lane. Disabling collision...");
317 
318  TArray<UPrimitiveComponent*> Primitives;
319  TrafficLight->GetComponents(Primitives);
320  for (auto* Primitive : Primitives)
321  {
322  Primitive->SetCollisionProfileName(TEXT("NoCollision"));
323  }
324  }
325  }
326 
327  RegisterLightComponent(TrafficLightComponent);
328  TrafficLightComponent->InitializeSign(GetMap().get());
329  }
330 }
