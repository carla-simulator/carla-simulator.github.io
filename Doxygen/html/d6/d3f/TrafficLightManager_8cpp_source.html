26 void ATrafficLightManager::RegisterLightComponent(UTrafficLightComponent * TrafficLight)
27 {
28  // Cast to std::string
29  carla::road::SignId SignId(TCHAR_TO_UTF8(*(TrafficLight->GetSignId())));
30 
31  // Get OpenDRIVE signal
32  if (GetMap()->GetSignals().count(SignId) == 0)
33  {
34  return;
35  }
36  const auto &Signal = GetMap()->GetSignals().at(SignId);
37  if(Signal->GetControllers().empty())
38  {
39  return;
40  }
41  // Only one controller per signal
42  auto ControllerId = *(Signal->GetControllers().begin());
43 
44  // Get controller
45  const auto &Controller = GetMap()->GetControllers().at(ControllerId);
46  if(Controller->GetJunctions().empty())
47  {
48  return;
49  }
50  // Get junction of the controller
51  auto JunctionId = *(Controller->GetJunctions().begin());
52 
53  // Search/create TrafficGroup (junction traffic light manager)
54  if(!TrafficGroups.Contains(JunctionId))
55  {
56  auto * TrafficLightGroup =
57  GetWorld()->SpawnActor<ATrafficLightGroup>();
58  TrafficLightGroup->JunctionId = JunctionId;
59  TrafficGroups.Add(JunctionId, TrafficLightGroup);
60  }
61  auto * TrafficLightGroup = TrafficGroups[JunctionId];
62 
63  // Search/create controller in the junction
64  if(!TrafficControllers.Contains(ControllerId.c_str()))
65  {
66  auto *TrafficLightController = NewObject<UTrafficLightController>();
67  TrafficLightController->SetControllerId(ControllerId.c_str());
68  TrafficLightGroup->GetControllers().Add(TrafficLightController);
69  TrafficControllers.Add(ControllerId.c_str(), TrafficLightController);
70  }
71  auto *TrafficLightController = TrafficControllers[ControllerId.c_str()];
72 
73  TrafficLight->TrafficLightGroup = TrafficLightGroup;
74 
75  // Add signal to controller
76  TrafficLightController->AddTrafficLight(TrafficLight);
77  TrafficLightController->ResetState();
78 
79  // Add signal to map
80  TrafficLights.Add(TrafficLight->GetSignId(), TrafficLight);
81 
82  TrafficLightGroup->ResetGroup();
83 }

99 void ATrafficLightManager::GenerateTrafficLights()
100 {
101  if(!TrafficLightModel)
102  {
103  return;
104  }
105  const auto& Signals = GetMap()->GetSignals();
106  for(const auto& ControllerPair : GetMap()->GetControllers())
107  {
108  const auto& Controller = ControllerPair.second;
109  for(const auto& SignalId : Controller->GetSignals())
110  {
111  const auto& Signal = Signals.at(SignalId);
112  auto CarlaTransform = Signal->GetTransform();
113  FTransform SpawnTransform(CarlaTransform);
114 
115  FVector SpawnLocation = SpawnTransform.GetLocation();
116  FRotator SpawnRotation(SpawnTransform.GetRotation());
117  SpawnRotation.Yaw += 90;
118 
119  FActorSpawnParameters SpawnParams;
120  SpawnParams.Owner = this;
121  SpawnParams.SpawnCollisionHandlingOverride =
122  ESpawnActorCollisionHandlingMethod::AlwaysSpawn;
123  AActor * TrafficLight = GetWorld()->SpawnActor<AActor>(
124  TrafficLightModel,
125  SpawnLocation,
126  SpawnRotation,
127  SpawnParams);
128 
129  UTrafficLightComponent *TrafficLightComponent =
130  NewObject<UTrafficLightComponent>(TrafficLight);
131  TrafficLightComponent->SetSignId(SignalId.c_str());
132  TrafficLightComponent->RegisterComponent();
133  TrafficLightComponent->AttachToComponent(
134  TrafficLight->GetRootComponent(),
135  FAttachmentTransformRules::KeepRelativeTransform);
136  }
137  }
