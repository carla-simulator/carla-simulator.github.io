26 void ATrafficLightManager::RegisterLightComponent(UTrafficLightComponent * TrafficLightComponent)
27 {
28  // Cast to std::string
29  carla::road::SignId SignId(TCHAR_TO_UTF8(*(TrafficLightComponent->GetSignId())));
30 
31  // Get OpenDRIVE signal
32  if (GetMap()->GetSignals().count(SignId) == 0)
33  {
34  UE_LOG(LogCarla, Error, TEXT("Missing signal with id: %s"), *(SignId.c_str()) );
35  return;
36  }
37  const auto &Signal = GetMap()->GetSignals().at(SignId);
38  if(Signal->GetControllers().empty())
39  {
40  UE_LOG(LogCarla, Error, TEXT("No controllers in signal %s"), *(SignId.c_str()) );
41  return;
42  }
43  // Only one controller per signal
44  auto ControllerId = *(Signal->GetControllers().begin());
45 
46  // Get controller
47  const auto &Controller = GetMap()->GetControllers().at(ControllerId);
48  if(Controller->GetJunctions().empty())
49  {
50  UE_LOG(LogCarla, Error, TEXT("No junctions in controller %d"), *(ControllerId.c_str()) );
51  return;
52  }
53  // Get junction of the controller
54  auto JunctionId = *(Controller->GetJunctions().begin());
55 
56  // Search/create TrafficGroup (junction traffic light manager)
57  if(!TrafficGroups.Contains(JunctionId))
58  {
59  auto * TrafficLightGroup =
60  GetWorld()->SpawnActor<ATrafficLightGroup>();
61  TrafficLightGroup->JunctionId = JunctionId;
62  TrafficGroups.Add(JunctionId, TrafficLightGroup);
63  }
64  auto * TrafficLightGroup = TrafficGroups[JunctionId];
65 
66  // Search/create controller in the junction
67  if(!TrafficControllers.Contains(ControllerId.c_str()))
68  {
69  auto *TrafficLightController = NewObject<UTrafficLightController>();
70  TrafficLightController->SetControllerId(ControllerId.c_str());
71  TrafficLightGroup->GetControllers().Add(TrafficLightController);
72  TrafficControllers.Add(ControllerId.c_str(), TrafficLightController);
73  }
74  auto *TrafficLightController = TrafficControllers[ControllerId.c_str()];
75 
76  TrafficLightComponent->TrafficLightGroup = TrafficLightGroup;
77  TrafficLightComponent->TrafficLightController = TrafficLightController;
78 
79  // Add signal to controller
80  TrafficLightController->AddTrafficLight(TrafficLightComponent);
81  TrafficLightController->ResetState();
82 
83  // Add signal to map
84  TrafficLightComponents.Add(TrafficLightComponent->GetSignId(), TrafficLightComponent);
85 
86  TrafficLightGroup->ResetGroup();
87 }

103 void ATrafficLightManager::GenerateTrafficLights()
104 {
105  if(!TrafficLightsGenerated){
106  if(!TrafficLightModel)
107  {
108  UE_LOG(LogCarla, Error, TEXT("Missing TrafficLightModel"));
109  return;
110  }
111  const auto& Signals = GetMap()->GetSignals();
112  for(const auto& ControllerPair : GetMap()->GetControllers())
113  {
114  const auto& Controller = ControllerPair.second;
115  for(const auto& SignalId : Controller->GetSignals())
116  {
117  const auto& Signal = Signals.at(SignalId);
118  auto CarlaTransform = Signal->GetTransform();
119  FTransform SpawnTransform(CarlaTransform);
120 
121  FVector SpawnLocation = SpawnTransform.GetLocation();
122  FRotator SpawnRotation(SpawnTransform.GetRotation());
123  SpawnRotation.Yaw += 90;
124 
125  FActorSpawnParameters SpawnParams;
126  SpawnParams.Owner = this;
127  SpawnParams.SpawnCollisionHandlingOverride =
128  ESpawnActorCollisionHandlingMethod::AlwaysSpawn;
129  ATrafficLightBase * TrafficLight = GetWorld()->SpawnActor<ATrafficLightBase>(
130  TrafficLightModel,
131  SpawnLocation,
132  SpawnRotation,
133  SpawnParams);
134 
135  TrafficSigns.Add(TrafficLight);
136 
137  UTrafficLightComponent *TrafficLightComponent =
138  NewObject<UTrafficLightComponent>(TrafficLight);
139  TrafficLightComponent->SetSignId(SignalId.c_str());
140  TrafficLightComponent->RegisterComponent();
141  TrafficLightComponent->AttachToComponent(
142  TrafficLight->GetRootComponent(),
143  FAttachmentTransformRules::KeepRelativeTransform);
144 
145  RegisterLightComponent(TrafficLightComponent);
146  }
147  }
148  TrafficLightsGenerated = true;
149  }
150 }
