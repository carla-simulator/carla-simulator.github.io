50  void Episode::Listen() {
51  std::weak_ptr<Episode> weak = shared_from_this();
52  _client.SubscribeToStream(_token, [weak](auto buffer) {
53  auto self = weak.lock();
54  if (self != nullptr) {
55  auto data = sensor::Deserializer::Deserialize(std::move(buffer));
56 
57  auto next = std::make_shared<const EpisodeState>(CastData(*data));
58  auto prev = self->GetState();
59  do {
60  if (prev->GetFrame() >= next->GetFrame()) {
61  self->_on_tick_callbacks.Call(next);
62  return;
63  }
64  } while (!self->_state.compare_exchange(&prev, next));
65 
66  if (next->GetEpisodeId() != prev->GetEpisodeId()) {
67  self->OnEpisodeStarted();
68  }
69 
70  // Notify waiting threads and do the callbacks.
71  self->_snapshot.SetValue(next);
72 
73  // Tick navigation.
74  auto navigation = self->_navigation.load();
75  if (navigation != nullptr) {
76  navigation->Tick(*next);
77  }
78 
79  // Call user callbacks.
80  self->_on_tick_callbacks.Call(next);
81  }
82  });
83  }
