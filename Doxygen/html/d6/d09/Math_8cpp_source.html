14  std::pair<float, float> Math::DistanceSegmentToPoint(
15  const Vector3D &p,
16  const Vector3D &v,
17  const Vector3D &w) {
18  const float l2 = DistanceSquared2D(v, w);
19  const float l = std::sqrt(l2);
20  if (l2 == 0.0f) {
21  return std::make_pair(0.0f, Distance2D(v, p));
22  }
23  const float dot_p_w = Dot2D(p - v, w - v);
24  const float t = Clamp(dot_p_w / l2);
25  const Vector3D projection = v + t * (w - v);
26  return std::make_pair(t * l, Distance2D(projection, p));
27  }

29  std::pair<float, float> Math::DistanceArcToPoint(
30  Vector3D p,
31  Vector3D start_pos,
32  const float length,
33  float heading, // [radians]
34  float curvature) {
35 
36  /// @todo: Because Unreal's coordinates, hacky way to correct
37  /// the -y, this must be changed in the future
38  p.y = -p.y;
39  start_pos.y = -start_pos.y;
40  heading = -heading;
41  curvature = -curvature;
42 
43  // since this algorithm is working for positive curvatures,
44  // and we are only calculating distances, we can invert the y
45  // axis (along with the curvature and the heading), so if the
46  // curvature is negative, the algorithm will work as expected
47  if (curvature < 0.0f) {
48  p.y = -p.y;
49  start_pos.y = -start_pos.y;
50  heading = -heading;
51  curvature = -curvature;
52  }
53 
54  // transport point relative to the arc starting poistion and rotation
55  const Vector3D rotated_p(RotatePointOnOrigin2D(p - start_pos, -heading));
56 
57  const float radius = 1.0f / curvature;
58  const Vector3D circ_center(0.0f, radius, 0.0f);
59 
60  // check if the point is in the center of the circle, so we know p
61  // is in the same distance of every possible point in the arc
62  if (rotated_p == circ_center) {
63  return std::make_pair(0.0f, radius);
64  }
65 
66  // find intersection position using the unit vector from the center
67  // of the circle to the point and multiplying by the radius
68  const Vector3D intersection = ((rotated_p - circ_center).MakeUnitVector() * radius) + circ_center;
69 
70  // use the arc length to calculate the angle in the last point of it
71  // circumference of a circle = 2 * PI * r
72  // last_point_angle = (length / circumference) * 2 * PI
73  // so last_point_angle = length / radius
74  const float last_point_angle = length / radius;
75 
76  constexpr float pi_half = Pi<float>() / 2.0f;
77 
78  // move the point relative to the center of the circle and find
79  // the angle between the point and the center of coords in rad
80  float angle = std::atan2(intersection.y - radius, intersection.x) + pi_half;
81 
82  if (angle < 0.0f) {
83  angle += Pi<float>() * 2.0f;
84  }
85 
86  // see if the angle is between 0 and last_point_angle
87  DEBUG_ASSERT(angle >= 0.0f);
88  if (angle <= last_point_angle) {
89  return std::make_pair(
90  angle * radius,
91  Distance2D(intersection, rotated_p));
92  }
93 
94  // find the nearest point, start or end to intersection
95  const float start_dist = Distance2D(Vector3D(), rotated_p);
96 
97  const Vector3D end_pos(
98  radius * std::cos(last_point_angle - pi_half),
99  radius * std::sin(last_point_angle - pi_half) + circ_center.y,
100  0.0f);
101  const float end_dist = Distance2D(end_pos, rotated_p);
102  return (start_dist < end_dist) ?
103  std::make_pair(0.0f, start_dist) :
104  std::make_pair(length, end_dist);
105  }

107  Vector3D Math::RotatePointOnOrigin2D(Vector3D p, float angle) {
108  const float s = std::sin(angle);
109  const float c = std::cos(angle);
110  return Vector3D(p.x * c - p.y * s, p.x * s + p.y * c, 0.0f);
111  }

113  Vector3D Math::GetForwardVector(const Rotation &rotation) {
