220  /// Creates an actor instance out of a description of an existing actor.
221  /// Note that this does not spawn an actor.
222  ///
223  /// If @a gc is GarbageCollectionPolicy::Enabled, the shared pointer
224  /// returned is provided with a custom deleter that calls Destroy() on the
225  /// actor. This method does not support GarbageCollectionPolicy::Inherit.
226  SharedPtr<Actor> MakeActor(
