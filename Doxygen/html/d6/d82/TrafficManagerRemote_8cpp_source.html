32 void TrafficManagerRemote::Start() {
33  _keep_alive = true;
34 
35  std::thread _thread = std::thread([this] () {
36  std::chrono::milliseconds wait_time(TM_TIMEOUT);
37  try {
38  do {
39  std::this_thread::sleep_for(wait_time);
40 
41  client.HealthCheckRemoteTM();
42 
43  /// Until connection active
44  } while (_keep_alive);
45  } catch (...) {
46 
47  std::string rhost("");
48  uint16_t rport = 0;
49 
50  client.getServerDetails(rhost, rport);
51 
52  std::string strtmserver(rhost + ":" + std::to_string(rport));
53 
54  /// Create error msg
55  std::string errmsg(
56  "Trying to connect rpc server of traffic manager; "
57  "but the system failed to connect at " + strtmserver);
58 
59  /// TSet the error message
60  if(_keep_alive) {
61  this->episodeProxyTM.Lock()->AddPendingException(errmsg);
62  }
63  }
64  _keep_alive = false;
65  _cv.notify_one();
66  });
67 
68  _thread.detach();
69 }
