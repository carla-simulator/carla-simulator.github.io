133 void FCarlaEngine::OnPostTick(UWorld *World, ELevelTick TickType, float DeltaSeconds)
134 {
135  // tick the recorder/replayer system
136  if (GetCurrentEpisode())
137  {
138  auto* EpisodeRecorder = GetCurrentEpisode()->GetRecorder();
139  if (EpisodeRecorder)
140  {
141  EpisodeRecorder->Ticking(DeltaSeconds);
142  }
143  }
144 
145  if ((TickType == ELevelTick::LEVELTICK_All) && (CurrentEpisode != nullptr))
146  {
147  // Look for lightsubsystem
148  bool LightUpdatePending = false;
149  if (World)
150  {
151  UCarlaLightSubsystem* CarlaLightSubsystem = World->GetSubsystem<UCarlaLightSubsystem>();
152  if (CarlaLightSubsystem)
153  {
154  LightUpdatePending = CarlaLightSubsystem->IsUpdatePending();
155  }
156  }
157 
158  // send the worldsnapshot
159  WorldObserver.BroadcastTick(*CurrentEpisode, DeltaSeconds, bMapChanged, LightUpdatePending);
160  ResetSimulationState();
161  }
162 }
