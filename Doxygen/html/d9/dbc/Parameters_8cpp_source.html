25 void Parameters::SetCollisionDetection(
26  const ActorPtr &reference_actor,
27  const ActorPtr &other_actor,
28  const bool detect_collision) {
29 
30  const ActorId reference_id = reference_actor->GetId();
31  const ActorId other_id = other_actor->GetId();
32 
33  if (detect_collision) {
34 
35  if (ignore_collision.Contains(reference_id)) {
36  std::shared_ptr<AtomicActorSet> actor_set = ignore_collision.GetValue(reference_id);
37  if (actor_set->Contains(other_id)) {
38  actor_set->Remove({other_actor});
39  }
40  }
41  } else {
42 
43  if (ignore_collision.Contains(reference_id)) {
44  std::shared_ptr<AtomicActorSet> actor_set = ignore_collision.GetValue(reference_id);
45  if (!actor_set->Contains(other_id)) {
46  actor_set->Insert({other_actor});
47  }
48  } else {
49  std::shared_ptr<AtomicActorSet> actor_set = std::make_shared<AtomicActorSet>();
50  actor_set->Insert({other_actor});
51  auto entry = std::make_pair(reference_id, actor_set);
52  ignore_collision.AddEntry(entry);
53  }
54  }
55 }
