26 void Parameters::SetCollisionDetection(
27  const ActorPtr &reference_actor,
28  const ActorPtr &other_actor,
29  const bool detect_collision) {
30 
31  const ActorId reference_id = reference_actor->GetId();
32  const ActorId other_id = other_actor->GetId();
33 
34  if (detect_collision) {
35 
36  if (ignore_collision.Contains(reference_id)) {
37  std::shared_ptr<AtomicActorSet> actor_set = ignore_collision.GetValue(reference_id);
38  if (actor_set->Contains(other_id)) {
39  actor_set->Remove({other_actor});
40  }
41  }
42  } else {
43 
44  if (ignore_collision.Contains(reference_id)) {
45  std::shared_ptr<AtomicActorSet> actor_set = ignore_collision.GetValue(reference_id);
46  if (!actor_set->Contains(other_id)) {
47  actor_set->Insert({other_actor});
48  }
49  } else {
50  std::shared_ptr<AtomicActorSet> actor_set = std::make_shared<AtomicActorSet>();
51  actor_set->Insert({other_actor});
52  auto entry = std::make_pair(reference_id, actor_set);
53  ignore_collision.AddEntry(entry);
54  }
55  }
56 }

78 float Parameters::GetVehicleTargetVelocity(const ActorPtr &actor) {
79 
80  const ActorId actor_id = actor->GetId();
