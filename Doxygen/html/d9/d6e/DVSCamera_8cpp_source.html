149 void ADVSCamera::imageToGray(const TArray<FColor> &image, TArray<float> &gray)
150 {
151  /** Sanity check **/
152  if (image.Num() != (this->GetImageHeight() * this->GetImageWidth()))
153  return;
154 
155  /** Reserve HxW elements **/
156  gray.SetNumUninitialized(image.Num());
157 
158  /** Convert image to gray raw image values **/
159  for (size_t i = 0; i<image.Num(); ++i)
160  {
161  gray[i] = 0.2989*image[i].R + 0.5870*image[i].G + 0.1140*image[i].B;

181 ADVSCamera::DVSEventArray ADVSCamera::simulation (float DeltaTime)
182 {
183  /** Array of events **/
184  ADVSCamera::DVSEventArray events;
185 
186  /** Sanity check **/
187  if (this->last_image.Num() != (this->GetImageHeight() * this->GetImageWidth()))
188  return events;
189 
190  /** Check initialization **/
191  if(this->prev_image.Num() == 0)
192  {
193  /** Set the first rendered image **/
194  this->ref_values = this->last_image;
195  this->prev_image = this->last_image;
196 
197  /** Resizes array to given number of elements. New elements will be zeroed.**/
198  this->last_event_timestamp.SetNumZeroed(this->last_image.Num());
199 
200  /** Reset current time **/
201  this->current_time = dvs::secToNanosec(this->GetEpisode().GetElapsedGameTime());
202 
203  return events;
204  }
205 
206  static constexpr float tolerance = 1e-6;
207 
208  /** delta time in nanoseconds **/
209  const std::uint64_t delta_t_ns = dvs::secToNanosec(
210  this->GetEpisode().GetElapsedGameTime()) - this->current_time;
211 
212  /** Loop along the image size **/
213  for (uint32 y = 0; y < this->GetImageHeight(); ++y)
214  {
215  for (uint32 x = 0; x < this->GetImageWidth(); ++x)
216  {
217  const uint32 i = (this->GetImageWidth() * y) + x;
218  const float itdt = this->last_image[i];
219  const float it = this->prev_image[i];
220  const float prev_cross = this->ref_values[i];
221 
222  if (std::fabs (it - itdt) > tolerance)
223  {
224  const float pol = (itdt >= it) ? +1.0 : -1.0;
225  float C = (pol > 0) ? this->config.Cp : this->config.Cm;
226  const float sigma_C = (pol > 0) ? this->config.sigma_Cp : this->config.sigma_Cm;
227 
228  if(sigma_C > 0)
229  {
230  C += RandomEngine->GetNormalDistribution(0, sigma_C);
231  constexpr float minimum_contrast_threshold = 0.01;
232  C = std::max(minimum_contrast_threshold, C);
233  }
234  float curr_cross = prev_cross;
235  bool all_crossings = false;
236 
237  do
238  {
239  curr_cross += pol * C;
240 
241  if ((pol > 0 && curr_cross > it && curr_cross <= itdt)
242  || (pol < 0 && curr_cross < it && curr_cross >= itdt))
243  {
244  const std::uint64_t edt = (curr_cross - it) * delta_t_ns / (itdt - it);
245  const std::int64_t t = this->current_time + edt;
246 
247  // check that pixel (x,y) is not currently in a "refractory" state
248  // i.e. |t-that last_timestamp(x,y)| >= refractory_period
249  const std::int64_t last_stamp_at_xy = dvs::secToNanosec(this->last_event_timestamp[i]);
250  if (t >= last_stamp_at_xy)
251  {
252  const std::uint64_t dt = t - last_stamp_at_xy;
253  if(this->last_event_timestamp[i] == 0 || dt >= this->config.refractory_period_ns)
254  {
255  events.push_back(::carla::sensor::data::DVSEvent(x, y, t, pol > 0));
256  this->last_event_timestamp[i] = dvs::nanosecToSecTrunc(t);
257  }
258  else
259  {
260  /** Dropping event because time since last event < refractory_period_ns **/
261  }
262  this->ref_values[i] = curr_cross;
263  }
264  }
265  else
266  {
267  all_crossings = true;
268  }
269  } while (!all_crossings);
270  } // end tolerance
271  } // end for each pixel
272  }
273 
274  /** Update current time **/
275  this->current_time = dvs::secToNanosec(this->GetEpisode().GetElapsedGameTime());
276 
277  this->prev_image = this->last_image;
278 
279  // Sort the events by increasing timestamps, since this is what
280  // most event processing algorithms expect
281  std::sort(events.begin(), events.end(), [](const ::carla::sensor::data::DVSEvent& it1, const ::carla::sensor::data::DVSEvent& it2){return it1.t < it2.t;});
282 
283  return events;
284 }
