62  template <typename FuncT>
63  static void ForEachDrivableLaneImpl(
64  RoadId road_id,
65  const LaneSection &lane_section,
66  double distance,
67  FuncT &&func) {

104  /// Returns a pair containing first = width, second = tangent,
105  /// for an specific Lane given an s and a iterator over lanes
106  template <typename T>
107  static std::pair<double, double> ComputeTotalLaneWidth(
108  const T container,
109  const double s,
110  const LaneId lane_id) {
111 
112  // lane_id can't be 0
113  RELEASE_ASSERT(lane_id != 0);
114 
115  const bool negative_lane_id = lane_id < 0;
116  double dist = 0.0;
117  double tangent = 0.0;
118  for (const auto &lane : container) {
119  auto info = lane.second.template GetInfo<RoadInfoLaneWidth>(s);
120  RELEASE_ASSERT(info != nullptr);
121  const auto current_polynomial = info->GetPolynomial();
122  auto current_dist = current_polynomial.Evaluate(s);
123  auto current_tang = current_polynomial.Tangent(s);
124  if (lane.first != lane_id) {
125  dist += negative_lane_id ? current_dist : -current_dist;
126  tangent += current_tang;
127  } else if (lane.first == lane_id) {
128  current_dist *= 0.5;
129  dist += negative_lane_id ? current_dist : -current_dist;
130  tangent += current_tang * 0.5;
131  break;
132  }
133  }
134  return std::make_pair(dist, tangent);
135  }

147  boost::optional<Waypoint> Map::GetClosestWaypointOnRoad(
148  const geom::Location &pos,
149  uint32_t lane_type) const {
150  // max_nearests represents the max nearests roads
151  // where we will search for nearests lanes
152  constexpr size_t max_nearests = 50u;
153  // in case that map has less than max_nearests lanes,
154  // we will use the maximum lanes
155  const size_t max_nearest_allowed = std::min(_data.GetRoadCount(), max_nearests);
156 
157  // Unreal's Y axis hack
158  const auto pos_inverted_y = geom::Location(pos.x, -pos.y, pos.z);
159 
160  double nearest_dist[max_nearests];
161  std::fill(nearest_dist, nearest_dist + max_nearest_allowed,
162  std::numeric_limits<double>::max());
163 
164  RoadId ids[max_nearests];
165  std::fill(ids, ids + max_nearest_allowed, 0);
166 
167  double dists[max_nearests];
168  std::fill(dists, dists + max_nearest_allowed, 0.0);
169 
170  for (const auto &road_pair : _data.GetRoads()) {
171  const auto road = &road_pair.second;
172  const auto current_dist = road->GetNearestPoint(pos_inverted_y);
173 
174  for (size_t i = 0u; i < max_nearest_allowed; ++i) {
175  if (current_dist.second < nearest_dist[i]) {
176  // reorder nearest_dist
177  for (size_t j = max_nearest_allowed - 1u; j > i; --j) {
178  DEBUG_ASSERT(j > 0u);
179  nearest_dist[j] = nearest_dist[j - 1];
180  ids[j] = ids[j - 1];
181  dists[j] = dists[j - 1];
182  }
183  nearest_dist[i] = current_dist.second;
184  ids[i] = road->GetId();
185  dists[i] = current_dist.first;
186  break;
187  }
188  }
189  }
190 
191  // search for the nearest lane in nearest_dist
192  Waypoint waypoint;
193  auto nearest_lane_dist = std::numeric_limits<double>::max();
194  for (size_t i = 0u; i < max_nearest_allowed; ++i) {
195  auto lane_dist = _data.GetRoad(ids[i]).GetNearestLane(dists[i], pos_inverted_y, lane_type);
196 
197  if (lane_dist.second < nearest_lane_dist) {
198  nearest_lane_dist = lane_dist.second;
199  waypoint.lane_id = lane_dist.first->GetId();
200  waypoint.road_id = ids[i];
201  waypoint.s = dists[i];
202  }
203  }
204 
205  if (nearest_lane_dist == std::numeric_limits<double>::max()) {
206  return boost::optional<Waypoint>{};
207  }
208 
209  const auto &road = _data.GetRoad(waypoint.road_id);
210 
211  // Make sure 0.0 < waipoint.s < Road's length
212  constexpr double margin = 5.0 * EPSILON;
213  DEBUG_ASSERT(margin < road.GetLength() - margin);
214  waypoint.s = geom::Math::Clamp(waypoint.s, margin, road.GetLength() - margin);
215 
216  auto &lane = road.GetLaneByDistance(waypoint.s, waypoint.lane_id);
217 
218  const auto lane_section = lane.GetLaneSection();
219  RELEASE_ASSERT(lane_section != nullptr);
220  const auto lane_section_id = lane_section->GetId();
221  waypoint.section_id = lane_section_id;
222 
223  return waypoint;
224  }

226  boost::optional<Waypoint> Map::GetWaypoint(
227  const geom::Location &pos,
228  uint32_t lane_type) const {
229  boost::optional<Waypoint> w = GetClosestWaypointOnRoad(pos, lane_type);
230 
231  if (!w.has_value()) {
232  return w;
233  }
234 
235  const auto dist = geom::Math::Distance2D(ComputeTransform(*w).location, pos);
236  const auto lane_width_info = GetLane(*w).GetInfo<RoadInfoLaneWidth>(w->s);
237  const auto half_lane_width =
238  lane_width_info->GetPolynomial().Evaluate(w->s) * 0.5;
239 
240  if (dist < half_lane_width) {
241  return w;
242  }
243 
244  return boost::optional<Waypoint>{};
245  }

247  geom::Transform Map::ComputeTransform(Waypoint waypoint) const {
248  const auto &road = _data.GetRoad(waypoint.road_id);
249 
250  // must s be smaller (or eq) than road lenght and bigger (or eq) than 0?
251  RELEASE_ASSERT(waypoint.s <= road.GetLength());
252  RELEASE_ASSERT(waypoint.s >= 0.0);
253 
254  const auto &lane_section = road.GetLaneSectionById(waypoint.section_id);
255  const std::map<LaneId, Lane> &lanes = lane_section.GetLanes();
256 
257  // check that lane_id exists on the current s
258  RELEASE_ASSERT(!lanes.empty());
259  RELEASE_ASSERT(waypoint.lane_id >= lanes.begin()->first);
260  RELEASE_ASSERT(waypoint.lane_id <= lanes.rbegin()->first);
261 
262  float lane_width = 0.0f;
263  float lane_tangent = 0.0f;
264 
265  if (waypoint.lane_id < 0) {
266  // right lane
267  const auto side_lanes = MakeListView(
268  std::make_reverse_iterator(lanes.lower_bound(0)), lanes.rend());
269  const auto computed_width =
270  ComputeTotalLaneWidth(side_lanes, waypoint.s, waypoint.lane_id);
271  lane_width = static_cast<float>(computed_width.first);
272  lane_tangent = static_cast<float>(computed_width.second);
273  } else if (waypoint.lane_id > 0) {
274  // left lane
275  const auto side_lanes = MakeListView(lanes.lower_bound(1), lanes.end());
276  const auto computed_width =
277  ComputeTotalLaneWidth(side_lanes, waypoint.s, waypoint.lane_id);
278  lane_width = static_cast<float>(computed_width.first);
279  lane_tangent = static_cast<float>(computed_width.second);
280  }
281 
282  // get a directed point in s and apply the computed lateral offet
283  DirectedPoint dp = road.GetDirectedPointIn(waypoint.s);
284 
285  // compute the tangent of the laneOffset
286  const auto lane_offset_info = road.GetInfo<RoadInfoLaneOffset>(waypoint.s);
287  const auto lane_offset_tangent = static_cast<float>(lane_offset_info->GetPolynomial().Tangent(waypoint.s));
288 
289  lane_tangent -= lane_offset_tangent;
290 
291  // Unreal's Y axis hack
292  lane_tangent *= -1;
293 
294  geom::Rotation rot(
295  geom::Math::ToDegrees(static_cast<float>(dp.pitch)),
296  geom::Math::ToDegrees(-static_cast<float>(dp.tangent)), // Unreal's Y axis hack
297  0.0f);
298 
299  dp.ApplyLateralOffset(lane_width);
300 
301  if (waypoint.lane_id > 0) {
302  rot.yaw += 180.0f + geom::Math::ToDegrees(lane_tangent);
303  rot.pitch = 360.0f - rot.pitch;
304  } else {
305  rot.yaw -= geom::Math::ToDegrees(lane_tangent);
306  }
307 
308  // Unreal's Y axis hack
309  dp.location.y *= -1;
310 
311  return geom::Transform(dp.location, rot);
312  }

369  std::vector<geom::Location> Map::GetAllCrosswalkZones() const {
370  std::vector<geom::Location> result;
371 
372  for (const auto &pair : _data.GetRoads()) {
373  const auto &road = pair.second;
374  std::vector<const RoadObjectCrosswalk *> crosswalks = road.GetObjects<RoadObjectCrosswalk>();
375  if (crosswalks.size() > 0) {
376  for (auto crosswalk : crosswalks) {
377  // waypoint only at start position
378  std::vector<geom::Location> points;
379  Waypoint waypoint;
380  geom::Transform base;
381  for (const auto &section : road.GetLaneSectionsAt(crosswalk->GetS())) {
382  // get the section with the center lane
383  for (const auto &lane : section.GetLanes()) {
384  // is the center line
385  if (lane.first == 0) {
386  // get the center point
387  waypoint.road_id = pair.first;
388  waypoint.section_id = section.GetId();
389  waypoint.lane_id = 0;
390  waypoint.s = crosswalk->GetS();
391  base = ComputeTransform(waypoint);
392  }
393  }
394  }
395 
396  // move perpendicular ('t')
397  geom::Transform pivot = base;
398  pivot.rotation.yaw -= geom::Math::ToDegrees<float>(static_cast<float>(crosswalk->GetHeading()));
399  pivot.rotation.yaw -= 90; // move perpendicular to 's' for the lateral offset
400  geom::Vector3D v(static_cast<float>(crosswalk->GetT()), 0.0f, 0.0f);
401  pivot.TransformPoint(v);
402  // restore pivot position and orientation
403  pivot = base;
404  pivot.location = v;
405  pivot.rotation.yaw -= geom::Math::ToDegrees<float>(static_cast<float>(crosswalk->GetHeading()));
406 
407  // calculate all the corners
408  for (auto corner : crosswalk->GetPoints()) {
409  geom::Vector3D v2(static_cast<float>(corner.u), static_cast<float>(corner.v), static_cast<float>(corner.z));
410  // set the width larger to contact with the sidewalk (in case they have gutter area)
411  if (corner.u < 0)
412  v2.x -= 1.0f;
413  else
414  v2.x += 1.0f;
415  pivot.TransformPoint(v2);
416  result.push_back(v2);
417  }
418  }
419  }
420  }
421  return result;
422  }

464  std::vector<Waypoint> Map::GetNext(
465  const Waypoint waypoint,
466  const double distance) const {
467  RELEASE_ASSERT(distance > 0.0);
468  const auto &lane = GetLane(waypoint);
469  const bool forward = (waypoint.lane_id <= 0);
470  const double signed_distance = forward ? distance : -distance;
471  const double relative_s = waypoint.s - lane.GetDistance() + EPSILON;
472  const double remaining_lane_length = forward ? lane.GetLength() - relative_s : relative_s;
473  DEBUG_ASSERT(remaining_lane_length >= 0.0);
474 
475  // If after subtracting the distance we are still in the same lane, return
476  // same waypoint with the extra distance.
477  if (distance <= remaining_lane_length) {
478  Waypoint result = waypoint;
479  result.s += signed_distance;
480  result.s += forward ? -EPSILON : EPSILON;
481  RELEASE_ASSERT(result.s > 0.0);
482  return { result };
483  }
484 
485  // If we run out of remaining_lane_length we have to go to the successors.
486  std::vector<Waypoint> result;
487  for (const auto &successor : GetSuccessors(waypoint)) {
488  DEBUG_ASSERT(
489  successor.road_id != waypoint.road_id ||
490  successor.section_id != waypoint.section_id ||
491  successor.lane_id != waypoint.lane_id);
492  result = ConcatVectors(result, GetNext(successor, distance - remaining_lane_length));
493  }
494  return result;
495  }
