65  /// Return a waypoint for each drivable lane on @a lane_section.
66  template <typename FuncT>
67  static void ForEachDrivableLaneImpl(
68  RoadId road_id,
69  const LaneSection &lane_section,
70  double distance,
71  FuncT &&func) {
72  for (const auto &pair : lane_section.GetLanes()) {
73  const auto &lane = pair.second;
74  if (lane.GetId() == 0) {
75  continue;
76  }
77  if ((static_cast<uint32_t>(lane.GetType()) & static_cast<uint32_t>(Lane::LaneType::Driving)) > 0) {
78  std::forward<FuncT>(func)(Waypoint{
79  road_id,
80  lane_section.GetId(),
81  lane.GetId(),
82  distance < 0.0 ? GetDistanceAtStartOfLane(lane) : distance});
83  }
84  }
85  }

156  boost::optional<Waypoint> Map::GetClosestWaypointOnRoad(
157  const geom::Location &pos,
158  uint32_t lane_type) const {
159  std::vector<Rtree::TreeElement> query_result =
160  _rtree.GetNearestNeighboursWithFilter(Rtree::BPoint(pos.x, pos.y, pos.z),
161  [&](Rtree::TreeElement const &element) {
162  const Lane &lane = GetLane(element.second.first);
163  return (lane_type & static_cast<uint32_t>(lane.GetType())) > 0;
164  });
165 
166  if (query_result.size() == 0) {
167  return boost::optional<Waypoint>{};
168  }
169 
170  Rtree::BSegment segment = query_result.front().first;
171  Rtree::BPoint s1 = segment.first;
172  Rtree::BPoint s2 = segment.second;
173  auto distance_to_segment = geom::Math::DistanceSegmentToPoint(pos,
174  geom::Vector3D(s1.get<0>(), s1.get<1>(), s1.get<2>()),
175  geom::Vector3D(s2.get<0>(), s2.get<1>(), s2.get<2>()));
176 
177  Waypoint result_start = query_result.front().second.first;
178  Waypoint result_end = query_result.front().second.second;
179 
180  if (result_start.lane_id < 0) {
181  double delta_s = distance_to_segment.first;
182  double final_s = result_start.s + delta_s;
183  if (final_s >= result_end.s) {
184  return result_end;
185  } else if (delta_s <= 0) {
186  return result_start;
187  } else {
188  return GetNext(result_start, delta_s).front();
189  }
190  } else {
191  double delta_s = distance_to_segment.first;
192  double final_s = result_start.s - delta_s;
193  if (final_s <= result_end.s) {
194  return result_end;
195  } else if (delta_s <= 0) {
196  return result_start;
197  } else {
198  return GetNext(result_start, delta_s).front();
199  }
200  }
201  }

203  boost::optional<Waypoint> Map::GetWaypoint(
204  const geom::Location &pos,
205  uint32_t lane_type) const {
206  boost::optional<Waypoint> w = GetClosestWaypointOnRoad(pos, lane_type);
207 
208  if (!w.has_value()) {
209  return w;
210  }
211 
212  const auto dist = geom::Math::Distance2D(ComputeTransform(*w).location, pos);
213  const auto lane_width_info = GetLane(*w).GetInfo<RoadInfoLaneWidth>(w->s);
214  const auto half_lane_width =
215  lane_width_info->GetPolynomial().Evaluate(w->s) * 0.5;
216 
217  if (dist < half_lane_width) {
218  return w;
219  }
220 
221  return boost::optional<Waypoint>{};
222  }

224  boost::optional<Waypoint> Map::GetWaypoint(
225  RoadId road_id,
226  LaneId lane_id,
227  float s) const {
228 
229  // define the waypoint with the known parameters
230  Waypoint waypoint;
231  waypoint.road_id = road_id;
232  waypoint.lane_id = lane_id;
233  waypoint.s = s;
234 
235  // check the road
236  if (!_data.ContainsRoad(waypoint.road_id)) {
237  return boost::optional<Waypoint>{};
238  }
239  const Road &road = _data.GetRoad(waypoint.road_id);
240 
241  // check the 's' distance
242  if (s < 0.0f || s >= road.GetLength()) {
243  return boost::optional<Waypoint>{};
244  }
245 
246  // check the section
247  bool lane_found = false;
248  for (auto &section : road.GetLaneSectionsAt(s)) {
249  if (section.ContainsLane(lane_id)) {
250  waypoint.section_id = section.GetId();
251  lane_found = true;
252  break;
253  }
254  }
255 
256  // check the lane id
257  if (!lane_found) {
258  return boost::optional<Waypoint>{};
259  }
260 
261  return waypoint;
262  }

297  std::pair<const RoadInfoMarkRecord *, const RoadInfoMarkRecord *>
298  Map::GetMarkRecord(const Waypoint waypoint) const {
299  // if lane Id is 0, just return a pair of nulls
300  if (waypoint.lane_id == 0)
301  return std::make_pair(nullptr, nullptr);
302 
303  const auto s = waypoint.s;
304 
305  const auto &current_lane = GetLane(waypoint);
306  RELEASE_ASSERT(current_lane.GetRoad() != nullptr);
307  RELEASE_ASSERT(s <= current_lane.GetRoad()->GetLength());
308 
309  const auto inner_lane_id = waypoint.lane_id < 0 ?
310  waypoint.lane_id + 1 :
311  waypoint.lane_id - 1;
312 
313  const auto &inner_lane = current_lane.GetRoad()->GetLaneById(waypoint.section_id, inner_lane_id);
314 
315  auto current_lane_info = current_lane.GetInfo<RoadInfoMarkRecord>(s);
316  auto inner_lane_info = inner_lane.GetInfo<RoadInfoMarkRecord>(s);
317 
318  return std::make_pair(current_lane_info, inner_lane_info);

321  std::vector<Map::SignalSearchData> Map::GetSignalsInDistance(
322  Waypoint waypoint, double distance, bool stop_at_junction) const {
323 
324  const auto &lane = GetLane(waypoint);
325  const bool forward = (waypoint.lane_id <= 0);
326  const double signed_distance = forward ? distance : -distance;
327  const double relative_s = waypoint.s - lane.GetDistance();
328  const double remaining_lane_length = forward ? lane.GetLength() - relative_s : relative_s;
329  DEBUG_ASSERT(remaining_lane_length >= 0.0);
330 
331  auto &road =_data.GetRoad(waypoint.road_id);
332  std::vector<SignalSearchData> result;
333 
334  // If after subtracting the distance we are still in the same lane, return
335  // same waypoint with the extra distance.
336  if (distance <= remaining_lane_length) {
337  auto signals = road.GetInfosInRange<RoadInfoSignal>(
338  waypoint.s, waypoint.s + signed_distance);
339  for(auto* signal : signals){
340  double distance_to_signal = 0;
341  if (waypoint.lane_id < 0){
342  distance_to_signal = signal->GetDistance() - waypoint.s;
343  } else {
344  distance_to_signal = waypoint.s - signal->GetDistance();
345  }
346  Waypoint signal_waypoint = GetNext(waypoint, distance_to_signal).front();
347  SignalSearchData signal_data{signal, signal_waypoint, distance_to_signal};
348  result.emplace_back(signal_data);
349  }
350  return result;
351  }
352  const double signed_remaining_length = forward ? remaining_lane_length : -remaining_lane_length;
353 
354  //result = road.GetInfosInRange<RoadInfoSignal>(waypoint.s, waypoint.s + signed_remaining_length);
355  auto signals = road.GetInfosInRange<RoadInfoSignal>(
356  waypoint.s, waypoint.s + signed_remaining_length);
357  for(auto* signal : signals){
358  double distance_to_signal = 0;
359  if (waypoint.lane_id < 0){
360  distance_to_signal = signal->GetDistance() - waypoint.s;
361  } else {
362  distance_to_signal = waypoint.s - signal->GetDistance();
363  }
364  Waypoint signal_waypoint = GetNext(waypoint, distance_to_signal).front();
365  SignalSearchData signal_data{signal, signal_waypoint, distance_to_signal};
366  result.emplace_back(signal_data);
367  }
368  // If we run out of remaining_lane_length we have to go to the successors.
369  for (const auto &successor : GetSuccessors(waypoint)) {
370  if(_data.GetRoad(successor.road_id).IsJunction() && stop_at_junction){
371  continue;
372  }
373  auto sucessor_signals = GetSignalsInDistance(
374  successor, distance - remaining_lane_length, stop_at_junction);
375  for(auto& signal : sucessor_signals){
376  signal.accumulated_s += remaining_lane_length;
377  }
378  result = ConcatVectors(result, sucessor_signals);
379  }
380  return result;
381  }

402  std::vector<geom::Location> Map::GetAllCrosswalkZones() const {
403  std::vector<geom::Location> result;
404 
405  for (const auto &pair : _data.GetRoads()) {
406  const auto &road = pair.second;
407  std::vector<const RoadInfoCrosswalk *> crosswalks = road.GetInfos<RoadInfoCrosswalk>();
408  if (crosswalks.size() > 0) {
409  for (auto crosswalk : crosswalks) {
410  // waypoint only at start position
411  std::vector<geom::Location> points;
412  Waypoint waypoint;
413  geom::Transform base;
414  for (const auto &section : road.GetLaneSectionsAt(crosswalk->GetS())) {
415  // get the section with the center lane
416  for (const auto &lane : section.GetLanes()) {
417  // is the center line
418  if (lane.first == 0) {
419  // get the center point
420  waypoint.road_id = pair.first;
421  waypoint.section_id = section.GetId();
422  waypoint.lane_id = 0;
423  waypoint.s = crosswalk->GetS();
424  base = ComputeTransform(waypoint);
425  }
426  }
427  }
428 
429  // move perpendicular ('t')
430  geom::Transform pivot = base;
431  pivot.rotation.yaw -= geom::Math::ToDegrees<float>(static_cast<float>(crosswalk->GetHeading()));
432  pivot.rotation.yaw -= 90; // move perpendicular to 's' for the lateral offset
433  geom::Vector3D v(static_cast<float>(crosswalk->GetT()), 0.0f, 0.0f);
434  pivot.TransformPoint(v);
435  // restore pivot position and orientation
436  pivot = base;
437  pivot.location = v;
438  pivot.rotation.yaw -= geom::Math::ToDegrees<float>(static_cast<float>(crosswalk->GetHeading()));
439 
440  // calculate all the corners
441  for (auto corner : crosswalk->GetPoints()) {
442  geom::Vector3D v2(
443  static_cast<float>(corner.u),
444  static_cast<float>(corner.v),
445  static_cast<float>(corner.z));
446  // set the width larger to contact with the sidewalk (in case they have gutter area)
447  if (corner.u < 0) {
448  v2.x -= 1.0f;
449  } else {
450  v2.x += 1.0f;
451  }
452  pivot.TransformPoint(v2);
453  result.push_back(v2);
454  }
455  }
456  }
457  }
458  return result;
459  }

501  std::vector<Waypoint> Map::GetNext(
502  const Waypoint waypoint,
503  const double distance) const {
504  RELEASE_ASSERT(distance > 0.0);
505  const auto &lane = GetLane(waypoint);
506  const bool forward = (waypoint.lane_id <= 0);
507  const double signed_distance = forward ? distance : -distance;
508  const double relative_s = waypoint.s - lane.GetDistance();
509  const double remaining_lane_length = forward ? lane.GetLength() - relative_s : relative_s;
510  DEBUG_ASSERT(remaining_lane_length >= 0.0);
511 
512  // If after subtracting the distance we are still in the same lane, return
513  // same waypoint with the extra distance.
514  if (distance <= remaining_lane_length) {
515  Waypoint result = waypoint;
516  result.s += signed_distance;
517  result.s += forward ? -EPSILON : EPSILON;
518  RELEASE_ASSERT(result.s > 0.0);
519  return { result };
520  }
521 
522  // If we run out of remaining_lane_length we have to go to the successors.
523  std::vector<Waypoint> result;
524  for (const auto &successor : GetSuccessors(waypoint)) {
525  DEBUG_ASSERT(
526  successor.road_id != waypoint.road_id ||
527  successor.section_id != waypoint.section_id ||
528  successor.lane_id != waypoint.lane_id);
529  result = ConcatVectors(result, GetNext(successor, distance - remaining_lane_length));
530  }
531  return result;
532  }

590  RELEASE_ASSERT(distance > 0.0);
591  std::vector<Waypoint> result;

694  std::unordered_map<road::RoadId, std::unordered_set<road::RoadId>>
695  Map::ComputeJunctionConflicts(JuncId id) const {
696 
697  const float epsilon = 0.0001f; // small delta in the road (set to 0.1
698  // millimeters to prevent numeric errors)
699  const Junction *junction = GetJunction(id);
700  std::unordered_map<road::RoadId, std::unordered_set<road::RoadId>>
701  conflicts;
702 
703  // 2d typedefs
704  typedef boost::geometry::model::point
705  <float, 2, boost::geometry::cs::cartesian> Point2d;
706  typedef boost::geometry::model::segment<Point2d> Segment2d;
707  typedef boost::geometry::model::box<Rtree::BPoint> Box;
708 
709  // box range
710  auto bbox_pos = junction->GetBoundingBox().location;
711  auto bbox_ext = junction->GetBoundingBox().extent;
712  auto min_corner = geom::Vector3D(
713  bbox_pos.x - bbox_ext.x,
714  bbox_pos.y - bbox_ext.y,
715  bbox_pos.z - bbox_ext.z - epsilon);
716  auto max_corner = geom::Vector3D(
717  bbox_pos.x + bbox_ext.x,
718  bbox_pos.y + bbox_ext.y,
719  bbox_pos.z + bbox_ext.z + epsilon);
720  Box box({min_corner.x, min_corner.y, min_corner.z},
721  {max_corner.x, max_corner.y, max_corner.z});
722  auto segments = _rtree.GetIntersections(box);
723 
724  for (size_t i = 0; i < segments.size(); ++i){
725  auto &segment1 = segments[i];
726  auto waypoint1 = segment1.second.first;
727  JuncId junc_id1 = _data.GetRoad(waypoint1.road_id).GetJunctionId();
728  // only segments in the junction
729  if(junc_id1 != id){
730  continue;
731  }
732  Segment2d seg1{{segment1.first.first.get<0>(), segment1.first.first.get<1>()},
733  {segment1.first.second.get<0>(), segment1.first.second.get<1>()}};
734  for (size_t j = i + 1; j < segments.size(); ++j){
735  auto &segment2 = segments[j];
736  auto waypoint2 = segment2.second.first;
737  JuncId junc_id2 = _data.GetRoad(waypoint2.road_id).GetJunctionId();
738  // only segments in the junction
739  if(junc_id2 != id){
740  continue;
741  }
742  // discard same road
743  if(waypoint1.road_id == waypoint2.road_id){
744  continue;
745  }
746  Segment2d seg2{{segment2.first.first.get<0>(), segment2.first.first.get<1>()},
747  {segment2.first.second.get<0>(), segment2.first.second.get<1>()}};
748 
749  double distance = boost::geometry::distance(seg1, seg2);
750  // better to set distance to lanewidth
751  if(distance > 2.0){
752  continue;
753  }
754  if(conflicts[waypoint1.road_id].count(waypoint2.road_id) == 0){
755  conflicts[waypoint1.road_id].insert(waypoint2.road_id);
756  }
757  if(conflicts[waypoint2.road_id].count(waypoint1.road_id) == 0){
758  conflicts[waypoint2.road_id].insert(waypoint1.road_id);
759  }
760  }
761  }
762  return conflicts;

818  void Map::CreateRtree() {
819  const double epsilon = 0.000001; // small delta in the road (set to 1
820  // micrometer to prevent numeric errors)
821  const double min_delta_s = 1; // segments of minimum 1m through the road
822 
823  // 1.8 degrees, maximum angle in a curve to place a segment
824  constexpr double angle_threshold = geom::Math::Pi<double>() / 100.0;
825  // maximum distance of a segment
826  constexpr double max_segment_length = 100.0;
827 
828  // Generate waypoints at start of every lane
829  std::vector<Waypoint> topology;
830  for (const auto &pair : _data.GetRoads()) {
831  const auto &road = pair.second;
832  ForEachLane(road, Lane::LaneType::Any, [&](auto &&waypoint) {
833  if(waypoint.lane_id != 0) {
834  topology.push_back(waypoint);
835  }
836  });
837  }
838 
839  // Container of segments and waypoints
840  std::vector<Rtree::TreeElement> rtree_elements;
841  // Loop through all lanes
842  for (auto &waypoint : topology) {
843  auto &lane_start_waypoint = waypoint;
844 
845  auto current_waypoint = lane_start_waypoint;
846 
847  const Lane &lane = GetLane(current_waypoint);
848 
849  geom::Transform current_transform = ComputeTransform(current_waypoint);
850 
851  // Save computation time in straight lines
852  if (lane.IsStraight()) {
853  double delta_s = min_delta_s;
854  double remaining_length =
855  GetRemainingLength(lane, current_waypoint.s);
856  remaining_length -= epsilon;
857  delta_s = remaining_length;
858  if (delta_s < epsilon) {
859  continue;
860  }
861  auto next = GetNext(current_waypoint, delta_s);
862 
863  RELEASE_ASSERT(next.size() == 1);
864  RELEASE_ASSERT(next.front().road_id == current_waypoint.road_id);
865  auto next_waypoint = next.front();
866 
867  AddElementToRtreeAndUpdateTransforms(
868  rtree_elements,
869  current_transform,
870  current_waypoint,
871  next_waypoint);
872  // end of lane
873  } else {
874  auto next_waypoint = current_waypoint;
875 
876  // Loop until the end of the lane
877  // Advance in small s-increments
878  while (true) {
879  double delta_s = min_delta_s;
880  double remaining_length =
881  GetRemainingLength(lane, next_waypoint.s);
882  remaining_length -= epsilon;
883  delta_s = std::min(delta_s, remaining_length);
884 
885  if (delta_s < epsilon) {
886  AddElementToRtreeAndUpdateTransforms(
887  rtree_elements,
888  current_transform,
889  current_waypoint,
890  next_waypoint);
891  break;
892  }
893 
894  auto next = GetNext(next_waypoint, delta_s);
895  if (next.size() != 1 ||
896  current_waypoint.section_id != next.front().section_id) {
897  AddElementToRtreeAndUpdateTransforms(
898  rtree_elements,
899  current_transform,
900  current_waypoint,
901  next_waypoint);
902  break;
903  }
904 
905  next_waypoint = next.front();
906  geom::Transform next_transform = ComputeTransform(next_waypoint);
907  double angle = geom::Math::GetVectorAngle(
908  current_transform.GetForwardVector(), next_transform.GetForwardVector());
909 
910  if (std::abs(angle) > angle_threshold ||
911  std::abs(current_waypoint.s - next_waypoint.s) > max_segment_length) {
912  AddElementToRtree(
913  rtree_elements,
914  current_transform,
915  next_transform,
916  current_waypoint,
917  next_waypoint);
918  current_waypoint = next_waypoint;
919  current_transform = next_transform;
920  }
921  }
922  }
923  }
924  // Add segments to Rtree
925  _rtree.InsertElements(rtree_elements);
926  }

1005  std::vector<std::unique_ptr<geom::Mesh>> lane_meshes;

1016  sidewalk_lane_meshes.push_back(mesh_factory.Generate(lane));
1017  }
1018  }
1019  }
1020  }
1021  if(params.smooth_junctions) {
1022  auto merged_mesh = mesh_factory.MergeAndSmooth(lane_meshes);
1023  for(auto& lane : sidewalk_lane_meshes) {

1026  out_mesh_list.push_back(std::move(merged_mesh));

1029  for(auto& lane : lane_meshes) {
1030  *junction_mesh += *lane;
1031  }
1032  for(auto& lane : sidewalk_lane_meshes) {

1036  }
1037  }

1040  out_mesh_list.front()->GetVertices().front().x,
1041  out_mesh_list.front()->GetVertices().front().y);
1042  auto max_pos = min_pos;

1048  max_pos.y = std::max(max_pos.y, vertex.y);

1054  for (size_t i = 0; i < mesh_amount_x*mesh_amount_y; ++i) {

1060  size_t y_pos = static_cast<size_t>((vertex.y - min_pos.y) / params.max_road_length);
1061  *(result[x_pos + mesh_amount_x*y_pos]) += *mesh;
1062  }
1063
