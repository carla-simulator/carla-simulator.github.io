60  /// Return a waypoint for each drivable lane on @a lane_section.
61  template <typename FuncT>
62  static void ForEachDrivableLaneImpl(
63  RoadId road_id,
64  const LaneSection &lane_section,
65  double distance,
66  FuncT &&func) {
67  for (const auto &pair : lane_section.GetLanes()) {
68  const auto &lane = pair.second;
69  if ((static_cast<uint32_t>(lane.GetType()) & static_cast<uint32_t>(Lane::LaneType::Driving)) > 0) {
70  std::forward<FuncT>(func)(Waypoint{
71  road_id,
72  lane_section.GetId(),
73  lane.GetId(),
74  distance < 0.0 ? GetDistanceAtStartOfLane(lane) : distance});
75  }
76  }
77  }

103  /// Returns a pair containing first = width, second = tangent,
104  /// for an specific Lane given an s and a iterator over lanes
105  template <typename T>
106  static std::pair<double, double> ComputeTotalLaneWidth(
107  const T container,
108  const double s,
109  const LaneId lane_id) {
110  const bool negative_lane_id = lane_id < 0;
111  double dist = 0.0;
112  double tangent = 0.0;
113  for (const auto &lane : container) {
114  auto info = lane.second.template GetInfo<RoadInfoLaneWidth>(s);
115  RELEASE_ASSERT(info != nullptr);
116  const auto current_polynomial = info->GetPolynomial();
117  auto current_dist = current_polynomial.Evaluate(s);
118  auto current_tang = current_polynomial.Tangent(s);
119  if (lane.first != lane_id) {
120  dist += negative_lane_id ? current_dist : -current_dist;
121  tangent += current_tang;
122  } else if (lane.first == lane_id) {
123  current_dist *= 0.5;
124  dist += negative_lane_id ? current_dist : -current_dist;
125  tangent += current_tang * 0.5;
126  break;
127  }
128  }
129  return std::make_pair(dist, tangent);
130  }

142  boost::optional<Waypoint> Map::GetClosestWaypointOnRoad(
143  const geom::Location &pos,
144  uint32_t lane_type) const {
145  // max_nearests represents the max nearests roads
146  // where we will search for nearests lanes
147  constexpr size_t max_nearests = 50u;
148  // in case that map has less than max_nearests lanes,
149  // we will use the maximum lanes
150  const size_t max_nearest_allowed = std::min(_data.GetRoadCount(), max_nearests);
151 
152  // Unreal's Y axis hack
153  const auto pos_inverted_y = geom::Location(pos.x, -pos.y, pos.z);
154 
155  double nearest_dist[max_nearests];
156  std::fill(nearest_dist, nearest_dist + max_nearest_allowed,
157  std::numeric_limits<double>::max());
158 
159  RoadId ids[max_nearests];
160  std::fill(ids, ids + max_nearest_allowed, 0);
161 
162  double dists[max_nearests];
163  std::fill(dists, dists + max_nearest_allowed, 0.0);
164 
165  for (const auto &road_pair : _data.GetRoads()) {
166  const auto road = &road_pair.second;
167  const auto current_dist = road->GetNearestPoint(pos_inverted_y);
168 
169  for (size_t i = 0u; i < max_nearest_allowed; ++i) {
170  if (current_dist.second < nearest_dist[i]) {
171  // reorder nearest_dist
172  for (size_t j = max_nearest_allowed - 1u; j > i; --j) {
173  DEBUG_ASSERT(j > 0u);
174  nearest_dist[j] = nearest_dist[j - 1];
175  ids[j] = ids[j - 1];
176  dists[j] = dists[j - 1];
177  }
178  nearest_dist[i] = current_dist.second;
179  ids[i] = road->GetId();
180  dists[i] = current_dist.first;
181  break;
182  }
183  }
184  }
185 
186  // search for the nearest lane in nearest_dist
187  Waypoint waypoint;
188  auto nearest_lane_dist = std::numeric_limits<double>::max();
189  for (size_t i = 0u; i < max_nearest_allowed; ++i) {
190  auto lane_dist = _data.GetRoad(ids[i]).GetNearestLane(dists[i], pos_inverted_y, lane_type);
191 
192  if (lane_dist.second < nearest_lane_dist) {
193  nearest_lane_dist = lane_dist.second;
194  waypoint.lane_id = lane_dist.first->GetId();
195  waypoint.road_id = ids[i];
196  waypoint.s = dists[i];
197  }
198  }
199 
200  if (nearest_lane_dist == std::numeric_limits<double>::max()) {
201  return boost::optional<Waypoint>{};
202  }
203 
204  const auto &road = _data.GetRoad(waypoint.road_id);
205 
206  // Make sure 0.0 < waipoint.s < Road's length
207  constexpr double margin = 5.0 * EPSILON;
208  DEBUG_ASSERT(margin < road.GetLength() - margin);
209  waypoint.s = geom::Math::Clamp(waypoint.s, margin, road.GetLength() - margin);
210 
211  auto &lane = road.GetLaneByDistance(waypoint.s, waypoint.lane_id);
212 
213  const auto lane_section = lane.GetLaneSection();
214  RELEASE_ASSERT(lane_section != nullptr);
215  const auto lane_section_id = lane_section->GetId();
216  waypoint.section_id = lane_section_id;
217 
218  return waypoint;
219  }

221  boost::optional<Waypoint> Map::GetWaypoint(
222  const geom::Location &pos,
223  uint32_t lane_type) const {
224  boost::optional<Waypoint> w = GetClosestWaypointOnRoad(pos, lane_type);
225 
226  if (!w.has_value()) {
227  return w;
228  }
229 
230  const auto dist = geom::Math::Distance2D(ComputeTransform(*w).location, pos);
231  const auto lane_width_info = GetLane(*w).GetInfo<RoadInfoLaneWidth>(w->s);
232  const auto half_lane_width =
233  lane_width_info->GetPolynomial().Evaluate(w->s) * 0.5;
234 
235  if (dist < half_lane_width) {
236  return w;
237  }
238 
239  return boost::optional<Waypoint>{};
240  }

242  geom::Transform Map::ComputeTransform(Waypoint waypoint) const {
243  // lane_id can't be 0
244  RELEASE_ASSERT(waypoint.lane_id != 0);
245 
246  const auto &road = _data.GetRoad(waypoint.road_id);
247 
248  // must s be smaller (or eq) than road lenght and bigger (or eq) than 0?
249  RELEASE_ASSERT(waypoint.s <= road.GetLength());
250  RELEASE_ASSERT(waypoint.s >= 0.0);
251 
252  const auto &lane_section = road.GetLaneSectionById(waypoint.section_id);
253  const std::map<LaneId, Lane> &lanes = lane_section.GetLanes();
254 
255  // check that lane_id exists on the current s
256  RELEASE_ASSERT(!lanes.empty());
257  RELEASE_ASSERT(waypoint.lane_id >= lanes.begin()->first);
258  RELEASE_ASSERT(waypoint.lane_id <= lanes.rbegin()->first);
259 
260  float lane_width = 0.0f;
261  float lane_tangent = 0.0f;
262  if (waypoint.lane_id < 0) {
263  // right lane
264  const auto side_lanes = MakeListView(
265  std::make_reverse_iterator(lanes.lower_bound(0)), lanes.rend());
266  const auto computed_width =
267  ComputeTotalLaneWidth(side_lanes, waypoint.s, waypoint.lane_id);
268  lane_width = static_cast<float>(computed_width.first);
269  lane_tangent = static_cast<float>(computed_width.second);
270  } else {
271  // left lane
272  const auto side_lanes = MakeListView(lanes.lower_bound(1), lanes.end());
273  const auto computed_width =
274  ComputeTotalLaneWidth(side_lanes, waypoint.s, waypoint.lane_id);
275  lane_width = static_cast<float>(computed_width.first);
276  lane_tangent = static_cast<float>(computed_width.second);
277  }
278 
279  // get a directed point in s and apply the computed lateral offet
280  DirectedPoint dp = road.GetDirectedPointIn(waypoint.s);
281 
282  // compute the tangent of the laneOffset
283  const auto lane_offset_info = road.GetInfo<RoadInfoLaneOffset>(waypoint.s);
284  const auto lane_offset_tangent = static_cast<float>(lane_offset_info->GetPolynomial().Tangent(waypoint.s));
285 
286  lane_tangent -= lane_offset_tangent;
287 
288  // Unreal's Y axis hack
289  lane_tangent *= -1;
290 
291  geom::Rotation rot(
292  geom::Math::ToDegrees(static_cast<float>(dp.pitch)),
293  geom::Math::ToDegrees(-static_cast<float>(dp.tangent)), // Unreal's Y axis hack
294  0.0f);
295 
296  dp.ApplyLateralOffset(lane_width);
297 
298  if (waypoint.lane_id > 0) {
299  rot.yaw += 180.0f + geom::Math::ToDegrees(lane_tangent);
300  rot.pitch = 360.0f - rot.pitch;
301  } else {
302  rot.yaw -= geom::Math::ToDegrees(lane_tangent);
303  }
304 
305  // Unreal's Y axis hack
306  dp.location.y *= -1;
307 
308  return geom::Transform(dp.location, rot);
309  }

371  const auto &next_lanes = GetLane(waypoint).GetNextLanes();
372  std::vector<Waypoint> result;
373  result.reserve(next_lanes.size());

375  RELEASE_ASSERT(next_lane != nullptr);

377  RELEASE_ASSERT(lane_id != 0);

383  result.emplace_back(Waypoint{road->GetId(), section->GetId(), lane_id, distance});

406  std::vector<Waypoint> Map::GetNext(
407  const Waypoint waypoint,
408  const double distance) const {
409  RELEASE_ASSERT(distance > 0.0);
410  const auto &lane = GetLane(waypoint);
411  const bool forward = (waypoint.lane_id <= 0);
412  const double signed_distance = forward ? distance : -distance;
413  const double relative_s = waypoint.s - lane.GetDistance() + EPSILON;
414  const double remaining_lane_length = forward ? lane.GetLength() - relative_s : relative_s;
415  DEBUG_ASSERT(remaining_lane_length >= 0.0);
416 
417  // If after subtracting the distance we are still in the same lane, return
418  // same waypoint with the extra distance.
419  if (distance <= remaining_lane_length) {
420  Waypoint result = waypoint;
421  result.s += signed_distance;
422  result.s += forward ? -EPSILON : EPSILON;
423  RELEASE_ASSERT(result.s > 0.0);
424  return { result };
425  }
426 
427  // If we run out of remaining_lane_length we have to go to the successors.
428  std::vector<Waypoint> result;
429  for (const auto &successor : GetSuccessors(waypoint)) {
430  DEBUG_ASSERT(
431  successor.road_id != waypoint.road_id ||
432  successor.section_id != waypoint.section_id ||
433  successor.lane_id != waypoint.lane_id);
434  result = ConcatVectors(result, GetNext(successor, distance - remaining_lane_length));
435  }
436  return result;
437  }
