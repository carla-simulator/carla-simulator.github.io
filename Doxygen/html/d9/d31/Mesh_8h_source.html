161  operator FProceduralCustomMesh() const {
162  FProceduralCustomMesh Mesh;
163 
164  // Build the mesh
165  for (const auto Vertex : GetVertices())
166  {
167  // From meters to centimeters
168  Mesh.Vertices.Add(FVector{1e2f * Vertex.x, 1e2f * Vertex.y, 1e2f * Vertex.z});
169  }
170 
171  const auto Indexes = GetIndexes();
172  TArray<FTriIndices> TriIndices;
173  for (auto i = 0u; i < Indexes.size(); i += 3)
174  {
175  FTriIndices Triangle;
176  // "-1" since mesh indexes in Unreal starts from index 0.
177  Mesh.Triangles.Add(Indexes[i] - 1);
178  // Since Unreal's coords are left handed, invert the last 2 indices.
179  Mesh.Triangles.Add(Indexes[i + 2] - 1);
180  Mesh.Triangles.Add(Indexes[i + 1] - 1);
181 
182  Triangle.v0 = Indexes[i] - 1;
183  Triangle.v1 = Indexes[i + 2] - 1;
184  Triangle.v2 = Indexes[i + 1] - 1;
185  TriIndices.Add(Triangle);
186  }
187 
188  // Compute the normals
189  TArray<FVector> Normals;
190  Mesh.Normals.Init(FVector::UpVector, Mesh.Vertices.Num());
191 
192  for (const auto &Triangle : TriIndices) {
193  FVector Normal;
194  const FVector U = Mesh.Vertices[Triangle.v1] - Mesh.Vertices[Triangle.v0];
195  const FVector V = Mesh.Vertices[Triangle.v2] - Mesh.Vertices[Triangle.v0];
196  Normal.X = (U.Y * V.Z) - (U.Z * V.Y);
197  Normal.Y = (U.Z * V.X) - (U.X * V.Z);
198  Normal.Z = (U.X * V.Y) - (U.Y * V.X);
199  Normal = -Normal;
200  Normal = Normal.GetSafeNormal(.0001f);
201  if (Normal != FVector::ZeroVector)
202  {
203  // fix to prevent ugly x-fighting in geometries with very large curvatures,
204  // ensures that all road geometry is facing upwards
205  if (FVector::DotProduct(Normal, FVector(0,0,1)) < 0)
206  {
207  Normal = -Normal;
208  }
209  Mesh.Normals[Triangle.v0] = Normal;
210  Mesh.Normals[Triangle.v1] = Normal;
211  Mesh.Normals[Triangle.v2] = Normal;
212  }
213  }
214 
215  return Mesh;
216  }
