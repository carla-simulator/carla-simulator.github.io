20 static auto FWorldObserver_GetActorState(const FActorView &View, const FActorRegistry &Registry)
21 {
22  using AType = FActorView::ActorType;
23 
24  carla::sensor::data::ActorDynamicState::TypeDependentState state;
25 
26  if (AType::Vehicle == View.GetActorType())
27  {
28  auto Vehicle = Cast<ACarlaWheeledVehicle>(View.GetActor());
29  if (Vehicle != nullptr)
30  {
31  state.vehicle_data.control = carla::rpc::VehicleControl{Vehicle->GetVehicleControl()};
32  auto Controller = Cast<AWheeledVehicleAIController>(Vehicle->GetController());
33  if (Controller != nullptr)
34  {
35  using TLS = carla::rpc::TrafficLightState;
36  state.vehicle_data.traffic_light_state = static_cast<TLS>(Controller->GetTrafficLightState());
37  state.vehicle_data.speed_limit = Controller->GetSpeedLimit();
38  auto TrafficLight = Controller->GetTrafficLight();
39  if (TrafficLight != nullptr)
40  {
41  state.vehicle_data.has_traffic_light = true;
42  auto TrafficLightView = Registry.Find(TrafficLight);
43  state.vehicle_data.traffic_light_id = TrafficLightView.GetActorId();
44  }
45  else
46  {
47  state.vehicle_data.has_traffic_light = false;
48  }
49  }
50  }
51  }
52 
53  else if (AType::Walker == View.GetActorType())
54  {
55  auto Walker = Cast<APawn>(View.GetActor());
56  auto Controller = Walker != nullptr ? Cast<AWalkerController>(Walker->GetController()) : nullptr;
57  if (Controller != nullptr)
58  {
59  state.walker_control = carla::rpc::WalkerControl{Controller->GetWalkerControl()};
60  }
61  }
62  else if (AType::TrafficLight == View.GetActorType())
63  {
64  auto TrafficLight = Cast<ATrafficLightBase>(View.GetActor());
65  if (TrafficLight != nullptr)
66  {
67  using TLS = carla::rpc::TrafficLightState;

71  state.traffic_light_data.red_time = TrafficLight->GetRedTime();
72  state.traffic_light_data.elapsed_time = TrafficLight->GetElapsedTime();

102 static carla::Buffer FWorldObserver_Serialize(
103  carla::Buffer &&buffer,
104  const UCarlaEpisode &Episode,
105  float DeltaSeconds)
106 {
107  using Serializer = carla::sensor::s11n::EpisodeStateSerializer;
108  using ActorDynamicState = carla::sensor::data::ActorDynamicState;
109 
110  const auto &Registry = Episode.GetActorRegistry();
111 
112  // Set up buffer for writing.
113  buffer.reset(sizeof(Serializer::Header) + sizeof(ActorDynamicState) * Registry.Num());
114  auto begin = buffer.begin();
115  auto write_data = [&begin](const auto &data)
116  {
117  std::memcpy(begin, &data, sizeof(data));
118  begin += sizeof(data);
119  };
120 
121  // Write header.
122  Serializer::Header header;
123  header.episode_id = Episode.GetId();
124  header.platform_timestamp = FPlatformTime::Seconds();
125  header.delta_seconds = DeltaSeconds;
126  write_data(header);
127 
128  // Write every actor.
129  for (auto &&View : Registry)
130  {
131  check(View.IsValid());
132  constexpr float TO_METERS = 1e-2;
133  const auto Velocity = TO_METERS * View.GetActor()->GetVelocity();
134 
135  ActorDynamicState info = {
136  View.GetActorId(),
137  View.GetActor()->GetActorTransform(),
138  carla::geom::Vector3D{Velocity.X, Velocity.Y, Velocity.Z},
139  FWorldObserver_GetAngularVelocity(*View.GetActor()),
140  FWorldObserver_GetAcceleration(View, Velocity, DeltaSeconds),
141  FWorldObserver_GetActorState(View, Registry)
142  };
143  write_data(info);
144  }
145 
146  check(begin == buffer.end());
147  return std::move(buffer);
148 }
