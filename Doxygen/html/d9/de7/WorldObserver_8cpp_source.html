26 static auto FWorldObserver_GetActorState(const FActorView &View, const FActorRegistry &Registry)
27 {
28  using AType = FActorView::ActorType;
29 
30  carla::sensor::data::ActorDynamicState::TypeDependentState state;
31 
32  if (AType::Vehicle == View.GetActorType())
33  {
34  auto Vehicle = Cast<ACarlaWheeledVehicle>(View.GetActor());
35  if (Vehicle != nullptr)
36  {
37  state.vehicle_data.control = carla::rpc::VehicleControl{Vehicle->GetVehicleControl()};
38  auto Controller = Cast<AWheeledVehicleAIController>(Vehicle->GetController());
39  if (Controller != nullptr)
40  {
41  using TLS = carla::rpc::TrafficLightState;
42  state.vehicle_data.traffic_light_state = static_cast<TLS>(Controller->GetTrafficLightState());
43  state.vehicle_data.speed_limit = Controller->GetSpeedLimit();
44  auto TrafficLight = Controller->GetTrafficLight();
45  if (TrafficLight != nullptr)
46  {
47  state.vehicle_data.has_traffic_light = true;
48  auto TrafficLightView = Registry.Find(TrafficLight);
49  state.vehicle_data.traffic_light_id = TrafficLightView.GetActorId();
50  }
51  else
52  {
53  state.vehicle_data.has_traffic_light = false;
54  }
55  }
56  }
57  }
58 
59  else if (AType::Walker == View.GetActorType())
60  {
61  auto Walker = Cast<APawn>(View.GetActor());
62  auto Controller = Walker != nullptr ? Cast<AWalkerController>(Walker->GetController()) : nullptr;
63  if (Controller != nullptr)
64  {
65  state.walker_control = carla::rpc::WalkerControl{Controller->GetWalkerControl()};
66  }
67  }
68  else if (AType::TrafficLight == View.GetActorType())
69  {
70  auto TrafficLight = Cast<ATrafficLightBase>(View.GetActor());
71  if (TrafficLight != nullptr)
72  {
73  auto* TrafficLightComponent =
74  TrafficLight->GetTrafficLightComponent();
75 
76  using TLS = carla::rpc::TrafficLightState;
77 
78  if(TrafficLightComponent == nullptr)
79  {
80  // Old way: traffic lights are actors
81  state.traffic_light_data.sign_id[0] = '\0';
82  state.traffic_light_data.state = static_cast<TLS>(TrafficLight->GetTrafficLightState());
83  state.traffic_light_data.green_time = TrafficLight->GetGreenTime();
84  state.traffic_light_data.yellow_time = TrafficLight->GetYellowTime();
85  state.traffic_light_data.red_time = TrafficLight->GetRedTime();
86  state.traffic_light_data.elapsed_time = TrafficLight->GetElapsedTime();
87  state.traffic_light_data.time_is_frozen = TrafficLight->GetTimeIsFrozen();
88  state.traffic_light_data.pole_index = TrafficLight->GetPoleIndex();
89  }
90  else
91  {
92  const UTrafficLightController* Controller = TrafficLightComponent->GetController();
93  const ATrafficLightGroup* Group = TrafficLightComponent->GetGroup();
94 
95  if (!Controller)
96  {
97  UE_LOG(LogCarla, Error, TEXT("TrafficLightComponent doesn't have any Controller assigned"));
98  }
99  else if (!Group)
100  {
101  UE_LOG(LogCarla, Error, TEXT("TrafficLightComponent doesn't have any Group assigned"));
102  }
103  else
104  {
105  const FString fstring_sign_id = TrafficLightComponent->GetSignId();
106  const std::string sign_id = carla::rpc::FromFString(fstring_sign_id);
107  constexpr size_t max_size = sizeof(state.traffic_light_data.sign_id);
108  size_t sign_id_length = sign_id.length();
109  if(max_size < sign_id_length)
110  {
111  UE_LOG(LogCarla, Warning, TEXT("The max size of a signal id is 32. %s (%d)"), *fstring_sign_id, sign_id.length());
112  sign_id_length = max_size;
113  }
114  std::memset(state.traffic_light_data.sign_id, '\0', max_size);
115  std::memcpy(state.traffic_light_data.sign_id, sign_id.c_str(), sign_id_length);
116  state.traffic_light_data.state = static_cast<TLS>(TrafficLightComponent->GetLightState());
117  state.traffic_light_data.green_time = Controller->GetGreenTime();
118  state.traffic_light_data.yellow_time = Controller->GetYellowTime();
119  state.traffic_light_data.red_time = Controller->GetRedTime();
120  state.traffic_light_data.elapsed_time = Controller->GetElapsedTime();
121  state.traffic_light_data.time_is_frozen = Group->IsFrozen();
122  state.traffic_light_data.pole_index = TrafficLight->GetPoleIndex();
123  }
124  }
125  }
126  }
127  else if (AType::TrafficSign == View.GetActorType())
128  {
129  auto TrafficSign = Cast<ATrafficSignBase>(View.GetActor());
130  if (TrafficSign != nullptr)
131  {
132  USignComponent* TrafficSignComponent =
133  Cast<USignComponent>(TrafficSign->FindComponentByClass<USignComponent>());
134 
135  if(TrafficSignComponent)
136  {
137  const FString fstring_sign_id = TrafficSignComponent->GetSignId();
138  const std::string sign_id = carla::rpc::FromFString(fstring_sign_id);
139  constexpr size_t max_size = sizeof(state.traffic_sign_data.sign_id);
140  size_t sign_id_length = sign_id.length();
141  if(max_size < sign_id_length)
142  {
143  UE_LOG(LogCarla, Warning, TEXT("The max size of a signal id is 32. %s (%d)"), *fstring_sign_id, sign_id.length());
144  sign_id_length = max_size;
145  }
146  std::memset(state.traffic_light_data.sign_id, '\0', max_size);
147  std::memcpy(state.traffic_sign_data.sign_id, sign_id.c_str(), sign_id_length);
148  }
149  }
150  }
151  return state;
152 }

164 static carla::geom::Vector3D FWorldObserver_GetAcceleration(
165  const FActorView &View,
166  const FVector &Velocity,
167  const float DeltaSeconds)
168 {
169  FVector &PreviousVelocity = View.GetActorInfo()->Velocity;
170  const FVector Acceleration = (Velocity - PreviousVelocity) / DeltaSeconds;

175 static carla::Buffer FWorldObserver_Serialize(
176  carla::Buffer &&buffer,
177  const UCarlaEpisode &Episode,
178  float DeltaSeconds,
179  bool MapChange,
180  bool PendingLightUpdates)
181 {
182 
183  using Serializer = carla::sensor::s11n::EpisodeStateSerializer;
184  using SimulationState = carla::sensor::s11n::EpisodeStateSerializer::SimulationState;
185  using ActorDynamicState = carla::sensor::data::ActorDynamicState;
186 
187 
188  const auto &Registry = Episode.GetActorRegistry();
189 
190  auto total_size = sizeof(Serializer::Header) + sizeof(ActorDynamicState) * Registry.Num();
191  auto current_size = 0;
192  // Set up buffer for writing.
193  buffer.reset(total_size);
194  auto write_data = [&current_size, &buffer](const auto &data)
195  {
196  auto begin = buffer.begin() + current_size;
197  std::memcpy(begin, &data, sizeof(data));
198  current_size += sizeof(data);
199  };
200 
201  // Write header.
202  Serializer::Header header;
203  header.episode_id = Episode.GetId();
204  header.platform_timestamp = FPlatformTime::Seconds();
205  header.delta_seconds = DeltaSeconds;
206 
207  uint8_t simulation_state = (SimulationState::MapChange * MapChange);
208  simulation_state |= (SimulationState::PendingLightUpdate * PendingLightUpdates);
209 
210  header.simulation_state = static_cast<SimulationState>(simulation_state);
211 
212  write_data(header);
213 
214  // Write every actor.
215  for (auto &&View : Registry)
216  {
217  check(View.IsValid());
218  constexpr float TO_METERS = 1e-2;
219  const auto Velocity = TO_METERS * View.GetActor()->GetVelocity();
220 
221  ActorDynamicState info = {
222  View.GetActorId(),
223  View.GetActor()->GetActorTransform(),
224  carla::geom::Vector3D{Velocity.X, Velocity.Y, Velocity.Z},
225  FWorldObserver_GetAngularVelocity(*View.GetActor()),
226  FWorldObserver_GetAcceleration(View, Velocity, DeltaSeconds),
227  FWorldObserver_GetActorState(View, Registry),
228  };
229  write_data(info);
230  }
231 
232  // Shrink buffer
233  buffer.resize(current_size);
234 
235  check(buffer.size() == current_size);
236 
237  return std::move(buffer);
238 }
