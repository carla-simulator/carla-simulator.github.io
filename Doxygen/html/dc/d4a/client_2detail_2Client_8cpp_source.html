46  Pimpl(const std::string &host, uint16_t port, size_t worker_threads)
47  : endpoint(host + ":" + std::to_string(port)),
48  rpc_client(host, port),
49  streaming_client(host) {
50  rpc_client.set_timeout(5000u);
51  streaming_client.AsyncRun(
52  worker_threads > 0u ? worker_threads : std::thread::hardware_concurrency());

55  template <typename ... Args>
56  auto RawCall(const std::string &function, Args && ... args) {
57  try {
58  return rpc_client.call(function, std::forward<Args>(args) ...);
59  } catch (const ::rpc::timeout &) {
60  throw_exception(TimeoutException(endpoint, GetTimeout()));
61  }
62  }
63 
64  template <typename T, typename ... Args>
65  auto CallAndWait(const std::string &function, Args && ... args) {
66  auto object = RawCall(function, std::forward<Args>(args) ...);
67  using R = typename carla::rpc::Response<T>;
68  auto response = object.template as<R>();
69  if (response.HasError()) {
70  throw_exception(std::runtime_error(response.GetError().What()));
71  }
72  return Get(response);
73  }
74 
75  template <typename ... Args>
76  void AsyncCall(const std::string &function, Args && ... args) {
77  // Discard returned future.
78  rpc_client.async_call(function, std::forward<Args>(args) ...);
79  }
80 
81  time_duration GetTimeout() const {
82  auto timeout = rpc_client.get_timeout();
83  DEBUG_ASSERT(timeout.has_value());
84  return time_duration::milliseconds(static_cast<size_t>(*timeout));
85  }
86 
87  const std::string endpoint;
88 
89  rpc::Client rpc_client;
90 
91  streaming::Client streaming_client;
92  };
