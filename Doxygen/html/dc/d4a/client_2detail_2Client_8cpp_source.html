47  Pimpl(const std::string &host, uint16_t port, size_t worker_threads)
48  : endpoint(host + ":" + std::to_string(port)),
49  rpc_client(host, port),

56  template <typename ... Args>
57  auto RawCall(const std::string &function, Args && ... args) {
58  try {
59  return rpc_client.call(function, std::forward<Args>(args) ...);
60  } catch (const ::rpc::timeout &) {
61  throw_exception(TimeoutException(endpoint, GetTimeout()));
62  }
63  }
64 
65  template <typename T, typename ... Args>
66  auto CallAndWait(const std::string &function, Args && ... args) {
67  auto object = RawCall(function, std::forward<Args>(args) ...);
68  using R = typename carla::rpc::Response<T>;
69  auto response = object.template as<R>();
70  if (response.HasError()) {
71  throw_exception(std::runtime_error(response.GetError().What()));
72  }
73  return Get(response);
74  }
75 
76  template <typename ... Args>
77  void AsyncCall(const std::string &function, Args && ... args) {
78  // Discard returned future.
79  rpc_client.async_call(function, std::forward<Args>(args) ...);
80  }
81 
82  time_duration GetTimeout() const {
83  auto timeout = rpc_client.get_timeout();
84  DEBUG_ASSERT(timeout.has_value());
85  return time_duration::milliseconds(static_cast<size_t>(*timeout));
86  }
87 
88  const std::string endpoint;
89 
90  rpc::Client rpc_client;
91 
92  streaming::Client streaming_client;
93  };

227  rpc::Actor Client::SpawnActorWithParent(
228  const rpc::ActorDescription &description,
229  const geom::Transform &transform,
230  rpc::ActorId parent,
231  rpc::AttachmentType attachment_type) {
232 
233  if(attachment_type == rpc::AttachmentType::SpringArm) {
234  const auto a = transform.location.MakeSafeUnitVector(std::numeric_limits<float>::epsilon());
235  const auto z = geom::Vector3D(0.0f, 0.f, 1.0f);
236  constexpr float OneEps = 1.0f - std::numeric_limits<float>::epsilon();
237  if (geom::Math::Dot(a, z) > OneEps) {
238  std::cout << "WARNING: Transformations with translation only in the 'z' axis are ill-formed when \
239  using SprintArm attachment. Please, be careful with that." << std::endl;
240  }
241  }
242 
243  return _pimpl->CallAndWait<rpc::Actor>("spawn_actor_with_parent",
244  description,
245  transform,
246  parent,
247  attachment_type);
248  }
