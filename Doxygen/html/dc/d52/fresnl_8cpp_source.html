23  * | |
24  * -
25  * 0
26  *

29  * | |
30  * S(x) = | sin(pi/2 t**2) dt.
31  * | |
32  * -

34  *
35  *
36  * The integrals are evaluated by a power series for x < 1.
37  * For x >= 1 auxiliary functions f(x) and g(x) are employed

39  *

158 int fresnl(double xxa, double *ssa, double *cca)
159 {
160  double f, g, cc, ss, c, s, t, u;
161  double x, x2;
162 
163  x = fabs(xxa);
164  x2 = x * x;
165 
166  if (x2 < 2.5625)
167  {
168  t = x2 * x2;
169  ss = x * x2 * polevl(t, sn, 5) / p1evl(t, sd, 6);
170  cc = x * polevl(t, cn, 5) / polevl(t, cd, 6);
171  goto done;
172  }
173 
174  if (x > 36974.0)
175  {
176  cc = 0.5;
177  ss = 0.5;
178  goto done;
179  }
180 
181  /* Asymptotic power series auxiliary functions
182  * for large argument
183  */
184  x2 = x * x;
185  t = FULL_PI_FRESNEL * x2;
186  u = 1.0 / (t * t);
187  t = 1.0 / t;
188  f = 1.0 - u * polevl(u, fn, 9) / p1evl(u, fd, 10);
189  g = t * polevl(u, gn, 10) / p1evl(u, gd, 11);
190 
191  t = HALF_PI_FRESNEL * x2;
192  c = cos(t);
193  s = sin(t);
194  t = FULL_PI_FRESNEL * x;
195  cc = 0.5 + (f * s - g * c) / t;
196  ss = 0.5 - (f * c + g * s) / t;
197 
198 done:
199  if (xxa < 0.0)
200  {
201  cc = -cc;
202  ss = -ss;
203  }
204 
205  *cca = cc;
206  *ssa = ss;
207  return(0);
208 }
