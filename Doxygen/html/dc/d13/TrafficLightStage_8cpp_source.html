43  void TrafficLightStage::Action() {
44 
45  // Selecting the output frame based on the selection key.
46  const auto current_planner_frame = frame_selector ? planner_frame_a : planner_frame_b;
47  // Looping over registered actors.
48  for (uint i = 0u; i < number_of_vehicles; ++i) {
49 
50  bool traffic_light_hazard = false;
51  const LocalizationToTrafficLightData &data = localization_frame->at(i);
52  const Actor ego_actor = data.actor;
53  const ActorId ego_actor_id = ego_actor->GetId();
54  const SimpleWaypointPtr closest_waypoint = data.closest_waypoint;
55  const SimpleWaypointPtr look_ahead_point = data.junction_look_ahead_waypoint;
56 
57  const JunctionID junction_id = look_ahead_point->GetWaypoint()->GetJunctionId();
58  const TimeInstance current_time = chr::system_clock::now();
59 
60  const auto ego_vehicle = boost::static_pointer_cast<cc::Vehicle>(ego_actor);
61  TLS traffic_light_state = ego_vehicle->GetTrafficLightState();
62  // Generate number between 0 and 100
63  const int r = rand() % 101;
64 
65  // Set to green if random number is lower than percentage, default is 0
66  if (parameters.GetPercentageRunningLight(boost::shared_ptr<cc::Actor>(ego_actor)) > r)
67  traffic_light_state = TLS::Green;
68 
69  // We determine to stop if the current position of the vehicle is not a
70  // junction,
71  // a point on the path beyond a threshold (velocity-dependent) distance
72  // is inside the junction and there is a red or yellow light.
73  if (!closest_waypoint->CheckJunction() &&
74  look_ahead_point->CheckJunction() &&
75  ego_vehicle->IsAtTrafficLight() &&
76  traffic_light_state != TLS::Green) {
77 
78  traffic_light_hazard = true;
79  }
80  // Handle entry negotiation at non-signalised junction.
81  else if (!closest_waypoint->CheckJunction() &&
82  look_ahead_point->CheckJunction() &&
83  !ego_vehicle->IsAtTrafficLight() &&
84  traffic_light_state != TLS::Green) {
85 
86  std::lock_guard<std::mutex> lock(no_signal_negotiation_mutex);
87 
88  if (vehicle_last_junction.find(ego_actor_id) == vehicle_last_junction.end()) {
89  // Initializing new map entry for the actor with
90  // an arbitrary and different junction id.
91  vehicle_last_junction.insert({ego_actor_id, junction_id + 1});
92  }
93 
94  if (vehicle_last_junction.at(ego_actor_id) != junction_id) {
95  vehicle_last_junction.at(ego_actor_id) = junction_id;
96 
97  // Check if the vehicle has an outdated ticket or needs a new one.
98  bool need_to_issue_new_ticket = false;
99  if (vehicle_last_ticket.find(ego_actor_id) == vehicle_last_ticket.end()) {
100 
101  need_to_issue_new_ticket = true;
102  } else {
103 
104  const TimeInstance &previous_ticket = vehicle_last_ticket.at(ego_actor_id);
105  const chr::duration<double> diff = current_time - previous_ticket;
106  if (diff.count() > NO_SIGNAL_PASSTHROUGH_INTERVAL) {
107  need_to_issue_new_ticket = true;
108  }
109  }
110 
111  // If new ticket is needed for the vehicle, then query the junction
112  // ticket map
113  // and update the map value to the new ticket value.
114  if (need_to_issue_new_ticket) {
115  if (junction_last_ticket.find(junction_id) != junction_last_ticket.end()) {
116 
117  TimeInstance &last_ticket = junction_last_ticket.at(junction_id);
118  const chr::duration<double> diff = current_time - last_ticket;
119  if (diff.count() > 0.0) {
120  last_ticket = current_time + chr::seconds(NO_SIGNAL_PASSTHROUGH_INTERVAL);
121  } else {
122  last_ticket += chr::seconds(NO_SIGNAL_PASSTHROUGH_INTERVAL);
123  }
124  } else {
125 
126  junction_last_ticket.insert({junction_id, current_time +
127  chr::seconds(NO_SIGNAL_PASSTHROUGH_INTERVAL)});
128  }
129 
130  if (vehicle_last_ticket.find(ego_actor_id) != vehicle_last_ticket.end()) {
131  vehicle_last_ticket.at(ego_actor_id) = junction_last_ticket.at(junction_id);
132  } else {
133  vehicle_last_ticket.insert({ego_actor_id, junction_last_ticket.at(junction_id)});
134  }
135  }
136  }
137 
138  // If current time is behind ticket time, then do not enter junction.
139  const TimeInstance &current_ticket = vehicle_last_ticket.at(ego_actor_id);
140  const chr::duration<double> diff = current_ticket - current_time;
141  if (diff.count() > 0.0) {
142  traffic_light_hazard = true;
143  }
144  }
145 
146  TrafficLightToPlannerData &message = current_planner_frame->at(i);
147  message.traffic_light_hazard = traffic_light_hazard;
148  }

152  void TrafficLightStage::DataReceiver() {
153  const auto packet = localization_messenger->ReceiveData(localization_messenger_state);
154  localization_frame = packet.data;
155  localization_messenger_state = packet.id;
156 
157  // Allocating new containers for the changed number of registered vehicles.
158  if (localization_frame != nullptr &&
159  number_of_vehicles != (*localization_frame.get()).size()) {
160 
161  number_of_vehicles = static_cast<uint>((*localization_frame.get()).size());
162  // Allocating output frames.
163  planner_frame_a = std::make_shared<TrafficLightToPlannerFrame>(number_of_vehicles);
164  planner_frame_b = std::make_shared<TrafficLightToPlannerFrame>(number_of_vehicles);
165  }
166  }
