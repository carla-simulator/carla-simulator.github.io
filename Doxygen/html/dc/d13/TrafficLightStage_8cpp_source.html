37  void TrafficLightStage::Action() {
38 
39  // Selecting the output frame based on the selection key.
40  const auto current_planner_frame = frame_selector ? planner_frame_a : planner_frame_b;
41  // Looping over registered actors.
42  for (uint64_t i = 0u; i < number_of_vehicles && localization_frame != nullptr; ++i) {
43 
44  bool traffic_light_hazard = false;
45  const LocalizationToTrafficLightData &data = localization_frame->at(i);
46  const Actor ego_actor = data.actor;
47  const ActorId ego_actor_id = ego_actor->GetId();
48  const SimpleWaypointPtr closest_waypoint = data.closest_waypoint;
49  const SimpleWaypointPtr look_ahead_point = data.junction_look_ahead_waypoint;
50 
51  const JunctionID junction_id = look_ahead_point->GetWaypoint()->GetJunctionId();
52  const TimeInstance current_time = chr::system_clock::now();
53 
54  const auto ego_vehicle = boost::static_pointer_cast<cc::Vehicle>(ego_actor);
55  TLS traffic_light_state = ego_vehicle->GetTrafficLightState();
56  // Generate number between 0 and 100
57  const int r = rand() % 101;
58 
59  // Set to green if random number is lower than percentage, default is 0
60  if (parameters.GetPercentageRunningLight(boost::shared_ptr<cc::Actor>(ego_actor)) > r)
61  traffic_light_state = TLS::Green;
62 
63  // We determine to stop if the current position of the vehicle is not a
64  // junction and there is a red or yellow light.
65  if (!closest_waypoint->CheckJunction() &&
66  ego_vehicle->IsAtTrafficLight() &&
67  traffic_light_state != TLS::Green) {
68 
69  traffic_light_hazard = true;
70  }
71  // Handle entry negotiation at non-signalised junction.
72  else if (!closest_waypoint->CheckJunction() &&
73  look_ahead_point->CheckJunction() &&
74  !ego_vehicle->IsAtTrafficLight() &&
75  traffic_light_state != TLS::Green) {
76 
77  std::lock_guard<std::mutex> lock(no_signal_negotiation_mutex);
78 
79  if (vehicle_last_junction.find(ego_actor_id) == vehicle_last_junction.end()) {
80  // Initializing new map entry for the actor with
81  // an arbitrary and different junction id.
82  vehicle_last_junction.insert({ego_actor_id, junction_id + 1});
83  }
84 
85  if (vehicle_last_junction.at(ego_actor_id) != junction_id) {
86  vehicle_last_junction.at(ego_actor_id) = junction_id;
87 
88  // Check if the vehicle has an outdated ticket or needs a new one.
89  bool need_to_issue_new_ticket = false;
90  if (vehicle_last_ticket.find(ego_actor_id) == vehicle_last_ticket.end()) {
91 
92  need_to_issue_new_ticket = true;
93  } else {
94 
95  const TimeInstance &previous_ticket = vehicle_last_ticket.at(ego_actor_id);
96  const chr::duration<double> diff = current_time - previous_ticket;
97  if (diff.count() > NO_SIGNAL_PASSTHROUGH_INTERVAL) {
98  need_to_issue_new_ticket = true;
99  }
100  }
101 
102  // If new ticket is needed for the vehicle, then query the junction
103  // ticket map
104  // and update the map value to the new ticket value.
105  if (need_to_issue_new_ticket) {
106  if (junction_last_ticket.find(junction_id) != junction_last_ticket.end()) {
107 
108  TimeInstance &last_ticket = junction_last_ticket.at(junction_id);
109  const chr::duration<double> diff = current_time - last_ticket;
110  if (diff.count() > 0.0) {
111  last_ticket = current_time + chr::seconds(NO_SIGNAL_PASSTHROUGH_INTERVAL);
112  } else {
113  last_ticket += chr::seconds(NO_SIGNAL_PASSTHROUGH_INTERVAL);
114  }
115  } else {
116 
117  junction_last_ticket.insert({junction_id, current_time +
118  chr::seconds(NO_SIGNAL_PASSTHROUGH_INTERVAL)});
119  }
120 
121  if (vehicle_last_ticket.find(ego_actor_id) != vehicle_last_ticket.end()) {
122  vehicle_last_ticket.at(ego_actor_id) = junction_last_ticket.at(junction_id);
123  } else {
124  vehicle_last_ticket.insert({ego_actor_id, junction_last_ticket.at(junction_id)});
125  }
126  }
127  }
128 
129  // If current time is behind ticket time, then do not enter junction.
130  const TimeInstance &current_ticket = vehicle_last_ticket.at(ego_actor_id);
131  const chr::duration<double> diff = current_ticket - current_time;
132  if (diff.count() > 0.0) {
133  traffic_light_hazard = true;
134  }
135  }
136 
137  TrafficLightToPlannerData &message = current_planner_frame->at(i);
138  message.traffic_light_hazard = traffic_light_hazard;
139  }

143  void TrafficLightStage::DataReceiver() {
144  localization_frame = localization_messenger->Peek();
145 
146  // Allocating new containers for the changed number of registered vehicles.
147  if (localization_frame != nullptr &&
148  number_of_vehicles != (*localization_frame.get()).size()) {
149 
150  number_of_vehicles = static_cast<uint64_t>((*localization_frame.get()).size());
151  // Allocating output frames.
152  planner_frame_a = std::make_shared<TrafficLightToPlannerFrame>(number_of_vehicles);
153  planner_frame_b = std::make_shared<TrafficLightToPlannerFrame>(number_of_vehicles);
154  }
155  }
