40  // We determine to stop if the current position of the vehicle is not a

42  if (is_at_traffic_light &&

60 bool TrafficLightStage::HandleNonSignalisedJunction(const ActorId ego_actor_id, const JunctionID junction_id,
61  const TimeInstance current_time) {
62 
63  bool traffic_light_hazard = false;
64 
65  if (vehicle_last_junction.find(ego_actor_id) == vehicle_last_junction.end()) {
66  // Initializing new map entry for the actor with
67  // an arbitrary and different junction id.
68  vehicle_last_junction.insert({ego_actor_id, junction_id + 1});
69  }
70 
71  if (vehicle_last_junction.at(ego_actor_id) != junction_id) {
72  vehicle_last_junction.at(ego_actor_id) = junction_id;
73 
74  // Check if the vehicle has an outdated ticket or needs a new one.
75  bool need_to_issue_new_ticket = false;
76  if (vehicle_last_ticket.find(ego_actor_id) == vehicle_last_ticket.end()) {
77 
78  need_to_issue_new_ticket = true;
79  } else {
80 
81  const TimeInstance &previous_ticket = vehicle_last_ticket.at(ego_actor_id);
82  const chr::duration<double> diff = current_time - previous_ticket;
83  if (diff.count() > DOUBLE_NO_SIGNAL_PASSTHROUGH_INTERVAL) {
84  need_to_issue_new_ticket = true;
85  }
86  }
87 
88  // If new ticket is needed for the vehicle, then query the junction
89  // ticket map
90  // and update the map value to the new ticket value.
91  if (need_to_issue_new_ticket) {
92  if (junction_last_ticket.find(junction_id) != junction_last_ticket.end()) {
93 
94  TimeInstance &last_ticket = junction_last_ticket.at(junction_id);
95  const chr::duration<double> diff = current_time - last_ticket;
96  if (diff.count() > 0.0) {
97  last_ticket = current_time + chr::seconds(NO_SIGNAL_PASSTHROUGH_INTERVAL);
98  } else {
99  last_ticket += chr::seconds(NO_SIGNAL_PASSTHROUGH_INTERVAL);
100  }
101  } else {
102  junction_last_ticket.insert({junction_id, current_time +
103  chr::seconds(NO_SIGNAL_PASSTHROUGH_INTERVAL)});
104  }
105  if (vehicle_last_ticket.find(ego_actor_id) != vehicle_last_ticket.end()) {
106  vehicle_last_ticket.at(ego_actor_id) = junction_last_ticket.at(junction_id);
107  } else {
108  vehicle_last_ticket.insert({ego_actor_id, junction_last_ticket.at(junction_id)});
109  }
110  }
111  }
112 
113  // If current time is behind ticket time, then do not enter junction.
114  const TimeInstance &current_ticket = vehicle_last_ticket.at(ego_actor_id);
115  const chr::duration<double> diff = current_ticket - current_time;
116  if (diff.count() > 0.0) {
117  traffic_light_hazard = true;
118  }
119 
120  return traffic_light_hazard;
121 }
