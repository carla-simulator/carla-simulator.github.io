40  void TrafficLightStage::Action() {
41 
42  // Selecting the output frame based on the selection key.
43  const auto current_planner_frame = frame_selector ? planner_frame_a : planner_frame_b;
44 
45  // Looping over registered actors.
46  for (uint64_t i = 0u; i < number_of_vehicles && localization_frame != nullptr; ++i) {
47 
48  bool traffic_light_hazard = false;
49  const LocalizationToTrafficLightData &data = localization_frame->at(i);
50  if (!data.actor->IsAlive()) {
51  continue;
52  }
53 
54  const Actor ego_actor = data.actor;
55  const ActorId ego_actor_id = ego_actor->GetId();
56  const SimpleWaypointPtr closest_waypoint = data.closest_waypoint;
57  const SimpleWaypointPtr look_ahead_point = data.junction_look_ahead_waypoint;
58 
59  const JunctionID junction_id = look_ahead_point->GetWaypoint()->GetJunctionId();
60  const TimeInstance current_time = chr::system_clock::now();
61 
62  const auto ego_vehicle = boost::static_pointer_cast<cc::Vehicle>(ego_actor);
63  TLS traffic_light_state = ego_vehicle->GetTrafficLightState();
64 
65  // We determine to stop if the current position of the vehicle is not a
66  // junction and there is a red or yellow light.
67  if (ego_vehicle->IsAtTrafficLight() &&
68  traffic_light_state != TLS::Green &&
69  parameters.GetPercentageRunningLight(boost::shared_ptr<cc::Actor>(ego_actor)) <= (rand() % 101)) {
70 
71  traffic_light_hazard = true;
72  }
73 
74  // Handle entry negotiation at non-signalised junction.
75  else if (look_ahead_point->CheckJunction() &&
76  !ego_vehicle->IsAtTrafficLight() &&
77  traffic_light_state != TLS::Green &&
78  parameters.GetPercentageRunningSign(boost::shared_ptr<cc::Actor>(ego_actor)) <= (rand() % 101)) {
79 
80  std::lock_guard<std::mutex> lock(no_signal_negotiation_mutex);
81 
82  if (vehicle_last_junction.find(ego_actor_id) == vehicle_last_junction.end()) {
83  // Initializing new map entry for the actor with
84  // an arbitrary and different junction id.
85  vehicle_last_junction.insert({ego_actor_id, junction_id + 1});
86  }
87 
88  if (vehicle_last_junction.at(ego_actor_id) != junction_id) {
89  vehicle_last_junction.at(ego_actor_id) = junction_id;
90 
91  // Check if the vehicle has an outdated ticket or needs a new one.
92  bool need_to_issue_new_ticket = false;
93  if (vehicle_last_ticket.find(ego_actor_id) == vehicle_last_ticket.end()) {
94 
95  need_to_issue_new_ticket = true;
96  } else {
97 
98  const TimeInstance &previous_ticket = vehicle_last_ticket.at(ego_actor_id);
99  const chr::duration<double> diff = current_time - previous_ticket;
100  if (diff.count() > NO_SIGNAL_PASSTHROUGH_INTERVAL) {
101  need_to_issue_new_ticket = true;
102  }
103  }
104 
105  // If new ticket is needed for the vehicle, then query the junction
106  // ticket map
107  // and update the map value to the new ticket value.
108  if (need_to_issue_new_ticket) {
109  if (junction_last_ticket.find(junction_id) != junction_last_ticket.end()) {
110 
111  TimeInstance &last_ticket = junction_last_ticket.at(junction_id);
112  const chr::duration<double> diff = current_time - last_ticket;
113  if (diff.count() > 0.0) {
114  last_ticket = current_time + chr::seconds(NO_SIGNAL_PASSTHROUGH_INTERVAL);
115  } else {
116  last_ticket += chr::seconds(NO_SIGNAL_PASSTHROUGH_INTERVAL);
117  }
118  } else {
119  junction_last_ticket.insert({junction_id, current_time +
120  chr::seconds(NO_SIGNAL_PASSTHROUGH_INTERVAL)});
121  }
122  if (vehicle_last_ticket.find(ego_actor_id) != vehicle_last_ticket.end()) {
123  vehicle_last_ticket.at(ego_actor_id) = junction_last_ticket.at(junction_id);
124  } else {
125  vehicle_last_ticket.insert({ego_actor_id, junction_last_ticket.at(junction_id)});
126  }
127  }
128  }
129 
130  // If current time is behind ticket time, then do not enter junction.
131  const TimeInstance &current_ticket = vehicle_last_ticket.at(ego_actor_id);
132  const chr::duration<double> diff = current_ticket - current_time;
133  if (diff.count() > 0.0) {
134  traffic_light_hazard = true;
135  }
136  }
137 
138  TrafficLightToPlannerData &message = current_planner_frame->at(i);
139  message.traffic_light_hazard = traffic_light_hazard;
140  }
141  }

143  void TrafficLightStage::DataReceiver() {
144 
145  localization_frame = localization_messenger->Peek();
146 
147  // Allocating new containers for the changed number of registered vehicles.
148  if (localization_frame != nullptr && number_of_vehicles != (*localization_frame.get()).size()) {
149 
150  number_of_vehicles = static_cast<uint64_t>((*localization_frame.get()).size());
151 
152  // Allocating output frames.
153  planner_frame_a = std::make_shared<TrafficLightToPlannerFrame>(number_of_vehicles);
154  planner_frame_b = std::make_shared<TrafficLightToPlannerFrame>(number_of_vehicles);
155  }
