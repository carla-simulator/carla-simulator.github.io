137 void FPixelReader::WritePixelsToBuffer(
138  UTextureRenderTarget2D &RenderTarget,
139  carla::Buffer &Buffer,
140  uint32 Offset,
141  FRHICommandListImmediate &
142 #if CARLA_WITH_VULKAN_SUPPORT == 1
143  InRHICmdList
144 #endif // CARLA_WITH_VULKAN_SUPPORT
145  )
146 {
147  check(IsInRenderingThread());
148 
149 #if CARLA_WITH_VULKAN_SUPPORT == 1
150  if (IsVulkanPlatform(GMaxRHIShaderPlatform))
151  {
152  WritePixelsToBuffer_Vulkan(RenderTarget, Buffer, Offset, InRHICmdList);
153  return;
154  }
155 #endif // CARLA_WITH_VULKAN_SUPPORT
156 
157  FTextureRenderTargetResource* RenderTargetResource = RenderTarget.GetRenderTargetResource();
158  if(!RenderTargetResource)
159  {
160  return;
161  }
162  FRHITexture2D *Texture = RenderTargetResource->GetRenderTargetTexture();
163  checkf(Texture != nullptr, TEXT("FPixelReader: UTextureRenderTarget2D missing render target texture"));
164 
165  const uint32 BytesPerPixel = 4u; // PF_R8G8B8A8
166  const uint32 Width = Texture->GetSizeX();
167  const uint32 Height = Texture->GetSizeY();
168  const uint32 ExpectedStride = Width * BytesPerPixel;
169 
170  uint32 SrcStride;
171  LockTexture Lock(Texture, SrcStride);
172 
173 #ifdef PLATFORM_WINDOWS
174  // JB: Direct 3D uses additional rows in the buffer, so we need check the
175  // result stride from the lock:
176  if (IsD3DPlatform(GMaxRHIShaderPlatform, false) && (ExpectedStride != SrcStride))
177  {
178  Buffer.reset(Offset + ExpectedStride * Height);
179  auto DstRow = Buffer.begin() + Offset;
180  const uint8 *SrcRow = Lock.Source;
181  for (uint32 Row = 0u; Row < Height; ++Row)
182  {
183  FMemory::Memcpy(DstRow, SrcRow, ExpectedStride);
184  DstRow += ExpectedStride;
185  SrcRow += SrcStride;
186  }
187  }
188  else
189 #endif // PLATFORM_WINDOWS
190  {
191  check(ExpectedStride == SrcStride);
192  const uint8 *Source = Lock.Source;
193  if(Source)
194  {
195  Buffer.copy_from(Offset, Source, ExpectedStride * Height);
196  }
197  }
198 }
