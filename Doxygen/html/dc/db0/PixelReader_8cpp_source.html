133 void FPixelReader::WritePixelsToBuffer(
134  UTextureRenderTarget2D &RenderTarget,
135  carla::Buffer &Buffer,
136  uint32 Offset,
137  FRHICommandListImmediate &
138 #if CARLA_WITH_VULKAN_SUPPORT == 1
139  InRHICmdList
140 #endif // CARLA_WITH_VULKAN_SUPPORT
141  )
142 {
143  check(IsInRenderingThread());
144 
145 #if CARLA_WITH_VULKAN_SUPPORT == 1
146  if (IsVulkanPlatform(GMaxRHIShaderPlatform))
147  {
148  WritePixelsToBuffer_Vulkan(RenderTarget, Buffer, Offset, InRHICmdList);
149  return;
150  }
151 #endif // CARLA_WITH_VULKAN_SUPPORT
152 
153  FRHITexture2D *Texture = RenderTarget.GetRenderTargetResource()->GetRenderTargetTexture();
154  checkf(Texture != nullptr, TEXT("FPixelReader: UTextureRenderTarget2D missing render target texture"));
155 
156  const uint32 BytesPerPixel = 4u; // PF_R8G8B8A8
157  const uint32 Width = Texture->GetSizeX();
158  const uint32 Height = Texture->GetSizeY();
159  const uint32 ExpectedStride = Width * BytesPerPixel;
160 
161  uint32 SrcStride;
162  LockTexture Lock(Texture, SrcStride);
163 
164 #ifdef PLATFORM_WINDOWS
165  // JB: Direct 3D uses additional rows in the buffer, so we need check the
166  // result stride from the lock:
167  if (IsD3DPlatform(GMaxRHIShaderPlatform, false) && (ExpectedStride != SrcStride))
168  {
169  Buffer.reset(Offset + ExpectedStride * Height);
170  auto DstRow = Buffer.begin() + Offset;
171  const uint8 *SrcRow = Lock.Source;
172  for (uint32 Row = 0u; Row < Height; ++Row)
173  {
174  FMemory::Memcpy(DstRow, SrcRow, ExpectedStride);
175  DstRow += ExpectedStride;
176  SrcRow += SrcStride;
177  }
178  }
179  else
180 #endif // PLATFORM_WINDOWS
181  {
182  check(ExpectedStride == SrcStride);
183  const uint8 *Source = Lock.Source;
184  Buffer.copy_from(Offset, Source, ExpectedStride * Height);
185  }
186 }
