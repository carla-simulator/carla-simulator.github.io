182  element::DirectedPoint Road::GetDirectedPointIn(const double s) const {
183  const auto clamped_s = geom::Math::Clamp(s, 0.0, _length);
184  const auto geometry = _info.GetInfo<element::RoadInfoGeometry>(clamped_s);
185 
186  const auto lane_offset = _info.GetInfo<element::RoadInfoLaneOffset>(clamped_s);
187  const auto offset = static_cast<float>(lane_offset->GetPolynomial().Evaluate(clamped_s));
188 
189  // Apply road's lane offset record
190  element::DirectedPoint p = geometry->GetGeometry().PosFromDist(clamped_s - geometry->GetDistance());
191  // Unreal's Y axis hack (the minus on the offset)
192  p.ApplyLateralOffset(-offset);
193 
194  // Apply road's elevation record
195  const auto elevation_info = GetElevationOn(s);
196  p.location.z = static_cast<float>(elevation_info.Evaluate(s));
197  p.pitch = elevation_info.Tangent(s);
198 
199  return p;
200  }

202  const std::pair<double, double> Road::GetNearestPoint(const geom::Location &loc) const {
203  std::pair<double, double> last = { 0.0, std::numeric_limits<double>::max() };
204 
205  auto geom_info_list = _info.GetInfos<element::RoadInfoGeometry>();
206  decltype(geom_info_list)::iterator nearest_geom = geom_info_list.end();
207 
208  for (auto g = geom_info_list.begin(); g != geom_info_list.end(); ++g) {
209  DEBUG_ASSERT(*g != nullptr);
210  auto dist = (*g)->GetGeometry().DistanceTo(loc);
211  if (dist.second < last.second) {
212  last = dist;
213  nearest_geom = g;
214  }
215  }
216 
217  for (auto g = geom_info_list.begin();
218  g != geom_info_list.end() && g != nearest_geom;
219  ++g) {
220  DEBUG_ASSERT(*g != nullptr);
221  last.first += (*g)->GetGeometry().GetLength();
222  }
223 
224  return last;
225  }

227  const std::pair<const Lane *, double> Road::GetNearestLane(
228  const double s,
229  const geom::Location &loc,
230  uint32_t lane_type) const {
231  using namespace carla::road::element;
232  std::map<LaneId, const Lane *> lanes(GetLanesAt(s));
233  // negative right lanes
234  auto right_lanes = MakeListView(
235  std::make_reverse_iterator(lanes.lower_bound(0)), lanes.rend());
236  // positive left lanes
237  auto left_lanes = MakeListView(
238  lanes.lower_bound(1), lanes.end());
239 
240  const DirectedPoint dp_lane_zero = GetDirectedPointIn(s);
241  std::pair<const Lane *, double> result =
242  std::make_pair(nullptr, std::numeric_limits<double>::max());
243 
244  DirectedPoint current_dp = dp_lane_zero;
245  for (const auto &lane : right_lanes) {
246  const auto lane_width_info = lane.second->GetInfo<RoadInfoLaneWidth>(s);
247  const auto half_width = static_cast<float>(lane_width_info->GetPolynomial().Evaluate(s)) * 0.5f;
248 
249  current_dp.ApplyLateralOffset(half_width);
250  const auto current_dist = geom::Math::Distance(current_dp.location, loc);
251 
252  // if the current_dp is near to loc, we are in the right way
253  if (current_dist <= result.second) {
254  // only consider the lanes that match the type flag for result
255  // candidates
256  if ((static_cast<uint32_t>(lane.second->GetType()) & lane_type) > 0) {
257  result.first = &(*lane.second);
258  result.second = current_dist;
259  }
260  } else {
261  // elsewhere, we are be moving away
262  break;
263  }
264  current_dp.ApplyLateralOffset(half_width);
265  }
266 
267  current_dp = dp_lane_zero;
268  for (const auto &lane : left_lanes) {
269  const auto lane_width_info = lane.second->GetInfo<RoadInfoLaneWidth>(s);
270  const auto half_width = -static_cast<float>(lane_width_info->GetPolynomial().Evaluate(s)) * 0.5f;
271 
272  current_dp.ApplyLateralOffset(half_width);
273  const auto current_dist = geom::Math::Distance(current_dp.location, loc);
274 
275  // if the current_dp is near to loc, we are in the right way
276  if (current_dist <= result.second) {
277  // only consider the lanes that match the type flag for result
278  // candidates
279  if ((static_cast<uint32_t>(lane.second->GetType()) & lane_type) > 0) {
280  result.first = &(*lane.second);
281  result.second = current_dist;
282  }
283  } else {
284  // elsewhere, we are be moving away
285  break;
286  }
287  current_dp.ApplyLateralOffset(half_width);
288  }
289 
290  return result;
291  }
