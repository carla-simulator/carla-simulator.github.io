60  void MotionPlannerStage::Action() {
61 
62  // Selecting an output frame.
63  const auto current_control_frame = frame_selector ? control_frame_a : control_frame_b;
64 
65  // Looping over all vehicles.
66  for (uint i = 0u; i < number_of_vehicles; ++i) {
67 
68  const LocalizationToPlannerData &localization_data = localization_frame->at(i);
69  const Actor actor = localization_data.actor;
70  const float current_deviation = localization_data.deviation;
71  const float current_distance = localization_data.distance;
72 
73  const ActorId actor_id = actor->GetId();
74 
75  const auto vehicle = boost::static_pointer_cast<cc::Vehicle>(actor);
76  const float current_velocity = vehicle->GetVelocity().Length();
77 
78  const auto current_time = chr::system_clock::now();
79 
80  if (pid_state_map.find(actor_id) == pid_state_map.end()) {
81  const auto initial_state = StateEntry{0.0f, 0.0f, 0.0f, chr::system_clock::now(), 0.0f, 0.0f, 0.0f};
82  pid_state_map.insert({actor_id, initial_state});
83  }
84 
85  // Retrieving the previous state.
86  traffic_manager::StateEntry previous_state;
87  previous_state = pid_state_map.at(actor_id);
88 
89  // Increase speed if on highway.
90  const float speed_limit = vehicle->GetSpeedLimit() / 3.6f;
91 
92  const float dynamic_target_velocity = parameters.GetVehicleTargetVelocity(actor) / 3.6f;
93 
94  if (speed_limit > HIGHWAY_SPEED) {
95  longitudinal_parameters = highway_longitudinal_parameters;
96  lateral_parameters = highway_lateral_parameters;
97  } else {
98  longitudinal_parameters = urban_longitudinal_parameters;
99  lateral_parameters = urban_lateral_parameters;
100  }
101 
102  // State update for vehicle.
103  StateEntry current_state = controller.StateUpdate(
104  previous_state,
105  current_velocity,
106  dynamic_target_velocity,
107  current_deviation,
108  current_distance,
109  current_time);
110 
111  // Controller actuation.
112  ActuationSignal actuation_signal = controller.RunStep(
113  current_state,
114  previous_state,
115  longitudinal_parameters,
116  lateral_parameters);
117 
118  // In case of collision or traffic light
119  if ((collision_frame != nullptr && traffic_light_frame != nullptr) &&
120  ((collision_messenger_state != 0 && collision_frame->at(i).hazard) ||
121  (traffic_light_messenger_state != 0 &&
122  traffic_light_frame->at(i).traffic_light_hazard))) {
123 
124  current_state.deviation_integral = 0.0f;
125  current_state.velocity_integral = 0.0f;
126  actuation_signal.throttle = 0.0f;
127  actuation_signal.brake = 1.0f;
128  }
129 
130  // Updating PID state.
131  StateEntry &state = pid_state_map.at(actor_id);
132  state = current_state;
133 
134  // Constructing the actuation signal.
135 
136  PlannerToControlData &message = current_control_frame->at(i);
137  message.actor_id = actor_id;
138  message.throttle = actuation_signal.throttle;
139  message.brake = actuation_signal.brake;
140  message.steer = actuation_signal.steer;
141  }
142  }

144  void MotionPlannerStage::DataReceiver() {
145 
146  const auto localization_packet = localization_messenger->ReceiveData(localization_messenger_state);
147  localization_frame = localization_packet.data;
148  localization_messenger_state = localization_packet.id;
149 
150  // Block on receive call only if new data is available on the messenger.
151  const int collision_messenger_current_state = collision_messenger->GetState();
152  if (collision_messenger_current_state != collision_messenger_state) {
153  const auto collision_packet = collision_messenger->ReceiveData(collision_messenger_state);
154  collision_frame = collision_packet.data;
155  collision_messenger_state = collision_packet.id;
156  }
157 
158  // Block on receive call only if new data is available on the messenger.
159  const int traffic_light_messenger_current_state = traffic_light_messenger->GetState();
160  if (traffic_light_messenger_current_state != traffic_light_messenger_state) {
161  const auto traffic_light_packet = traffic_light_messenger->ReceiveData(traffic_light_messenger_state);
162  traffic_light_frame = traffic_light_packet.data;
163  traffic_light_messenger_state = traffic_light_packet.id;
164  }
165 
166  // Allocating new containers for the changed number of registered vehicles.
167  if (localization_frame != nullptr &&
168  number_of_vehicles != (*localization_frame.get()).size()) {
169 
170  number_of_vehicles = static_cast<uint>((*localization_frame.get()).size());
171  // Allocate output frames.
172  control_frame_a = std::make_shared<PlannerToControlFrame>(number_of_vehicles);
173  control_frame_b = std::make_shared<PlannerToControlFrame>(number_of_vehicles);
174  }
175  }
