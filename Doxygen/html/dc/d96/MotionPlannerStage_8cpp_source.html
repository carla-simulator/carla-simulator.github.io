53  void MotionPlannerStage::Action() {
54 
55  // Selecting an output frame.
56  const auto current_control_frame = frame_selector ? control_frame_a : control_frame_b;
57 
58  // Looping over all vehicles.
59  for (uint64_t i = 0u;
60  i < number_of_vehicles &&
61  localization_frame != nullptr &&
62  collision_frame != nullptr &&
63  traffic_light_frame != nullptr;
64  ++i) {
65 
66  const LocalizationToPlannerData &localization_data = localization_frame->at(i);
67  if (!localization_data.actor->IsAlive()) {
68  continue;
69  }
70 
71  const Actor actor = localization_data.actor;
72  const float current_deviation = localization_data.deviation;
73  const float current_distance = localization_data.distance;
74 
75  const ActorId actor_id = actor->GetId();
76 
77  const auto vehicle = boost::static_pointer_cast<cc::Vehicle>(actor);
78  const float current_velocity = vehicle->GetVelocity().Length();
79 
80  const auto current_time = chr::system_clock::now();
81 
82  if (pid_state_map.find(actor_id) == pid_state_map.end()) {
83  const auto initial_state = StateEntry{0.0f, 0.0f, 0.0f, chr::system_clock::now(), 0.0f, 0.0f, 0.0f};
84  pid_state_map.insert({actor_id, initial_state});
85  }
86 
87  // Retrieving the previous state.
88  traffic_manager::StateEntry previous_state;
89  previous_state = pid_state_map.at(actor_id);
90 
91  // Change PID parameters if on highway.
92 
93  const float dynamic_target_velocity = parameters.GetVehicleTargetVelocity(actor) / 3.6f;
94 
95  if (current_velocity > HIGHWAY_SPEED) {
96  longitudinal_parameters = highway_longitudinal_parameters;
97  lateral_parameters = highway_lateral_parameters;
98  } else {
99  longitudinal_parameters = urban_longitudinal_parameters;
100  lateral_parameters = urban_lateral_parameters;
101  }
102 
103  // State update for vehicle.
104  StateEntry current_state = controller.StateUpdate(
105  previous_state,
106  current_velocity,
107  dynamic_target_velocity,
108  current_deviation,
109  current_distance,
110  current_time);
111 
112  // Controller actuation.
113  ActuationSignal actuation_signal = controller.RunStep(
114  current_state,
115  previous_state,
116  longitudinal_parameters,
117  lateral_parameters);
118 
119  // In case of collision or traffic light
120  if ((collision_frame != nullptr && collision_frame->at(i).hazard) ||
121  (traffic_light_frame != nullptr && traffic_light_frame->at(i).traffic_light_hazard)) {
122 
123  current_state.deviation_integral = 0.0f;
124  current_state.velocity_integral = 0.0f;
125  actuation_signal.throttle = 0.0f;
126  actuation_signal.brake = 1.0f;
127  }
128 
129  // Updating PID state.
130  StateEntry &state = pid_state_map.at(actor_id);
131  state = current_state;
132 
133  // Constructing the actuation signal.
134  PlannerToControlData &message = current_control_frame->at(i);
135  message.actor = actor;
136  message.throttle = actuation_signal.throttle;
137  message.brake = actuation_signal.brake;
138  message.steer = actuation_signal.steer;
139  }
140  }

142  void MotionPlannerStage::DataReceiver() {
143 
144  localization_frame = localization_messenger->Peek();
145  collision_frame = collision_messenger->Peek();
146  traffic_light_frame = traffic_light_messenger->Peek();
147 
148  // Allocating new containers for the changed number of registered vehicles.
149  if (localization_frame != nullptr &&
150  number_of_vehicles != (*localization_frame.get()).size()) {
151 
152  number_of_vehicles = static_cast<uint64_t>((*localization_frame.get()).size());
153  // Allocate output frames.
154  control_frame_a = std::make_shared<PlannerToControlFrame>(number_of_vehicles);
155  control_frame_b = std::make_shared<PlannerToControlFrame>(number_of_vehicles);
156  }
157  }

159  void MotionPlannerStage::DataSender() {
