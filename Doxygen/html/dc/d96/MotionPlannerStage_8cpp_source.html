60  void MotionPlannerStage::Action() {
61 
62  // Selecting an output frame.
63  const auto current_control_frame = frame_selector ? control_frame_a : control_frame_b;
64 
65  // Looping over all vehicles.
66  for (uint64_t i = 0u;
67  i < number_of_vehicles &&
68  localization_frame != nullptr &&
69  collision_frame != nullptr &&
70  traffic_light_frame != nullptr;
71  ++i) {
72 
73  const LocalizationToPlannerData &localization_data = localization_frame->at(i);
74  const CollisionToPlannerData& collision_data = collision_frame->at(i);
75 
76  if (!localization_data.actor->IsAlive()) {
77  continue;
78  }
79 
80  const Actor actor = localization_data.actor;
81  const float current_deviation = localization_data.deviation;
82  const float current_distance = localization_data.distance;
83  const cg::Vector3D current_velocity_vector = localization_data.velocity;
84  const float current_velocity = current_velocity_vector.Length();
85 
86  const ActorId actor_id = actor->GetId();
87  const cg::Vector3D ego_heading = actor->GetTransform().GetForwardVector();
88  const auto vehicle = boost::static_pointer_cast<cc::Vehicle>(actor);
89 
90  const auto current_time = chr::system_clock::now();
91 
92  // If previous state for vehicle not found, initialize state entry.
93  if (pid_state_map.find(actor_id) == pid_state_map.end()) {
94  const auto initial_state = StateEntry{0.0f, 0.0f, 0.0f, chr::system_clock::now(), 0.0f, 0.0f, 0.0f};
95  pid_state_map.insert({actor_id, initial_state});
96  }
97 
98  // Retrieving the previous state.
99  traffic_manager::StateEntry previous_state;
100  previous_state = pid_state_map.at(actor_id);
101 
102  // Change PID parameters if on highway.
103  if (current_velocity > HIGHWAY_SPEED) {
104  longitudinal_parameters = highway_longitudinal_parameters;
105  lateral_parameters = highway_lateral_parameters;
106  } else {
107  longitudinal_parameters = urban_longitudinal_parameters;
108  lateral_parameters = urban_lateral_parameters;
109  }
110 
111  // Target velocity for vehicle.
112  float max_target_velocity = parameters.GetVehicleTargetVelocity(actor) / 3.6f;
113  float dynamic_target_velocity = max_target_velocity;
114  //////////////////////// Collision related data handling ///////////////////////////
115  bool collision_emergency_stop = false;
116  if (collision_data.hazard) {
117  cg::Vector3D other_vehicle_velocity = collision_data.other_vehicle_velocity;
118  float ego_relative_velocity = (current_velocity_vector - other_vehicle_velocity).Length();
119  float other_velocity_along_heading = cg::Math::Dot(other_vehicle_velocity, ego_heading);
120 
121  // Consider collision avoidance decisions only if there is positive relative velocity
122  // of the ego vehicle (meaning, ego vehicle is closing the gap to the lead vehicle).
123  if (ego_relative_velocity > EPSILON_VELOCITY) {
124  // If other vehicle is approaching lead vehicle and lead vehicle is further
125  // than follow_lead_distance 0 kmph -> 5m, 100 kmph -> 10m.
126  float follow_lead_distance = ego_relative_velocity * FOLLOW_DISTANCE_RATE + MIN_FOLLOW_LEAD_DISTANCE;
127  if (collision_data.distance_to_other_vehicle > follow_lead_distance) {
128  // Then reduce the gap between the vehicles till FOLLOW_LEAD_DISTANCE
129  // by maintaining a relative speed of RELATIVE_APPROACH_SPEED
130  dynamic_target_velocity = other_velocity_along_heading + RELATIVE_APPROACH_SPEED;
131  }
132  // If vehicle is approaching a lead vehicle and the lead vehicle is further
133  // than CRITICAL_BRAKING_MARGIN but closer than FOLLOW_LEAD_DISTANCE.
134  else if (collision_data.distance_to_other_vehicle > CRITICAL_BRAKING_MARGIN) {
135  // Then follow the lead vehicle by acquiring it's speed along current heading.
136  dynamic_target_velocity = std::max(other_velocity_along_heading, RELATIVE_APPROACH_SPEED);
137  } else {
138  // If lead vehicle closer than CRITICAL_BRAKING_MARGIN, initiate emergency stop.
139  collision_emergency_stop = true;
140  }
141  }
142  if (collision_data.distance_to_other_vehicle < CRITICAL_BRAKING_MARGIN) {
143  collision_emergency_stop = true;
144  }
145  }
146  ///////////////////////////////////////////////////////////////////////////////////
147 
148  // Clip dynamic target velocity to maximum allowed speed for the vehicle.
149  dynamic_target_velocity = std::min(max_target_velocity, dynamic_target_velocity);
150 
151  bool emergency_stop = false;
152  // In case of collision or traffic light hazard.
153  if (traffic_light_frame->at(i).traffic_light_hazard
154  || collision_emergency_stop) {
155  emergency_stop = true;
156  }
157 
158  // Message items to be sent to batch control stage.
159  ActuationSignal actuation_signal {0.0f, 0.0f, 0.0f};
160  bool physics_enabled = true;
161  cg::Transform teleportation_transform;
162 
163  // If physics is enabled for the vehicle, use PID controller.
164  StateEntry current_state;
165  if (localization_data.physics_enabled) {
166 
167  // State update for vehicle.
168  current_state = controller.StateUpdate(previous_state, current_velocity,
169  dynamic_target_velocity, current_deviation,
170  current_distance, current_time);
171 
172  // Controller actuation.
173  actuation_signal = controller.RunStep(current_state, previous_state,
174  longitudinal_parameters, lateral_parameters);
175 
176  if (emergency_stop) {
177 
178  current_state.deviation_integral = 0.0f;
179  current_state.velocity_integral = 0.0f;
180  actuation_signal.throttle = 0.0f;
181  actuation_signal.brake = 1.0f;
182  }
183 
184  }
185  // For physics-less vehicles, determine position and orientation for teleportation.
186  else if (hybrid_physics_mode) {
187 
188  physics_enabled = false;
189 
190  // Flushing controller state for vehicle.
191  current_state = {0.0f, 0.0f, 0.0f,
192  chr::system_clock::now(),
193  0.0f, 0.0f, 0.0f};
194 
195  // Add entry to teleportation duration clock table if not present.
196  if (teleportation_instance.find(actor_id) == teleportation_instance.end()) {
197  teleportation_instance.insert({actor_id, chr::system_clock::now()});
198  }
199 
200  // Measuring time elapsed since last teleportation for the vehicle.
201  chr::duration<float> elapsed_time = current_time - teleportation_instance.at(actor_id);
202 
203  // Find a location ahead of the vehicle for teleportation to achieve intended velocity.
204  if (!emergency_stop && (parameters.GetSynchronousMode() || elapsed_time.count() > HYBRID_MODE_DT)) {
205 
206  // Target displacement magnitude to achieve target velocity.
207  const float target_displacement = dynamic_target_velocity * HYBRID_MODE_DT;
208  const float target_displacement_square = std::pow(target_displacement, 2.0f);
209 
210  SimpleWaypointPtr target_interval_begin = nullptr;
211  SimpleWaypointPtr target_interval_end = nullptr;
212  bool teleportation_interval_found = false;
213  cg::Location vehicle_location = actor->GetLocation();
214 
215  // Find the interval containing position to achieve target displacement.
216  for (uint32_t j = 0u;
217  j+1 < localization_data.position_window.size() && !teleportation_interval_found;
218  ++j) {
219 
220  target_interval_begin = localization_data.position_window.at(j);
221  target_interval_end = localization_data.position_window.at(j+1);
222 
223  cg::Vector3D relative_position = target_interval_begin->GetLocation() - vehicle_location;
224  if (cg::Math::Dot(relative_position, ego_heading) > 0.0f
225  && ((target_interval_begin->DistanceSquared(vehicle_location) > target_displacement_square)
226  || (target_interval_begin->DistanceSquared(vehicle_location) < target_displacement_square
227  && target_interval_end->DistanceSquared(vehicle_location) > target_displacement_square))) {
228  teleportation_interval_found = true;
229  }
230  }
231 
232  if (target_interval_begin != nullptr && target_interval_end != nullptr) {
233 
234  // Construct target transform to accurately achieve desired velocity.
235  float missing_displacement = 0.0f;
236  const float base_displacement = target_interval_begin->Distance(vehicle_location);
237  if (base_displacement < target_displacement) {
238  missing_displacement = target_displacement - base_displacement;
239  }
240  cg::Transform target_base_transform = target_interval_begin->GetTransform();
241  cg::Location target_base_location = target_base_transform.location;
242  cg::Vector3D target_heading = target_base_transform.GetForwardVector();
243  cg::Location teleportation_location = target_base_location
244  + cg::Location(target_heading * missing_displacement);
245  teleportation_transform = cg::Transform(teleportation_location, target_base_transform.rotation);
246 
247  } else {
248 
249  teleportation_transform = actor->GetTransform();
250 
251  }
252 
253  }
254  // In case of an emergency stop, stay in the same location.
255  // Also, teleport only once every dt in asynchronous mode.
256  else {
257 
258  teleportation_transform = actor->GetTransform();
259 
260  }
261 
262  }
263 
264  // Updating PID state.
265  StateEntry &state = pid_state_map.at(actor_id);
266  state = current_state;
267 
268  // Constructing the actuation signal.
269  PlannerToControlData &message = current_control_frame->at(i);
270  message.actor = actor;
271  message.throttle = actuation_signal.throttle;
272  message.brake = actuation_signal.brake;
273  message.steer = actuation_signal.steer;
274  message.physics_enabled = physics_enabled;
275  message.transform = teleportation_transform;
276  }
277  }

279  void MotionPlannerStage::DataReceiver() {
280 
281  localization_frame = localization_messenger->Peek();
282  collision_frame = collision_messenger->Peek();
283  traffic_light_frame = traffic_light_messenger->Peek();
284 
285  // Allocating new containers for the changed number of registered vehicles.
286  if (localization_frame != nullptr &&
287  number_of_vehicles != (*localization_frame.get()).size()) {
288 
289  number_of_vehicles = static_cast<uint64_t>((*localization_frame.get()).size());
290  // Allocate output frames.
291  control_frame_a = std::make_shared<PlannerToControlFrame>(number_of_vehicles);
292  control_frame_b = std::make_shared<PlannerToControlFrame>(number_of_vehicles);
293  }
294 
295  hybrid_physics_mode = parameters.GetHybridPhysicsMode();
296  }
