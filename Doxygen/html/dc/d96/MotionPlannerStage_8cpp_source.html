53  void MotionPlannerStage::Action() {
54 
55  // Selecting an output frame.
56  const auto current_control_frame = frame_selector ? control_frame_a : control_frame_b;
57 
58  // Looping over all vehicles.
59  for (uint64_t i = 0u;
60  i < number_of_vehicles &&
61  localization_frame != nullptr &&
62  collision_frame != nullptr &&
63  traffic_light_frame != nullptr;
64  ++i) {
65 
66  const LocalizationToPlannerData &localization_data = localization_frame->at(i);
67  const Actor actor = localization_data.actor;
68  const float current_deviation = localization_data.deviation;
69  const float current_distance = localization_data.distance;
70 
71  const ActorId actor_id = actor->GetId();
72 
73  const auto vehicle = boost::static_pointer_cast<cc::Vehicle>(actor);
74  const float current_velocity = vehicle->GetVelocity().Length();
75 
76  const auto current_time = chr::system_clock::now();
77 
78  if (pid_state_map.find(actor_id) == pid_state_map.end()) {
79  const auto initial_state = StateEntry{0.0f, 0.0f, 0.0f, chr::system_clock::now(), 0.0f, 0.0f, 0.0f};
80  pid_state_map.insert({actor_id, initial_state});
81  }
82 
83  // Retrieving the previous state.
84  traffic_manager::StateEntry previous_state;
85  previous_state = pid_state_map.at(actor_id);
86 
87  // Change PID parameters if on highway.
88 
89  const float dynamic_target_velocity = parameters.GetVehicleTargetVelocity(actor) / 3.6f;
90 
91  if (current_velocity > HIGHWAY_SPEED) {
92  longitudinal_parameters = highway_longitudinal_parameters;
93  lateral_parameters = highway_lateral_parameters;
94  } else {
95  longitudinal_parameters = urban_longitudinal_parameters;
96  lateral_parameters = urban_lateral_parameters;
97  }
98 
99  // State update for vehicle.
100  StateEntry current_state = controller.StateUpdate(
101  previous_state,
102  current_velocity,
103  dynamic_target_velocity,
104  current_deviation,
105  current_distance,
106  current_time);
107 
108  // Controller actuation.
109  ActuationSignal actuation_signal = controller.RunStep(
110  current_state,
111  previous_state,
112  longitudinal_parameters,
113  lateral_parameters);
114 
115  // In case of collision or traffic light
116  if ((collision_frame != nullptr && collision_frame->at(i).hazard) ||
117  (traffic_light_frame != nullptr && traffic_light_frame->at(i).traffic_light_hazard)) {
118 
119  current_state.deviation_integral = 0.0f;
120  current_state.velocity_integral = 0.0f;
121  actuation_signal.throttle = 0.0f;
122  actuation_signal.brake = 1.0f;
123  }
124 
125  // Updating PID state.
126  StateEntry &state = pid_state_map.at(actor_id);
127  state = current_state;
128 
129  // Constructing the actuation signal.
130 
131  PlannerToControlData &message = current_control_frame->at(i);
132  message.actor_id = actor_id;
133  message.throttle = actuation_signal.throttle;
134  message.brake = actuation_signal.brake;
135  message.steer = actuation_signal.steer;
136  }
137  }

139  void MotionPlannerStage::DataReceiver() {
140 
141  localization_frame = localization_messenger->Peek();
142  collision_frame = collision_messenger->Peek();
143  traffic_light_frame = traffic_light_messenger->Peek();
144 
145  // Allocating new containers for the changed number of registered vehicles.
146  if (localization_frame != nullptr &&
147  number_of_vehicles != (*localization_frame.get()).size()) {
148 
149  number_of_vehicles = static_cast<uint64_t>((*localization_frame.get()).size());
150  // Allocate output frames.
151  control_frame_a = std::make_shared<PlannerToControlFrame>(number_of_vehicles);
152  control_frame_b = std::make_shared<PlannerToControlFrame>(number_of_vehicles);
153  }
154  }
