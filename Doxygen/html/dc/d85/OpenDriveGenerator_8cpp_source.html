64 void AOpenDriveGenerator::GenerateRoadMesh()
65 {
66  if (!IsOpenDriveValid())
67  {
68  UE_LOG(LogCarla, Error, TEXT("The OpenDrive has not been loaded"));
69  return;
70  }
71 
72  const auto MeshData = CarlaMap->GenerateMesh(2);
73 
74  // Build the mesh
75  TArray<FVector> Vertices;
76  for (const auto vertex : MeshData.GetVertices())
77  {
78  // From meters to centimeters
79  Vertices.Add(FVector(vertex.x, vertex.y, vertex.z) * 1e2f);
80  }
81 
82  const auto Indexes = MeshData.GetIndexes();
83  TArray<int32> Triangles;
84  TArray<FTriIndices> TriIndices;
85  FTriIndices Triangle;
86  for (auto i = 0u; i < Indexes.size(); i += 3)
87  {
88  // "-1" since mesh indexes in Unreal starts from index 0.
89  Triangles.Add(Indexes[i] - 1);
90  // Since Unreal's coords are left handed, invert the last 2 indices.
91  Triangles.Add(Indexes[i + 2] - 1);
92  Triangles.Add(Indexes[i + 1] - 1);
93 
94  Triangle.v0 = Indexes[i] - 1;
95  Triangle.v1 = Indexes[i + 2] - 1;
96  Triangle.v2 = Indexes[i + 1] - 1;
97  TriIndices.Add(Triangle);
98  }
99 
100  RoadMesh->CreateMeshSection_LinearColor(
101  0,
102  Vertices,
103  Triangles,
104  TArray<FVector>(),
105  TArray<FVector2D>(),
106  TArray<FLinearColor>(),
107  TArray<FProcMeshTangent>(),
108  true);
109 
110  // Build collision data
111  FTriMeshCollisionData CollisitonData;
112  CollisitonData.bDeformableMesh = false;
113  CollisitonData.bDisableActiveEdgePrecompute = false;
114  CollisitonData.bFastCook = false;
115  CollisitonData.bFlipNormals = false;
116  CollisitonData.Indices = TriIndices;
117  CollisitonData.Vertices = Vertices;
118 
119  RoadMesh->ContainsPhysicsTriMeshData(true);
120  bool Success = RoadMesh->GetPhysicsTriMeshData(&CollisitonData, true);
121  if (!Success)
122  {
123  UE_LOG(LogCarla, Error, TEXT("The road collision mesh could not be generated!"));
124  }
125 }
