35 void ACarlaWheeledVehicle::BeginPlay()
36 {
37  Super::BeginPlay();
38 
39  float FrictionScale = 3.5f;
40 
41  UWheeledVehicleMovementComponent4W *Vehicle4W = Cast<UWheeledVehicleMovementComponent4W>(
42  GetVehicleMovementComponent());
43  check(Vehicle4W != nullptr);
44 
45  // Setup Tire Configs with default value. This is needed to avoid getting
46  // friction values of previously created TireConfigs for the same vehicle
47  // blueprint.
48  TArray<float> OriginalFrictions;
49  OriginalFrictions.Init(FrictionScale, Vehicle4W->Wheels.Num());
50  SetWheelsFrictionScale(OriginalFrictions);
51 
52  // Check if it overlaps with a Friction trigger, if so, update the friction
53  // scale.
54  TArray<AActor *> OverlapActors;
55  GetOverlappingActors(OverlapActors, AFrictionTrigger::StaticClass());
56  for (const auto &Actor : OverlapActors)
57  {
58  AFrictionTrigger *FrictionTrigger = Cast<AFrictionTrigger>(Actor);
59  if (FrictionTrigger)
60  {
61  FrictionScale = FrictionTrigger->Friction;
62  }
63  }
64 
65  // Set the friction scale to Wheel CDO and update wheel setups
66  TArray<FWheelSetup> NewWheelSetups = Vehicle4W->WheelSetups;
67 
68  for (const auto &WheelSetup : NewWheelSetups)
69  {
70  UVehicleWheel *Wheel = WheelSetup.WheelClass.GetDefaultObject();
71  check(Wheel != nullptr);
72 
73  // Assigning new tire config
74  Wheel->TireConfig = NewObject<UTireConfig>();
75 
76  // Setting a new value to friction
77  Wheel->TireConfig->SetFrictionScale(FrictionScale);
78  }
79 
80  Vehicle4W->WheelSetups = NewWheelSetups;
81 }

125 void ACarlaWheeledVehicle::FlushVehicleControl()
126 {
127  auto *MovementComponent = GetVehicleMovementComponent();
128  MovementComponent->SetThrottleInput(InputControl.Control.Throttle);
129  MovementComponent->SetSteeringInput(InputControl.Control.Steer);
130  MovementComponent->SetBrakeInput(InputControl.Control.Brake);
131  MovementComponent->SetHandbrakeInput(InputControl.Control.bHandBrake);
132  if (LastAppliedControl.bReverse != InputControl.Control.bReverse)
133  {
134  MovementComponent->SetUseAutoGears(!InputControl.Control.bReverse);
135  MovementComponent->SetTargetGear(InputControl.Control.bReverse ? -1 : 1, true);
136  }
137  else
138  {
139  MovementComponent->SetUseAutoGears(!InputControl.Control.bManualGearShift);
140  if (InputControl.Control.bManualGearShift)
141  {
142  MovementComponent->SetTargetGear(InputControl.Control.Gear, true);
143  }
144  }
145  InputControl.Control.Gear = MovementComponent->GetCurrentGear();
146  InputControl.Control.bReverse = InputControl.Control.Gear < 0;
147  LastAppliedControl = InputControl.Control;
148  InputControl.Priority = EVehicleInputPriority::INVALID;
149 }

213 FVehiclePhysicsControl ACarlaWheeledVehicle::GetVehiclePhysicsControl()
214 {
215  UWheeledVehicleMovementComponent4W *Vehicle4W = Cast<UWheeledVehicleMovementComponent4W>(
216  GetVehicleMovement());
217  check(Vehicle4W != nullptr);
218 
219  FVehiclePhysicsControl PhysicsControl;
220 
221  // Engine Setup
222  PhysicsControl.TorqueCurve = Vehicle4W->EngineSetup.TorqueCurve.EditorCurveData;
223  PhysicsControl.MaxRPM = Vehicle4W->EngineSetup.MaxRPM;
224  PhysicsControl.MOI = Vehicle4W->EngineSetup.MOI;
225  PhysicsControl.DampingRateFullThrottle = Vehicle4W->EngineSetup.DampingRateFullThrottle;
226  PhysicsControl.DampingRateZeroThrottleClutchEngaged =
227  Vehicle4W->EngineSetup.DampingRateZeroThrottleClutchEngaged;
228  PhysicsControl.DampingRateZeroThrottleClutchDisengaged =
229  Vehicle4W->EngineSetup.DampingRateZeroThrottleClutchDisengaged;
230 
231  // Transmission Setup
232  PhysicsControl.bUseGearAutoBox = Vehicle4W->TransmissionSetup.bUseGearAutoBox;
233  PhysicsControl.GearSwitchTime = Vehicle4W->TransmissionSetup.GearSwitchTime;
234  PhysicsControl.ClutchStrength = Vehicle4W->TransmissionSetup.ClutchStrength;
235  PhysicsControl.FinalRatio = Vehicle4W->TransmissionSetup.FinalRatio;
236 
237  TArray<FGearPhysicsControl> ForwardGears;
238 
239  for (const auto &Gear : Vehicle4W->TransmissionSetup.ForwardGears)
240  {
241  FGearPhysicsControl GearPhysicsControl;
242 
243  GearPhysicsControl.Ratio = Gear.Ratio;
244  GearPhysicsControl.UpRatio = Gear.UpRatio;
245  GearPhysicsControl.DownRatio = Gear.DownRatio;
246 
247  ForwardGears.Add(GearPhysicsControl);
248  }
249 
250  PhysicsControl.ForwardGears = ForwardGears;
251 
252  // Vehicle Setup
253  PhysicsControl.Mass = Vehicle4W->Mass;
254  PhysicsControl.DragCoefficient = Vehicle4W->DragCoefficient;
255 
256  // Center of mass offset (Center of mass is always zero vector in local
257  // position)
258  UPrimitiveComponent *UpdatedPrimitive = Cast<UPrimitiveComponent>(Vehicle4W->UpdatedComponent);
259  check(UpdatedPrimitive != nullptr);
260 
261  PhysicsControl.CenterOfMass = UpdatedPrimitive->BodyInstance.COMNudge;
262 
263  // Transmission Setup
264  PhysicsControl.SteeringCurve = Vehicle4W->SteeringCurve.EditorCurveData;
265 
266  // Wheels Setup
267  TArray<FWheelPhysicsControl> Wheels;
268 
269  for (int32 i = 0; i < Vehicle4W->WheelSetups.Num(); ++i)
270  {
271  FWheelPhysicsControl PhysicsWheel;
272 
273  PxVehicleWheelData PWheelData = Vehicle4W->PVehicle->mWheelsSimData.getWheelData(i);
274 
275  PhysicsWheel.TireFriction = Vehicle4W->Wheels[i]->TireConfig->GetFrictionScale();
276  PhysicsWheel.DampingRate = Cm2ToM2(PWheelData.mDampingRate);
277  PhysicsWheel.MaxSteerAngle = FMath::RadiansToDegrees(PWheelData.mMaxSteer);
278  PhysicsWheel.Radius = PWheelData.mRadius;
279  PhysicsWheel.MaxBrakeTorque = Cm2ToM2(PWheelData.mMaxBrakeTorque);
280  PhysicsWheel.MaxHandBrakeTorque = Cm2ToM2(PWheelData.mMaxHandBrakeTorque);
281 
282  PhysicsWheel.Position = Vehicle4W->Wheels[i]->Location;
283 
284  Wheels.Add(PhysicsWheel);
285  }
286 
287  PhysicsControl.Wheels = Wheels;
288 
289  return PhysicsControl;
290 }

297 void ACarlaWheeledVehicle::ApplyVehiclePhysicsControl(const FVehiclePhysicsControl &PhysicsControl)
298 {
299  UWheeledVehicleMovementComponent4W *Vehicle4W = Cast<UWheeledVehicleMovementComponent4W>(
300  GetVehicleMovement());
301  check(Vehicle4W != nullptr);
302 
303  // Engine Setup
304  Vehicle4W->EngineSetup.TorqueCurve.EditorCurveData = PhysicsControl.TorqueCurve;
305  Vehicle4W->EngineSetup.MaxRPM = PhysicsControl.MaxRPM;
306 
307  Vehicle4W->EngineSetup.MOI = PhysicsControl.MOI;
308 
309  Vehicle4W->EngineSetup.DampingRateFullThrottle = PhysicsControl.DampingRateFullThrottle;
310  Vehicle4W->EngineSetup.DampingRateZeroThrottleClutchEngaged =
311  PhysicsControl.DampingRateZeroThrottleClutchEngaged;
312  Vehicle4W->EngineSetup.DampingRateZeroThrottleClutchDisengaged =
313  PhysicsControl.DampingRateZeroThrottleClutchDisengaged;
314 
315  // Transmission Setup
316  Vehicle4W->TransmissionSetup.bUseGearAutoBox = PhysicsControl.bUseGearAutoBox;
317  Vehicle4W->TransmissionSetup.GearSwitchTime = PhysicsControl.GearSwitchTime;
318  Vehicle4W->TransmissionSetup.ClutchStrength = PhysicsControl.ClutchStrength;
319  Vehicle4W->TransmissionSetup.FinalRatio = PhysicsControl.FinalRatio;
320 
321  TArray<FVehicleGearData> ForwardGears;
322 
323  for (const auto &Gear : PhysicsControl.ForwardGears)
324  {
325  FVehicleGearData GearData;
326 
327  GearData.Ratio = Gear.Ratio;
328  GearData.UpRatio = Gear.UpRatio;
329  GearData.DownRatio = Gear.DownRatio;
330 
331  ForwardGears.Add(GearData);
332  }
333 
334  Vehicle4W->TransmissionSetup.ForwardGears = ForwardGears;
335 
336 
337 
338  // Vehicle Setup
339  Vehicle4W->Mass = PhysicsControl.Mass;
340  Vehicle4W->DragCoefficient = PhysicsControl.DragCoefficient;
341 
342  // Center of mass
343  UPrimitiveComponent *UpdatedPrimitive = Cast<UPrimitiveComponent>(Vehicle4W->UpdatedComponent);
344  check(UpdatedPrimitive != nullptr);
345 
346  UpdatedPrimitive->BodyInstance.COMNudge = PhysicsControl.CenterOfMass;
347 
348  // Transmission Setup
349  Vehicle4W->SteeringCurve.EditorCurveData = PhysicsControl.SteeringCurve;
350 
351  // Recreate Physics State only for vehicle setup
352  Vehicle4W->RecreatePhysicsState();
353 
354  // Wheels Setup
355  const int PhysicsWheelsNum = PhysicsControl.Wheels.Num();
356  if (PhysicsWheelsNum != 4)
357  {
358  UE_LOG(LogCarla, Error, TEXT("Number of WheelPhysicsControl is not 4."));
359  return;
360  }
361 
362  for (int32 i = 0; i < PhysicsWheelsNum; ++i)
363  {
364  PxVehicleWheelData PWheelData = Vehicle4W->PVehicle->mWheelsSimData.getWheelData(i);
365 
366  PWheelData.mRadius = PhysicsControl.Wheels[i].Radius;
367  PWheelData.mMaxSteer = FMath::DegreesToRadians(PhysicsControl.Wheels[i].MaxSteerAngle);
368  PWheelData.mDampingRate = M2ToCm2(PhysicsControl.Wheels[i].DampingRate);
369  PWheelData.mMaxBrakeTorque = M2ToCm2(PhysicsControl.Wheels[i].MaxBrakeTorque);
370  PWheelData.mMaxHandBrakeTorque = M2ToCm2(PhysicsControl.Wheels[i].MaxHandBrakeTorque);
371 
372  Vehicle4W->PVehicle->mWheelsSimData.setWheelData(i, PWheelData);
373  Vehicle4W->Wheels[i]->TireConfig->SetFrictionScale(PhysicsControl.Wheels[i].TireFriction);
374  }
375 
376 }
