55  Map::TopologyList Map::GetTopology() const {
56  namespace re = carla::road::element;
57  std::unordered_map<re::Waypoint, SharedPtr<Waypoint>> waypoints;
58 
59  auto get_or_make_waypoint = [&](const auto &waypoint) {
60  auto it = waypoints.find(waypoint);
61  if (it == waypoints.end()) {
62  it = waypoints.emplace(
63  waypoint,
64  SharedPtr<Waypoint>(new Waypoint{shared_from_this(), waypoint})).first;
65  }
66  return it->second;
67  };
68 
69  TopologyList result;
70  auto topology = _map.GenerateTopology();
71  result.reserve(topology.size());
72  for (const auto &pair : topology) {
73  result.emplace_back(
74  get_or_make_waypoint(pair.first),
75  get_or_make_waypoint(pair.second));
76  }
77  return result;
78  }
