57  void CollisionStage::Action() {
58  const auto current_planner_frame = frame_selector ? planner_frame_a : planner_frame_b;
59 
60  // Handle vehicles not spawned by TrafficManager.
61  const auto current_time = chr::system_clock::now();
62  const chr::duration<double> diff = current_time - last_world_actors_pass_instance;
63 
64  // Periodically check for actors not spawned by TrafficManager.
65  if (diff.count() > 1.0f) {
66 
67  const auto world_actors = world.GetActors()->Filter("vehicle.*");
68  const auto world_walker = world.GetActors()->Filter("walker.*");
69  // Scanning for vehicles.
70  for (auto actor: *world_actors.get()) {
71  const auto unregistered_id = actor->GetId();
72  if (vehicle_id_to_index.find(unregistered_id) == vehicle_id_to_index.end() &&
73  unregistered_actors.find(unregistered_id) == unregistered_actors.end()) {
74  unregistered_actors.insert({unregistered_id, actor});
75  }
76  }
77  // Scanning for pedestrians.
78  for (auto walker: *world_walker.get()) {
79  const auto unregistered_id = walker->GetId();
80  if (unregistered_actors.find(unregistered_id) == unregistered_actors.end()) {
81  unregistered_actors.insert({unregistered_id, walker});
82  }
83  }
84  // Regularly update unregistered actors.
85  std::vector<ActorId> actor_ids_to_erase;
86  for (auto actor_info: unregistered_actors) {
87  if (actor_info.second->IsAlive()) {
88  vicinity_grid.UpdateGrid(actor_info.second);
89  } else {
90  vicinity_grid.EraseActor(actor_info.first);
91  actor_ids_to_erase.push_back(actor_info.first);
92  }
93  }
94  for (auto actor_id: actor_ids_to_erase) {
95  unregistered_actors.erase(actor_id);
96  }
97 
98  last_world_actors_pass_instance = current_time;
99  }
100 
101  // Looping over registered actors.
102  for (uint64_t i = 0u; i < number_of_vehicles; ++i) {
103 
104  const LocalizationToCollisionData &data = localization_frame->at(i);
105  const Actor ego_actor = data.actor;
106  const ActorId ego_actor_id = ego_actor->GetId();
107 
108  // Retrieve actors around the path of the ego vehicle.
109  std::unordered_set<ActorId> actor_id_list = GetPotentialVehicleObstacles(ego_actor);
110 
111  bool collision_hazard = false;
112 
113  // Generate number between 0 and 100
114  const int r = rand() % 101;
115 
116  // Continue only if random number is lower than our %, default is 0.
117  if (parameters.GetPercentageIgnoreActors(boost::shared_ptr<cc::Actor>(ego_actor)) <= r) {
118  // Check every actor in the vicinity if it poses a collision hazard.
119  for (auto j = actor_id_list.begin(); (j != actor_id_list.end()) && !collision_hazard; ++j) {
120  const ActorId actor_id = *j;
121  try {
122 
123  Actor actor = nullptr;
124  if (vehicle_id_to_index.find(actor_id) != vehicle_id_to_index.end()) {
125  actor = localization_frame->at(vehicle_id_to_index.at(actor_id)).actor;
126  } else if (unregistered_actors.find(actor_id) != unregistered_actors.end()) {
127  actor = unregistered_actors.at(actor_id);
128  }
129 
130  const cg::Location ego_location = ego_actor->GetLocation();
131  const cg::Location other_location = actor->GetLocation();
132 
133  if (actor_id != ego_actor_id &&
134  (cg::Math::DistanceSquared(ego_location, other_location)
135  < std::pow(MAX_COLLISION_RADIUS, 2)) &&
136  (std::abs(ego_location.z - other_location.z) < VERTICAL_OVERLAP_THRESHOLD)) {
137 
138  if (parameters.GetCollisionDetection(ego_actor, actor) &&
139  NegotiateCollision(ego_actor, actor)) {
140 
141  collision_hazard = true;
142  }
143  }
144 
145  } catch (const std::exception &e) {
146  carla::log_warning("Encountered problem while determining collision \n");
147  carla::log_info("Actor might not be alive \n");
148  }
149 
150  }
151  }
152 
153  CollisionToPlannerData &message = current_planner_frame->at(i);
154  message.hazard = collision_hazard;
155 
156  }

196  bool CollisionStage::NegotiateCollision(const Actor &reference_vehicle, const Actor &other_vehicle) const {
197 
198  bool hazard = false;
199 
200  auto& data_packet = localization_frame->at(vehicle_id_to_index.at(reference_vehicle->GetId()));
201  Buffer& waypoint_buffer = data_packet.buffer;
202 
203  auto& other_packet = localization_frame->at(vehicle_id_to_index.at(other_vehicle->GetId()));
204  Buffer& other_buffer = other_packet.buffer;
205 
206  const cg::Location reference_location = reference_vehicle->GetLocation();
207  const cg::Location other_location = other_vehicle->GetLocation();
208 
209  const cg::Vector3D reference_heading = reference_vehicle->GetTransform().GetForwardVector();
210  cg::Vector3D reference_to_other = other_location - reference_location;
211  reference_to_other = reference_to_other.MakeUnitVector();
212 
213  const auto reference_vehicle_ptr = boost::static_pointer_cast<cc::Vehicle>(reference_vehicle);
214  const auto other_vehicle_ptr = boost::static_pointer_cast<cc::Vehicle>(other_vehicle);
215 
216  if (waypoint_buffer.front()->CheckJunction() &&
217  other_buffer.front()->CheckJunction()) {
218 
219  const Polygon reference_geodesic_polygon = GetPolygon(GetGeodesicBoundary(reference_vehicle));
220  const Polygon other_geodesic_polygon = GetPolygon(GetGeodesicBoundary(other_vehicle));
221  const Polygon reference_polygon = GetPolygon(GetBoundary(reference_vehicle));
222  const Polygon other_polygon = GetPolygon(GetBoundary(other_vehicle));
223 
224  const double reference_vehicle_to_other_geodesic = bg::distance(reference_polygon, other_geodesic_polygon);
225  const double other_vehicle_to_reference_geodesic = bg::distance(other_polygon, reference_geodesic_polygon);
226 
227  const auto inter_geodesic_distance = bg::distance(reference_geodesic_polygon, other_geodesic_polygon);
228  const auto inter_bbox_distance = bg::distance(reference_polygon, other_polygon);
229 
230  const cg::Vector3D other_heading = other_vehicle->GetTransform().GetForwardVector();
231  cg::Vector3D other_to_reference = reference_vehicle->GetLocation() - other_vehicle->GetLocation();
232  other_to_reference = other_to_reference.MakeUnitVector();
233 
234  // Whichever vehicle's path is farthest away from the other vehicle gets
235  // priority to move.
236  if (inter_geodesic_distance < 0.1 &&
237  ((
238  inter_bbox_distance > 0.1 &&
239  reference_vehicle_to_other_geodesic > other_vehicle_to_reference_geodesic
240  ) || (
241  inter_bbox_distance < 0.1 &&
242  cg::Math::Dot(reference_heading, reference_to_other) > cg::Math::Dot(other_heading, other_to_reference)
243  )) ) {
244 
245  hazard = true;
246  }
247 
248  } else if (!waypoint_buffer.front()->CheckJunction()) {
249 
250  const float reference_vehicle_length = reference_vehicle_ptr->GetBoundingBox().extent.x;
251  const float other_vehicle_length = other_vehicle_ptr->GetBoundingBox().extent.x;
252  const float vehicle_length_sum = reference_vehicle_length + other_vehicle_length;
253 
254  const float bbox_extension_length = GetBoundingBoxExtention(reference_vehicle);
255 
256  if ((cg::Math::Dot(reference_heading, reference_to_other) > 0.0f) &&
257  (cg::Math::DistanceSquared(reference_location, other_location) <
258  std::pow(bbox_extension_length+vehicle_length_sum, 2))) {
259 
260  hazard = true;
261  }
262  }
263 
264  return hazard;
265  }

281  LocationList CollisionStage::GetGeodesicBoundary(const Actor &actor) const {
282 
283  const LocationList bbox = GetBoundary(actor);
284 
285  if (vehicle_id_to_index.find(actor->GetId()) != vehicle_id_to_index.end()) {
286 
287  const cg::Location vehicle_location = actor->GetLocation();
288 
289  float bbox_extension = GetBoundingBoxExtention(actor);
290 
291  const float specific_distance_margin = parameters.GetDistanceToLeadingVehicle(actor);
292  if (specific_distance_margin > 0.0f) {
293  bbox_extension = std::max(specific_distance_margin, bbox_extension);
294  }
295 
296  const auto &waypoint_buffer = localization_frame->at(vehicle_id_to_index.at(actor->GetId())).buffer;
297 
298  LocationList left_boundary;
299  LocationList right_boundary;
300  const auto vehicle = boost::static_pointer_cast<cc::Vehicle>(actor);
301  const float width = vehicle->GetBoundingBox().extent.y;
302  const float length = vehicle->GetBoundingBox().extent.x;
303 
304  SimpleWaypointPtr boundary_start = waypoint_buffer.front();
305  uint64_t boundary_start_index = 0u;
306  while (boundary_start->DistanceSquared(vehicle_location) < std::pow(length, 2) &&
307  boundary_start_index < waypoint_buffer.size() -1) {
308  boundary_start = waypoint_buffer.at(boundary_start_index);
309  ++boundary_start_index;
310  }
311  SimpleWaypointPtr boundary_end = nullptr;
312  SimpleWaypointPtr current_point = waypoint_buffer.at(boundary_start_index);
313 
314  const auto vehicle_reference = boost::static_pointer_cast<cc::Vehicle>(actor);
315  // At non-signalized junctions, we extend the boundary across the junction
316  // and in all other situations, boundary length is velocity-dependent.
317  bool reached_distance = false;
318  for (uint64_t j = boundary_start_index; !reached_distance && (j < waypoint_buffer.size()); ++j) {
319 
320  if (boundary_start->DistanceSquared(current_point) > std::pow(bbox_extension, 2)) {
321  reached_distance = true;
322  }
323 
324  if (boundary_end == nullptr ||
325  boundary_end->DistanceSquared(current_point) > std::pow(BOUNDARY_EDGE_LENGTH, 2) ||
326  reached_distance) {
327 
328  const cg::Vector3D heading_vector = current_point->GetForwardVector();
329  const cg::Location location = current_point->GetLocation();
330  cg::Vector3D perpendicular_vector = cg::Vector3D(-heading_vector.y, heading_vector.x, 0.0f);
331  perpendicular_vector = perpendicular_vector.MakeUnitVector();
332  // Direction determined for the left-handed system.
333  const cg::Vector3D scaled_perpendicular = perpendicular_vector * width;
334  left_boundary.push_back(location + cg::Location(scaled_perpendicular));
335  right_boundary.push_back(location + cg::Location(-1.0f * scaled_perpendicular));
336 
337  boundary_end = current_point;
338  }
339 
340  current_point = waypoint_buffer.at(j);
341  }
342 
343  // Connecting the geodesic path boundary with the vehicle bounding box.
344  LocationList geodesic_boundary;
345  // Reversing right boundary to construct clockwise (left-hand system)
346  // boundary. This is so because both left and right boundary vectors have
347  // the closest point to the vehicle at their starting index for the right
348  // boundary,
349  // we want to begin at the farthest point to have a clockwise trace.
350  std::reverse(right_boundary.begin(), right_boundary.end());
351  geodesic_boundary.insert(geodesic_boundary.end(), right_boundary.begin(), right_boundary.end());
352  geodesic_boundary.insert(geodesic_boundary.end(), bbox.begin(), bbox.end());
353  geodesic_boundary.insert(geodesic_boundary.end(), left_boundary.begin(), left_boundary.end());
354 
355  return geodesic_boundary;
356  } else {
357 
358  return bbox;
359  }
360  }

362  float CollisionStage::GetBoundingBoxExtention(const Actor &actor) const {
363 
364  const float velocity = actor->GetVelocity().Length();
365  float bbox_extension = BOUNDARY_EXTENSION_MINIMUM;
366  if (velocity > HIGHWAY_SPEED) {
367  bbox_extension = HIGHWAY_TIME_HORIZON * velocity;
368  } else if (velocity < CRAWL_SPEED) {
369  bbox_extension = BOUNDARY_EXTENSION_MINIMUM;
370  } else {
371  bbox_extension = std::sqrt(
372  EXTENSION_SQUARE_POINT * velocity) +
373  velocity * TIME_HORIZON +
374  BOUNDARY_EXTENSION_MINIMUM;
375  }
376 
377  return bbox_extension;
378  }

398  LocationList CollisionStage::GetBoundary(const Actor &actor) const {
399  const auto actor_type = actor->GetTypeId();
400 
401  cg::BoundingBox bbox;
402  cg::Location location;
403  cg::Vector3D heading_vector;
404 
405  if (actor_type[0] == 'v') {
406 
407  const auto vehicle = boost::static_pointer_cast<cc::Vehicle>(actor);
408  bbox = vehicle->GetBoundingBox();
409  location = vehicle->GetLocation();
410  heading_vector = vehicle->GetTransform().GetForwardVector();
411  } else if (actor_type[0] == 'w') {
412 
413  const auto walker = boost::static_pointer_cast<cc::Walker>(actor);
414  bbox = walker->GetBoundingBox();
415  location = walker->GetLocation();
416  heading_vector = walker->GetTransform().GetForwardVector();
417  }
418 
419  const cg::Vector3D extent = bbox.extent;
420  heading_vector.z = 0.0f;
421  const cg::Vector3D perpendicular_vector = cg::Vector3D(-heading_vector.y, heading_vector.x, 0.0f);
422 
423  // Four corners of the vehicle in top view clockwise order (left-handed
424  // system).
425  const cg::Vector3D x_boundary_vector = heading_vector * extent.x;
426  const cg::Vector3D y_boundary_vector = perpendicular_vector * extent.y;
427  return {
428  location + cg::Location(x_boundary_vector - y_boundary_vector),
429  location + cg::Location(-1.0f * x_boundary_vector - y_boundary_vector),
430  location + cg::Location(-1.0f * x_boundary_vector + y_boundary_vector),
431  location + cg::Location(x_boundary_vector + y_boundary_vector),
432  };
433  }
