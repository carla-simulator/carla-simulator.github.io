56  void CollisionStage::Action() {
57  const auto current_planner_frame = frame_selector ? planner_frame_a : planner_frame_b;
58 
59  // Handle vehicles not spawned by TrafficManager.
60  const auto current_time = chr::system_clock::now();
61  const chr::duration<double> diff = current_time - last_world_actors_pass_instance;
62 
63  // Periodically check for actors not spawned by TrafficManager.
64  if (diff.count() > 1.0f) {
65 
66  const auto world_actors = world.GetActors()->Filter("vehicle.*");
67  const auto world_walker = world.GetActors()->Filter("walker.*");
68  // Scanning for vehicles.
69  for (auto actor: *world_actors.get()) {
70  const auto unregistered_id = actor->GetId();
71  if (vehicle_id_to_index.find(unregistered_id) == vehicle_id_to_index.end() &&
72  unregistered_actors.find(unregistered_id) == unregistered_actors.end()) {
73  unregistered_actors.insert({unregistered_id, actor});
74  }
75  }
76  // Scanning for pedestrians.
77  for (auto walker: *world_walker.get()) {
78  const auto unregistered_id = walker->GetId();
79  if (unregistered_actors.find(unregistered_id) == unregistered_actors.end()) {
80  unregistered_actors.insert({unregistered_id, walker});
81  }
82  }
83  // Regularly update unregistered actors.
84  std::vector<ActorId> actor_ids_to_erase;
85  for (auto actor_info: unregistered_actors) {
86  if (actor_info.second->IsAlive()) {
87  vicinity_grid.UpdateGrid(actor_info.second);
88  } else {
89  vicinity_grid.EraseActor(actor_info.first);
90  actor_ids_to_erase.push_back(actor_info.first);
91  }
92  }
93  for (auto actor_id: actor_ids_to_erase) {
94  unregistered_actors.erase(actor_id);
95  }
96 
97  last_world_actors_pass_instance = current_time;
98  }
99 
100  // Looping over registered actors.
101  for (uint i = 0u; i < number_of_vehicles; ++i) {
102 
103  const LocalizationToCollisionData &data = localization_frame->at(i);
104  const Actor ego_actor = data.actor;
105  const ActorId ego_actor_id = ego_actor->GetId();
106 
107  // Retrieve actors around the path of the ego vehicle.
108  std::unordered_set<ActorId> actor_id_list = GetPotentialVehicleObstacles(ego_actor);
109 
110  bool collision_hazard = false;
111 
112  // Generate number between 0 and 100
113  const int r = rand() % 101;
114 
115  // Continue only if random number is lower than our %, default is 0.
116  if (parameters.GetPercentageIgnoreActors(boost::shared_ptr<cc::Actor>(ego_actor)) <= r) {
117  // Check every actor in the vicinity if it poses a collision hazard.
118  for (auto j = actor_id_list.begin(); (j != actor_id_list.end()) && !collision_hazard; ++j) {
119  const ActorId actor_id = *j;
120  try {
121 
122  Actor actor = nullptr;
123  if (vehicle_id_to_index.find(actor_id) != vehicle_id_to_index.end()) {
124  actor = localization_frame->at(vehicle_id_to_index.at(actor_id)).actor;
125  } else if (unregistered_actors.find(actor_id) != unregistered_actors.end()) {
126  actor = unregistered_actors.at(actor_id);
127  }
128 
129  const cg::Location ego_location = ego_actor->GetLocation();
130  const cg::Location other_location = actor->GetLocation();
131 
132  if (actor_id != ego_actor_id &&
133  (cg::Math::DistanceSquared(ego_location, other_location)
134  < std::pow(MAX_COLLISION_RADIUS, 2)) &&
135  (std::abs(ego_location.z - other_location.z) < VERTICAL_OVERLAP_THRESHOLD)) {
136 
137  if (parameters.GetCollisionDetection(ego_actor, actor) &&
138  NegotiateCollision(ego_actor, actor)) {
139 
140  collision_hazard = true;
141  }
142  }
143 
144  } catch (const std::exception &e) {
145  carla::log_warning("Encountered problem while determining collision \n");
146  carla::log_info("Actor might not be alive \n");
147  }
148 
149  }
150  }
151 
152  CollisionToPlannerData &message = current_planner_frame->at(i);
153  message.hazard = collision_hazard;
154 
155  }

195  bool CollisionStage::NegotiateCollision(const Actor &reference_vehicle, const Actor &other_vehicle) const {
196 
197  bool hazard = false;
198 
199  auto& data_packet = localization_frame->at(vehicle_id_to_index.at(reference_vehicle->GetId()));
200  Buffer& waypoint_buffer = data_packet.buffer;
201 
202  auto& other_packet = localization_frame->at(vehicle_id_to_index.at(other_vehicle->GetId()));
203  Buffer& other_buffer = other_packet.buffer;
204 
205  const cg::Location reference_location = reference_vehicle->GetLocation();
206  const cg::Location other_location = other_vehicle->GetLocation();
207 
208  const cg::Vector3D reference_heading = reference_vehicle->GetTransform().GetForwardVector();
209  cg::Vector3D reference_to_other = other_location - reference_location;
210  reference_to_other = reference_to_other.MakeUnitVector();
211 
212  const auto reference_vehicle_ptr = boost::static_pointer_cast<cc::Vehicle>(reference_vehicle);
213  const auto other_vehicle_ptr = boost::static_pointer_cast<cc::Vehicle>(other_vehicle);
214 
215  if (waypoint_buffer.front()->CheckJunction() &&
216  other_buffer.front()->CheckJunction()) {
217 
218  const Polygon reference_geodesic_polygon = GetPolygon(GetGeodesicBoundary(reference_vehicle));
219  const Polygon other_geodesic_polygon = GetPolygon(GetGeodesicBoundary(other_vehicle));
220  const Polygon reference_polygon = GetPolygon(GetBoundary(reference_vehicle));
221  const Polygon other_polygon = GetPolygon(GetBoundary(other_vehicle));
222 
223  const double reference_vehicle_to_other_geodesic = bg::distance(reference_polygon, other_geodesic_polygon);
224  const double other_vehicle_to_reference_geodesic = bg::distance(other_polygon, reference_geodesic_polygon);
225 
226  const auto inter_geodesic_distance = bg::distance(reference_geodesic_polygon, other_geodesic_polygon);
227  const auto inter_bbox_distance = bg::distance(reference_polygon, other_polygon);
228 
229  const cg::Vector3D other_heading = other_vehicle->GetTransform().GetForwardVector();
230  cg::Vector3D other_to_reference = reference_vehicle->GetLocation() - other_vehicle->GetLocation();
231  other_to_reference = other_to_reference.MakeUnitVector();
232 
233  // Whichever vehicle's path is farthest away from the other vehicle gets
234  // priority to move.
235  if (inter_geodesic_distance < 0.1 &&
236  ((
237  inter_bbox_distance > 0.1 &&
238  reference_vehicle_to_other_geodesic > other_vehicle_to_reference_geodesic
239  ) || (
240  inter_bbox_distance < 0.1 &&
241  cg::Math::Dot(reference_heading, reference_to_other) > cg::Math::Dot(other_heading, other_to_reference)
242  )) ) {
243 
244  hazard = true;
245  }
246 
247  } else if (!waypoint_buffer.front()->CheckJunction()) {
248 
249  const float reference_vehicle_length = reference_vehicle_ptr->GetBoundingBox().extent.x;
250  const float other_vehicle_length = other_vehicle_ptr->GetBoundingBox().extent.x;
251  const float vehicle_length_sum = reference_vehicle_length + other_vehicle_length;
252 
253  const float bbox_extension_length = GetBoundingBoxExtention(reference_vehicle);
254 
255  if ((cg::Math::Dot(reference_heading, reference_to_other) > 0.0f) &&
256  (cg::Math::DistanceSquared(reference_location, other_location) <
257  std::pow(bbox_extension_length+vehicle_length_sum, 2))) {
258 
259  hazard = true;
260  }
261  }
262 
263  return hazard;
264  }

280  LocationList CollisionStage::GetGeodesicBoundary(const Actor &actor) const {
281 
282  const LocationList bbox = GetBoundary(actor);
283 
284  if (vehicle_id_to_index.find(actor->GetId()) != vehicle_id_to_index.end()) {
285 
286  const cg::Location vehicle_location = actor->GetLocation();
287 
288  float bbox_extension = GetBoundingBoxExtention(actor);
289 
290  const float specific_distance_margin = parameters.GetDistanceToLeadingVehicle(actor);
291  if (specific_distance_margin > 0.0f) {
292  bbox_extension = std::max(specific_distance_margin, bbox_extension);
293  }
294 
295  const auto &waypoint_buffer = localization_frame->at(vehicle_id_to_index.at(actor->GetId())).buffer;
296 
297  LocationList left_boundary;
298  LocationList right_boundary;
299  const auto vehicle = boost::static_pointer_cast<cc::Vehicle>(actor);
300  const float width = vehicle->GetBoundingBox().extent.y;
301  const float length = vehicle->GetBoundingBox().extent.x;
302 
303  SimpleWaypointPtr boundary_start = waypoint_buffer.front();
304  uint boundary_start_index = 0u;
305  while (boundary_start->DistanceSquared(vehicle_location) < std::pow(length, 2) &&
306  boundary_start_index < waypoint_buffer.size() -1) {
307  boundary_start = waypoint_buffer.at(boundary_start_index);
308  ++boundary_start_index;
309  }
310  SimpleWaypointPtr boundary_end = nullptr;
311  SimpleWaypointPtr current_point = waypoint_buffer.at(boundary_start_index);
312 
313  const auto vehicle_reference = boost::static_pointer_cast<cc::Vehicle>(actor);
314  // At non-signalized junctions, we extend the boundary across the junction
315  // and in all other situations, boundary length is velocity-dependent.
316  bool reached_distance = false;
317  for (uint j = boundary_start_index; !reached_distance && (j < waypoint_buffer.size()); ++j) {
318 
319  if (boundary_start->DistanceSquared(current_point) > std::pow(bbox_extension, 2)) {
320  reached_distance = true;
321  }
322 
323  if (boundary_end == nullptr ||
324  boundary_end->DistanceSquared(current_point) > std::pow(BOUNDARY_EDGE_LENGTH, 2) ||
325  reached_distance) {
326 
327  const cg::Vector3D heading_vector = current_point->GetForwardVector();
328  const cg::Location location = current_point->GetLocation();
329  cg::Vector3D perpendicular_vector = cg::Vector3D(-heading_vector.y, heading_vector.x, 0.0f);
330  perpendicular_vector = perpendicular_vector.MakeUnitVector();
331  // Direction determined for the left-handed system.
332  const cg::Vector3D scaled_perpendicular = perpendicular_vector * width;
333  left_boundary.push_back(location + cg::Location(scaled_perpendicular));
334  right_boundary.push_back(location + cg::Location(-1.0f * scaled_perpendicular));
335 
336  boundary_end = current_point;
337  }
338 
339  current_point = waypoint_buffer.at(j);
340  }
341 
342  // Connecting the geodesic path boundary with the vehicle bounding box.
343  LocationList geodesic_boundary;
344  // Reversing right boundary to construct clockwise (left-hand system)
345  // boundary. This is so because both left and right boundary vectors have
346  // the closest point to the vehicle at their starting index for the right
347  // boundary,
348  // we want to begin at the farthest point to have a clockwise trace.
349  std::reverse(right_boundary.begin(), right_boundary.end());
350  geodesic_boundary.insert(geodesic_boundary.end(), right_boundary.begin(), right_boundary.end());
351  geodesic_boundary.insert(geodesic_boundary.end(), bbox.begin(), bbox.end());
352  geodesic_boundary.insert(geodesic_boundary.end(), left_boundary.begin(), left_boundary.end());
353 
354  return geodesic_boundary;
355  } else {
356 
357  return bbox;
358  }
359  }

361  float CollisionStage::GetBoundingBoxExtention(const Actor &actor) const {
362 
363  const float velocity = actor->GetVelocity().Length();
364  float bbox_extension = BOUNDARY_EXTENSION_MINIMUM;
365  if (velocity > HIGHWAY_SPEED) {
366  bbox_extension = HIGHWAY_TIME_HORIZON * velocity;
367  } else if (velocity < CRAWL_SPEED) {
368  bbox_extension = BOUNDARY_EXTENSION_MINIMUM;
369  } else {
370  bbox_extension = std::sqrt(
371  EXTENSION_SQUARE_POINT * velocity) +
372  velocity * TIME_HORIZON +
373  BOUNDARY_EXTENSION_MINIMUM;
374  }
375 
376  return bbox_extension;
377  }

397  LocationList CollisionStage::GetBoundary(const Actor &actor) const {
398  const auto actor_type = actor->GetTypeId();
399 
400  cg::BoundingBox bbox;
401  cg::Location location;
402  cg::Vector3D heading_vector;
403 
404  if (actor_type[0] == 'v') {
405 
406  const auto vehicle = boost::static_pointer_cast<cc::Vehicle>(actor);
407  bbox = vehicle->GetBoundingBox();
408  location = vehicle->GetLocation();
409  heading_vector = vehicle->GetTransform().GetForwardVector();
410  } else if (actor_type[0] == 'w') {
411 
412  const auto walker = boost::static_pointer_cast<cc::Walker>(actor);
413  bbox = walker->GetBoundingBox();
414  location = walker->GetLocation();
415  heading_vector = walker->GetTransform().GetForwardVector();
416  }
417 
418  const cg::Vector3D extent = bbox.extent;
419  heading_vector.z = 0.0f;
420  const cg::Vector3D perpendicular_vector = cg::Vector3D(-heading_vector.y, heading_vector.x, 0.0f);
421 
422  // Four corners of the vehicle in top view clockwise order (left-handed
423  // system).
424  const cg::Vector3D x_boundary_vector = heading_vector * extent.x;
425  const cg::Vector3D y_boundary_vector = perpendicular_vector * extent.y;
426  return {
427  location + cg::Location(x_boundary_vector - y_boundary_vector),
428  location + cg::Location(-1.0f * x_boundary_vector - y_boundary_vector),
429  location + cg::Location(-1.0f * x_boundary_vector + y_boundary_vector),
430  location + cg::Location(x_boundary_vector + y_boundary_vector),
431  };
432  }
