34 void CollisionStage::Update(const unsigned long index) {
35  ActorId obstacle_id = 0u;
36  bool collision_hazard = false;
37  float available_distance_margin = std::numeric_limits<float>::infinity();
38 
39  const ActorId ego_actor_id = vehicle_id_list.at(index);
40  if (simulation_state.ContainsActor(ego_actor_id)) {
41  const cg::Location ego_location = simulation_state.GetLocation(ego_actor_id);
42  const Buffer &ego_buffer = buffer_map.at(ego_actor_id);
43  const unsigned long look_ahead_index = GetTargetWaypoint(ego_buffer, JUNCTION_LOOK_AHEAD).second;
44  const float velocity = simulation_state.GetVelocity(ego_actor_id).Length();
45 
46  ActorIdSet overlapping_actors = track_traffic.GetOverlappingVehicles(ego_actor_id);
47  std::vector<ActorId> collision_candidate_ids;
48  // Run through vehicles with overlapping paths and filter them;
49  const float distance_to_leading = parameters.GetDistanceToLeadingVehicle(ego_actor_id);
50  float collision_radius_square = SQUARE(COLLISION_RADIUS_RATE * velocity + COLLISION_RADIUS_MIN) + distance_to_leading;;
51  if (velocity < 2.0f) {
52  const float length = simulation_state.GetDimensions(ego_actor_id).x;
53  const float collision_radius_stop = COLLISION_RADIUS_STOP + length;
54  collision_radius_square = SQUARE(collision_radius_stop);
55  if (distance_to_leading > collision_radius_stop) {
56  collision_radius_square = SQUARE(collision_radius_stop) + distance_to_leading;
57  }
58  }
59  for (ActorId overlapping_actor_id : overlapping_actors) {
60  // If actor is within maximum collision avoidance and vertical overlap range.
61  const cg::Location &overlapping_actor_location = simulation_state.GetLocation(overlapping_actor_id);
62  if (overlapping_actor_id != ego_actor_id
63  && cg::Math::DistanceSquared(overlapping_actor_location, ego_location) < collision_radius_square
64  && std::abs(ego_location.z - overlapping_actor_location.z) < VERTICAL_OVERLAP_THRESHOLD) {
65  collision_candidate_ids.push_back(overlapping_actor_id);
66  }
67  }
68 
69  // Sorting collision candidates in accending order of distance to current vehicle.
70  std::sort(collision_candidate_ids.begin(), collision_candidate_ids.end(),
71  [this, &ego_location](const ActorId &a_id_1, const ActorId &a_id_2) {
72  const cg::Location &e_loc = ego_location;
73  const cg::Location &loc_1 = simulation_state.GetLocation(a_id_1);
74  const cg::Location &loc_2 = simulation_state.GetLocation(a_id_2);
75  return (cg::Math::DistanceSquared(e_loc, loc_1) < cg::Math::DistanceSquared(e_loc, loc_2));
76  });
77 
78  // Check every actor in the vicinity if it poses a collision hazard.
79  for (auto iter = collision_candidate_ids.begin();
80  iter != collision_candidate_ids.end() && !collision_hazard;
81  ++iter) {
82  const ActorId other_actor_id = *iter;
83  const ActorType other_actor_type = simulation_state.GetType(other_actor_id);
84 
85  if (parameters.GetCollisionDetection(ego_actor_id, other_actor_id)
86  && buffer_map.find(ego_actor_id) != buffer_map.end()
87  && simulation_state.ContainsActor(other_actor_id)) {
88  std::pair<bool, float> negotiation_result = NegotiateCollision(ego_actor_id,
89  other_actor_id,
90  look_ahead_index);
91  if (negotiation_result.first) {
92  if ((other_actor_type == ActorType::Vehicle
93  && parameters.GetPercentageIgnoreVehicles(ego_actor_id) <= random_devices.at(ego_actor_id).next())
94  || (other_actor_type == ActorType::Pedestrian
95  && parameters.GetPercentageIgnoreWalkers(ego_actor_id) <= random_devices.at(ego_actor_id).next())) {
96  collision_hazard = true;
97  obstacle_id = other_actor_id;
98  available_distance_margin = negotiation_result.second;
99  }
100  }
101  }
102  }
103  }
104 
105  CollisionHazardData &output_element = output_array.at(index);
106  output_element.hazard_actor_id = obstacle_id;
107  output_element.hazard = collision_hazard;
108  output_element.available_distance_margin = available_distance_margin;
109 }

119 float CollisionStage::GetBoundingBoxExtention(const ActorId actor_id) {
120 
121  const float velocity = cg::Math::Dot(simulation_state.GetVelocity(actor_id), simulation_state.GetHeading(actor_id));
122  float bbox_extension;
123  // Using a function to calculate boundary length.
124  float velocity_extension = VEL_EXT_FACTOR*velocity;
125  bbox_extension = BOUNDARY_EXTENSION_MINIMUM + velocity_extension * velocity_extension;
126  // If a valid collision lock present, change boundary length to maintain lock.
127  if (collision_locks.find(actor_id) != collision_locks.end()) {
128  const CollisionLock &lock = collision_locks.at(actor_id);
129  float lock_boundary_length = static_cast<float>(lock.distance_to_lead_vehicle + LOCKING_DISTANCE_PADDING);
130  // Only extend boundary track vehicle if the leading vehicle
131  // if it is not further than velocity dependent extension by MAX_LOCKING_EXTENSION.
132  if ((lock_boundary_length - lock.initial_lock_distance) < MAX_LOCKING_EXTENSION) {
133  bbox_extension = lock_boundary_length;
134  }
135  }
136 
137  return bbox_extension;
138 }

140 LocationVector CollisionStage::GetBoundary(const ActorId actor_id) {
141  const ActorType actor_type = simulation_state.GetType(actor_id);
142  const cg::Vector3D heading_vector = simulation_state.GetHeading(actor_id);
143 
144  float forward_extension = 0.0f;
145  if (actor_type == ActorType::Pedestrian) {
146  // Extend the pedestrians bbox to "predict" where they'll be and avoid collisions.
147  forward_extension = simulation_state.GetVelocity(actor_id).Length() * WALKER_TIME_EXTENSION;
148  }
149 
150  cg::Vector3D dimensions = simulation_state.GetDimensions(actor_id);
151 
152  float bbox_x = dimensions.x;
153  float bbox_y = dimensions.y;
154 
155  const cg::Vector3D x_boundary_vector = heading_vector * (bbox_x + forward_extension);
156  const auto perpendicular_vector = cg::Vector3D(-heading_vector.y, heading_vector.x, 0.0f).MakeSafeUnitVector(EPSILON);
157  const cg::Vector3D y_boundary_vector = perpendicular_vector * (bbox_y + forward_extension);
158 
159  // Four corners of the vehicle in top view clockwise order (left-handed system).
160  const cg::Location location = simulation_state.GetLocation(actor_id);
161  LocationVector bbox_boundary = {
162  location + cg::Location(x_boundary_vector - y_boundary_vector),
163  location + cg::Location(-1.0f * x_boundary_vector - y_boundary_vector),
164  location + cg::Location(-1.0f * x_boundary_vector + y_boundary_vector),
165  location + cg::Location(x_boundary_vector + y_boundary_vector),
166  };
167 
168  return bbox_boundary;
169 }

171 LocationVector CollisionStage::GetGeodesicBoundary(const ActorId actor_id) {
172  LocationVector geodesic_boundary;
173 
174  if (geodesic_boundary_map.find(actor_id) != geodesic_boundary_map.end()) {
175  geodesic_boundary = geodesic_boundary_map.at(actor_id);
176  } else {
177  const LocationVector bbox = GetBoundary(actor_id);
178 
179  if (buffer_map.find(actor_id) != buffer_map.end()) {
180  float bbox_extension = GetBoundingBoxExtention(actor_id);
181  const float specific_lead_distance = parameters.GetDistanceToLeadingVehicle(actor_id);
182  bbox_extension = std::max(specific_lead_distance, bbox_extension);
183  const float bbox_extension_square = SQUARE(bbox_extension);
184 
185  LocationVector left_boundary;
186  LocationVector right_boundary;
187  cg::Vector3D dimensions = simulation_state.GetDimensions(actor_id);
188  const float width = dimensions.y;
189  const float length = dimensions.x;
190 
191  const Buffer &waypoint_buffer = buffer_map.at(actor_id);
192  const TargetWPInfo target_wp_info = GetTargetWaypoint(waypoint_buffer, length);
193  const SimpleWaypointPtr boundary_start = target_wp_info.first;
194  const uint64_t boundary_start_index = target_wp_info.second;
195 
196  // At non-signalized junctions, we extend the boundary across the junction
197  // and in all other situations, boundary length is velocity-dependent.
198  SimpleWaypointPtr boundary_end = nullptr;
199  SimpleWaypointPtr current_point = waypoint_buffer.at(boundary_start_index);
200  bool reached_distance = false;
201  for (uint64_t j = boundary_start_index; !reached_distance && (j < waypoint_buffer.size()); ++j) {
202  if (boundary_start->DistanceSquared(current_point) > bbox_extension_square || j == waypoint_buffer.size() - 1) {
203  reached_distance = true;
204  }
205  if (boundary_end == nullptr
206  || cg::Math::Dot(boundary_end->GetForwardVector(), current_point->GetForwardVector()) < COS_10_DEGREES
207  || reached_distance) {
208 
209  const cg::Vector3D heading_vector = current_point->GetForwardVector();
210  const cg::Location location = current_point->GetLocation();
211  cg::Vector3D perpendicular_vector = cg::Vector3D(-heading_vector.y, heading_vector.x, 0.0f);
212  perpendicular_vector = perpendicular_vector.MakeSafeUnitVector(EPSILON);
213  // Direction determined for the left-handed system.
214  const cg::Vector3D scaled_perpendicular = perpendicular_vector * width;
215  left_boundary.push_back(location + cg::Location(scaled_perpendicular));
216  right_boundary.push_back(location + cg::Location(-1.0f * scaled_perpendicular));
217 
218  boundary_end = current_point;
219  }
220 
221  current_point = waypoint_buffer.at(j);
222  }
223 
224  // Reversing right boundary to construct clockwise (left-hand system)
225  // boundary. This is so because both left and right boundary vectors have
226  // the closest point to the vehicle at their starting index for the right
227  // boundary,
228  // we want to begin at the farthest point to have a clockwise trace.
229  std::reverse(right_boundary.begin(), right_boundary.end());
230  geodesic_boundary.insert(geodesic_boundary.end(), right_boundary.begin(), right_boundary.end());
231  geodesic_boundary.insert(geodesic_boundary.end(), bbox.begin(), bbox.end());
232  geodesic_boundary.insert(geodesic_boundary.end(), left_boundary.begin(), left_boundary.end());
233  } else {
234 
235  geodesic_boundary = bbox;
236  }
237 
238  geodesic_boundary_map.insert({actor_id, geodesic_boundary});
239  }
240 
241  return geodesic_boundary;
242 }

255 GeometryComparison CollisionStage::GetGeometryBetweenActors(const ActorId reference_vehicle_id,
256  const ActorId other_actor_id) {
257 
258 
259  std::pair<ActorId, ActorId> key_parts;
260  if (reference_vehicle_id < other_actor_id) {
261  key_parts = {reference_vehicle_id, other_actor_id};
262  } else {
263  key_parts = {other_actor_id, reference_vehicle_id};
264  }
265 
266  uint64_t actor_id_key = 0u;
267  actor_id_key |= key_parts.first;
268  actor_id_key <<= 32;
269  actor_id_key |= key_parts.second;
270 
271  GeometryComparison comparision_result{-1.0, -1.0, -1.0, -1.0};
272 
273  if (geometry_cache.find(actor_id_key) != geometry_cache.end()) {
274 
275  comparision_result = geometry_cache.at(actor_id_key);
276  double mref_veh_other = comparision_result.reference_vehicle_to_other_geodesic;
277  comparision_result.reference_vehicle_to_other_geodesic = comparision_result.other_vehicle_to_reference_geodesic;
278  comparision_result.other_vehicle_to_reference_geodesic = mref_veh_other;
279  } else {
280 
281  const Polygon reference_polygon = GetPolygon(GetBoundary(reference_vehicle_id));
282  const Polygon other_polygon = GetPolygon(GetBoundary(other_actor_id));
283 
284  const Polygon reference_geodesic_polygon = GetPolygon(GetGeodesicBoundary(reference_vehicle_id));
285 
286  const Polygon other_geodesic_polygon = GetPolygon(GetGeodesicBoundary(other_actor_id));
287 
288  const double reference_vehicle_to_other_geodesic = bg::distance(reference_polygon, other_geodesic_polygon);
289  const double other_vehicle_to_reference_geodesic = bg::distance(other_polygon, reference_geodesic_polygon);
290  const auto inter_geodesic_distance = bg::distance(reference_geodesic_polygon, other_geodesic_polygon);
291  const auto inter_bbox_distance = bg::distance(reference_polygon, other_polygon);
292 
293  comparision_result = {reference_vehicle_to_other_geodesic,
294  other_vehicle_to_reference_geodesic,
295  inter_geodesic_distance,
296  inter_bbox_distance};
297 
298  geometry_cache.insert({actor_id_key, comparision_result});
299  }
300 
301  return comparision_result;
302 }

304 std::pair<bool, float> CollisionStage::NegotiateCollision(const ActorId reference_vehicle_id,
305  const ActorId other_actor_id,
306  const uint64_t reference_junction_look_ahead_index) {
307  // Output variables for the method.
308  bool hazard = false;
309  float available_distance_margin = std::numeric_limits<float>::infinity();
310 
311  const cg::Location reference_location = simulation_state.GetLocation(reference_vehicle_id);
312  const cg::Location other_location = simulation_state.GetLocation(other_actor_id);
313 
314  // Ego and other vehicle heading.
315  const cg::Vector3D reference_heading = simulation_state.GetHeading(reference_vehicle_id);
316  // Vector from ego position to position of the other vehicle.
317  cg::Vector3D reference_to_other = other_location - reference_location;
318  reference_to_other = reference_to_other.MakeSafeUnitVector(EPSILON);
319 
320  // Other vehicle heading.
321  const cg::Vector3D other_heading = simulation_state.GetHeading(other_actor_id);
322  // Vector from other vehicle position to ego position.
323  cg::Vector3D other_to_reference = reference_location - other_location;
324  other_to_reference = other_to_reference.MakeSafeUnitVector(EPSILON);
325 
326  float reference_vehicle_length = simulation_state.GetDimensions(reference_vehicle_id).x * SQUARE_ROOT_OF_TWO;
327  float other_vehicle_length = simulation_state.GetDimensions(other_actor_id).x * SQUARE_ROOT_OF_TWO;
328 
329  float inter_vehicle_distance = cg::Math::DistanceSquared(reference_location, other_location);
330  float ego_bounding_box_extension = GetBoundingBoxExtention(reference_vehicle_id);
331  float other_bounding_box_extension = GetBoundingBoxExtention(other_actor_id);
332  // Calculate minimum distance between vehicle to consider collision negotiation.
333  float inter_vehicle_length = reference_vehicle_length + other_vehicle_length;
334  float ego_detection_range = SQUARE(ego_bounding_box_extension + inter_vehicle_length);
335  float cross_detection_range = SQUARE(ego_bounding_box_extension + inter_vehicle_length + other_bounding_box_extension);
336 
337  // Conditions to consider collision negotiation.
338  bool other_vehicle_in_ego_range = inter_vehicle_distance < ego_detection_range;
339  bool other_vehicles_in_cross_detection_range = inter_vehicle_distance < cross_detection_range;
340  float reference_heading_to_other_dot = cg::Math::Dot(reference_heading, reference_to_other);
341  bool other_vehicle_in_front = reference_heading_to_other_dot > 0;
342  const Buffer &reference_vehicle_buffer = buffer_map.at(reference_vehicle_id);
343  SimpleWaypointPtr closest_point = reference_vehicle_buffer.front();
344  bool ego_inside_junction = closest_point->CheckJunction();
345  TrafficLightState reference_tl_state = simulation_state.GetTLS(reference_vehicle_id);
346  bool ego_at_traffic_light = reference_tl_state.at_traffic_light;
347  bool ego_stopped_by_light = reference_tl_state.tl_state != TLS::Green && reference_tl_state.tl_state != TLS::Off;
348  SimpleWaypointPtr look_ahead_point = reference_vehicle_buffer.at(reference_junction_look_ahead_index);
349  bool ego_at_junction_entrance = !closest_point->CheckJunction() && look_ahead_point->CheckJunction();
350 
351  // Conditions to consider collision negotiation.
352  if (!(ego_at_junction_entrance && ego_at_traffic_light && ego_stopped_by_light)
353  && ((ego_inside_junction && other_vehicles_in_cross_detection_range)
354  || (!ego_inside_junction && other_vehicle_in_front && other_vehicle_in_ego_range))) {
355  GeometryComparison geometry_comparison = GetGeometryBetweenActors(reference_vehicle_id, other_actor_id);
356 
357  // Conditions for collision negotiation.
358  bool geodesic_path_bbox_touching = geometry_comparison.inter_geodesic_distance < OVERLAP_THRESHOLD;
359  bool vehicle_bbox_touching = geometry_comparison.inter_bbox_distance < OVERLAP_THRESHOLD;
360  bool ego_path_clear = geometry_comparison.other_vehicle_to_reference_geodesic > OVERLAP_THRESHOLD;
361  bool other_path_clear = geometry_comparison.reference_vehicle_to_other_geodesic > OVERLAP_THRESHOLD;
362  bool ego_path_priority = geometry_comparison.reference_vehicle_to_other_geodesic < geometry_comparison.other_vehicle_to_reference_geodesic;
363  bool other_path_priority = geometry_comparison.reference_vehicle_to_other_geodesic > geometry_comparison.other_vehicle_to_reference_geodesic;
364  bool ego_angular_priority = reference_heading_to_other_dot< cg::Math::Dot(other_heading, other_to_reference);
365 
366  // Whichever vehicle's path is farthest away from the other vehicle gets priority to move.
367  bool lower_priority = !ego_path_priority && (other_path_priority || !ego_angular_priority);
368  bool blocked_by_other_or_lower_priority = !ego_path_clear || (other_path_clear && lower_priority);
369  bool yield_pre_crash = !vehicle_bbox_touching && blocked_by_other_or_lower_priority;
370  bool yield_post_crash = vehicle_bbox_touching && !ego_angular_priority;
371 
372  if (geodesic_path_bbox_touching && (yield_pre_crash || yield_post_crash)) {
373 
374  hazard = true;
375 
376  const float reference_lead_distance = parameters.GetDistanceToLeadingVehicle(reference_vehicle_id);
377  const float specific_distance_margin = std::max(reference_lead_distance, MIN_REFERENCE_DISTANCE);
378  available_distance_margin = static_cast<float>(std::max(geometry_comparison.reference_vehicle_to_other_geodesic
379  - static_cast<double>(specific_distance_margin), 0.0));
380 
381  ///////////////////////////////////// Collision locking mechanism /////////////////////////////////
382  // The idea is, when encountering a possible collision,
383  // we should ensure that the bounding box extension doesn't decrease too fast and loose collision tracking.
384  // This enables us to smoothly approach the lead vehicle.
385 
386  // When possible collision found, check if an entry for collision lock present.
387  if (collision_locks.find(reference_vehicle_id) != collision_locks.end()) {
388  CollisionLock &lock = collision_locks.at(reference_vehicle_id);
389  // Check if the same vehicle is under lock.
390  if (other_actor_id == lock.lead_vehicle_id) {
391  // If the body of the lead vehicle is touching the reference vehicle bounding box.
392  if (geometry_comparison.other_vehicle_to_reference_geodesic < OVERLAP_THRESHOLD) {
393  // Distance between the bodies of the vehicles.
394  lock.distance_to_lead_vehicle = geometry_comparison.inter_bbox_distance;
395  } else {
396  // Distance from reference vehicle body to other vehicle path polygon.
397  lock.distance_to_lead_vehicle = geometry_comparison.reference_vehicle_to_other_geodesic;
398  }
399  } else {
400  // If possible collision with a new vehicle, re-initialize with new lock entry.
401  lock = {geometry_comparison.inter_bbox_distance, geometry_comparison.inter_bbox_distance, other_actor_id};
402  }
403  } else {
404  // Insert and initialize lock entry if not present.
405  collision_locks.insert({reference_vehicle_id,
406  {geometry_comparison.inter_bbox_distance,
407  geometry_comparison.inter_bbox_distance,
408  other_actor_id}});
409  }
410  }
411  }
412 
413  // If no collision hazard detected, then flush collision lock held by the vehicle.
414  if (!hazard && collision_locks.find(reference_vehicle_id) != collision_locks.end()) {
415  collision_locks.erase(reference_vehicle_id);
416  }
417 
418  return {hazard, available_distance_margin};
419 }
