51  void CollisionStage::Action() {
52  const auto current_planner_frame = frame_selector ? planner_frame_a : planner_frame_b;
53 
54  // Looping over registered actors.
55  for (uint64_t i = 0u; i < number_of_vehicles && localization_frame != nullptr; ++i) {
56 
57  LocalizationToCollisionData &data = localization_frame->at(i);
58  const Actor ego_actor = data.actor;
59  const ActorId ego_actor_id = ego_actor->GetId();
60  const std::unordered_map<ActorId, Actor> overlapping_actors = data.overlapping_actors;
61  const cg::Location ego_location = ego_actor->GetLocation();
62  const SimpleWaypointPtr& closest_point = data.closest_waypoint;
63  const SimpleWaypointPtr& junction_look_ahead = data.junction_look_ahead_waypoint;
64 
65  // Retrieve actors around the path of the ego vehicle.
66  bool collision_hazard = false;
67 
68  // Generate number between 0 and 100
69  const int r = rand() % 101;
70 
71  // Continue only if random number is lower than our %, default is 0.
72  if (parameters.GetPercentageIgnoreActors(boost::shared_ptr<cc::Actor>(ego_actor)) <= r) {
73 
74  // Check every actor in the vicinity if it poses a collision hazard.
75  for (auto j = overlapping_actors.begin(); (j != overlapping_actors.end()) && !collision_hazard; ++j) {
76  const Actor actor = j->second;
77  const ActorId actor_id = j->first;
78  const cg::Location other_location = actor->GetLocation();
79 
80  try {
81  // Collision checks increase with speed (Official formula used)
82  float collision_distance = std::pow(floor(ego_actor->GetVelocity().Length()*3.6f/10.0f),2.0f);
83  collision_distance = cg::Math::Clamp(collision_distance, MIN_COLLISION_RADIUS, MAX_COLLISION_RADIUS);
84  // Temporary fix to (0,0,0) bug
85  if (other_location.x != 0 && other_location.y != 0 && other_location.z != 0){
86  if (actor_id != ego_actor_id &&
87  (cg::Math::DistanceSquared(ego_location, other_location)
88  < std::pow(MAX_COLLISION_RADIUS, 2)) &&
89  (std::abs(ego_location.z - other_location.z) < VERTICAL_OVERLAP_THRESHOLD)) {
90 
91  if (parameters.GetCollisionDetection(ego_actor, actor) &&
92  NegotiateCollision(ego_actor, actor, closest_point, junction_look_ahead)) {
93  collision_hazard = true;
94  }
95  }
96  }
97  } catch (const std::exception &e) {
98  carla::log_info("Actor might not be alive \n");
99  }
100 
101  }
102  }
103 
104  CollisionToPlannerData &message = current_planner_frame->at(i);
105  message.hazard = collision_hazard;
106  }
107  }

109  void CollisionStage::DataReceiver() {

114  // This map also provides us the additional benefit of being able to

123  // vehicles.

125

142  bool CollisionStage::NegotiateCollision(const Actor &reference_vehicle, const Actor &other_vehicle,
143  const SimpleWaypointPtr& closest_point,
144  const SimpleWaypointPtr& junction_look_ahead) {
145 
146  bool hazard = false;
147 
148  const cg::Location reference_location = reference_vehicle->GetLocation();
149  const cg::Location other_location = other_vehicle->GetLocation();
150 
151  const cg::Vector3D reference_heading = reference_vehicle->GetTransform().GetForwardVector();
152  cg::Vector3D reference_to_other = other_location - reference_location;
153  reference_to_other = reference_to_other.MakeUnitVector();
154 
155  const cg::Vector3D other_heading = other_vehicle->GetTransform().GetForwardVector();
156  cg::Vector3D other_to_reference = reference_vehicle->GetLocation() - other_vehicle->GetLocation();
157  other_to_reference = other_to_reference.MakeUnitVector();
158 
159  const auto &waypoint_buffer = localization_frame->at(
160  vehicle_id_to_index.at(reference_vehicle->GetId())).buffer;
161  const SimpleWaypointPtr& reference_front_wp = waypoint_buffer.front();
162 
163  const auto reference_vehicle_ptr = boost::static_pointer_cast<cc::Vehicle>(reference_vehicle);
164  const auto other_vehicle_ptr = boost::static_pointer_cast<cc::Vehicle>(other_vehicle);
165 
166  float reference_vehicle_length = reference_vehicle_ptr->GetBoundingBox().extent.x * 1.414f;
167  float other_vehicle_length = other_vehicle_ptr->GetBoundingBox().extent.x * 1.414f;
168  float inter_vehicle_length = reference_vehicle_length + other_vehicle_length;
169 
170  if (!(!reference_front_wp->CheckJunction() &&
171  cg::Math::Dot(reference_heading, reference_to_other) < 0) &&
172 
173  !(!closest_point->CheckJunction() && junction_look_ahead->CheckJunction() &&
174  reference_vehicle_ptr->GetVelocity().SquaredLength() < 0.1 &&
175  reference_vehicle_ptr->GetTrafficLightState() != carla::rpc::TrafficLightState::Green) &&
176 
177  !(!reference_front_wp->CheckJunction() &&
178  cg::Math::Dot(reference_heading, reference_to_other) > 0 &&
179  (cg::Math::DistanceSquared(reference_location, other_location) >
180  std::pow(GetBoundingBoxExtention(reference_vehicle) + inter_vehicle_length, 2)))) {
181 
182  const Polygon reference_geodesic_polygon = GetPolygon(GetGeodesicBoundary(reference_vehicle));
183  const Polygon other_geodesic_polygon = GetPolygon(GetGeodesicBoundary(other_vehicle));
184  const Polygon reference_polygon = GetPolygon(GetBoundary(reference_vehicle));
185  const Polygon other_polygon = GetPolygon(GetBoundary(other_vehicle));
186 
187  const double reference_vehicle_to_other_geodesic = bg::distance(reference_polygon, other_geodesic_polygon);
188  const double other_vehicle_to_reference_geodesic = bg::distance(other_polygon, reference_geodesic_polygon);
189 
190  const auto inter_geodesic_distance = bg::distance(reference_geodesic_polygon, other_geodesic_polygon);
191  const auto inter_bbox_distance = bg::distance(reference_polygon, other_polygon);
192 
193  // Whichever vehicle's path is farthest away from the other vehicle gets
194  // priority to move.
195  if (inter_geodesic_distance < 0.1 &&
196  ((inter_bbox_distance > 0.1 &&
197  reference_vehicle_to_other_geodesic > other_vehicle_to_reference_geodesic
198  ) || (
199  inter_bbox_distance < 0.1 &&
200  (cg::Math::Dot(reference_heading, reference_to_other) >
201  cg::Math::Dot(other_heading, other_to_reference))
202  ))
203  ) {
204 
205  hazard = true;
206  }
207  }
208  return hazard;
209  }

225  LocationList CollisionStage::GetGeodesicBoundary(const Actor &actor) {
226 
227  const LocationList bbox = GetBoundary(actor);
228 
229  if (vehicle_id_to_index.find(actor->GetId()) != vehicle_id_to_index.end()) {
230 
231  const cg::Location vehicle_location = actor->GetLocation();
232 
233  float bbox_extension = GetBoundingBoxExtention(actor);
234 
235  const float specific_distance_margin = parameters.GetDistanceToLeadingVehicle(actor);
236  if (specific_distance_margin > 0.0f) {
237  bbox_extension = std::max(specific_distance_margin, bbox_extension);
238  }
239 
240  const auto &waypoint_buffer = localization_frame->at(vehicle_id_to_index.at(actor->GetId())).buffer;
241 
242  LocationList left_boundary;
243  LocationList right_boundary;
244  const auto vehicle = boost::static_pointer_cast<cc::Vehicle>(actor);
245  const float width = vehicle->GetBoundingBox().extent.y;
246  const float length = vehicle->GetBoundingBox().extent.x*2;
247 
248  SimpleWaypointPtr boundary_start = waypoint_buffer.front();
249  uint64_t boundary_start_index = 0u;
250  while (boundary_start->DistanceSquared(vehicle_location) < std::pow(length, 2) &&
251  boundary_start_index < waypoint_buffer.size() -1) {
252  boundary_start = waypoint_buffer.at(boundary_start_index);
253  ++boundary_start_index;
254  }
255  SimpleWaypointPtr boundary_end = nullptr;
256  SimpleWaypointPtr current_point = waypoint_buffer.at(boundary_start_index);
257 
258  const auto vehicle_reference = boost::static_pointer_cast<cc::Vehicle>(actor);
259  // At non-signalized junctions, we extend the boundary across the junction
260  // and in all other situations, boundary length is velocity-dependent.
261  bool reached_distance = false;
262  for (uint64_t j = boundary_start_index; !reached_distance && (j < waypoint_buffer.size()); ++j) {
263 
264  if (boundary_start->DistanceSquared(current_point) > std::pow(bbox_extension, 2)) {
265  reached_distance = true;
266  }
267 
268  if (boundary_end == nullptr ||
269  boundary_end->DistanceSquared(current_point) > std::pow(BOUNDARY_EDGE_LENGTH, 2) ||
270  reached_distance) {
271 
272  const cg::Vector3D heading_vector = current_point->GetForwardVector();
273  const cg::Location location = current_point->GetLocation();
274  cg::Vector3D perpendicular_vector = cg::Vector3D(-heading_vector.y, heading_vector.x, 0.0f);
275  perpendicular_vector = perpendicular_vector.MakeUnitVector();
276  // Direction determined for the left-handed system.
277  const cg::Vector3D scaled_perpendicular = perpendicular_vector * width;
278  left_boundary.push_back(location + cg::Location(scaled_perpendicular));
279  right_boundary.push_back(location + cg::Location(-1.0f * scaled_perpendicular));
280 
281  boundary_end = current_point;
282  }
283 
284  current_point = waypoint_buffer.at(j);
285  }
286 
287  // Connecting the geodesic path boundary with the vehicle bounding box.
288  LocationList geodesic_boundary;
289  // Reversing right boundary to construct clockwise (left-hand system)
290  // boundary. This is so because both left and right boundary vectors have
291  // the closest point to the vehicle at their starting index for the right
292  // boundary,
293  // we want to begin at the farthest point to have a clockwise trace.
294  std::reverse(right_boundary.begin(), right_boundary.end());
295  geodesic_boundary.insert(geodesic_boundary.end(), right_boundary.begin(), right_boundary.end());
296  geodesic_boundary.insert(geodesic_boundary.end(), bbox.begin(), bbox.end());
297  geodesic_boundary.insert(geodesic_boundary.end(), left_boundary.begin(), left_boundary.end());
298 
299  return geodesic_boundary;
300  } else {
301 
302  return bbox;
303  }
304 
305  }

307  float CollisionStage::GetBoundingBoxExtention(const Actor &actor) {
308 
309  const float velocity = actor->GetVelocity().Length();
310  float bbox_extension = BOUNDARY_EXTENSION_MINIMUM;
311  if (velocity > HIGHWAY_SPEED) {
312  bbox_extension = HIGHWAY_TIME_HORIZON * velocity;
313  } else if (velocity < CRAWL_SPEED) {
314  bbox_extension = BOUNDARY_EXTENSION_MINIMUM;
315  } else {
316  bbox_extension = std::sqrt(
317  EXTENSION_SQUARE_POINT * velocity) +
318  velocity * TIME_HORIZON +
319  BOUNDARY_EXTENSION_MINIMUM;
320  }
321 
322  return bbox_extension;
323  }

325  LocationList CollisionStage::GetBoundary(const Actor &actor) {
326 
327  const auto actor_type = actor->GetTypeId();
328  cg::Location location = actor->GetLocation();
329  cg::Vector3D heading_vector = actor->GetTransform().GetForwardVector();
330  heading_vector.z = 0.0f;
331  heading_vector = heading_vector.MakeUnitVector();
332 
333  cg::BoundingBox bbox;
334  float forward_extension = 0.0f;
335  if (actor_type[0] == 'v') {
336  const auto vehicle = boost::static_pointer_cast<cc::Vehicle>(actor);
337  bbox = vehicle->GetBoundingBox();
338  } else if (actor_type[0] == 'w') {
339  const auto walker = boost::static_pointer_cast<cc::Walker>(actor);
340  bbox = walker->GetBoundingBox();
341  // Extend the pedestrians bbox to "predict" where they'll be and avoid collisions.
342  forward_extension = walker->GetVelocity().Length() * WALKER_TIME_EXTENSION;
343  }
344 
345  const cg::Vector3D extent = bbox.extent;
346  const cg::Vector3D perpendicular_vector = cg::Vector3D(-heading_vector.y, heading_vector.x, 0.0f);
347 
348  const cg::Vector3D x_boundary_vector = heading_vector * (extent.x + forward_extension);
349  const cg::Vector3D y_boundary_vector = perpendicular_vector * (extent.y + forward_extension);
350 
351  // Four corners of the vehicle in top view clockwise order (left-handed
352  // system).
353  LocationList bbox_boundary = {
354  location + cg::Location(x_boundary_vector - y_boundary_vector),
355  location + cg::Location(-1.0f * x_boundary_vector - y_boundary_vector),
356  location + cg::Location(-1.0f * x_boundary_vector + y_boundary_vector),
357  location + cg::Location(x_boundary_vector + y_boundary_vector),
358  };
359 
360  return bbox_boundary;
361  }
