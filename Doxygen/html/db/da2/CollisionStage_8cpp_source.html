58  void CollisionStage::Action() {
59 
60  // Clearing the old chache.
61  vehicle_cache.clear();
62  const auto current_planner_frame = frame_selector ? planner_frame_a : planner_frame_b;
63 
64  // Looping over registered actors.
65  for (uint64_t i = 0u; i < number_of_vehicles && localization_frame != nullptr; ++i) {
66 
67  LocalizationToCollisionData &data = localization_frame->at(i);
68  if (!data.actor->IsAlive()) {
69  continue;
70  }
71 
72  const Actor ego_actor = data.actor;
73  const ActorId ego_actor_id = ego_actor->GetId();
74  const cg::Location ego_location = ego_actor->GetLocation();
75  const cg::Vector3D ego_velocity = data.velocity;
76 
77  const SimpleWaypointPtr& closest_point = data.closest_waypoint;
78  const SimpleWaypointPtr& junction_look_ahead = data.junction_look_ahead_waypoint;
79  using OverlappingActorInfo = std::vector<std::tuple<ActorId, Actor, cg::Vector3D>>;
80  const OverlappingActorInfo &collision_candidates = data.overlapping_actors;
81 
82  bool collision_hazard = false;
83  float available_distance_margin = std::numeric_limits<float>::infinity();
84  cg::Vector3D obstacle_velocity;
85  const SimpleWaypointPtr safe_point_junction = data.safe_point_after_junction;
86 
87  try {
88  // Check every actor in the vicinity if it poses a collision hazard.
89  for (auto actor_info = collision_candidates.begin();
90  actor_info != collision_candidates.end() && !collision_hazard;
91  ++actor_info) {
92 
93  ActorId other_actor_id;
94  Actor other_actor;
95  cg::Vector3D other_velocity;
96  std::tie(other_actor_id, other_actor, other_velocity) = *actor_info;
97 
98  if (!other_actor->IsAlive()) continue;
99  const auto other_actor_type = other_actor->GetTypeId();
100  const cg::Location other_location = other_actor->GetLocation();
101 
102  // Collision checks increase with speed
103  float collision_distance = std::pow(floor(ego_velocity.Length()*3.6f/10.0f),2.0f);
104  collision_distance = cg::Math::Clamp(collision_distance, MIN_COLLISION_RADIUS, MAX_COLLISION_RADIUS);
105 
106  // Temporary fix to (0,0,0) bug
107  if (!(other_location.x == 0 && other_location.y == 0 && other_location.z == 0)) {
108 
109  if (other_actor_id != ego_actor_id &&
110  (cg::Math::DistanceSquared(ego_location, other_location)
111  < std::pow(MAX_COLLISION_RADIUS, 2)) &&
112  (std::abs(ego_location.z - other_location.z) < VERTICAL_OVERLAP_THRESHOLD)) {
113 
114  if (parameters.GetCollisionDetection(ego_actor, other_actor)) {
115 
116  std::pair<bool, float> negotiation_result = NegotiateCollision(ego_actor, other_actor, ego_location,
117  other_location, closest_point, junction_look_ahead,
118  ego_velocity, other_velocity);
119  if ((safe_point_junction != nullptr
120  && !IsLocationAfterJunctionSafe(ego_actor, other_actor, safe_point_junction,
121  other_location, other_velocity))
122  || negotiation_result.first)
123  {
124 
125  if ((other_actor_type[0] == 'v' && parameters.GetPercentageIgnoreVehicles(ego_actor) <= (rand() % 101)) ||
126  (other_actor_type[0] == 'w' && parameters.GetPercentageIgnoreWalkers(ego_actor) <= (rand() % 101)))
127  {
128  collision_hazard = true;
129  available_distance_margin = negotiation_result.second;
130  obstacle_velocity = other_velocity;
131  }
132  }
133  }
134  }
135  }
136  }
137 
138  } catch (const std::exception &e) {
139  carla::log_info("Actor might not be alive \n");
140  }
141 
142  CollisionToPlannerData &message = current_planner_frame->at(i);
143  message.hazard = collision_hazard;
144  message.distance_to_other_vehicle = available_distance_margin;
145  message.other_vehicle_velocity = obstacle_velocity;
146  }
147  }

190  std::pair<bool, float> CollisionStage::NegotiateCollision(const Actor &reference_vehicle, const Actor &other_vehicle,
191  const cg::Location &reference_location,
192  const cg::Location &other_location,
193  const SimpleWaypointPtr &closest_point,
194  const SimpleWaypointPtr &junction_look_ahead,
195  const cg::Vector3D reference_velocity,
196  const cg::Vector3D other_velocity) {
197 
198  // Vehicle IDs
199  const ActorId reference_vehicle_id = reference_vehicle->GetId();
200  const ActorId other_vehicle_id = other_vehicle->GetId();
201 
202  // Output variables for the method.
203  bool hazard = false;
204  float available_distance_margin = std::numeric_limits<float>::infinity();
205 
206  // Ego and other vehicle heading.
207  const cg::Vector3D reference_heading = reference_vehicle->GetTransform().GetForwardVector();
208  // Vector from ego position to position of the other vehicle.
209  cg::Vector3D reference_to_other = other_location - reference_location;
210  reference_to_other = reference_to_other.MakeUnitVector();
211 
212  // Other vehicle heading.
213  const cg::Vector3D other_heading = other_vehicle->GetTransform().GetForwardVector();
214  // Vector from other vehicle position to ego position.
215  cg::Vector3D other_to_reference = reference_location - other_location;
216  other_to_reference = other_to_reference.MakeUnitVector();
217 
218  // Obtain cc::Vehicle pointers and calculate half diagonal length of vehicle bounding box.
219  const auto reference_vehicle_ptr = boost::static_pointer_cast<cc::Vehicle>(reference_vehicle);
220  const auto other_vehicle_ptr = boost::static_pointer_cast<cc::Vehicle>(other_vehicle);
221  float reference_vehicle_length = reference_vehicle_ptr->GetBoundingBox().extent.x * SQUARE_ROOT_OF_TWO;
222  float other_vehicle_length = other_vehicle_ptr->GetBoundingBox().extent.x * SQUARE_ROOT_OF_TWO;
223 
224  float inter_vehicle_distance = cg::Math::DistanceSquared(reference_location, other_location);
225  float ego_bounding_box_extension = GetBoundingBoxExtention(reference_vehicle_id,
226  reference_velocity, reference_heading);
227  float other_bounding_box_extension = GetBoundingBoxExtention(other_vehicle_id, other_velocity, other_heading);
228  // Calculate minimum distance between vehicle to consider collision negotiation.
229  float inter_vehicle_length = reference_vehicle_length + other_vehicle_length;
230  float ego_detection_range = std::pow(ego_bounding_box_extension + inter_vehicle_length, 2.0f);
231  float cross_detection_range = std::pow(ego_bounding_box_extension + inter_vehicle_length
232  + other_bounding_box_extension, 2.0f);
233 
234  // Conditions to consider collision negotiation.
235  bool other_vehicle_in_ego_range = inter_vehicle_distance < ego_detection_range;
236  bool other_vehicles_in_cross_detection_range = inter_vehicle_distance < cross_detection_range;
237  bool other_vehicle_in_front = cg::Math::Dot(reference_heading, reference_to_other) > 0;
238  bool ego_inside_junction = closest_point->CheckJunction();
239  bool ego_at_traffic_light = reference_vehicle_ptr->IsAtTrafficLight();
240  bool ego_stopped_by_light = reference_vehicle_ptr->GetTrafficLightState() != carla::rpc::TrafficLightState::Green;
241  bool ego_at_junction_entrance = !closest_point->CheckJunction() && junction_look_ahead->CheckJunction();
242 
243  // Conditions to consider collision negotiation.
244  if (!(ego_at_junction_entrance && ego_at_traffic_light && ego_stopped_by_light)
245  && ((ego_inside_junction && other_vehicles_in_cross_detection_range)
246  || (!ego_inside_junction && other_vehicle_in_front && other_vehicle_in_ego_range))) {
247 
248  GeometryComparisonCache cache = GetGeometryBetweenActors(reference_vehicle, other_vehicle,
249  reference_location, other_location,
250  reference_velocity, other_velocity);
251 
252  // Conditions for collision negotiation.
253  bool geodesic_path_bbox_touching = cache.inter_geodesic_distance < 0.1;
254  bool vehicle_bbox_touching = cache.inter_bbox_distance < 0.1;
255  bool ego_path_clear = cache.other_vehicle_to_reference_geodesic > 0.1;
256  bool other_path_clear = cache.reference_vehicle_to_other_geodesic > 0.1;
257  bool ego_path_priority = cache.reference_vehicle_to_other_geodesic < cache.other_vehicle_to_reference_geodesic;
258  bool ego_angular_priority = cg::Math::Dot(reference_heading, reference_to_other)
259  < cg::Math::Dot(other_heading, other_to_reference);
260 
261  // Whichever vehicle's path is farthest away from the other vehicle gets priority to move.
262  if (geodesic_path_bbox_touching
263  && ((!vehicle_bbox_touching
264  && (!ego_path_clear || (ego_path_clear && other_path_clear && !ego_angular_priority && !ego_path_priority)))
265  || (vehicle_bbox_touching && !ego_angular_priority && !ego_path_priority))) {
266  hazard = true;
267 
268  const float specific_distance_margin = MAX(parameters.GetDistanceToLeadingVehicle(reference_vehicle),
269  BOUNDARY_EXTENSION_MINIMUM);
270  available_distance_margin = static_cast<float>(MAX(cache.reference_vehicle_to_other_geodesic
271  - specific_distance_margin, 0.0));
272 
273  ///////////////////////////////////// Collision locking mechanism /////////////////////////////////
274  // The idea is, when encountering a possible collision,
275  // we should ensure that the bounding box extension doesn't decrease too fast and loose collision tracking.
276  // This enables us to smoothly approach the lead vehicle.
277 
278  // When possible collision found, check if an entry for collision lock present.
279  if (collision_locks.find(reference_vehicle_id) != collision_locks.end()) {
280  CollisionLock &lock = collision_locks.at(reference_vehicle_id);
281  // Check if the same vehicle is under lock.
282  if (other_vehicle_id == lock.lead_vehicle_id) {
283  // If the body of the lead vehicle is touching the reference vehicle bounding box.
284  if (cache.other_vehicle_to_reference_geodesic < 0.1) {
285  // Distance between the bodies of the vehicles.
286  lock.distance_to_lead_vehicle = cache.inter_bbox_distance;
287  } else {
288  // Distance from reference vehicle body to other vehicle path polygon.
289  lock.distance_to_lead_vehicle = cache.reference_vehicle_to_other_geodesic;
290  }
291  } else {
292  // If possible collision with a new vehicle, re-initialize with new lock entry.
293  lock = {other_vehicle_id, cache.inter_bbox_distance, cache.inter_bbox_distance};
294  }
295  } else {
296  // Insert and initialize lock entry if not present.
297  collision_locks.insert({reference_vehicle_id,
298  {other_vehicle_id, cache.inter_bbox_distance, cache.inter_bbox_distance}});
299  }
300  }
301  }
302 
303  // If no collision hazard detected, then flush collision lock held by the vehicle.
304  if (!hazard && collision_locks.find(reference_vehicle_id) != collision_locks.end()) {
305  collision_locks.erase(reference_vehicle_id);
306  }
307 
308  return {hazard, available_distance_margin};
309  }

322  LocationList CollisionStage::GetGeodesicBoundary(const Actor &actor, const cg::Location &vehicle_location,
323  const cg::Vector3D velocity) {
324 
325  const ActorId actor_id = actor->GetId();
326 
327  if (geodesic_boundaries.find(actor_id) != geodesic_boundaries.end()) {
328  return geodesic_boundaries.at(actor_id);
329  }
330 
331  const LocationList bbox = GetBoundary(actor, vehicle_location, velocity);
332 
333  if (vehicle_id_to_index.find(actor_id) != vehicle_id_to_index.end()) {
334 
335  float bbox_extension = GetBoundingBoxExtention(actor_id, velocity, actor->GetTransform().GetForwardVector());
336 
337  const float specific_distance_margin = parameters.GetDistanceToLeadingVehicle(actor);
338  if (specific_distance_margin > 0.0f) {
339  bbox_extension = MAX(specific_distance_margin, bbox_extension);
340  }
341 
342  const auto &waypoint_buffer = localization_frame->at(vehicle_id_to_index.at(actor_id)).buffer;
343 
344  LocationList left_boundary;
345  LocationList right_boundary;
346  const auto vehicle = boost::static_pointer_cast<cc::Vehicle>(actor);
347  const float width = vehicle->GetBoundingBox().extent.y;
348  const float length_squared = std::pow(vehicle->GetBoundingBox().extent.x*2, 2.0f);
349 
350  SimpleWaypointPtr boundary_start = waypoint_buffer.front();
351  SimpleWaypointPtr front_waypoint = waypoint_buffer.front();
352  uint64_t boundary_start_index = 0u;
353  while (boundary_start->DistanceSquared(front_waypoint) < length_squared &&
354  boundary_start_index < waypoint_buffer.size() -1) {
355  boundary_start = waypoint_buffer.at(boundary_start_index);
356  ++boundary_start_index;
357  }
358  SimpleWaypointPtr boundary_end = nullptr;
359  SimpleWaypointPtr current_point = waypoint_buffer.at(boundary_start_index);
360 
361  const auto vehicle_reference = boost::static_pointer_cast<cc::Vehicle>(actor);
362  // At non-signalized junctions, we extend the boundary across the junction
363  // and in all other situations, boundary length is velocity-dependent.
364  bool reached_distance = false;
365  for (uint64_t j = boundary_start_index; !reached_distance && (j < waypoint_buffer.size()); ++j) {
366 
367  if (boundary_start->DistanceSquared(current_point) > std::pow(bbox_extension, 2)
368  || j == waypoint_buffer.size() - 1) {
369  reached_distance = true;
370  }
371 
372  if (boundary_end == nullptr
373  || cg::Math::Dot(boundary_end->GetForwardVector(), current_point->GetForwardVector()) < COS_10_DEGREES
374  || reached_distance) {
375 
376  const cg::Vector3D heading_vector = current_point->GetForwardVector();
377  const cg::Location location = current_point->GetLocation();
378  cg::Vector3D perpendicular_vector = cg::Vector3D(-heading_vector.y, heading_vector.x, 0.0f);
379  perpendicular_vector = perpendicular_vector.MakeUnitVector();
380  // Direction determined for the left-handed system.
381  const cg::Vector3D scaled_perpendicular = perpendicular_vector * width;
382  left_boundary.push_back(location + cg::Location(scaled_perpendicular));
383  right_boundary.push_back(location + cg::Location(-1.0f * scaled_perpendicular));
384 
385  boundary_end = current_point;
386  }
387 
388  current_point = waypoint_buffer.at(j);
389  }
390 
391  // Connecting the geodesic path boundary with the vehicle bounding box.
392  LocationList geodesic_boundary;
393  // Reversing right boundary to construct clockwise (left-hand system)
394  // boundary. This is so because both left and right boundary vectors have
395  // the closest point to the vehicle at their starting index for the right
396  // boundary,
397  // we want to begin at the farthest point to have a clockwise trace.
398  std::reverse(right_boundary.begin(), right_boundary.end());
399  geodesic_boundary.insert(geodesic_boundary.end(), right_boundary.begin(), right_boundary.end());
400  geodesic_boundary.insert(geodesic_boundary.end(), bbox.begin(), bbox.end());
401  geodesic_boundary.insert(geodesic_boundary.end(), left_boundary.begin(), left_boundary.end());
402 
403  geodesic_boundaries.insert({actor->GetId(), geodesic_boundary});
404  return geodesic_boundary;
405  } else {
406 
407  geodesic_boundaries.insert({actor->GetId(), bbox});
408  return bbox;
409  }
410 
411  }

413  float CollisionStage::GetBoundingBoxExtention(const ActorId actor_id,
414  const cg::Vector3D velocity_vector,
415  const cg::Vector3D heading_vector) {
416 
417  const float velocity = cg::Math::Dot(velocity_vector, heading_vector);
418  float bbox_extension;
419  // Using a linear function to calculate boundary length.
420  // min speed : 0kmph -> BOUNDARY_EXTENSION_MINIMUM
421  // max speed : 100kmph -> BOUNDARY_EXTENSION_MAXIMUM
422  float slope = (BOUNDARY_EXTENSION_MAXIMUM - BOUNDARY_EXTENSION_MINIMUM) / ARBITRARY_MAX_SPEED;
423  bbox_extension = slope * velocity + BOUNDARY_EXTENSION_MINIMUM;
424  // If a valid collision lock present, change boundary length to maintain lock.
425  if (collision_locks.find(actor_id) != collision_locks.end())
426  {
427  CollisionLock &lock = collision_locks.at(actor_id);
428  float lock_boundary_length = static_cast<float>(lock.distance_to_lead_vehicle
429  + LOCKING_DISTANCE_PADDING);
430  // Only extend boundary track vehicle if the leading vehicle
431  // if it is not further than velocity dependent extension by MAX_LOCKING_EXTENSION.
432  if ((lock_boundary_length - lock.initial_lock_distance) < MAX_LOCKING_EXTENSION)
433  {
434  bbox_extension = lock_boundary_length;
435  }
436  }
437 
438  return bbox_extension;
439  }

441  LocationList CollisionStage::GetBoundary(const Actor &actor,
442  const cg::Location &location,
443  const cg::Vector3D velocity) {
444 
445  const auto actor_type = actor->GetTypeId();
446  cg::Vector3D heading_vector = actor->GetTransform().GetForwardVector();
447  heading_vector.z = 0.0f;
448  heading_vector = heading_vector.MakeUnitVector();
449 
450  cg::BoundingBox bbox;
451  float forward_extension = 0.0f;
452  if (actor_type[0] == 'v') {
453  const auto vehicle = boost::static_pointer_cast<cc::Vehicle>(actor);
454  bbox = vehicle->GetBoundingBox();
455  } else if (actor_type[0] == 'w') {
456  const auto walker = boost::static_pointer_cast<cc::Walker>(actor);
457  bbox = walker->GetBoundingBox();
458  // Extend the pedestrians bbox to "predict" where they'll be and avoid collisions.
459  forward_extension = velocity.Length() * WALKER_TIME_EXTENSION;
460  }
461 
462  const cg::Vector3D extent = bbox.extent;
463  const cg::Vector3D perpendicular_vector = cg::Vector3D(-heading_vector.y, heading_vector.x, 0.0f);
464 
465  const cg::Vector3D x_boundary_vector = heading_vector * (extent.x + forward_extension);
466  const cg::Vector3D y_boundary_vector = perpendicular_vector * (extent.y + forward_extension);
467 
468  // Four corners of the vehicle in top view clockwise order (left-handed
469  // system).
470  LocationList bbox_boundary = {
471  location + cg::Location(x_boundary_vector - y_boundary_vector),
472  location + cg::Location(-1.0f * x_boundary_vector - y_boundary_vector),
473  location + cg::Location(-1.0f * x_boundary_vector + y_boundary_vector),
474  location + cg::Location(x_boundary_vector + y_boundary_vector),
475  };
476 
477  return bbox_boundary;
478  }

480  bool CollisionStage::IsLocationAfterJunctionSafe(const Actor &ego_actor,
481  const Actor &other_actor,
482  const SimpleWaypointPtr safe_point,
483  const cg::Location &other_location,
484  const cg::Vector3D other_velocity){
485 
486  bool safe_junction = true;
487 
488  if (other_velocity.Length() < EPSILON_VELOCITY){
489 
490  cg::Location safe_location = safe_point->GetLocation();
491  cg::Vector3D heading_vector = safe_point->GetForwardVector();
492  heading_vector.z = 0.0f;
493  heading_vector = heading_vector.MakeUnitVector();
494 
495  cg::BoundingBox bbox;
496  const auto vehicle = boost::static_pointer_cast<cc::Vehicle>(ego_actor);
497  bbox = vehicle->GetBoundingBox();
498  const cg::Vector3D extent = bbox.extent;
499 
500  const cg::Vector3D perpendicular_vector = cg::Vector3D(-heading_vector.y, heading_vector.x, 0.0f);
501 
502  const cg::Vector3D x_boundary_vector = heading_vector * extent.x;
503  const cg::Vector3D y_boundary_vector = perpendicular_vector * extent.y;
504 
505  LocationList ego_actor_boundary = {
506  safe_location + cg::Location(x_boundary_vector - y_boundary_vector),
507  safe_location + cg::Location(-1.0f * x_boundary_vector - y_boundary_vector),
508  safe_location + cg::Location(-1.0f * x_boundary_vector + y_boundary_vector),
509  safe_location + cg::Location(x_boundary_vector + y_boundary_vector),
510  };
511 
512  const Polygon reference_polygon = GetPolygon(ego_actor_boundary);
513  const Polygon other_polygon = GetPolygon(GetBoundary(other_actor, other_location, other_velocity));
514 
515  const auto inter_bbox_distance = bg::distance(reference_polygon, other_polygon);
516  if (inter_bbox_distance < INTER_BBOX_DISTANCE_THRESHOLD){
517  safe_junction = false;
518  }
519  }
520 
521  return safe_junction;
522  }
