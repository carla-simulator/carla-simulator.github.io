54 FRotator GetDeltaRotator(const FRotator & Rotator1, const FRotator & Rotator2)
55 {
56  float HalfCircle = 180.f;
57  float FullCircle = 360.f;
58  auto GetDeltaAngle = [&](float Angle1, float Angle2) -> float
59  {
60  if (Angle1 < 0)
61  {
62  Angle1 = FullCircle + Angle1;
63  }
64  if (Angle2 < 0)
65  {
66  Angle2 = FullCircle + Angle2;
67  }
68  float Diff = fmod( Angle1 - Angle2 + HalfCircle , FullCircle) - HalfCircle;
69  return Diff < -HalfCircle ? Diff + FullCircle : Diff;
70  };
71 
72  return FRotator(
73  GetDeltaAngle(Rotator1.Pitch, Rotator2.Pitch),
74  GetDeltaAngle(Rotator1.Yaw, Rotator2.Yaw),
75  GetDeltaAngle(Rotator1.Roll, Rotator2.Roll)
76  );
77 }

167 {
168  TRACE_CPUPROFILER_EVENT_SCOPE(FSkeletonHierarchy::ComputeEndJoints);
169  EndJoints.Empty();
170  for (int i = 0; i < Joints.Num(); i++)
171  {
172  FSkeletonJoint& Joint = Joints[i];
173  if (!Joint.ChildrenIds.Num())
174  {
175  EndJoints.Add(i);

177  }
178 
179  // build traversal order so that parent nodes are visited before any children
180  RootToEndOrder.Empty();
181  std::vector<int> JointsToVisit;
182  JointsToVisit.emplace_back(0); // root element in the hierarchy
183  RootToEndOrder.Add(0);
184  while (JointsToVisit.size())
185  {
186  FSkeletonJoint& Joint = Joints[JointsToVisit.back()];
187  JointsToVisit.pop_back();
188

191  RootToEndOrder.Add(ChildrenId);

193  }
194  }
195

197  EndToRootOrder.Empty();

426 // Compute a single joint properties (Center of Mass, Inertia, Forces and Torque)
427 void USpringBasedVegetationComponent::ComputePerJointProperties(
428  std::vector<FJointProperties>& JointLocalPropertiesList,
429  std::vector<FJointProperties>& JointPropertiesList)
430 {
431  TRACE_CPUPROFILER_EVENT_SCOPE(USpringBasedVegetationComponent::ComputePerJointProperties);
432  for(FSkeletonJoint& Joint : Skeleton.Joints)
433  {
434  FJointProperties& Properties = JointLocalPropertiesList[Joint.JointId];
435  Eigen::Vector3d JointGlobalPosition = ToEigenVector(Joint.GlobalTransform.GetLocation())/100.f;
436  Properties.JointToGlobalMatrix = ToEigenMatrix(Joint.GlobalTransform);
437  JointPropertiesList[Joint.JointId].JointToGlobalMatrix = Properties.JointToGlobalMatrix;
438  if (!Joint.Bones.Num())
439  continue;
440  // COM and mass
441  for (FSkeletonBone& Bone : Joint.Bones)
442  {
443  Properties.Mass += Bone.Mass;
444  FVector GlobalCenterOfMass = Joint.GlobalTransform.TransformPositionNoScale(Bone.CenterOfMass);
445  Properties.CenterOfMass += Bone.Mass*ToEigenVector(GlobalCenterOfMass)/100.f;
446  }
447  Properties.CenterOfMass = Properties.CenterOfMass/Properties.Mass;
448 
449  // force
450  Eigen::Vector3d GravityForce = ToEigenVector(Gravity)/100.f; // world space gravity
451  Properties.Force = Properties.Mass * GravityForce + ToEigenVector(Joint.ExternalForces)/100.f;
452  // torque
453  Properties.Torque = (Properties.CenterOfMass - JointGlobalPosition).cross(Properties.Force);
454  // inertia tensor
455  for (FSkeletonBone& Bone : Joint.Bones)
456  {
457  if (Bone.Length < 1)
458  continue;
459  float CylinderRadius = 0.1f;
460  float CylinderHeight = Bone.Length/100.f;
461  Eigen::Matrix3d LocalCylinderInertia;
462  LocalCylinderInertia <<
463  0.5 * Bone.Mass*CylinderRadius*CylinderRadius, 0.f, 0.f,
464  0.f, 1.f/12.f * Bone.Mass*(3*CylinderRadius*CylinderRadius + CylinderHeight*CylinderHeight), 0.f,
465  0.f, 0.f, 1.f/12.f * Bone.Mass*(3*CylinderRadius*CylinderRadius + CylinderHeight*CylinderHeight);
466  Eigen::Vector3d BoneVector = ToEigenVector(Bone.CenterOfMass)/100.f;
467  Eigen::Vector3d LocalV1 = BoneVector.normalized();
468  Eigen::Vector3d LocalV2 = LocalV1.cross(Eigen::Vector3d(0,1,0));
469  if (LocalV2.norm() == 0)
470  LocalV2 = LocalV1.cross(Eigen::Vector3d(0,0,1));
471  LocalV2.normalize();
472  Eigen::Vector3d LocalV3 = LocalV1.cross(LocalV2);
473  Eigen::Matrix3d LocalToJointMatrix;
474  LocalToJointMatrix << LocalV1, LocalV2, LocalV3;
475  Eigen::Matrix3d JointSpaceInertia = LocalToJointMatrix*LocalCylinderInertia*(LocalToJointMatrix.transpose());
476  Properties.InertiaTensor += Properties.JointToGlobalMatrix*JointSpaceInertia*Properties.JointToGlobalMatrix.transpose();
477  Eigen::Vector3d BoneCenterWorldSpace = ToEigenVector(Joint.GlobalTransform.TransformPositionNoScale(Bone.CenterOfMass))/100.0;
478  Properties.InertiaTensor += Properties.Mass*OuterProduct(Properties.CenterOfMass - BoneCenterWorldSpace, Properties.CenterOfMass - BoneCenterWorldSpace);
479  }
480  ACC_LOG(Log, "Local Joint: %s \n Inertia \n %s \n Force \n %s \n Torque \n %s \n COM: \n %s \n Mass %f", *Joint.JointName, *EigenToFString(Properties.InertiaTensor), *EigenToFString(Properties.Force), *EigenToFString(Properties.Torque), *EigenToFString(Properties.CenterOfMass), Properties.Mass);
481  }
482 }

484 // Compute accumulated properties (Center of Mass, Inertia, Forces and Torque)
485 void USpringBasedVegetationComponent::ComputeCompositeBodyContribution(
486  std::vector<FJointProperties>& JointLocalPropertiesList,
487  std::vector<FJointProperties>& JointPropertiesList)
488 {
489  TRACE_CPUPROFILER_EVENT_SCOPE(USpringBasedVegetationComponent::ComputeCompositeBodyContribution);
490  for (int Id : Skeleton.EndToRootOrder)
491  {
492  FSkeletonJoint& Joint = Skeleton.Joints[Id];
493  FJointProperties& JointProperties = JointPropertiesList[Joint.JointId];
494  FJointProperties& JointLocalProperties = JointLocalPropertiesList[Joint.JointId];
495 
496  if (!Joint.ChildrenIds.Num())
497  {
498  continue;
499  }
500 
501  Eigen::Vector3d CenterOfMass = JointLocalProperties.Mass*JointLocalProperties.CenterOfMass;
502  // compute COM and accumulate mass
503  JointProperties.Mass = JointLocalProperties.Mass;
504  for(int ChildrenId : Joint.ChildrenIds)
505  {
506  FSkeletonJoint& ChildrenJoint = Skeleton.Joints[ChildrenId];
507  FJointProperties& ChildrenProperties = JointPropertiesList[ChildrenId];
508  CenterOfMass += ChildrenProperties.Mass*ChildrenProperties.CenterOfMass;
509  JointProperties.Mass += ChildrenProperties.Mass;
510  }
511  JointProperties.CenterOfMass = CenterOfMass / JointProperties.Mass;
512 
513  // compute forces
514  JointProperties.Force = JointLocalProperties.Force;
515  for(int ChildrenId : Joint.ChildrenIds)
516  {
517  FSkeletonJoint& ChildrenJoint = Skeleton.Joints[ChildrenId];
518  FJointProperties& ChildrenProperties = JointPropertiesList[ChildrenId];
519  JointProperties.Force += ChildrenProperties.Force;
520  }
521 
522  // compute torque
523  JointProperties.Torque = JointLocalProperties.Torque;
524  for(int ChildrenId : Joint.ChildrenIds)
525  {
526  FSkeletonJoint& ChildrenJoint = Skeleton.Joints[ChildrenId];
527  FJointProperties& ChildrenProperties = JointPropertiesList[ChildrenId];
528  Eigen::Vector3d ChildrenJointGlobalPosition = ToEigenVector(ChildrenJoint.GlobalTransform.GetLocation())/100.f;
529  Eigen::Vector3d ParentJointGlobalPosition = ToEigenVector(Joint.GlobalTransform.GetLocation())/100.f;
530  JointProperties.Torque += ChildrenProperties.Torque + (ChildrenJointGlobalPosition - ParentJointGlobalPosition).cross(ChildrenProperties.Force);
531  }
532 
533  // compute inertia tensor
534  JointProperties.InertiaTensor = JointLocalProperties.InertiaTensor + JointLocalProperties.Mass*OuterProduct(JointProperties.CenterOfMass - JointLocalProperties.CenterOfMass, JointProperties.CenterOfMass - JointLocalProperties.CenterOfMass);
535  for(int ChildrenId : Joint.ChildrenIds)
536  {
537  FSkeletonJoint& ChildrenJoint = Skeleton.Joints[ChildrenId];
538  FJointProperties& ChildrenProperties = JointPropertiesList[ChildrenId];
539  Eigen::Vector3d ChildrenToCenterOfMass = JointProperties.CenterOfMass - ChildrenProperties.CenterOfMass;
540 
541  JointProperties.InertiaTensor += ChildrenProperties.InertiaTensor + ChildrenProperties.Mass*OuterProduct(ChildrenToCenterOfMass, ChildrenToCenterOfMass);
542  }
543  ACC_LOG(Log, "Accumulated Joint: %s \n Inertia \n %s \n Force \n %s \n Torque \n %s \n COM: \n %s \n Mass %f", *Joint.JointName, *EigenToFString(JointProperties.InertiaTensor), *EigenToFString(JointProperties.Force), *EigenToFString(JointProperties.Torque), *EigenToFString(JointProperties.CenterOfMass), JointProperties.Mass);

564  Eigen::Matrix3d GlobalToJointMatrix = JointToGlobalMatrix.transpose();
565  Eigen::Matrix3d ParentJointToGlobalMatrix = ParentProperties.JointToGlobalMatrix;
566  Eigen::Matrix3d GlobalToParentJointMatrix = ParentJointToGlobalMatrix.transpose();
567  Eigen::Vector3d JointGlobalPosition = ToEigenVector(Joint.GlobalTransform.GetLocation())/100.f;
568  Eigen::Vector3d ParentJointGlobalPosition = ToEigenVector(ParentJoint.GlobalTransform.GetLocation())/100.f;
569 
570  JointProperties.AngularVelocity = ParentProperties.AngularVelocity
571  + ParentJointToGlobalMatrix*RotatorToEigenVector(ParentJoint.AngularVelocity);
572  JointProperties.LinearVelocity = ParentProperties.LinearVelocity
573  + ParentProperties.AngularVelocity.cross(JointGlobalPosition - ParentJointGlobalPosition);
574  JointProperties.AngularAcceleration = ParentProperties.AngularAcceleration
575  + ParentJointToGlobalMatrix*RotatorToEigenVector(ParentJoint.AngularAcceleration)
576  + ParentProperties.AngularVelocity.cross(JointProperties.AngularVelocity);
577  JointProperties.LinearAcceleration = ParentProperties.LinearAcceleration
578  + ParentProperties.LinearAcceleration.cross(JointGlobalPosition - ParentJointGlobalPosition)
579  + ParentProperties.AngularVelocity.cross(JointProperties.AngularVelocity - ParentProperties.AngularVelocity);
580 
581  Eigen::Vector3d CompositeCenterOfMass = JointProperties.CenterOfMass;
582  Eigen::Vector3d CompositeLinearVelocity = JointProperties.LinearVelocity
583  + JointProperties.AngularVelocity.cross(CompositeCenterOfMass - JointGlobalPosition);
584  Eigen::Vector3d CompositeLinearAcceleration = JointProperties.LinearAcceleration
585  + JointProperties.AngularAcceleration.cross(CompositeCenterOfMass - JointGlobalPosition)
586  + JointProperties.AngularVelocity.cross(CompositeLinearVelocity - JointProperties.LinearVelocity);
587 
588  Eigen::Vector3d TotalForces = -Joint.Mass * JointProperties.LinearAcceleration;
589  Eigen::Vector3d TotalTorque = (CompositeCenterOfMass - JointGlobalPosition).cross(TotalForces);
590  JointProperties.FictitiousTorque = GlobalToJointMatrix*TotalTorque;
591 
592  FICT_LOG(Log, "Joint: %s \n Position \n %s \n Velocity \n %s \n Acceleration \n %s \n Fictitious forces: \n %s", *Joint.JointName, *EigenToFString(CompositeCenterOfMass), *EigenToFString(CompositeLinearVelocity), *EigenToFString(CompositeLinearAcceleration), *EigenToFString(JointProperties.FictitiousTorque));
593  }

596 void USpringBasedVegetationComponent::ResolveContactsAndCollisions(
597  std::vector<FJointProperties>& JointLocalPropertiesList,
598  std::vector<FJointProperties>& JointPropertiesList)

602  // set all joints that can rest
603  for (auto &Joint : JointCollisionList)
604  {
605  Joint.CanRest = true;
606  }
607 
608  for (auto& ActorCapsules : OverlappingActors)
609  {
610  TRACE_CPUPROFILER_EVENT_SCOPE(ActorLoop);
611  AActor* CollidingActor = ActorCapsules.Key;
612  if (!IsValid(CollidingActor))
613  continue;
614  UPrimitiveComponent* Primitive = Cast<UPrimitiveComponent>(CollidingActor->GetRootComponent());
615  if (!IsValid(Primitive))
616  continue;
617  // force transferring momentum (for the initial collision frame)
618  const FVector PrimitiveVelocity = Primitive->GetComponentVelocity();
619  const Eigen::Vector3d ColliderVelocity = ToEigenVector(PrimitiveVelocity) / 100.f;
620  const FVector Impulse = (Primitive->GetMass() * PrimitiveVelocity);
621  const Eigen::Vector3d CollisionImpulse = 0*ToEigenVector(Impulse) / 100.f;
622  TArray<UPrimitiveComponent*>& CollidingCapsules = ActorCapsules.Value;
623  for (UPrimitiveComponent* Capsule : CollidingCapsules)
624  {
625  TRACE_CPUPROFILER_EVENT_SCOPE(CapsuleLoop);

632  float DistanceOnCapsule;
633  FHitResult HitResult;
634  bool HitFound;

636  float DistanceToCollider;

638  TRACE_CPUPROFILER_EVENT_SCOPE(ColliderPenetrationDistance);
639  DistanceOnCapsule = Capsule->GetClosestPointOnCollision(PrimitiveLocation, ClosestPointOnCapsule);

642  FVector LineTraceEnd = ClosestPointOnCapsule;

644  LineTraceStart, LineTraceEnd, FCollisionQueryParams());

647  if (DebugEnableVisualization)

650  }
651  }

655  }
656 
657  if (DebugEnableVisualization)
658  {

660  DrawDebugLine(GetWorld(), ClosestPointOnCapsule, ClosestPointOnCollider, FColor::Green, false, 0.1f, 0.0f, 1.f);

665  DrawDebugSphere(GetWorld(), PrimitiveLocation, DEBUG_SPHERE_SIZE, 64, FColor(255, 255, 255, 255));

668  const int JointId = CapsuleToJointId[Capsule];

670  FJointProperties& JointProperties = JointLocalPropertiesList[Joint.JointId];
671  FJointCollision& JointCollision = JointCollisionList[Joint.JointId];

682  GetDeltaRotator(CurrRotator, RestRotator);

684  const Eigen::Vector3d JointCapsuleVector = JointGlobalPosition - CapsulePosition;
685  const Eigen::Vector3d RepulsionForce = SpringTorque.cross(JointCapsuleVector) * JointCapsuleVector.squaredNorm();

688  Primitive->AddForceAtLocation(RepulsionForceUE, ClosestPointOnCollider);

691  float ForceFactor = 1.f;
692  // from eq f = 1 - a*d^p, f: ForceFactor, a: ProportionalConstant, p: ForceDistanceFalloffExponent, d: DistanceOnCapsule

694  // ForceFactor = 1.f - ProportionalConstant * FMath::Pow(DistanceOnCapsule, ForceDistanceFalloffExponent);

708  //UE_LOG(LogCarla, Display, TEXT("DistanceToCollider: %f, ForceFactor: %f"), DistanceToCollider, ForceFactor);
709 
710  // block forces to go to rest angles
711  int TempId = JointId;
712  do
713  {

715  TempJointCollision.CanRest = false;
716 
717  FSkeletonJoint &TempJoint = Skeleton.Joints[TempId];
718  TempId = TempJoint.ParentId;
719  }

722  if (DebugEnableVisualization)

731  FVector UEOverlapForces = ToUnrealVector(OverlappingForces)*100.f;
732  DrawDebugLine(GetWorld(), ClosestPointOnCapsule, ClosestPointOnCapsule+UEOverlapForces.GetSafeNormal()*20.f, FColor::Turquoise, false, 0.1f, 0.0f, 1.f);
733  FVector UECOM = ToUnrealVector(JointProperties.CenterOfMass )*100.f;
734  DrawDebugSphere(GetWorld(), UECOM, DEBUG_SPHERE_SIZE, 64, FColor::Emerald);
735  FVector UEJointPos = ToUnrealVector(JointGlobalPosition )*100.f;
736  DrawDebugSphere(GetWorld(), UEJointPos, DEBUG_SPHERE_SIZE, 64, FColor::Purple);
737  DrawDebugLine(GetWorld(), ClosestPointOnCapsule, UEJointPos, FColor::Cyan, false, 0.1f, 0.0f, 1.f);
738  }
739  }
740  }
741 
742  // reset iteration
743  for (auto &Joint : JointCollisionList)
744  {
745  if (Joint.CanRest)
746  {
747  if (Joint.Iteration > 1)
748  {
749  --Joint.Iteration;
750  }
751  }
752  else
753  {
754  if (Joint.Iteration <= 95)
755  {
756  Joint.Iteration += 5;
757  }
758  }

762 void USpringBasedVegetationComponent::SolveEquationOfMotion(
763  std::vector<FJointProperties>& JointPropertiesList,
764  float DeltaTime)
765 {
766  TRACE_CPUPROFILER_EVENT_SCOPE(USpringBasedVegetationComponent::SolveEquationOfMotion);
767  // solver
768  for (FSkeletonJoint& Joint : Skeleton.Joints)
769  {
770  if (!Joint.Bones.Num() || Joint.bIsStatic)
771  {
772  continue;
773  }
774  FJointProperties& JointProperties = JointPropertiesList[Joint.JointId];
775  FJointCollision& JointCollision = JointCollisionList[Joint.JointId];
776 
777  // debug drawing
778  if (DebugEnableVisualization)
779  {
780  if (Joint.ParentId != -1)
781  {
782  FVector Start = Joint.GlobalTransform.GetLocation();
783  FVector End = Skeleton.Joints[Joint.ParentId].GlobalTransform.GetLocation();
784  if (!JointCollision.CanRest)
785  {
786  DrawDebugLine(GetWorld(), Start, End, FColor::Red, false, 0.3f, 0.0f, 1.f);
787  }
788  else
789  {
790  DrawDebugLine(GetWorld(), Start, End, FColor::Blue, false, 0.1f, 0.0f, 1.f);
791  }
792  }
793  }
794 
795  float Mass = JointProperties.Mass;
796  Eigen::Vector3d CenterToJoint = JointProperties.CenterOfMass - ToEigenVector(Joint.GlobalTransform.GetLocation())/100.f;
797  Eigen::Matrix3d GlobalToJointMatrix = JointProperties.JointToGlobalMatrix.transpose();
798  Eigen::Matrix3d JointSpaceIntertiaTensor = JointProperties.Mass*OuterProduct(CenterToJoint, CenterToJoint)
799  + GlobalToJointMatrix*JointProperties.InertiaTensor*GlobalToJointMatrix.transpose();
800  Eigen::Vector3d Torque = GlobalToJointMatrix*JointProperties.Torque + JointProperties.FictitiousTorque;
801  // build differential equation
802  Eigen::Matrix3d I = JointSpaceIntertiaTensor;
803  float SpringStrength = Joint.SpringStrength;
804  float beta = Beta;
805  float alpha = Alpha;
806  Eigen::Matrix3d K;
807  K << SpringStrength*SpringStrengthMulFactor.X,0.f,0.f,
808  0.f,SpringStrength*SpringStrengthMulFactor.Y,0.f,
809  0.f,0.f,SpringStrength*SpringStrengthMulFactor.Z;
810  Eigen::LLT<Eigen::Matrix3d> lltofI (I);
811  Eigen::Matrix3d L = lltofI.matrixL();
812  Eigen::Matrix3d Linv = L.inverse();
813  Eigen::EigenSolver<Eigen::Matrix3d> EigenDecomposition(Linv*K*Linv.transpose());
814  Eigen::Matrix3d Lambda = EigenDecomposition.eigenvalues().real().asDiagonal();
815  Eigen::Matrix3d X = EigenDecomposition.eigenvectors().real();
816  // Eigen::Matrix3d Lambda = U.transpose()*K*U;
817  Eigen::Matrix3d U = Linv.transpose()*X;
818  // Eigen::Matrix3d Uinv = U.inverse();
819  Eigen::Matrix3d Uinv = X.transpose()*L.transpose();
820  Eigen::Vector3d Coeffsb = Eigen::Vector3d(alpha,alpha,alpha) + beta*Lambda*Eigen::Vector3d(1,1,1);
821  Eigen::Vector3d Coeffsk = Lambda*Eigen::Vector3d(1,1,1);
822  Eigen::Vector3d Coeffsf = U.transpose()*Torque;
823  FString StringI = EigenToFString(I);
824  FString StringTorque = EigenToFString(Torque);
825  FString StringU = EigenToFString(U);
826  FString StringLambda = EigenToFString(Lambda);
827  FString StringGlobalToJoint = EigenToFString(GlobalToJointMatrix);
828  FString StringCenterToJoint = EigenToFString(CenterToJoint);
829  SOLVER_LOG(Log, "Bone %s: \n I: \n %s \n Tau: \n %s \n U: \n %s \n Lambda: \n %s \n X: \n %s \n GlobalToJoint \n %s \n CenterToJoint \n %s", *Joint.JointName, *StringI, *StringTorque, *StringU, *StringLambda, *EigenToFString(X), *StringGlobalToJoint, *StringCenterToJoint);
830 
831  FRotator CurrRotator = Joint.Transform.Rotator();
832  FRotator RestRotator = Joint.RestingAngles;
833  FRotator AngularVelocity = Joint.AngularVelocity;
834  FRotator DeltaRotator =
835  GetDeltaRotator(CurrRotator, RestRotator);
836  if (!JointCollision.CanRest)
837  {
838  float Factor = 1.0f - ((JointCollision.Iteration / 100.0f) * RestFactor);
839  // DeltaRotator *= Factor;
840  AngularVelocity *= Factor;
841  }
842  Eigen::Vector3d InitialTheta = Uinv*RotatorToEigenVector(DeltaRotator);
843  Eigen::Vector3d InitialThetaVelocity = Uinv*RotatorToEigenVector(AngularVelocity);
844  SOLVER_LOG(Log, "Old angle for joint %s, %s", *Joint.JointName, *CurrRotator.ToString());
845  SOLVER_LOG(Log, "InitialTheta \n %s", *EigenToFString(InitialTheta));
846  Eigen::Vector3d NewTheta (0.f,0.f,0.f);
847  Eigen::Vector3d NewThetaVelocity (0.f,0.f,0.f);
848  Eigen::Vector3d NewThetaAccel (0.f,0.f,0.f);
849  Eigen::Vector3d Discriminant = Coeffsb.cwiseProduct(Coeffsb) - 4*Coeffsk;
850  // 1st row
851  for (int i = 0; i < 3; i++)
852  {
853  double b = Coeffsb(i);
854  double k = Coeffsk(i);
855  double f = Coeffsf(i);
856  double discriminant = Discriminant(i);
857  double theta0 = InitialTheta(i);
858 
859  SOLVER_LOG(Log, "Bone %s, component %d, b %f, k %f, f %f, discriminant %f, theta0 %f", *Joint.JointName, i, b, k, f, discriminant, theta0);
860  if(k == 0)
861  {
862  NewTheta(i) = theta0;
863  SOLVER_LOG(Log, "In Bone %s, K is 0 in component %d", *Joint.JointName, i);
864  }
865  double dtheta0 = InitialThetaVelocity(i); // compute
866  double deltatheta = 0;
867  double angularvelocity = 0;
868  float angularaccel = 0;
869  if (discriminant > 0)
870  {
871  double r1 = (-b + FMath::Sqrt(discriminant))*0.5f;
872  double r2 = (-b - FMath::Sqrt(discriminant))*0.5f;
873  double c1 = (r2*(theta0 - f/k) - dtheta0)/(r2-r1);
874  double c2 = (dtheta0 - c1*r1)/r2;
875  deltatheta = c1*std::exp(r1*DeltaTime) + c2*std::exp(r2*DeltaTime) + f/k;
876  angularvelocity = c1*r1*std::exp(r1*DeltaTime) + c2*r2*std::exp(r2*DeltaTime);
877  SOLVER_LOG(Log, "r1 %f, r2 %f, c1 %f, c2 %f, deltatheta %f", r1, r2, c1, c2, deltatheta);
878  }
879  else if (discriminant == 0)
880  {
881  double r = -b/2.f;
882  double c1 = theta0 - f/k;
883  double c2 = dtheta0 - c1*r;
884  deltatheta = (c1 + c2*DeltaTime)*std::exp(r*DeltaTime) + f/k;
885  angularvelocity = (c1*r + c2 + c2*r*DeltaTime)*std::exp(r*DeltaTime);
886  SOLVER_LOG(Log, "r %f, c1 %f, c2 %f, deltatheta %f", r, c1, c2, deltatheta);
887  }
888  else
889  {
890  double gamma = -b/2.f;
891  double mu = FMath::Sqrt(FMath::Abs(discriminant))/2.f;
892  double c1 = theta0 - f/k;
893  double c2 = (c1*gamma - dtheta0)/mu;
894  deltatheta =
895  c1*std::exp(gamma*DeltaTime)*std::cos(mu*DeltaTime) +
896  c2*std::exp(gamma*DeltaTime)*std::sin(mu*DeltaTime) + f/k;
897  angularvelocity =
898  c1*std::exp(gamma*DeltaTime)*(gamma*std::cos(mu*DeltaTime) + mu*std::sin(mu*DeltaTime)) +
899  c2*std::exp(gamma*DeltaTime)*(gamma*std::sin(mu*DeltaTime) - mu*std::cos(mu*DeltaTime));
900  SOLVER_LOG(Log, "gamma %f, mu %f, c1 %f, c2 %f, deltatheta %f", gamma, mu, c1, c2, deltatheta);
901  }
902  angularaccel = f - b*angularvelocity - k*deltatheta;
903  if (!FMath::IsNaN(deltatheta))
904  {
905  NewTheta(i) = deltatheta;
906  if (angularvelocity > 1e-4)
907  {
908  NewThetaVelocity(i) = angularvelocity;
909  }
910  if (angularaccel > 1e-2)
911  {
912  NewThetaAccel(i) = angularaccel;
913  }
914  }
915  }
916  Eigen::Vector3d FinalNewTheta = U*NewTheta;
917  Eigen::Vector3d FinalNewThetaVelocity = U*NewThetaVelocity;
918  Eigen::Vector3d FinalNewThetaAccel = U*NewThetaAccel;
919 
920  auto NewPitch = FMath::RadiansToDegrees(FinalNewTheta(1));
921  auto NewYaw = FMath::RadiansToDegrees(FinalNewTheta(2));
922  auto NewRoll = FMath::RadiansToDegrees(FinalNewTheta(0));
923 
924  FRotator NewAngularVelocity = EigenVectorToRotator(FinalNewThetaVelocity);
925  FRotator NewAngularAccel = EigenVectorToRotator(FinalNewThetaAccel);
926 
927  FRotator NewAngle(
928  RestRotator.Pitch + NewPitch,
929  RestRotator.Yaw - NewYaw,
930  RestRotator.Roll + NewRoll);
931  SOLVER_LOG(Log, "FinalNewTheta \n %s \n FinalNewThetaVelocity \n %s \n FinalNewThetaAccel \n %s",
932  *EigenToFString(FinalNewTheta), *EigenToFString(FinalNewThetaVelocity), *EigenToFString(FinalNewThetaAccel));
933  SOLVER_LOG(Log, "New angle %s, new angular velocity %s, new angular accel %s",
934  *NewAngle.ToString(), *NewAngularVelocity.ToString(), *NewAngularAccel.ToString());
935  Joint.Transform.SetRotation(NewAngle.Quaternion());
936  Joint.AngularVelocity = NewAngularVelocity;
937  Joint.AngularAcceleration = NewAngularAccel;
938  }
