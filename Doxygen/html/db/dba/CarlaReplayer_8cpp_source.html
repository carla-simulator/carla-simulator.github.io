246 void CarlaReplayer::ProcessToTime(double Time, bool IsFirstTime)
247 {
248  double Per = 0.0f;
249  double NewTime = CurrentTime + Time;
250  bool bFrameFound = false;
251  bool bExitAtNextFrame = false;
252  bool bExitLoop = false;
253 
254  // check if we are in the right frame
255  if (NewTime >= Frame.Elapsed && NewTime < Frame.Elapsed + Frame.DurationThis)
256  {
257  Per = (NewTime - Frame.Elapsed) / Frame.DurationThis;
258  bFrameFound = true;
259  bExitLoop = true;
260  }
261 
262  // process all frames until time we want or end
263  while (!File.eof() && !bExitLoop)
264  {
265  // get header
266  ReadHeader();
267 
268  // check for a frame packet
269  switch (Header.Id)
270  {
271  // frame
272  case static_cast<char>(CarlaRecorderPacketId::FrameStart):
273  // only read if we are not in the right frame
274  Frame.Read(File);
275  // check if target time is in this frame
276  if (NewTime < Frame.Elapsed + Frame.DurationThis)
277  {
278  Per = (NewTime - Frame.Elapsed) / Frame.DurationThis;
279  bFrameFound = true;
280  }
281  break;
282 
283  // events add
284  case static_cast<char>(CarlaRecorderPacketId::EventAdd):
285  ProcessEventsAdd();
286  break;
287 
288  // events del
289  case static_cast<char>(CarlaRecorderPacketId::EventDel):
290  ProcessEventsDel();
291  break;
292 
293  // events parent
294  case static_cast<char>(CarlaRecorderPacketId::EventParent):
295  ProcessEventsParent();
296  break;
297 
298  // collisions
299  case static_cast<char>(CarlaRecorderPacketId::Collision):
300  SkipPacket();
301  break;
302 
303  // positions
304  case static_cast<char>(CarlaRecorderPacketId::Position):
305  if (bFrameFound)
306  ProcessPositions(IsFirstTime);
307  else
308  SkipPacket();
309  break;
310 
311  // states
312  case static_cast<char>(CarlaRecorderPacketId::State):
313  if (bFrameFound)
314  ProcessStates();
315  else
316  SkipPacket();
317  break;
318 
319  // vehicle animation
320  case static_cast<char>(CarlaRecorderPacketId::AnimVehicle):
321  if (bFrameFound)
322  ProcessAnimVehicle();
323  else
324  SkipPacket();
325  break;
326 
327  // walker animation
328  case static_cast<char>(CarlaRecorderPacketId::AnimWalker):
329  if (bFrameFound)
330  ProcessAnimWalker();
331  else
332  SkipPacket();
333  break;
334 
335  // frame end
336  case static_cast<char>(CarlaRecorderPacketId::FrameEnd):
337  if (bFrameFound)
338  bExitLoop = true;
339  break;
340 
341  // unknown packet, just skip
342  default:
343  // skip packet
344  SkipPacket();
345  break;
346 
347  }
348  }
349 
350  // update all positions
351  if (Enabled && bFrameFound)
352  {
353  UpdatePositions(Per, Time);
354  }
355 
356  // save current time
357  CurrentTime = NewTime;
358 
359  // stop replay?
360  if (CurrentTime >= TimeToStop)
361  {
362  // keep actors in scene and let them continue with autopilot
363  Stop(true);
364  }
365 }

397 
398  // actor reused from existing

525 void CarlaReplayer::UpdatePositions(double Per, double DeltaTime)
526 {
527  unsigned int i;
528  uint32_t NewFollowId = 0;
529  std::unordered_map<int, int> TempMap;
530 
531  // map the id of all previous positions to its index
532  for (i = 0; i < PrevPos.size(); ++i)
533  {
534  TempMap[PrevPos[i].DatabaseId] = i;
535  }
536 
537  // get the Id of the actor to follow
538  if (FollowId != 0)
539  {
540  auto NewId = MappedId.find(FollowId);
541  if (NewId != MappedId.end())
542  {
543  NewFollowId = NewId->second;
544  }
545  }
546 
547  // go through each actor and update
548  for (i = 0; i < CurrPos.size(); ++i)
549  {
550  // check if exist a previous position
551  auto Result = TempMap.find(CurrPos[i].DatabaseId);
552  if (Result != TempMap.end())
553  {
554  // check if time factor is high
555  if (TimeFactor >= 2.0)
556  // assign first position
557  InterpolatePosition(PrevPos[Result->second], CurrPos[i], 0.0, DeltaTime);
558  else
559  // interpolate
560  InterpolatePosition(PrevPos[Result->second], CurrPos[i], Per, DeltaTime);
561  }
562  else
563  {
564  // assign last position (we don't have previous one)
565  InterpolatePosition(CurrPos[i], CurrPos[i], 0.0, DeltaTime);
566  }
567 
568  // move the camera to follow this actor if required
569  if (NewFollowId != 0)
570  {
571  if (NewFollowId == CurrPos[i].DatabaseId)
572  Helper.SetCameraPosition(NewFollowId, FVector(-1000, 0, 500), FQuat::MakeFromEuler({0, -25, 0}));
573  }
574  }
575 }
