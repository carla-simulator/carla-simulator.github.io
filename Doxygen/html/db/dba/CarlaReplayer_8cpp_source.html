262 void CarlaReplayer::ProcessToTime(double Time, bool IsFirstTime)
263 {
264  double Per = 0.0f;
265  double NewTime = CurrentTime + Time;
266  bool bFrameFound = false;
267  bool bExitAtNextFrame = false;
268  bool bExitLoop = false;
269 
270  // check if we are in the right frame
271  if (NewTime >= Frame.Elapsed && NewTime < Frame.Elapsed + Frame.DurationThis)
272  {
273  Per = (NewTime - Frame.Elapsed) / Frame.DurationThis;
274  bFrameFound = true;
275  bExitLoop = true;
276  }
277 
278  // process all frames until time we want or end
279  while (!File.eof() && !bExitLoop)
280  {
281  // get header
282  ReadHeader();
283 
284  // check for a frame packet
285  switch (Header.Id)
286  {
287  // frame
288  case static_cast<char>(CarlaRecorderPacketId::FrameStart):
289  // only read if we are not in the right frame
290  Frame.Read(File);
291  // check if target time is in this frame
292  if (NewTime < Frame.Elapsed + Frame.DurationThis)
293  {
294  Per = (NewTime - Frame.Elapsed) / Frame.DurationThis;
295  bFrameFound = true;
296  }
297  break;
298 
299  // visual time for FX
300  case static_cast<char>(CarlaRecorderPacketId::VisualTime):
301  ProcessVisualTime();
302  break;
303 
304  // events add
305  case static_cast<char>(CarlaRecorderPacketId::EventAdd):
306  ProcessEventsAdd();
307  break;
308 
309  // events del
310  case static_cast<char>(CarlaRecorderPacketId::EventDel):
311  ProcessEventsDel();
312  break;
313 
314  // events parent
315  case static_cast<char>(CarlaRecorderPacketId::EventParent):
316  ProcessEventsParent();
317  break;
318 
319  // collisions
320  case static_cast<char>(CarlaRecorderPacketId::Collision):
321  SkipPacket();
322  break;
323 
324  // positions
325  case static_cast<char>(CarlaRecorderPacketId::Position):
326  if (bFrameFound)
327  ProcessPositions(IsFirstTime);
328  else
329  SkipPacket();
330  break;
331 
332  // states
333  case static_cast<char>(CarlaRecorderPacketId::State):
334  if (bFrameFound)
335  ProcessStates();
336  else
337  SkipPacket();
338  break;
339 
340  // vehicle animation
341  case static_cast<char>(CarlaRecorderPacketId::AnimVehicle):
342  if (bFrameFound)
343  ProcessAnimVehicle();
344  else
345  SkipPacket();
346  break;
347 
348  // vehicle wheels animation
349  case static_cast<char>(CarlaRecorderPacketId::AnimVehicleWheels):
350  if (bFrameFound)
351  ProcessAnimVehicleWheels();
352  else
353  SkipPacket();
354  break;
355 
356  // walker animation
357  case static_cast<char>(CarlaRecorderPacketId::AnimWalker):
358  if (bFrameFound)
359  ProcessAnimWalker();
360  else
361  SkipPacket();
362  break;
363 
364  // biker animation
365  case static_cast<char>(CarlaRecorderPacketId::AnimBiker):
366  if (bFrameFound)
367  ProcessAnimBiker();
368  else
369  SkipPacket();
370  break;
371 
372  // vehicle light animation
373  case static_cast<char>(CarlaRecorderPacketId::VehicleLight):
374  if (bFrameFound)
375  ProcessLightVehicle();
376  else
377  SkipPacket();
378  break;
379 
380  // scene lights animation
381  case static_cast<char>(CarlaRecorderPacketId::SceneLight):
382  if (bFrameFound)
383  ProcessLightScene();
384  else
385  SkipPacket();
386  break;
387 
388  // walker bones
389  case static_cast<char>(CarlaRecorderPacketId::WalkerBones):
390  if (bFrameFound)
391  ProcessWalkerBones();
392  else
393  SkipPacket();
394  break;
395 
396  // frame end
397  case static_cast<char>(CarlaRecorderPacketId::FrameEnd):
398  if (bFrameFound)
399  bExitLoop = true;
400  break;
401 
402  // unknown packet, just skip
403  default:
404  // skip packet
405  SkipPacket();
406  break;
407 
408  }
409  }
410 
411  // update all positions
412  if (Enabled && bFrameFound)
413  {
414  UpdatePositions(Per, Time);
415  }
416 
417  // save current time
418  CurrentTime = NewTime;
419 
420  // stop replay?
421  if (CurrentTime >= TimeToStop)
422  {
423  // keep actors in scene and let them continue with autopilot
424  Stop(true);
425  }
426 }

437 void CarlaReplayer::ProcessEventsAdd(void)
438 {
439  uint16_t i, Total;
440  CarlaRecorderEventAdd EventAdd;
441 
442  // process creation events
443  ReadValue<uint16_t>(File, Total);
444  for (i = 0; i < Total; ++i)
445  {
446  EventAdd.Read(File);
447 
448  // auto Result = CallbackEventAdd(
449  auto Result = Helper.ProcessReplayerEventAdd(
450  EventAdd.Location,
451  EventAdd.Rotation,
452  EventAdd.Description,
453  EventAdd.DatabaseId,
454  IgnoreHero,
455  IgnoreSpectator,
456  bReplaySensors);
457 
458  switch (Result.first)
459  {
460  // actor not created
461  case 0:
462  UE_LOG(LogCarla, Log, TEXT("actor could not be created"));
463  break;
464 
465  // actor created but with different id
466  case 1:
467  // mapping id (recorded Id is a new Id in replayer)
468  MappedId[EventAdd.DatabaseId] = Result.second;
469  break;
470 
471  // actor reused from existing
472  case 2:
473  // mapping id (say desired Id is mapped to what)
474  MappedId[EventAdd.DatabaseId] = Result.second;
475  break;
476 
477  // actor ignored (either Hero or Spectator)
478  case 3:
479  UE_LOG(LogCarla, Log, TEXT("ignoring actor from replayer (Hero or Spectator)"));
480  break;
481 
482  }
483 
484  // check to mark if actor is a hero vehicle or not
485  if (Result.first > 0 && Result.first < 3)
486  {
487  // init
488  IsHeroMap[Result.second] = false;
489  for (const auto &Item : EventAdd.Description.Attributes)
490  {
491  if (Item.Id == "role_name" && Item.Value == "hero")
492  {
493  // mark as hero
494  IsHeroMap[Result.second] = true;
495  break;
496  }
497  }
498  }
499  }

718 void CarlaReplayer::UpdatePositions(double Per, double DeltaTime)
719 {
720  unsigned int i;
721  uint32_t NewFollowId = 0;
722  std::unordered_map<int, int> TempMap;
723 
724  // map the id of all previous positions to its index
725  for (i = 0; i < PrevPos.size(); ++i)
726  {
727  TempMap[PrevPos[i].DatabaseId] = i;
728  }
729 
730  // get the Id of the actor to follow
731  if (FollowId != 0)
732  {
733  auto NewId = MappedId.find(FollowId);
734  if (NewId != MappedId.end())
735  {
736  NewFollowId = NewId->second;
737  }
738  }
739 
740  // go through each actor and update
741  for (auto &Pos : CurrPos)
742  {
743  // check if ignore this actor (hero) or the spectator (id == 1)
744  if (!(IgnoreHero && IsHeroMap[Pos.DatabaseId]) &&
745  !(IgnoreSpectator && Pos.DatabaseId == 1))
746  {
747  // check if exist a previous position
748  auto Result = TempMap.find(Pos.DatabaseId);
749  if (Result != TempMap.end())
750  {
751  // check if time factor is high
752  if (TimeFactor >= 2.0)
753  // assign first position
754  InterpolatePosition(PrevPos[Result->second], Pos, 0.0, DeltaTime);
755  else
756  // interpolate
757  InterpolatePosition(PrevPos[Result->second], Pos, Per, DeltaTime);
758  }
759  else
760  {
761  // assign last position (we don't have previous one)
762  InterpolatePosition(Pos, Pos, 0.0, DeltaTime);
763  }
764  }
765 
766  // move the camera to follow this actor if required
767  if (NewFollowId != 0)
768  {
769  if (NewFollowId == Pos.DatabaseId)
770  Helper.SetCameraPosition(NewFollowId, FVector(-1000, 0, 500), FQuat::MakeFromEuler({0, -25, 0}));
771  }
772  }
773 }
