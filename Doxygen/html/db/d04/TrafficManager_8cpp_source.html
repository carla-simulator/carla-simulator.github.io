216  void TrafficManager::ResetAllTrafficLights() {
217  const auto world_traffic_lights = world.GetActors()->Filter("*traffic_light*");
218 
219  std::vector<TLGroup> list_of_all_groups;
220  TLGroup tl_to_freeze;
221  std::vector<carla::ActorId> list_of_ids;
222  for (auto tl : *world_traffic_lights.get()) {
223  if (!(std::find(list_of_ids.begin(), list_of_ids.end(), tl->GetId()) != list_of_ids.end())) {
224  const TLGroup tl_group = boost::static_pointer_cast<cc::TrafficLight>(tl)->GetGroupTrafficLights();
225  list_of_all_groups.push_back(tl_group);
226  for (uint i=0u; i<tl_group.size(); i++) {
227  list_of_ids.push_back(tl_group.at(i).get()->GetId());
228  if(i!=0u) {
229  tl_to_freeze.push_back(tl_group.at(i));
230  }
231  }
232  }
233  }
234 
235  for (TLGroup& tl_group : list_of_all_groups) {
236  tl_group.front()->SetState(TLS::Green);
237  std::for_each(
238  tl_group.begin()+1, tl_group.end(),
239  [] (auto& tl) {tl->SetState(TLS::Red);});
240  }
241 
242  while (!CheckAllFrozen(tl_to_freeze)) {
243  for (auto& tln : tl_to_freeze) {
244  tln->SetState(TLS::Red);
245  tln->Freeze(true);
246  }
247  }
248  }
