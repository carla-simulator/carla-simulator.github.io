208  bool TrafficManager::CheckAllFrozen(TLGroup tl_to_freeze) {

217  void TrafficManager::ResetAllTrafficLights() {
218  const auto world_traffic_lights = world.GetActors()->Filter("*traffic_light*");
219 
220  std::vector<TLGroup> list_of_all_groups;
221  TLGroup tl_to_freeze;
222  std::vector<carla::ActorId> list_of_ids;
223  for (auto tl : *world_traffic_lights.get()) {
224  if (!(std::find(list_of_ids.begin(), list_of_ids.end(), tl->GetId()) != list_of_ids.end())) {
225  const TLGroup tl_group = boost::static_pointer_cast<cc::TrafficLight>(tl)->GetGroupTrafficLights();
226  list_of_all_groups.push_back(tl_group);
227  for (uint64_t i=0u; i<tl_group.size(); i++) {
228  list_of_ids.push_back(tl_group.at(i).get()->GetId());
229  if(i!=0u) {
230  tl_to_freeze.push_back(tl_group.at(i));
231  }
232  }
233  }
234  }
235 
236  for (TLGroup& tl_group : list_of_all_groups) {
237  tl_group.front()->SetState(TLS::Green);
238  std::for_each(
239  tl_group.begin()+1, tl_group.end(),
240  [] (auto& tl) {tl->SetState(TLS::Red);});
241  }
242 
243  while (!CheckAllFrozen(tl_to_freeze)) {
244  for (auto& tln : tl_to_freeze) {
245  tln->SetState(TLS::Red);
246  tln->Freeze(true);
247  }
248  }
249  }
