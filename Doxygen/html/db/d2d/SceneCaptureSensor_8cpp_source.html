128 void ASceneCaptureSensor::BeginPlay()
129 {
130  using namespace SceneCaptureSensor_local_ns;
131 
132  // Setup render target.
133 
134  // Determine the gamma of the player.
135 
136  const bool bInForceLinearGamma = !bEnablePostProcessingEffects;
137 
138  CaptureRenderTarget->InitCustomFormat(ImageWidth, ImageHeight, PF_B8G8R8A8, bInForceLinearGamma);
139 
140  if (bEnablePostProcessingEffects)
141  {
142  CaptureRenderTarget->TargetGamma = TargetGamma;
143  }
144 
145  check(IsValid(CaptureComponent2D) && !CaptureComponent2D->IsPendingKill());
146 
147  CaptureComponent2D->Deactivate();
148  CaptureComponent2D->TextureTarget = CaptureRenderTarget;
149 
150  // Call derived classes to set up their things.
151  SetUpSceneCaptureComponent(*CaptureComponent2D);
152 
153  if (bEnablePostProcessingEffects &&
154  (SceneCaptureSensor_local_ns::GetQualitySettings(GetWorld()) == EQualityLevel::Low))
155  {
156  CaptureComponent2D->CaptureSource = ESceneCaptureSource::SCS_SceneColorHDRNoAlpha;
157  }
158  else
159  {
160  // LDR is faster than HDR (smaller bitmap array).
161  CaptureComponent2D->CaptureSource = ESceneCaptureSource::SCS_FinalColorLDR;
162  }
163 
164  CaptureComponent2D->UpdateContent();
165  CaptureComponent2D->Activate();
166 
167  // Make sure that there is enough time in the render queue.
168  UKismetSystemLibrary::ExecuteConsoleCommand(
169  GetWorld(),
170  FString("g.TimeoutForBlockOnRenderFence 300000"));
171 
172  SceneCaptureSensor_local_ns::ConfigureShowFlags(CaptureComponent2D->ShowFlags,
173  bEnablePostProcessingEffects);
174 
175  Super::BeginPlay();
176 }
