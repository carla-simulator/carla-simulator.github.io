417 void ASceneCaptureSensor::BeginPlay()
418 {
419  using namespace SceneCaptureSensor_local_ns;
420 
421  // Setup render target.
422 
423  // Determine the gamma of the player.
424 
425  const bool bInForceLinearGamma = !bEnablePostProcessingEffects;
426 
427  CaptureRenderTarget->InitCustomFormat(ImageWidth, ImageHeight, PF_B8G8R8A8, bInForceLinearGamma);
428 
429  if (bEnablePostProcessingEffects)
430  {
431  CaptureRenderTarget->TargetGamma = TargetGamma;
432  }
433 
434  check(IsValid(CaptureComponent2D) && !CaptureComponent2D->IsPendingKill());
435 
436  CaptureComponent2D->Deactivate();
437  CaptureComponent2D->TextureTarget = CaptureRenderTarget;
438 
439  // Call derived classes to set up their things.
440  SetUpSceneCaptureComponent(*CaptureComponent2D);
441 
442  if (bEnablePostProcessingEffects &&
443  (SceneCaptureSensor_local_ns::GetQualitySettings(GetWorld()) == EQualityLevel::Low))
444  {
445  CaptureComponent2D->CaptureSource = ESceneCaptureSource::SCS_SceneColorHDRNoAlpha;
446  }
447  else
448  {
449  // LDR is faster than HDR (smaller bitmap array).
450  CaptureComponent2D->CaptureSource = ESceneCaptureSource::SCS_FinalColorLDR;
451  }
452 
453  CaptureComponent2D->UpdateContent();
454  CaptureComponent2D->Activate();
455 
456  // Make sure that there is enough time in the render queue.
457  UKismetSystemLibrary::ExecuteConsoleCommand(
458  GetWorld(),
459  FString("g.TimeoutForBlockOnRenderFence 300000"));
460 
461  SceneCaptureSensor_local_ns::ConfigureShowFlags(CaptureComponent2D->ShowFlags,
462  bEnablePostProcessingEffects);
463 
464  // This ensures the camera is always spawning the rain drops in case the
465  // weather was previously set to has rain
466  GetEpisode().GetWeather()->NotifyWeather();
467 
468  Super::BeginPlay();
469 }
