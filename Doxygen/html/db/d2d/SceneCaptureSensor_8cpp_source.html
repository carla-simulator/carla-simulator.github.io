462 void ASceneCaptureSensor::BeginPlay()
463 {
464  using namespace SceneCaptureSensor_local_ns;
465 
466  // Determine the gamma of the player.
467  const bool bInForceLinearGamma = !bEnablePostProcessingEffects;
468 
469  CaptureRenderTarget->InitCustomFormat(ImageWidth, ImageHeight, PF_B8G8R8A8, bInForceLinearGamma);
470 
471  if (bEnablePostProcessingEffects)
472  {
473  CaptureRenderTarget->TargetGamma = TargetGamma;
474  }
475 
476  check(IsValid(CaptureComponent2D) && !CaptureComponent2D->IsPendingKill());
477 
478  CaptureComponent2D->Deactivate();
479  CaptureComponent2D->TextureTarget = CaptureRenderTarget;
480 
481  // Call derived classes to set up their things.
482  SetUpSceneCaptureComponent(*CaptureComponent2D);
483 
484  CaptureComponent2D->CaptureSource = ESceneCaptureSource::SCS_FinalColorLDR;
485 
486  CaptureComponent2D->UpdateContent();
487  CaptureComponent2D->Activate();
488 
489  // Make sure that there is enough time in the render queue.
490  UKismetSystemLibrary::ExecuteConsoleCommand(
491  GetWorld(),
492  FString("g.TimeoutForBlockOnRenderFence 300000"));
493 
494  SceneCaptureSensor_local_ns::ConfigureShowFlags(CaptureComponent2D->ShowFlags,
495  bEnablePostProcessingEffects);
496 
497  // This ensures the camera is always spawning the raindrops in case the
498  // weather was previously set to have rain.
499  GetEpisode().GetWeather()->NotifyWeather();
500 
501  Super::BeginPlay();
502 }

504 void ASceneCaptureSensor::PrePhysTick(float DeltaSeconds)
