56  void Run(size_t number_of_messages) {
57  _threads.CreateThread([this]() { _client_callback.run(); });
58  _server.AsyncRun(_streams.size());
59  _client.AsyncRun(_streams.size());
60 
61  std::this_thread::sleep_for(1s); // the client needs to be ready so we make
62  // sure we get all the messages.
63 
64  for (auto &&stream : _streams) {
65  _threads.CreateThread([=]() mutable {
66  for (auto i = 0u; i < number_of_messages; ++i) {
67  std::this_thread::sleep_for(11ms); // ~90FPS.
68  {
69  CARLA_PROFILE_SCOPE(game, write_to_stream);
70  stream << _message.buffer();
71  }
72  }
73  });
74  }

76  const auto expected_number_of_messages = _streams.size() * number_of_messages;
77  const auto threshold =
78  static_cast<size_t>(_success_ratio * static_cast<double>(expected_number_of_messages));
79 
80  for (auto i = 0u; i < 10; ++i) {
81  std::cout << "received " << _number_of_messages_received
82  << " of " << expected_number_of_messages
83  << " messages,";
84  if (_number_of_messages_received >= expected_number_of_messages) {
85  break;

88  std::this_thread::sleep_for(1s);
89  }
