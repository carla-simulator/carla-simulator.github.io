20  template <typename StreamStateT, typename StreamMapT>
21  static auto MakeStreamState(const token_type &cached_token, StreamMapT &stream_map) {
22  auto ptr = std::make_shared<StreamStateT>(cached_token);
23  auto result = stream_map.emplace(std::make_pair(cached_token.get_stream_id(), ptr));
24  if (!result.second) {
25  throw_exception(std::runtime_error("failed to create stream!"));
26  }
27  return ptr;
28  }

30  Dispatcher::~Dispatcher() {
31  // Disconnect all the sessions from their streams, this should kill any
32  // session remaining since at this point the io_context should be already
33  // stopped.
34  for (auto &pair : _stream_map) {
35 #ifndef LIBCARLA_NO_EXCEPTIONS
36  try {
37 #endif // LIBCARLA_NO_EXCEPTIONS
38  auto stream_state = pair.second.lock();
39  if (stream_state != nullptr) {
40  stream_state->ClearSessions();
41  }
42 #ifndef LIBCARLA_NO_EXCEPTIONS
43  } catch (const std::exception &e) {
44  log_error("failed to clear sessions:", e.what());
45  }
46 #endif // LIBCARLA_NO_EXCEPTIONS
47  }
48  }

62  bool Dispatcher::RegisterSession(std::shared_ptr<Session> session) {
63  DEBUG_ASSERT(session != nullptr);
64  std::lock_guard<std::mutex> lock(_mutex);
65  auto search = _stream_map.find(session->get_stream_id());
66  if (search != _stream_map.end()) {
67  auto stream_state = search->second.lock();
68  if (stream_state != nullptr) {
69  stream_state->ConnectSession(std::move(session));
70  return true;
71  }
72  }
73  log_error("Invalid session: no stream available with id", session->get_stream_id());
74  return false;
75  }
