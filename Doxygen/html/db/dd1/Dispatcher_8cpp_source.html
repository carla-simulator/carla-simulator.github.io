19  Dispatcher::~Dispatcher() {
20  // Disconnect all the sessions from their streams, this should kill any
21  // session remaining since at this point the io_context should be already
22  // stopped.
23  for (auto &pair : _stream_map) {
24 #ifndef LIBCARLA_NO_EXCEPTIONS

26 #endif // LIBCARLA_NO_EXCEPTIONS

31  log_error("failed to clear sessions:", e.what());
32  }
33 #endif // LIBCARLA_NO_EXCEPTIONS
34  }
35  }

37  carla::streaming::Stream Dispatcher::MakeStream() {
38  std::lock_guard<std::mutex> lock(_mutex);
39  ++_cached_token._token.stream_id; // id zero only happens in overflow.
40  log_info("New stream:", _cached_token._token.stream_id);
41  std::shared_ptr<MultiStreamState> ptr;
42  auto search = _stream_map.find(_cached_token.get_stream_id());
43  if (search == _stream_map.end()) {
44  // creating new stream
45  ptr = std::make_shared<MultiStreamState>(_cached_token);
46  auto result = _stream_map.emplace(std::make_pair(_cached_token.get_stream_id(), ptr));
47  if (!result.second) {
48  throw_exception(std::runtime_error("failed to create stream!"));
49  }
50  log_info("Stream created");
51  return carla::streaming::Stream(ptr);
52  } else {
53  // reusing existing stream
54  log_info("Stream reused");
55  ptr = search->second;
56  return carla::streaming::Stream(ptr);
57  }
58  }

60  bool Dispatcher::RegisterSession(std::shared_ptr<Session> session) {
61  DEBUG_ASSERT(session != nullptr);
62  std::lock_guard<std::mutex> lock(_mutex);
63  auto search = _stream_map.find(session->get_stream_id());
64  if (search != _stream_map.end()) {
65  auto stream_state = search->second;
66  log_info("Connecting session (stream ", session->get_stream_id(), ")");
67  stream_state->ConnectSession(std::move(session));
68  return true;
69  }
70  log_error("Invalid session: no stream available with id", session->get_stream_id());
71  return false;
72  }

85  token_type Dispatcher::GetToken(stream_id_type sensor_id) {
86  std::lock_guard<std::mutex> lock(_mutex);
87  log_info("Searching sensor id: ", sensor_id);
88  auto search = _stream_map.find(sensor_id);
89  if (search != _stream_map.end()) {
90  log_info("Found sensor id: ", sensor_id);
91  auto stream_state = search->second;
92  log_info("Getting token from stream ", sensor_id, " on port ", stream_state->token().get_port());
93  return stream_state->token();
94  } else {
95  log_info("Not Found sensor id, creating sensor stream: ", sensor_id);
96  token_type temp_token(_cached_token);
97  temp_token.set_stream_id(sensor_id);
98  auto ptr = std::make_shared<MultiStreamState>(temp_token);
99  auto result = _stream_map.emplace(std::make_pair(temp_token.get_stream_id(), ptr));
100  if (!result.second) {
101  log_info("Failed to create multistream for stream ", sensor_id, " on port ", temp_token.get_port());
102  }
103  log_info("Created token from stream ", sensor_id, " on port ", temp_token.get_port());
104  return temp_token;
105  }
106  return token_type();
107  }
