124  template <typename MultiMapT>
125  static auto GetLessEqualRange(MultiMapT &map, double s) {
126  if (map.find(s) == map.end()) {
127  auto it = map.lower_bound(s);
128  if (it == map.begin()) {
129  return std::make_pair(map.end(), map.end());
130  }
131  s = (--it)->first;
132  }
133  return std::make_pair(map.lower_bound(s), map.upper_bound(s));
134  }
