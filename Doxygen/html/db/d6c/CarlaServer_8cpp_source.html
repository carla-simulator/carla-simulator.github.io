217  /// Looks for a Traffic Manager running on port
218  BIND_SYNC(is_traffic_manager_running) << [this] (uint16_t port) ->R<bool>
219  {
220  return (TrafficManagerInfo.find(port) != TrafficManagerInfo.end());
221  };
222 
223  /// Gets a pair filled with the <IP, port> of the Trafic Manager running on port.
224  /// If there is no Traffic Manager running the pair will be ("", 0)
225  BIND_SYNC(get_traffic_manager_running) << [this] (uint16_t port) ->R<std::pair<std::string, uint16_t>>
226  {
227  auto it = TrafficManagerInfo.find(port);
228  if(it != TrafficManagerInfo.end()) {
229  return std::pair<std::string, uint16_t>(it->second, it->first);
230  }
231  return std::pair<std::string, uint16_t>("",0);
232  };
233 
234  /// Add a new Traffic Manager running on <IP, port>
235  BIND_SYNC(add_traffic_manager_running) << [this] (std::pair<std::string, uint16_t> trafficManagerInfo) ->R<bool>
236  {
237  uint16_t port = trafficManagerInfo.second;
238  auto it = TrafficManagerInfo.find(port);
239  if(it == TrafficManagerInfo.end()) {
240  TrafficManagerInfo.insert(
241  std::pair<uint16_t, std::string>(port, trafficManagerInfo.first));
242  return true;
243  }
244  return false;
245 
246  };

248  BIND_SYNC(destroy_traffic_manager) << [this] (uint16_t port) ->R<bool>

250  auto it = TrafficManagerInfo.find(port);
251  if(it != TrafficManagerInfo.end()) {
252  TrafficManagerInfo.erase(it);
253  return true;
254  }
