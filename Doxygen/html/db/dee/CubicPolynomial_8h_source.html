96  /// Evaluates f(x) = a + bx + cx^2 + dx^3
97  value_type Evaluate(const value_type &x) const {
98  // return _v[0] + _v[1] * (x) + _v[2] * (x * x) + _v[3] * (x * x * x);
99  return _v[0] + x * (_v[1] + x * (_v[2] + x * _v[3]));
100  }

102  /// Evaluates the tangent using df/dx = b + 2cx + 3dx^2
103  value_type Tangent(const value_type &x) const {
104  return _v[1] + x * (2 * _v[2] + x * 3 * _v[3]);
105  }
