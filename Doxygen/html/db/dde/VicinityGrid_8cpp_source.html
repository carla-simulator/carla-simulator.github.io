22  std::pair<int, int> VicinityGrid::UpdateGrid(Actor actor) {
23 
24  const ActorId actor_id = actor->GetId();
25  const cg::Location location = actor->GetLocation();
26  const int first = static_cast<int>(std::floor(location.x / GRID_SIZE));
27  const int second = static_cast<int>(std::floor(location.y / GRID_SIZE));
28 
29  const std::string new_grid_id = MakeKey({first, second});
30 
31  // If the actor exists in the grid.
32  if (actor_to_grid_id.find(actor_id) != actor_to_grid_id.end()) {
33 
34  const std::string old_grid_id = actor_to_grid_id.at(actor_id);
35 
36  // If the actor has moved into a new road/section/lane.
37  if (old_grid_id != new_grid_id) {
38  std::unique_lock<std::shared_timed_mutex> lock(modification_mutex);
39 
40  // Update the actor's grid id.
41  actor_to_grid_id.at(actor_id) = new_grid_id;
42 
43  // Remove the actor from the old grid position and update the new position.
44  grid_to_actor_id.at(old_grid_id).erase(actor_id);
45  if (grid_to_actor_id.find(new_grid_id) != grid_to_actor_id.end()) {
46  grid_to_actor_id.at(new_grid_id).insert(actor_id);
47  } else {
48  grid_to_actor_id.insert({new_grid_id, {actor_id}});
49  }
50  }
51  }
52  // If the actor is new, then add entries to map.
53  else {
54 
55  std::unique_lock<std::shared_timed_mutex> lock(modification_mutex);
56  actor_to_grid_id.insert({actor_id, new_grid_id});
57  if (grid_to_actor_id.find(new_grid_id) != grid_to_actor_id.end()) {
58  grid_to_actor_id.at(new_grid_id).insert(actor_id);
59  } else {
60  grid_to_actor_id.insert({new_grid_id, {actor_id}});
61  }
62  }
63 
64  // Return updated grid position.
65  return {first, second};
66  }

68  std::unordered_set<carla::ActorId> VicinityGrid::GetActors(Actor actor) {
69 
70  const std::pair<int, int> grid_ids = UpdateGrid(actor);
71 
72  std::shared_lock<std::shared_timed_mutex> lock(modification_mutex);
73  std::unordered_set<ActorId> actors;
74 
75  // Search all surrounding grids and find any vehicles in them.
76  for (int i = -1; i <= 1; ++i) {
77  for (int j = -1; j <= 1; ++j) {
78 
79  const std::string grid_key = MakeKey({grid_ids.first + i, grid_ids.second + j});
80  if (grid_to_actor_id.find(grid_key) != grid_to_actor_id.end()) {
81  const auto &grid_actor_set = grid_to_actor_id.at(grid_key);
82  actors.insert(grid_actor_set.begin(), grid_actor_set.end());
83  }
84  }
85  }
86 
87  return actors;
88  }
