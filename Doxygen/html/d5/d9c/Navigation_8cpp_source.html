90  // load navigation data from memory
91  bool Navigation::Load(std::vector<uint8_t> content) {
92  const int NAVMESHSET_MAGIC = 'M' << 24 | 'S' << 16 | 'E' << 8 | 'T'; // 'MSET';
93  const int NAVMESHSET_VERSION = 1;
94 #pragma pack(push, 1)
95 
96  struct NavMeshSetHeader {
97  int magic;
98  int version;
99  int numTiles;
100  dtNavMeshParams params;
101  } header;
102  struct NavMeshTileHeader {
103  dtTileRef tileRef;
104  int dataSize;
105  };
106 #pragma pack(pop)
107 
108  // check size for header
109  if (content.size() < sizeof(header)) {
110  logging::log("Nav: failed loading binary");
111  return false;
112  }
113 
114  // read the file header
115  unsigned long pos = 0;
116  memcpy(&header, &content[pos], sizeof(header));
117  pos += sizeof(header);
118 
119  // check file magic and version
120  if (header.magic != NAVMESHSET_MAGIC || header.version != NAVMESHSET_VERSION) {
121  return false;
122  }
123 
124  // allocate object
125  dtNavMesh *mesh = dtAllocNavMesh();
126  if (!mesh) {
127  return false;
128  }
129 
130  // set number of tiles and origin
131  dtStatus status = mesh->init(&header.params);
132  if (dtStatusFailed(status)) {
133  return false;
134  }
135 
136  // read the tiles data
137  for (int i = 0; i < header.numTiles; ++i) {
138  NavMeshTileHeader tileHeader;
139 
140  // read the tile header
141  memcpy(&tileHeader, &content[pos], sizeof(tileHeader));
142  pos += sizeof(tileHeader);
143  if (pos >= content.size()) {
144  dtFreeNavMesh(mesh);
145  return false;
146  }
147 
148  // check for valid tile
149  if (!tileHeader.tileRef || !tileHeader.dataSize) {
150  break;
151  }
152 
153  // allocate the buffer
154  char *data = static_cast<char *>(dtAlloc(static_cast<size_t>(tileHeader.dataSize), DT_ALLOC_PERM));
155  if (!data) {
156  break;
157  }
158 
159  // read the tile
160  memcpy(data, &content[pos], static_cast<size_t>(tileHeader.dataSize));
161  pos += static_cast<unsigned long>(tileHeader.dataSize);
162  if (pos > content.size()) {
163  dtFree(data);
164  dtFreeNavMesh(mesh);
165  return false;
166  }
167 
168  // add the tile data
169  mesh->addTile(reinterpret_cast<unsigned char *>(data), tileHeader.dataSize, DT_TILE_FREE_DATA,
170  tileHeader.tileRef, 0);
171  }
172 
173  // exchange
174  dtFreeNavMesh(_navMesh);
175  _navMesh = mesh;
176 
177  // prepare the query object
178  dtFreeNavMeshQuery(_navQuery);
179  _navQuery = dtAllocNavMeshQuery();
180  _navQuery->init(_navMesh, MAX_QUERY_SEARCH_NODES);
181 
182  // copy
183  _binaryMesh = std::move(content);
184  _ready = true;
185 
186  // create and init the crowd manager
187  CreateCrowd();
188 
189  return true;
190  }

246  // return the path points to go from one position to another
247  bool Navigation::GetPath(carla::geom::Location from, carla::geom::Location to,
248  dtQueryFilter * filter, std::vector<carla::geom::Location> &path) {
249  // path found
250  float m_straightPath[MAX_POLYS * 3];
251  unsigned char m_straightPathFlags[MAX_POLYS];
252  dtPolyRef m_straightPathPolys[MAX_POLYS];
253  int m_nstraightPath;
254  int m_straightPathOptions = 0;
255  // polys in path
256  dtPolyRef m_polys[MAX_POLYS];
257  int m_npolys;
258 
259  // force single thread running this
260  std::lock_guard<std::mutex> lock(_mutex);
261 
262  // check if all is ready
263  if (!_ready) {
264  return false;
265  }
266 
267  DEBUG_ASSERT(_navQuery != nullptr);
268 
269  // point extension
270  float m_polyPickExt[3];
271  m_polyPickExt[0] = 2;
272  m_polyPickExt[1] = 4;
273  m_polyPickExt[2] = 2;
274 
275  // filter
276  dtQueryFilter filter2;
277  if (filter == nullptr) {
278  filter2.setIncludeFlags(SAMPLE_POLYFLAGS_ALL ^ SAMPLE_POLYFLAGS_DISABLED);
279  filter2.setExcludeFlags(0);
280  filter = &filter2;
281  }
282 
283  // set the points
284  dtPolyRef m_startRef = 0;
285  dtPolyRef m_endRef = 0;
286  float m_spos[3] = { from.x, from.z, from.y };
287  float m_epos[3] = { to.x, to.z, to.y };
288  _navQuery->findNearestPoly(m_spos, m_polyPickExt, filter, &m_startRef, 0);
289  _navQuery->findNearestPoly(m_epos, m_polyPickExt, filter, &m_endRef, 0);
290  if (!m_startRef || !m_endRef) {
291  return false;
292  }
293 
294  // get the path of nodes
295  _navQuery->findPath(m_startRef, m_endRef, m_spos, m_epos, filter, m_polys, &m_npolys, MAX_POLYS);
296 
297  // get the path of points
298  m_nstraightPath = 0;
299  if (m_npolys == 0) {
300  return false;
301  }
302 
303  // in case of partial path, make sure the end point is clamped to the last
304  // polygon
305  float epos[3];
306  dtVcopy(epos, m_epos);
307  if (m_polys[m_npolys - 1] != m_endRef) {
308  _navQuery->closestPointOnPoly(m_polys[m_npolys - 1], m_epos, epos, 0);
309  }
310 
311  // get the points
312  _navQuery->findStraightPath(m_spos, epos, m_polys, m_npolys,
313  m_straightPath, m_straightPathFlags,
314  m_straightPathPolys, &m_nstraightPath, MAX_POLYS, m_straightPathOptions);
315 
316  // copy the path to the output buffer
317  path.clear();
318  path.reserve(static_cast<unsigned long>(m_nstraightPath));
319  for (int i = 0; i < m_nstraightPath * 3; i += 3) {
320  // export for Unreal axis (x, z, y)
321  path.emplace_back(m_straightPath[i], m_straightPath[i + 2], m_straightPath[i + 1]);
322  }
323 
324  return true;
325  }

489  void Navigation::UpdateCrowd(const client::detail::EpisodeState &state) {
490 
491  // force single thread running this
492  std::lock_guard<std::mutex> lock(_mutex);
493 
494  // check if all is ready
495  if (!_ready) {
496  return;
497  }
498 
499  DEBUG_ASSERT(_crowd != nullptr);
500 
501  // update all
502  _delta_seconds = state.GetTimestamp().delta_seconds;
503  _crowd->update(static_cast<float>(_delta_seconds), nullptr);
504 
505  // check if walker has finished
506  for (int i = 0; i < _crowd->getAgentCount(); ++i) {
507  const dtCrowdAgent *ag = _crowd->getAgent(i);
508  if (!ag->active) {
509  continue;

511 
512  // check distance to the target point
513  const float *end = &ag->cornerVerts[(ag->ncorners - 1) * 3];
514  carla::geom::Vector3D dist(end[0] - ag->npos[0], end[1] - ag->npos[1], end[2] - ag->npos[2]);
515  if (dist.SquaredLength() <= 2) {
516  // set a new random target
517  carla::geom::Location location;
518  GetRandomLocation(location, 1, nullptr, false);
519  SetWalkerTargetIndex(i, location, false);

521  }
