49 void ARayCastSemanticLidar::CreateLasers()
50 {
51  const auto NumberOfLasers = Description.Channels;
52  check(NumberOfLasers > 0u);
53  const float DeltaAngle = NumberOfLasers == 1u ? 0.f :
54  (Description.UpperFovLimit - Description.LowerFovLimit) /
55  static_cast<float>(NumberOfLasers - 1);
56  LaserAngles.Empty(NumberOfLasers);
57  for(auto i = 0u; i < NumberOfLasers; ++i)
58  {
59  const float VerticalAngle =
60  Description.UpperFovLimit - static_cast<float>(i) * DeltaAngle;
61  LaserAngles.Emplace(VerticalAngle);
62  }
63 }

75 void ARayCastSemanticLidar::SimulateLidar(const float DeltaTime)
76 {
77  const uint32 ChannelCount = Description.Channels;
78  const uint32 PointsToScanWithOneLaser =
79  FMath::RoundHalfFromZero(
80  Description.PointsPerSecond * DeltaTime / float(ChannelCount));
81 
82  if (PointsToScanWithOneLaser <= 0)
83  {
84  UE_LOG(
85  LogCarla,
86  Warning,
87  TEXT("%s: no points requested this frame, try increasing the number of points per second."),
88  *GetName());
89  return;
90  }
91 
92  check(ChannelCount == LaserAngles.Num());
93 
94  const float CurrentHorizontalAngle = carla::geom::Math::ToDegrees(
95  SemanticLidarData.GetHorizontalAngle());
96  const float AngleDistanceOfTick = Description.RotationFrequency * 360.0f * DeltaTime;
97  const float AngleDistanceOfLaserMeasure = AngleDistanceOfTick / PointsToScanWithOneLaser;
98 
99  ResetRecordedHits(ChannelCount, PointsToScanWithOneLaser);
100 
101  GetWorld()->GetPhysicsScene()->GetPxScene()->lockRead();
102  ParallelFor(ChannelCount, [&](int32 idxChannel) {
103 
104  FCriticalSection Mutex;
105  ParallelFor(PointsToScanWithOneLaser, [&](int32 idxPtsOneLaser) {
106  FHitResult HitResult;
107  const float VertAngle = LaserAngles[idxChannel];
108  const float HorizAngle = CurrentHorizontalAngle + AngleDistanceOfLaserMeasure * idxPtsOneLaser;
109  const bool PreprocessResult = PreprocessRay();
110 
111  if (PreprocessResult && ShootLaser(VertAngle, HorizAngle, HitResult)) {
112  Mutex.Lock();
113  WritePointAsync(idxChannel, HitResult);
114  Mutex.Unlock();
115  }
116  });
117  });
118  GetWorld()->GetPhysicsScene()->GetPxScene()->unlockRead();
119 
120  FTransform ActorTransf = GetTransform();
121  ComputeAndSaveDetections(ActorTransf);
122 
123  const float HorizontalAngle = carla::geom::Math::ToRadians(
124  std::fmod(CurrentHorizontalAngle + AngleDistanceOfTick, 360.0f));
125  SemanticLidarData.SetHorizontalAngle(HorizontalAngle);
126 }

142  for (auto idxChannel = 0u; idxChannel < Description.Channels; ++idxChannel)
143  PointsPerChannel[idxChannel] = RecordedHits[idxChannel].size();
144  SemanticLidarData.ResetSerPoints(PointsPerChannel);

155 void ARayCastSemanticLidar::ComputeRawDetection(const FHitResult& HitInfo, const FTransform& SensorTransf, FSemanticDetection& Detection) const
156 {
157  const FVector HitPoint = HitInfo.ImpactPoint;
158  Detection.point = SensorTransf.Inverse().TransformPosition(HitPoint);
159 
160  const FVector VecInc = - (HitPoint - SensorTransf.GetLocation()).GetSafeNormal();
161  Detection.cos_inc_angle = FVector::DotProduct(VecInc, HitInfo.ImpactNormal);
162 
163  const FActorRegistry &Registry = GetEpisode().GetActorRegistry();
164 
165  AActor* actor = HitInfo.Actor.Get();
166  Detection.object_idx = 0;
167  Detection.object_tag = static_cast<uint32_t>(ECityObjectLabel::None);
168  if (actor != nullptr) {
169 
170  FActorView view = Registry.Find(actor);
171 
172  if(view.IsValid()) {

174  Detection.object_idx = ActorInfo->Description.UId;

180  }

188 bool ARayCastSemanticLidar::ShootLaser(const float VerticalAngle, const float HorizontalAngle, FHitResult& HitResult) const
189 {
190  FCollisionQueryParams TraceParams = FCollisionQueryParams(FName(TEXT("Laser_Trace")), true, this);
191  TraceParams.bTraceComplex = true;
192  TraceParams.bReturnPhysicalMaterial = false;
193 
194  FHitResult HitInfo(ForceInit);
195 
196  FTransform ActorTransf = GetTransform();
197  FVector LidarBodyLoc = ActorTransf.GetLocation();
198  FRotator LidarBodyRot = ActorTransf.Rotator();
199  FRotator LaserRot (VerticalAngle, HorizontalAngle, 0); // float InPitch, float InYaw, float InRoll
200  FRotator ResultRot = UKismetMathLibrary::ComposeRotators(
201  LaserRot,
202  LidarBodyRot
203  );
204  const auto Range = Description.Range;
205  FVector EndTrace = Range * UKismetMathLibrary::GetForwardVector(ResultRot) + LidarBodyLoc;
206 
207  GetWorld()->LineTraceSingleByChannel(
208  HitInfo,
209  LidarBodyLoc,
210  EndTrace,
211  ECC_GameTraceChannel2,
212  TraceParams,
213  FCollisionResponseParams::DefaultResponseParam
214  );
215 
216  if (HitInfo.bBlockingHit) {
217  HitResult = HitInfo;
218  return true;
219  } else {
220  return false;
221  }
