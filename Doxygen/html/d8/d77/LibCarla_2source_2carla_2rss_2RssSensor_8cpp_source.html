29  void RssSensor::Listen(CallbackFunctionType callback) {
30  if (_is_listening) {
31  log_error(GetDisplayId(), ": already listening");
32  return;
33  }
34 
35  if (GetParent() == nullptr) {
36  throw_exception(std::runtime_error(GetDisplayId() + ": not attached to vehicle"));
37  return;
38  }
39 
40  _map = GetWorld().GetMap();
41  DEBUG_ASSERT(_map != nullptr);
42 
43  mRssCheck = std::make_shared<::carla::rss::RssCheck>();
44 
45  auto self = boost::static_pointer_cast<RssSensor>(shared_from_this());
46 
47  log_debug(GetDisplayId(), ": subscribing to tick event");
48  GetEpisode().Lock()->RegisterOnTickEvent(
49  [cb = std::move(callback), weak_self = WeakPtr<RssSensor>(self)](const auto &snapshot) {
50  auto self = weak_self.lock();
51  if (self != nullptr) {
52  auto data = self->TickRssSensor(snapshot.GetTimestamp());
53  if (data != nullptr) {
54  cb(std::move(data));
55  }
56  }
57  });
58 
59  _is_listening = true;
60  }

62  SharedPtr<sensor::SensorData> RssSensor::TickRssSensor(const Timestamp &timestamp) {
63  try {
64  bool result = false;
65  ::ad_rss::state::ProperResponse response;
66  ::ad_rss::world::AccelerationRestriction accelerationRestriction;
67  ::ad_rss::world::Velocity egoVelocity;
68  if (_processing_lock.try_lock()) {
69  cc::World world = GetWorld();
70  SharedPtr<cc::ActorList> actors = world.GetActors();
71  SharedPtr<cc::ActorList> vehicles = actors->Filter("vehicle.*");
72 
73  // check all object<->ego pairs with RSS and calculate proper response
74  result =
75  mRssCheck->checkObjects(timestamp, world, vehicles, GetParent(), _map, response,
76  accelerationRestriction,
77  egoVelocity, visualize_results);
78  _processing_lock.unlock();
79 
80  csd::LongitudinalResponse longitudinalResponse;
81  csd::LateralResponse lateralResponseRight;
82  csd::LateralResponse lateralResponseLeft;
83 
84  switch (response.longitudinalResponse) {
85  case ::ad_rss::state::LongitudinalResponse::None:
86  longitudinalResponse = csd::LongitudinalResponse::None;
87  break;
88  case ::ad_rss::state::LongitudinalResponse::BrakeMinCorrect:
89  longitudinalResponse = csd::LongitudinalResponse::BrakeMinCorrect;
90  break;
91  case ::ad_rss::state::LongitudinalResponse::BrakeMin:
92  longitudinalResponse = csd::LongitudinalResponse::BrakeMin;
93  break;
94  }
95  switch (response.lateralResponseRight) {
96  case ::ad_rss::state::LateralResponse::None:
97  lateralResponseRight = csd::LateralResponse::None;
98  break;
99  case ::ad_rss::state::LateralResponse::BrakeMin:
100  lateralResponseRight = csd::LateralResponse::BrakeMin;
101  break;
102  }
103  switch (response.lateralResponseLeft) {
104  case ::ad_rss::state::LateralResponse::None:
105  lateralResponseLeft = csd::LateralResponse::None;
106  break;
107  case ::ad_rss::state::LateralResponse::BrakeMin:
108  lateralResponseLeft = csd::LateralResponse::BrakeMin;
109  break;
110  }
111  return MakeShared<sensor::data::RssResponse>(timestamp.frame, timestamp.elapsed_seconds,
112  GetTransform(),
113  result, longitudinalResponse, lateralResponseRight,
114  lateralResponseLeft, accelerationRestriction, egoVelocity);
115  } else {
116  return nullptr;
117  }
118  } catch (const std::exception &e) {
119  std::cout << e.what() << std::endl;
120  _processing_lock.unlock();
121  return nullptr;
122  }
123  }
