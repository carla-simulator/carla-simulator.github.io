29 void RssSensor::Listen(CallbackFunctionType callback) {
30  if (IsListening()) {
31  log_error(GetDisplayId(), ": already listening");
32  return;
33  }
34 
35  if (GetParent() == nullptr) {
36  throw_exception(std::runtime_error(GetDisplayId() + ": not attached to vehicle"));
37  return;
38  }
39 
40  _rss_check = std::make_shared<::carla::rss::RssCheck>();
41 
42  auto map = GetWorld().GetMap();
43  DEBUG_ASSERT(map != nullptr);
44  std::string const open_drive_content = map->GetOpenDrive();
45 
46  ::ad::map::access::cleanup();
47  ::ad::map::access::initFromOpenDriveContent(open_drive_content, 0.2,
48  ::ad::map::intersection::IntersectionType::TrafficLight,
49  ::ad::map::landmark::TrafficLightType::LEFT_STRAIGHT_RED_YELLOW_GREEN);
50 
51  auto self = boost::static_pointer_cast<RssSensor>(shared_from_this());
52 
53  log_debug(GetDisplayId(), ": subscribing to tick event");
54  _on_tick_register_id = GetEpisode().Lock()->RegisterOnTickEvent(
55  [ cb = std::move(callback), weak_self = WeakPtr<RssSensor>(self) ](const auto &snapshot) {
56  auto self = weak_self.lock();
57  if (self != nullptr) {
58  auto data = self->TickRssSensor(snapshot.GetTimestamp());
59  if (data != nullptr) {
60  cb(std::move(data));
61  }
62  }
63  });
64 }

66 void RssSensor::Stop() {
67  if (!IsListening()) {
68  log_error(GetDisplayId(), ": not listening at all");
69  return;
70  }
71 
72  // ensure there is no processing anymore
73  const std::lock_guard<std::mutex> lock(_processing_lock);
74 
75  log_debug(GetDisplayId(), ": unsubscribing from tick event");
76  GetEpisode().Lock()->RemoveOnTickEvent(_on_tick_register_id);
77  _on_tick_register_id = 0u;
78 
79  ::ad::map::access::cleanup();
80 }

168 SharedPtr<sensor::SensorData> RssSensor::TickRssSensor(const Timestamp &timestamp) {
169  try {
170  bool result = false;
171  ::ad::rss::state::ProperResponse response;
172  ::ad::rss::world::AccelerationRestriction acceleration_restriction;
173  ::ad::rss::state::RssStateSnapshot rss_state_snapshot;
174  carla::rss::EgoDynamicsOnRoute ego_dynamics_on_route;
175  if (_processing_lock.try_lock()) {
176  spdlog::debug("RssSensor tick: T={}", timestamp.frame);
177  carla::client::World world = GetWorld();
178  SharedPtr<carla::client::ActorList> actors = world.GetActors();
179 
180  if (_drop_route) {
181  _drop_route = false;
182  _rss_check->DropRoute();
183  }
184 
185  // check all object<->ego pairs with RSS and calculate proper response
186  result = _rss_check->CheckObjects(timestamp, actors, GetParent(), response, acceleration_restriction,
187  rss_state_snapshot, ego_dynamics_on_route);
188  _processing_lock.unlock();
189 
190  _rss_check->VisualizeResults(world, GetParent());
191 
192  spdlog::debug(
193  "RssSensor response: T={} S:{}->E:{} DeltaT:{}", timestamp.frame,
194  ego_dynamics_on_route.time_since_epoch_check_start_ms, ego_dynamics_on_route.time_since_epoch_check_end_ms,
195  ego_dynamics_on_route.time_since_epoch_check_end_ms - ego_dynamics_on_route.time_since_epoch_check_start_ms);
196  return MakeShared<sensor::data::RssResponse>(timestamp.frame, timestamp.elapsed_seconds, GetTransform(), result,
197  response, acceleration_restriction, rss_state_snapshot,
198  ego_dynamics_on_route);
199  } else {
200  spdlog::debug("RssSensor tick dropped: T={}", timestamp.frame);
201  return nullptr;
202  }
203  } catch (const std::exception &e) {
204  /// @todo do we need to unsubscribe the sensor here?
205  std::cout << e.what() << std::endl;
206  _processing_lock.unlock();
207  spdlog::error("RssSensor tick exception");
208  return nullptr;
209  } catch (...) {
210  _processing_lock.unlock();
211  spdlog::error("RssSensor tick exception");
212  return nullptr;
213  }
214 }
