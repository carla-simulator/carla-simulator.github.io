28 void BatchControlStage::Action() {
29 
30  // Looping over registered actors.
31  for (uint64_t i = 0u; i < number_of_vehicles && data_frame != nullptr; ++i) {
32 
33  carla::rpc::VehicleControl vehicle_control;
34 
35  const PlannerToControlData &element = data_frame->at(i);
36  if (!element.actor || !element.actor->IsAlive()) {
37  continue;
38  }
39  const carla::ActorId actor_id = element.actor->GetId();
40 
41  vehicle_control.throttle = element.throttle;
42  vehicle_control.brake = element.brake;
43  vehicle_control.steer = element.steer;
44 
45  commands->at(i) = carla::rpc::Command::ApplyVehicleControl(actor_id, vehicle_control);
46  }
47 }

63 void BatchControlStage::DataSender() {
64 
65  messenger->Pop();
66  bool synch_mode = parameters.GetSynchronousMode();
67 
68  // Asynchronous mode.
69  if (!synch_mode) {
70  // Apply batch command through an asynchronous RPC call.
71  if (commands != nullptr) {
72  episode_proxy_bcs.Lock()->ApplyBatch(*commands.get(), false);
73  }
74  // Applying an infinitesimal sleep statement for providing a system cancellation point.
75  std::this_thread::sleep_for(1us);
76  }
77  // Synchronous mode.
78  else {
79  std::unique_lock<std::mutex> lock(step_execution_mutex);
80  // TODO: Re-introduce timeout while waiting for RunStep() call.
81  while (!run_step.load()) {
82  step_execution_notifier.wait_for(lock, 1ms, [this]() {return run_step.load();});
83  }
84  // Apply batch command in synchronous RPC call.
85  if (commands != nullptr) {
86  episode_proxy_bcs.Lock()->ApplyBatchSync(*commands.get(), false);
87  }
88  // Set flag to false, unblock RunStep() call and release mutex lock.
89  run_step.store(false);
90  step_complete_notifier.notify_one();
91  lock.unlock();
92  }
93 }

96 bool BatchControlStage::RunStep() {
97 
98  bool synch_mode = parameters.GetSynchronousMode();
99  if (synch_mode) {
100  std::unique_lock<std::mutex> lock(step_execution_mutex);
101  // Set flag to true, notify DataSender and wait for a return notification
102  run_step.store(true);
103  step_execution_notifier.notify_one();
104  while (run_step.load()) {
105  step_complete_notifier.wait_for(lock, 1ms, [this]() {return !run_step.load();});
106  }
107  }
108 
109  return true;
