29  /// Can be used as flags
30  enum class LightState : flag_type {

35  Brake = 0x1 << 3,

40  Interior = 0x1 << 8,
41  Special1 = 0x1 << 9, // E.g: sirens

43  All = 0xFFFFFFFF
44  };

56  VehicleLightState(const FVehicleLightState &InLightState) {
57  light_state = static_cast<flag_type>(LightState::None);
58  if (InLightState.Position) SET_FLAG(light_state, LightState::Position);
59  if (InLightState.LowBeam) SET_FLAG(light_state, LightState::LowBeam);
60  if (InLightState.HighBeam) SET_FLAG(light_state, LightState::HighBeam);
61  if (InLightState.Brake) SET_FLAG(light_state, LightState::Brake);
62  if (InLightState.RightBlinker) SET_FLAG(light_state, LightState::RightBlinker);
63  if (InLightState.LeftBlinker) SET_FLAG(light_state, LightState::LeftBlinker);
64  if (InLightState.Reverse) SET_FLAG(light_state, LightState::Reverse);
65  if (InLightState.Fog) SET_FLAG(light_state, LightState::Fog);
66  if (InLightState.Interior) SET_FLAG(light_state, LightState::Interior);
67  if (InLightState.Special1) SET_FLAG(light_state, LightState::Special1);
68  if (InLightState.Special2) SET_FLAG(light_state, LightState::Special2);
69  }

71  operator FVehicleLightState() const {
72  FVehicleLightState Lights; // By default all False
73  if (FLAG_ENABLED(light_state, LightState::Position)) Lights.Position = true;
74  if (FLAG_ENABLED(light_state, LightState::LowBeam)) Lights.LowBeam = true;
75  if (FLAG_ENABLED(light_state, LightState::HighBeam)) Lights.HighBeam = true;
76  if (FLAG_ENABLED(light_state, LightState::Brake)) Lights.Brake = true;
77  if (FLAG_ENABLED(light_state, LightState::RightBlinker)) Lights.RightBlinker = true;
78  if (FLAG_ENABLED(light_state, LightState::LeftBlinker)) Lights.LeftBlinker = true;
79  if (FLAG_ENABLED(light_state, LightState::Reverse)) Lights.Reverse = true;
80  if (FLAG_ENABLED(light_state, LightState::Fog)) Lights.Fog = true;
81  if (FLAG_ENABLED(light_state, LightState::Interior)) Lights.Interior = true;
82  if (FLAG_ENABLED(light_state, LightState::Special1)) Lights.Special1 = true;
83  if (FLAG_ENABLED(light_state, LightState::Special2)) Lights.Special2 = true;
84  return Lights;
85  }
