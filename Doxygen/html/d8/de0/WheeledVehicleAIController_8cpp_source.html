42 static bool IsThereAnObstacleAhead(
43  const ACarlaWheeledVehicle &Vehicle,
44  const float Speed,
45  const FVector &Direction)
46 {
47  const auto ForwardVector = Vehicle.GetVehicleOrientation();
48  const auto VehicleBounds = Vehicle.GetVehicleBoundingBoxExtent();
49 
50  FVector NormDirection = Direction.GetSafeNormal();
51 
52  const float Distance = std::max(50.0f, Speed * Speed); // why?
53 
54  const FVector StartCenter = Vehicle.GetActorLocation() +
55  (ForwardVector * (250.0f + VehicleBounds.X / 2.0f)) + FVector(0.0f, 0.0f, 50.0f);
56  const FVector EndCenter = StartCenter + NormDirection * (Distance + VehicleBounds.X / 2.0f);
57 
58  const FVector StartRight = StartCenter +
59  (FVector(ForwardVector.Y, -ForwardVector.X, ForwardVector.Z) * 100.0f);
60  const FVector EndRight = StartRight + NormDirection * (Distance + VehicleBounds.X / 2.0f);
61 
62  const FVector StartLeft = StartCenter +
63  (FVector(-ForwardVector.Y, ForwardVector.X, ForwardVector.Z) * 100.0f);
64  const FVector EndLeft = StartLeft + NormDirection * (Distance + VehicleBounds.X / 2.0f);
65 
66  return
67  RayCast(Vehicle, StartCenter, EndCenter) ||
68  RayCast(Vehicle, StartRight, EndRight) ||
69  RayCast(Vehicle, StartLeft, EndLeft);
70 }

155 {

165  ClearQueue(TargetLocations);
166  Vehicle->SetAIVehicleState(
167  bAutopilotEnabled ?
168  ECarlaWheeledVehicleState::FreeDriving :
169  ECarlaWheeledVehicleState::AutopilotOff);
170  }
171 
172  TrafficLightState = ETrafficLightState::Green;
173 
174  /// @todo Workaround for a race condition between client and server when
175  /// enabling autopilot right after initializing a vehicle.
176  if (bAutopilotEnabled)
177  {
178  for (TActorIterator<ARoutePlanner> It(GetWorld()); It; ++It)
179  {
180  ARoutePlanner *RoutePlanner = *It;
181  // Check if we are inside this route planner.
182  TSet<AActor *> OverlappingActors;
183  RoutePlanner->TriggerVolume->GetOverlappingActors(
184  OverlappingActors,
185  ACarlaWheeledVehicle::StaticClass());
186  for (auto *Actor : OverlappingActors)
187  {
188  if (Actor == Vehicle)
189  {
190  RoutePlanner->AssignRandomRoute(*this);
191  return;
192  }
193  }
194  }

220 FVehicleControl AWheeledVehicleAIController::TickAutopilotController()
221 {
222 #if WITH_EDITOR // This happens in simulation mode in editor.
223  if (Vehicle == nullptr)
224  {
225  bAutopilotEnabled = false;
226  return {};
227  }
228 #endif // WITH_EDITOR
229 
230  check(Vehicle != nullptr);
231 
232  FVector Direction;
233 
234  float Steering;
235  if (!TargetLocations.empty())
236  {
237  Steering = GoToNextTargetLocation(Direction);
238  }
239  else
240  {
241  Steering = RoadMap != nullptr ? CalcStreeringValue(Direction) : 0.0f;
242  Direction = Vehicle->GetVehicleTransform().GetRotation().Rotator().Vector();
243  }
244 
245  // Speed in km/h.
246  const auto Speed = Vehicle->GetVehicleForwardSpeed() * 0.036f;
247 
248  float Throttle;
249  if (TrafficLightState != ETrafficLightState::Green)
250  {
251  Vehicle->SetAIVehicleState(ECarlaWheeledVehicleState::WaitingForRedLight);
252  Throttle = Stop(Speed);
253  }
254  else if (IsThereAnObstacleAhead(*Vehicle, Speed, Direction))
255  {
256  Vehicle->SetAIVehicleState(ECarlaWheeledVehicleState::ObstacleAhead);
257  Throttle = Stop(Speed);
258  }
259  else
260  {
261  Throttle = Move(Speed);
262  }
263 
264  FVehicleControl AutopilotControl;
265 
266  if (Throttle < 0.001f)
267  {
268  AutopilotControl.Brake = 1.0f;
269  AutopilotControl.Throttle = 0.0f;
270  }
271  else
272  {
273  AutopilotControl.Brake = 0.0f;
274  AutopilotControl.Throttle = Throttle;
275  }
276  AutopilotControl.Steer = Steering;
277 
278  return AutopilotControl;
279 }

281 float AWheeledVehicleAIController::GoToNextTargetLocation(FVector &Direction)
282 {
283  // Get middle point between the two front wheels.
284  const auto CurrentLocation = [&]() {
285  const auto &Wheels = Vehicle->GetVehicleMovementComponent()->Wheels;
286  check((Wheels.Num() > 1) && (Wheels[0u] != nullptr) && (Wheels[1u] != nullptr));
287  return (Wheels[0u]->Location + Wheels[1u]->Location) / 2.0f;
288  } ();
289 
290  const auto Target = [&]() {
291  const auto &Result = TargetLocations.front();
292  return FVector{Result.X, Result.Y, CurrentLocation.Z};
293  } ();
294 
295  if (Target.Equals(CurrentLocation, 200.0f))
296  {
297  TargetLocations.pop();
298  if (!TargetLocations.empty())
299  {
300  return GoToNextTargetLocation(Direction);
301  }
302  else
303  {
304  return RoadMap != nullptr ? CalcStreeringValue(Direction) : 0.0f;
305  }
306  }
307 
308  Direction = (Target - CurrentLocation).GetSafeNormal();
309 
310  const FVector &Forward = GetPawn()->GetActorForwardVector();
311 
312  float dirAngle = Direction.UnitCartesianToSpherical().Y;
313  float actorAngle = Forward.UnitCartesianToSpherical().Y;
314 
315  dirAngle *= (180.0f / PI);
316  actorAngle *= (180.0 / PI);
317 
318  float angle = dirAngle - actorAngle;
319 
320  if (angle > 180.0f)
321  {
322  angle -= 360.0f;
323  }
324  else if (angle < -180.0f)
325  {
326  angle += 360.0f;
327  }
328 
329  float Steering = 0.0f;
330  if (angle < -MaximumSteerAngle)
331  {
332  Steering = -1.0f;
333  }
334  else if (angle > MaximumSteerAngle)
335  {
336  Steering = 1.0f;
337  }
338  else
339  {
340  Steering += angle / MaximumSteerAngle;
341  }
342 
343  Vehicle->SetAIVehicleState(ECarlaWheeledVehicleState::FollowingFixedRoute);
344  return Steering;
345 }

347 float AWheeledVehicleAIController::CalcStreeringValue(FVector &direction)
348 {
349  float steering = 0;
350  FVector BoxExtent = Vehicle->GetVehicleBoundingBoxExtent();
351  FVector forward = Vehicle->GetActorForwardVector();
352 
353  FVector rightSensorPosition(BoxExtent.X / 2.0f, (BoxExtent.Y / 2.0f) + 100.0f, 0.0f);
354  FVector leftSensorPosition(BoxExtent.X / 2.0f, -(BoxExtent.Y / 2.0f) - 100.0f, 0.0f);
355 
356  float forwardMagnitude = BoxExtent.X / 2.0f;
357 
358  float Magnitude =
359  (float) sqrt(pow((double) leftSensorPosition.X, 2.0) + pow((double) leftSensorPosition.Y, 2.0));
360 
361  // same for the right and left
362  float offset = FGenericPlatformMath::Acos(forwardMagnitude / Magnitude);
363 
364  float actorAngle = forward.UnitCartesianToSpherical().Y;
365 
366  float sinR = FGenericPlatformMath::Sin(actorAngle + offset);
367  float cosR = FGenericPlatformMath::Cos(actorAngle + offset);
368 
369  float sinL = FGenericPlatformMath::Sin(actorAngle - offset);
370  float cosL = FGenericPlatformMath::Cos(actorAngle - offset);
371 
372  rightSensorPosition.Y = sinR * Magnitude;
373  rightSensorPosition.X = cosR * Magnitude;
374 
375  leftSensorPosition.Y = sinL * Magnitude;
376  leftSensorPosition.X = cosL * Magnitude;
377 
378  FVector rightPositon = GetPawn()->GetActorLocation() + FVector(rightSensorPosition.X,
379  rightSensorPosition.Y,
380  0.0f);
381  FVector leftPosition = GetPawn()->GetActorLocation() + FVector(leftSensorPosition.X,
382  leftSensorPosition.Y,
383  0.0f);
384 
385  FRoadMapPixelData rightRoadData = RoadMap->GetDataAt(rightPositon);
386  if (!rightRoadData.IsRoad())
387  {
388  steering -= 0.2f;
389  }
390 
391  FRoadMapPixelData leftRoadData = RoadMap->GetDataAt(leftPosition);
392  if (!leftRoadData.IsRoad())
393  {
394  steering += 0.2f;
395  }
396 
397  FRoadMapPixelData roadData = RoadMap->GetDataAt(GetPawn()->GetActorLocation());
398  if (!roadData.IsRoad())
399  {
400  steering = 0.0f;
401  }
402  else if (roadData.HasDirection())
403  {
404 
405  direction = roadData.GetDirection();
406  FVector right = rightRoadData.GetDirection();
407  FVector left = leftRoadData.GetDirection();
408 
409  forward.Z = 0.0f;
410 
411  float dirAngle = direction.UnitCartesianToSpherical().Y;
412  float rightAngle = right.UnitCartesianToSpherical().Y;
413  float leftAngle = left.UnitCartesianToSpherical().Y;
414 
415  dirAngle *= (180.0f / PI);
416  rightAngle *= (180.0 / PI);
417  leftAngle *= (180.0 / PI);
418  actorAngle *= (180.0 / PI);
419 
420  float min = dirAngle - 90.0f;
421  if (min < -180.0f)
422  {
423  min = 180.0f + (min + 180.0f);
424  }
425 
426  float max = dirAngle + 90.0f;
427  if (max > 180.0f)
428  {
429  max = -180.0f + (max - 180.0f);
430  }
431 
432  if (dirAngle < -90.0 || dirAngle > 90.0)
433  {
434  if (rightAngle < min && rightAngle > max)
435  {
436  steering -= 0.2f;
437  }
438  if (leftAngle < min && leftAngle > max)
439  {
440  steering += 0.2f;
441  }
442  }
443  else
444  {
445  if (rightAngle < min || rightAngle > max)
446  {
447  steering -= 0.2f;
448  }
449  if (leftAngle < min || leftAngle > max)
450  {
451  steering += 0.2f;
452  }
453  }
454 
455  float angle = dirAngle - actorAngle;
456 
457  if (angle > 180.0f)
458  {
459  angle -= 360.0f;
460  }
461  else if (angle < -180.0f)
462  {
463  angle += 360.0f;
464  }
465 
466  if (angle < -MaximumSteerAngle)
467  {
468  steering = -1.0f;
469  }
470  else if (angle > MaximumSteerAngle)
471  {
472  steering = 1.0f;
473  }
474  else
475  {
476  steering += angle / MaximumSteerAngle;
477  }
478  }
479 
480  Vehicle->SetAIVehicleState(ECarlaWheeledVehicleState::FreeDriving);
481  return steering;
482 }
