23  void WalkerNavigation::Tick(const EpisodeState &state) {
24  auto walkers = _walkers.Load();
25  if (walkers->empty()) {
26  return;
27  }
28 
29  // purge all possible dead walkers
30  CheckIfWalkerExist(*walkers, state);
31 
32  // update crowd in navigation module
33  _nav.UpdateCrowd(state);
34 
35  carla::geom::Transform trans;
36  using Cmd = rpc::Command;
37  std::vector<Cmd> commands;
38  commands.reserve(walkers->size());
39  for (auto handle : *walkers) {
40  // get the transform of the walker
41  if (_nav.GetWalkerTransform(handle.walker, trans)) {
42  float speed = _nav.GetWalkerSpeed(handle.walker);
43  commands.emplace_back(Cmd::ApplyWalkerState{ handle.walker, trans, speed });
44  }
45  }
46 
47  _client.ApplyBatch(std::move(commands), false);
48  }
