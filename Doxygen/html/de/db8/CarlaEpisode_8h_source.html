178  /// Spawns an actor based on @a ActorDescription at @a Transform. To properly
179  /// despawn an actor created with this function call DestroyActor.
180  ///
181  /// @return A pair containing the result of the spawn function and a view over
182  /// the actor and its properties. If the status is different of Success the
183  /// view is invalid.
184  TPair<EActorSpawnResultStatus, FActorView> SpawnActorWithInfo(
185  const FTransform &Transform,
186  FActorDescription thisActorDescription,
187  FActorView::IdType DesiredId = 0)
188  {
189  auto result = ActorDispatcher->SpawnActor(Transform, thisActorDescription, DesiredId);
190  if (Recorder->IsEnabled())
191  {
192  if (result.Key == EActorSpawnResultStatus::Success)
193  {
194  Recorder->CreateRecorderEventAdd(
195  result.Value.GetActorId(),
196  static_cast<uint8_t>(result.Value.GetActorType()),
197  Transform,
198  std::move(thisActorDescription)
199  );
200  }
201  }
202 
203  return result;
204  }
