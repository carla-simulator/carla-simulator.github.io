190  /// @return A pair containing the result of the spawn function and a view over

193  TPair<EActorSpawnResultStatus, FActorView> SpawnActorWithInfo(
194  const FTransform &Transform,
195  FActorDescription thisActorDescription,
196  FActorView::IdType DesiredId = 0)
197  {
198  auto result = ActorDispatcher->SpawnActor(Transform, thisActorDescription, DesiredId);
199  if (Recorder->IsEnabled())
200  {
201  if (result.Key == EActorSpawnResultStatus::Success)
202  {
203  Recorder->CreateRecorderEventAdd(
204  result.Value.GetActorId(),
205  static_cast<uint8_t>(result.Value.GetActorType()),
206  Transform,
207  std::move(thisActorDescription)

209  }
