284 std::string CarlaRecorderQuery::QueryCollisions(std::string Filename, char Category1, char Category2)
285 {
286  std::stringstream Info;
287 
288  // get the final path + filename
289  std::string Filename2 = GetRecorderFilename(Filename);
290 
291  // try to open
292  File.open(Filename2, std::ios::binary);
293  if (!File.is_open())
294  {
295  Info << "File " << Filename2 << " not found on server\n";
296  return Info.str();
297  }
298 
299  if (!CheckFileInfo(Info))
300  return Info.str();
301 
302  // other, vehicle, walkers, trafficLight, hero, any
303  char Categories[] = { 'o', 'v', 'w', 't', 'h', 'a' };
304  uint16_t i, Total;
305  struct ReplayerActorInfo
306  {
307  uint8_t Type;
308  FString Id;
309  };
310  std::unordered_map<uint32_t, ReplayerActorInfo> Actors;
311  struct PairHash
312  {
313  std::size_t operator()(const std::pair<uint32_t, uint32_t>& P) const
314  {
315  std::size_t hash = P.first;
316  hash <<= 32;
317  hash += P.second;
318  return hash;
319  }
320  };
321  std::unordered_set<std::pair<uint32_t, uint32_t>, PairHash > oldCollisions, newCollisions;
322 
323  // header
324  Info << std::setw(8) << "Time";
325  Info << " " << std::setw(6) << "Types";
326  Info << " " << std::setw(6) << std::right << "Id";
327  Info << " " << std::setw(35) << std::left << "Actor 1";
328  Info << " " << std::setw(6) << std::right << "Id";
329  Info << " " << std::setw(35) << std::left << "Actor 2";
330  Info << std::endl;
331 
332  // parse only frames
333  while (File)
334  {
335 
336  // get header
337  if (!ReadHeader())
338  {
339  break;
340  }
341 
342  // check for a frame packet
343  switch (Header.Id)
344  {
345  // frame
346  case static_cast<char>(CarlaRecorderPacketId::FrameStart):
347  Frame.Read(File);
348  // exchange sets of collisions (to know when a collision is new or continue from previous frame)
349  oldCollisions = std::move(newCollisions);
350  newCollisions.clear();
351  break;
352 
353  // events add
354  case static_cast<char>(CarlaRecorderPacketId::EventAdd):
355  ReadValue<uint16_t>(File, Total);
356  for (i = 0; i < Total; ++i)
357  {
358  // add
359  EventAdd.Read(File);
360  Actors[EventAdd.DatabaseId] = ReplayerActorInfo { EventAdd.Type, EventAdd.Description.Id };
361  }
362  break;
363 
364  // events del
365  case static_cast<char>(CarlaRecorderPacketId::EventDel):
366  ReadValue<uint16_t>(File, Total);
367  for (i = 0; i < Total; ++i)
368  {
369  EventDel.Read(File);
370  Actors.erase(EventAdd.DatabaseId);
371  }
372  break;
373 
374  // events parenting
375  case static_cast<char>(CarlaRecorderPacketId::EventParent):
376  SkipPacket();
377  break;
378 
379  // collisions
380  case static_cast<char>(CarlaRecorderPacketId::Collision):
381  ReadValue<uint16_t>(File, Total);
382  for (i = 0; i < Total; ++i)
383  {
384  Collision.Read(File);
385 
386  int Valid = 0;
387  // get categories for both actors
388  uint8_t Type1 = Categories[Actors[Collision.DatabaseId1].Type];
389  uint8_t Type2 = Categories[Actors[Collision.DatabaseId2].Type];
390 
391  // filter actor 1
392  if (Category1 == 'a')
393  ++Valid;
394  else if (Category1 == Type1)
395  ++Valid;
396  else if (Category1 == 'h' && Collision.IsActor1Hero)
397  ++Valid;
398 
399  // filter actor 2
400  if (Category2 == 'a')
401  ++Valid;
402  else if (Category2 == Type2)
403  ++Valid;
404  else if (Category2 == 'h' && Collision.IsActor2Hero)
405  ++Valid;
406 
407  // only show if both actors has passed the filter
408  if (Valid == 2)
409  {
410  // check if we need to show as a starting collision or it is a continuation one
411  auto collisionPair = std::make_pair(Collision.DatabaseId1, Collision.DatabaseId2);
412  if (oldCollisions.count(collisionPair) == 0)
413  {
414  Info << std::setw(8) << std::setprecision(0) << std::right << std::fixed << Frame.Elapsed;
415  Info << " " << " " << Type1 << " " << Type2 << " ";
416  Info << " " << std::setw(6) << std::right << Collision.DatabaseId1;
417  Info << " " << std::setw(35) << std::left << TCHAR_TO_UTF8(*Actors[Collision.DatabaseId1].Id);
418  Info << " " << std::setw(6) << std::right << Collision.DatabaseId2;
419  Info << " " << std::setw(35) << std::left << TCHAR_TO_UTF8(*Actors[Collision.DatabaseId2].Id);
420  Info << std::endl;
421  }
422  // save current collision
423  newCollisions.insert(collisionPair);
424  }
425  }
426  break;
427 
428  case static_cast<char>(CarlaRecorderPacketId::Position):
429  SkipPacket();
430  break;
431 
432  case static_cast<char>(CarlaRecorderPacketId::State):
433  SkipPacket();
434  break;
435 
436  // frame end
437  case static_cast<char>(CarlaRecorderPacketId::FrameEnd):
438  // do nothing, it is empty
439  break;
440 
441  default:
442  SkipPacket();
443  break;
444  }
445  }
446 
447  Info << "\nFrames: " << Frame.Id << "\n";
448  Info << "Duration: " << Frame.Elapsed << " seconds\n";
449 
450  File.close();
451 
452  return Info.str();
453 }

475  struct ReplayerActorInfo
476  {
477  uint8_t Type;
478  FString Id;
479  FVector LastPosition;
480  double Time;

485  std::multimap<double, std::string, std::greater<double>> Results;
486 
487  // header

489  Info << " " << std::setw(6) << "Id";
490  Info << " " << std::setw(35) << std::left << "Actor";
491  Info << " " << std::setw(10) << std::right << "Duration";
492  Info << std::endl;
493 
494  // parse only frames
495  while (File)
496  {
497 
498  // get header
499  if (!ReadHeader())
500  {
501  break;
502  }
503 
504  // check for a frame packet
505  switch (Header.Id)
506  {
507  // frame
508  case static_cast<char>(CarlaRecorderPacketId::FrameStart):
509  Frame.Read(File);
510  break;
511 
512  // events add
513  case static_cast<char>(CarlaRecorderPacketId::EventAdd):
514  ReadValue<uint16_t>(File, Total);
515  for (i = 0; i < Total; ++i)
516  {
517  // add
518  EventAdd.Read(File);
519  Actors[EventAdd.DatabaseId] = ReplayerActorInfo { EventAdd.Type, EventAdd.Description.Id, FVector(0, 0, 0), 0.0, 0.0 };
520  }
521  break;
522 
523  // events del
524  case static_cast<char>(CarlaRecorderPacketId::EventDel):
525  ReadValue<uint16_t>(File, Total);
526  for (i = 0; i < Total; ++i)
527  {
528  EventDel.Read(File);
529  Actors.erase(EventAdd.DatabaseId);
530  }
531  break;
532 
533  // events parenting
534  case static_cast<char>(CarlaRecorderPacketId::EventParent):
535  SkipPacket();
536  break;
537 
538  // collisions
539  case static_cast<char>(CarlaRecorderPacketId::Collision):
540  SkipPacket();
541  break;
542 
543  // positions
544  case static_cast<char>(CarlaRecorderPacketId::Position):
545  // read all positions
546  ReadValue<uint16_t>(File, Total);
547  for (i=0; i<Total; ++i)
548  {
549  Position.Read(File);
550  // check if actor moved less than a distance
551  if (FVector::Distance(Actors[Position.DatabaseId].LastPosition, Position.Location) < MinDistance)
552  {
553  // actor stopped
554  if (Actors[Position.DatabaseId].Duration == 0)
555  Actors[Position.DatabaseId].Time = Frame.Elapsed;
556  Actors[Position.DatabaseId].Duration += Frame.DurationThis;

565  Result << " " << std::setw(6) << Position.DatabaseId;
566  Result << " " << std::setw(35) << std::left << TCHAR_TO_UTF8(*Actors[Position.DatabaseId].Id);
567  Result << " " << std::setw(10) << std::setprecision(0) << std::fixed << std::right << Actors[Position.DatabaseId].Duration;
568  Result << std::endl;
569  Results.insert(std::make_pair(Actors[Position.DatabaseId].Duration, Result.str()));
570  }
571  // actor moving
572  Actors[Position.DatabaseId].Duration = 0;
573  Actors[Position.DatabaseId].LastPosition = Position.Location;
574  }
575  }
576  break;
577 
578  // traffic light
579  case static_cast<char>(CarlaRecorderPacketId::State):
580  SkipPacket();
581  break;
582 
583  // frame end
584  case static_cast<char>(CarlaRecorderPacketId::FrameEnd):
585  // do nothing, it is empty
586  break;
587 
588  default:
589  SkipPacket();
590  break;
591  }
592  }
593 
594  // show actors stopped that were not moving again
595  for (auto &Actor : Actors)
596  {
597  // check to show info
598  if (Actor.second.Duration >= MinTime)
599  {
600  std::stringstream Result;
601  Result << std::setw(8) << std::setprecision(0) << std::fixed << Actor.second.Time;
602  Result << " " << std::setw(6) << Actor.first;
603  Result << " " << std::setw(35) << std::left << TCHAR_TO_UTF8(*Actor.second.Id);
604  Result << " " << std::setw(10) << std::setprecision(0) << std::fixed << std::right << Actor.second.Duration;
605  Result << std::endl;
606  Results.insert(std::make_pair(Actor.second.Duration, Result.str()));
607  }
608  }
609 
610  // show the result
611  for (auto &Result : Results)
612  {
613  Info << Result.second;
614  }
615 
616  Info << "\nFrames: " << Frame.Id << "\n";
617  Info << "Duration: " << Frame.Elapsed << " seconds\n";
618 
619  File.close();
620 
621  return Info.str();
