18  /// A low-level streaming server. Each new stream has a token associated, this
19  /// token can be used by a client to subscribe to the stream. This server
20  /// requires an external io_context running.
21  ///
22  /// @warning This server cannot be destructed before its @a io_context is
23  /// stopped.

73  void StartServer() {
74  auto on_session_opened = [this](auto session) {
75  if (!_dispatcher.RegisterSession(session)) {
76  session->Close();
77  }
78  };
79  auto on_session_closed = [this](auto session) {
80  _dispatcher.DeregisterSession(session);

82  _server.Listen(on_session_opened, on_session_closed);
