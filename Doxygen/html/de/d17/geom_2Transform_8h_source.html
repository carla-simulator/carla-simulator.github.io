56  void TransformPoint(Vector3D &in_point) const {
57  // Rotate
58  const float cy = std::cos(Math::ToRadians(rotation.yaw));
59  const float sy = std::sin(Math::ToRadians(rotation.yaw));
60  const float cr = std::cos(Math::ToRadians(rotation.roll));
61  const float sr = std::sin(Math::ToRadians(rotation.roll));
62  const float cp = std::cos(Math::ToRadians(rotation.pitch));
63  const float sp = std::sin(Math::ToRadians(rotation.pitch));
64 
65  Vector3D out_point;
66  out_point.x = in_point.x * (cp * cy) +
67  in_point.y * (cy * sp * sr - sy * cr) +
68  in_point.z * (-cy * sp * cr - sy * sr);
69  out_point.y = in_point.x * (cp * sy) +
70  in_point.y * (sy * sp * sr + cy * cr) +
71  in_point.z * (-sy * sp * cr + cy * sr);
72 
73  out_point.z = in_point.x * (sp) +
74  in_point.y * -(cp * sr) +
75  in_point.z * (cp * cr);
76 
77  // Translate
78  out_point += location;
79 
80  in_point = out_point;
81  }
