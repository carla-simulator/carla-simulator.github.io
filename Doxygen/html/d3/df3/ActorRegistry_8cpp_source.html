21 static FCarlaActor::ActorType FActorRegistry_GetActorType(const AActor *Actor)
22 {
23  if (!Actor)
24  {
25  return FCarlaActor::ActorType::INVALID;
26  }
27 
28  if (nullptr != Cast<ACarlaWheeledVehicle>(Actor))
29  {
30  return FCarlaActor::ActorType::Vehicle;
31  }
32  else if (nullptr != Cast<ACharacter>(Actor))
33  {
34  return FCarlaActor::ActorType::Walker;
35  }
36  else if (nullptr != Cast<ATrafficLightBase>(Actor))
37  {
38  return FCarlaActor::ActorType::TrafficLight;
39  }
40  else if (nullptr != Cast<ATrafficSignBase>(Actor))
41  {
42  return FCarlaActor::ActorType::TrafficSign;
43  }
44  else if (nullptr != Cast<ASensor>(Actor))
45  {
46  return FCarlaActor::ActorType::Sensor;
47  }
48  else
49  {
50  return FCarlaActor::ActorType::Other;
51  }
52 }

54 static FString GetRelevantTagAsString(const TSet<crp::CityObjectLabel> &SemanticTags)
55 {
56  for (auto &&Tag : SemanticTags)
57  {
58  if ((Tag != crp::CityObjectLabel::None) && (Tag != crp::CityObjectLabel::Other))
59  {
60  auto Str = ATagger::GetTagAsString(Tag).ToLower();
61  return (Str.EndsWith(TEXT("s")) ? Str.LeftChop(1) : Str);
62  }
63  }
64  return TEXT("unknown");
65 }

67 FCarlaActor* FActorRegistry::Register(AActor &Actor, FActorDescription Description, IdType DesiredId)
68 {
69 
70  FCarlaActor* CarlaActor = FindCarlaActor(DesiredId);
71  bool IsDormant = CarlaActor && (CarlaActor->IsDormant());
72  if(IsDormant)
73  {
74  CarlaActor->TheActor = &Actor;
75  Actors.Add(DesiredId, &Actor);
76  Ids.Add(&Actor, DesiredId);
77  return CarlaActor;
78  }
79 
80  IdType Id = ++FActorRegistry::ID_COUNTER;
81 
82  if (DesiredId != 0 && Id != DesiredId) {
83  // check if the desired Id is free, then use it instead
84  if (!Actors.Contains(DesiredId))
85  {
86  Id = DesiredId;
87  if (ID_COUNTER < Id)
88  ID_COUNTER = Id;
89  }
90  }
91 
92  Actors.Emplace(Id, &Actor);
93  if (Ids.Contains(&Actor))
94  {
95  UE_LOG(
96  LogCarla,
97  Warning,
98  TEXT("This actor's memory address is already registered, "
99  "either you forgot to deregister the actor "
100  "or the actor was garbage collected."));
101  }
102  Ids.Emplace(&Actor, Id);
103 
104  TSharedPtr<FCarlaActor> View =
105  MakeCarlaActor(Id, Actor, std::move(Description), crp::ActorState::Active);
106 
107  TSharedPtr<FCarlaActor>& Result = ActorDatabase.Emplace(Id, MoveTemp(View));
108 
109  check(static_cast<size_t>(Actors.Num()) == ActorDatabase.Num());
110  return Result.Get();
111 }

138 TSharedPtr<FCarlaActor> FActorRegistry::MakeCarlaActor(
139  IdType Id,
140  AActor &Actor,
141  FActorDescription Description,
142  crp::ActorState InState) const
143 {
144  auto Info = MakeShared<FActorInfo>();
145  Info->Description = std::move(Description);
146  ATagger::GetTagsOfTaggedActor(Actor, Info->SemanticTags);
147  Info->BoundingBox = UBoundingBoxCalculator::GetActorBoundingBox(&Actor);
148 
149  if (Info->Description.Id.IsEmpty())
150  {
151  // This is a fake actor, let's fake the id based on their semantic tags.
152  Info->Description.Id = TEXT("static.") + GetRelevantTagAsString(Info->SemanticTags);
153  }
154 
155  Info->SerializedData.id = Id;
156  Info->SerializedData.description = Info->Description;
157  Info->SerializedData.bounding_box = Info->BoundingBox;
158  Info->SerializedData.semantic_tags.reserve(Info->SemanticTags.Num());
159  for (auto &&Tag : Info->SemanticTags)
160  {
161  using tag_t = decltype(Info->SerializedData.semantic_tags)::value_type;
162  Info->SerializedData.semantic_tags.emplace_back(static_cast<tag_t>(Tag));
163  }
164  auto *Sensor = Cast<ASensor>(&Actor);
165  if (Sensor != nullptr)
166  {
167  const auto &Token = Sensor->GetToken();
168  Info->SerializedData.stream_token = decltype(Info->SerializedData.stream_token)(
169  std::begin(Token.data),
170  std::end(Token.data));
171  }
172  auto Type = FActorRegistry_GetActorType(&Actor);
173  TSharedPtr<FCarlaActor> CarlaActor =
174  FCarlaActor::ConstructCarlaActor(
175  Id, &Actor,
176  std::move(Info), Type,
177  InState, Actor.GetWorld());
178  return CarlaActor;
179 }
