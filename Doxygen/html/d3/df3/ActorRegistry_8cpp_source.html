16 static FActorView::ActorType FActorRegistry_GetActorType(const FActorView &View)
17 {
18  if (!View.IsValid())
19  {
20  return FActorView::ActorType::INVALID;
21  }
22  else if (nullptr != Cast<ACarlaWheeledVehicle>(View.GetActor()))
23  {
24  return FActorView::ActorType::Vehicle;
25  }
26  else if (nullptr != Cast<ACharacter>(View.GetActor()))

28  return FActorView::ActorType::Walker;

30  else if (nullptr != Cast<ATrafficLightBase>(View.GetActor()))
31  {
32  return FActorView::ActorType::TrafficLight;
33  }
34  else
35  {
36  return FActorView::ActorType::Other;
37  }
38 }

40 static FString GetRelevantTagAsString(const TSet<ECityObjectLabel> &SemanticTags)
41 {
42  for (auto &&Tag : SemanticTags)
43  {
44  if ((Tag != ECityObjectLabel::None) && (Tag != ECityObjectLabel::Other))
45  {
46  auto Str = ATagger::GetTagAsString(Tag).ToLower();
47  return (Str.EndsWith(TEXT("s")) ? Str.LeftChop(1) : Str);

49  }
50  return TEXT("unknown");

53 FActorView FActorRegistry::Register(AActor &Actor, FActorDescription Description, IdType DesiredId)
54 {
55  static IdType ID_COUNTER = 0u;
56  auto Id = ++ID_COUNTER;
57 
58  if (DesiredId != 0 && Id != DesiredId) {
59  // check if the desired Id is free, then use it instead
60  if (!Actors.Contains(DesiredId))
61  Id = DesiredId;
62  if (ID_COUNTER < Id)
63  ID_COUNTER = Id;
64  }
65 
66  Actors.Emplace(Id, &Actor);
67  if (Ids.Contains(&Actor))
68  {
69  UE_LOG(
70  LogCarla,
71  Warning,
72  TEXT("This actor's memory address is already registered, "
73  "either you forgot to deregister the actor "
74  "or the actor was garbage collected."));
75  }
76  Ids.Emplace(&Actor, Id);
77 
78  auto View = MakeView(Id, Actor, std::move(Description));
79 
80  auto Result = ActorDatabase.emplace(Id, View);
81  check(Result.second);
82  check(static_cast<size_t>(Actors.Num()) == ActorDatabase.size());
83  return Result.first->second;

120  Info->BoundingBox = UBoundingBoxCalculator::GetActorBoundingBox(&Actor);
121 
122  if (Info->Description.Id.IsEmpty())
123  {
124  // This is a fake actor, let's fake the id based on their semantic tags.
125  Info->Description.Id = TEXT("static.") + GetRelevantTagAsString(Info->SemanticTags);
126  }
127

129  Info->SerializedData.description = Info->Description;
130  Info->SerializedData.bounding_box = Info->BoundingBox;
131  Info->SerializedData.semantic_tags.reserve(Info->SemanticTags.Num());
132  for (auto &&Tag : Info->SemanticTags)
133  {
134  using tag_t = decltype(Info->SerializedData.semantic_tags)::value_type;
135  Info->SerializedData.semantic_tags.emplace_back(static_cast<tag_t>(Tag));
136  }
137  auto *Sensor = Cast<ASensor>(&Actor);
138  if (Sensor != nullptr)
139  {
140  const auto &Token = Sensor->GetToken();
141  Info->SerializedData.stream_token = decltype(Info->SerializedData.stream_token)(
142  std::begin(Token.data),
143  std::end(Token.data));
144  }
145  auto View = FActorView{Id, Actor, std::move(Info)};
