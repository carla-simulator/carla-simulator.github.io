120 TSharedPtr<FCarlaActor> FCarlaActor::ConstructCarlaActor(
121  IdType ActorId,
122  AActor* Actor,
123  TSharedPtr<const FActorInfo> Info,
124  ActorType Type,
125  carla::rpc::ActorState InState,
126  UWorld* World)
127 {
128  switch(Type)
129  {
130  case ActorType::TrafficSign:
131  return MakeShared<FTrafficSignActor>(ActorId, Actor, std::move(Info), InState, World);
132  break;
133  case ActorType::TrafficLight:
134  return MakeShared<FTrafficLightActor>(ActorId, Actor, std::move(Info), InState, World);
135  break;
136  case ActorType::Vehicle:
137  return MakeShared<FVehicleActor>(ActorId, Actor, std::move(Info), InState, World);
138  break;
139  case ActorType::Walker:
140  return MakeShared<FWalkerActor>(ActorId, Actor, std::move(Info), InState, World);
141  break;
142  case ActorType::Sensor:
143  return MakeShared<FSensorActor>(ActorId, Actor, std::move(Info), InState, World);
144  break;
145  default:
146  return MakeShared<FOtherActor>(ActorId, Actor, std::move(Info), InState, World);
147  break;
148  }
149 }
