23 void UTrafficLightComponent::InitializeSign(const carla::road::Map &Map)
24 {
25  const double epsilon = 0.00001;
26 
27  auto References = GetAllReferencesToThisSignal(Map);
28 
29  for (auto& Reference : References)
30  {
31  auto RoadId = Reference.first;
32  const auto* SignalReference = Reference.second;
33  for(auto &validity : SignalReference->GetValidities())
34  {
35  for(auto lane : carla::geom::Math::GenerateRange(validity._from_lane, validity._to_lane))
36  {
37  if(lane == 0)
38  continue;
39 
40  auto signal_waypoint = Map.GetWaypoint(
41  RoadId, lane, SignalReference->GetS()).get();
42 
43  // Prevent adding the bounding box inside the intersection
44  if (Map.IsJunction(RoadId)) {
45  auto predecessors = Map.GetPredecessors(signal_waypoint);
46  if (predecessors.size() == 1) {
47  auto predecessor = predecessors.front();
48  if (!Map.IsJunction(predecessor.road_id)) {
49  signal_waypoint = predecessor;
50  }
51  }
52  }
53 
54  if(Map.GetLane(signal_waypoint).GetType() != cr::Lane::LaneType::Driving)
55  continue;
56 
57  // Get 50% of the half size of the width of the lane
58  float BoxWidth = static_cast<float>(
59  0.5f*Map.GetLaneWidth(signal_waypoint)*0.5);
60  float BoxLength = 1.5f;
61  float BoxHeight = 1.0f;
62 
63  // Prevent a situation where the road width is 0,
64  // this could happen in a lane that is just appearing
65  BoxWidth = std::max(0.01f, BoxWidth);
66  // Get min and max
67  double LaneLength = Map.GetLane(signal_waypoint).GetLength();
68  double LaneDistance = Map.GetLane(signal_waypoint).GetDistance();
69  // Safe distance to avoid overlapping the bounding box with the intersection
70  float AdditionalDistance = 1.5f;
71  if(lane < 0)
72  {
73  signal_waypoint.s = FMath::Clamp(signal_waypoint.s - (BoxLength + AdditionalDistance),
74  LaneDistance + epsilon, LaneDistance + LaneLength - epsilon);
75  }
76  else
77  {
78  signal_waypoint.s = FMath::Clamp(signal_waypoint.s + (BoxLength + AdditionalDistance),
79  LaneDistance + epsilon, LaneDistance + LaneLength - epsilon);
80  }
81  FTransform BoxTransform = Map.ComputeTransform(signal_waypoint);
82  ALargeMapManager* LargeMapManager = UCarlaStatics::GetLargeMapManager(GetWorld());
83  if (LargeMapManager)
84  {
85  BoxTransform = LargeMapManager->GlobalToLocalTransform(BoxTransform);
86  }
87  GenerateTrafficLightBox(BoxTransform, FVector(100*BoxLength, 100*BoxWidth, 100*BoxHeight));
88  }
89  }
90  }
91 }
