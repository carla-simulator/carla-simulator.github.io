20 void UTrafficLightComponent::InitializeSign(const carla::road::Map &Map)
21 {
22  const double epsilon = 0.00001;
23 
24  auto References = GetAllReferencesToThisSignal(Map);
25 
26  for (auto& Reference : References)
27  {
28  auto RoadId = Reference.first;
29  const auto* SignalReference = Reference.second;
30  for(auto &validity : SignalReference->GetValidities())
31  {
32  for(auto lane : carla::geom::Math::GenerateRange(validity._from_lane, validity._to_lane))
33  {
34  if(lane == 0)
35  continue;
36 
37  auto signal_waypoint = Map.GetWaypoint(
38  RoadId, lane, SignalReference->GetS()).get();
39 
40  if(Map.GetLane(signal_waypoint).GetType() != cr::Lane::LaneType::Driving)
41  continue;
42 
43  // Get 90% of the half size of the width of the lane
44  float BoxSize = static_cast<float>(
45  0.9f*Map.GetLaneWidth(signal_waypoint)*0.5);
46  // Get min and max
47  double LaneLength = Map.GetLane(signal_waypoint).GetLength();
48  double LaneDistance = Map.GetLane(signal_waypoint).GetDistance();
49  if(lane < 0)
50  {
51  signal_waypoint.s = FMath::Clamp(signal_waypoint.s - BoxSize,
52  LaneDistance + epsilon, LaneDistance + LaneLength - epsilon);
53  }
54  else
55  {
56  signal_waypoint.s = FMath::Clamp(signal_waypoint.s + BoxSize,
57  LaneDistance + epsilon, LaneDistance + LaneLength - epsilon);
58  }
59  float UnrealBoxSize = 100*BoxSize;
60  GenerateTrafficLightBox(Map.ComputeTransform(signal_waypoint), UnrealBoxSize);
61  }
62  }
63  }
64 }
