30  void InMemoryMap::SetUp(float sampling_resolution) {
31 
32  NodeList entry_node_list;
33  NodeList exit_node_list;
34 
35  auto distance_squared =
36  [](cg::Location l1, cg::Location l2) {
37  return cg::Math::DistanceSquared(l1, l2);
38  };
39  auto square = [](float input) {return std::pow(input, 2);};
40 
41  // Creating dense topology.
42  for (auto &pair : _topology) {
43 
44  // Looping through every topology segment.
45  const WaypointPtr begin_waypoint = pair.first;
46  const WaypointPtr end_waypoint = pair.second;
47  const cg::Location begin_location = begin_waypoint->GetTransform().location;
48  const cg::Location end_location = end_waypoint->GetTransform().location;
49 
50  if (distance_squared(begin_location, end_location) > square(ZERO_LENGTH)) {
51 
52  // Adding entry waypoint.
53  WaypointPtr current_waypoint = begin_waypoint;
54  dense_topology.push_back(std::make_shared<SimpleWaypoint>(current_waypoint));
55 
56  entry_node_list.push_back(dense_topology.back());
57 
58  // Populating waypoints from begin_waypoint to end_waypoint.
59  while (distance_squared(current_waypoint->GetTransform().location,
60  end_location) > square(sampling_resolution)) {
61 
62  current_waypoint = current_waypoint->GetNext(sampling_resolution)[0];
63  SimpleWaypointPtr previous_wp = dense_topology.back();
64  dense_topology.push_back(std::make_shared<SimpleWaypoint>(current_waypoint));
65 
66  previous_wp->SetNextWaypoint({dense_topology.back()});
67  }
68 
69  // Adding exit waypoint.
70  SimpleWaypointPtr previous_wp = dense_topology.back();
71  dense_topology.push_back(std::make_shared<SimpleWaypoint>(end_waypoint));
72 
73  previous_wp->SetNextWaypoint({dense_topology.back()});
74  exit_node_list.push_back(dense_topology.back());
75  }
76  }
77 
78  // Linking segments.
79  uint i = 0u, j = 0u;
80  for (SimpleWaypointPtr end_point : exit_node_list) {
81  for (SimpleWaypointPtr begin_point : entry_node_list) {
82  if (end_point->DistanceSquared(begin_point) < square(ZERO_LENGTH) and i != j) {
83  end_point->SetNextWaypoint({begin_point});
84  }
85  ++j;
86  }
87  ++i;
88  }
89 
90  // Tying up loose ends.
91  // Loop through all exit nodes of topology segments,
92  // connect any dangling endpoints to the nearest entry point
93  // of another topology segment.
94  i = 0u;
95  for (auto &end_point : exit_node_list) {
96  if (end_point->GetNextWaypoint().size() == 0) {
97  j = 0u;
98  float min_distance = INFINITE_DISTANCE;
99  SimpleWaypointPtr closest_connection;
100  for (auto &begin_point : entry_node_list) {
101  float new_distance = end_point->DistanceSquared(begin_point);
102  if (new_distance < min_distance and i != j) {
103  min_distance = new_distance;
104  closest_connection = begin_point;
105  }
106  ++j;
107  }
108  const cg::Vector3D end_point_vector = end_point->GetForwardVector();
109  cg::Vector3D relative_vector = closest_connection->GetLocation() - end_point->GetLocation();
110  relative_vector = relative_vector.MakeUnitVector();
111  const float relative_dot = cg::Math::Dot(end_point_vector, relative_vector);
112  if (relative_dot < LANE_CHANGE_ANGULAR_THRESHOLD) {
113  uint count = LANE_CHANGE_LOOK_AHEAD;
114  while (count > 0u) {
115  closest_connection = closest_connection->GetNextWaypoint()[0];
116  --count;
117  }
118  }
119  end_point->SetNextWaypoint({closest_connection});
120  }
121  ++i;
122  }
123 
124  // Localizing waypoints into grids.
125  for (auto &simple_waypoint: dense_topology) {
126  const cg::Location loc = simple_waypoint->GetLocation();
127  const std::string grid_key = MakeGridKey(MakeGridId(loc.x, loc.y));
128  if (waypoint_grid.find(grid_key) == waypoint_grid.end()) {
129  waypoint_grid.insert({grid_key, {simple_waypoint}});
130  } else {
131  waypoint_grid.at(grid_key).insert(simple_waypoint);
132  }
133  }
134 
135  // Linking lane change connections.
136  for (auto &simple_waypoint:dense_topology) {
137  if (!simple_waypoint->CheckJunction()) {
138  FindAndLinkLaneChange(simple_waypoint);
139  }
140  }
141  }

151  SimpleWaypointPtr InMemoryMap::GetWaypointInVicinity(cg::Location location) {
152 
153  const std::pair<int, int> grid_ids = MakeGridId(location.x, location.y);
154  SimpleWaypointPtr closest_waypoint = nullptr;
155  float closest_distance = INFINITE_DISTANCE;
156 
157  // Search all surrounding grids for closest waypoint.
158  for (int i = -1; i <= 1; ++i) {
159  for (int j = -1; j <= 1; ++j) {
160 
161  const std::string grid_key = MakeGridKey({grid_ids.first + i, grid_ids.second + j});
162  if (waypoint_grid.find(grid_key) != waypoint_grid.end()) {
163 
164  const auto& waypoint_set = waypoint_grid.at(grid_key);
165  if (closest_waypoint == nullptr) {
166  closest_waypoint = *waypoint_set.begin();
167  }
168 
169  for (auto &simple_waypoint: waypoint_set) {
170 
171  if (simple_waypoint->DistanceSquared(location) < std::pow(closest_distance, 2)) {
172  closest_waypoint = simple_waypoint;
173  closest_distance = simple_waypoint->DistanceSquared(location);
174  }
175  }
176  }
177  }
178  }
179 
180  // Return the closest waypoint in the surrounding grids
181  // only if it is in the same horizontal plane as the requested location.
182  if (closest_waypoint != nullptr &&
183  std::abs(closest_waypoint->GetLocation().z - location.z) > 1.0) { // Account for constant.
184 
185  closest_waypoint = nullptr;
186  }
187 
188  return closest_waypoint;
189  }

209  void InMemoryMap::FindAndLinkLaneChange(SimpleWaypointPtr reference_waypoint) {
210 
211  const WaypointPtr raw_waypoint = reference_waypoint->GetWaypoint();
212  const crd::element::LaneMarking::LaneChange lane_change = raw_waypoint->GetLaneChange();
213  const auto change_right = crd::element::LaneMarking::LaneChange::Right;
214  const auto change_left = crd::element::LaneMarking::LaneChange::Left;
215  const auto change_both = crd::element::LaneMarking::LaneChange::Both;
216 
217  try {
218  if (lane_change == change_right || lane_change == change_both) {
219 
220  const WaypointPtr right_waypoint = raw_waypoint->GetRight();
221  if (right_waypoint != nullptr &&
222  right_waypoint->GetType() == crd::Lane::LaneType::Driving &&
223  (right_waypoint->GetLaneId() * raw_waypoint->GetLaneId() > 0)) {
224 
225  SimpleWaypointPtr closest_simple_waypoint = GetWaypointInVicinity(right_waypoint->GetTransform().location);
226  if (closest_simple_waypoint == nullptr) {
227  closest_simple_waypoint = GetWaypoint(right_waypoint->GetTransform().location);
228  }
229  reference_waypoint->SetRightWaypoint(closest_simple_waypoint);
230  }
231  }
232 
233  if (lane_change == change_left || lane_change == change_both) {
234 
235  const WaypointPtr left_waypoint = raw_waypoint->GetLeft();
236  if (left_waypoint != nullptr &&
237  left_waypoint->GetType() == crd::Lane::LaneType::Driving &&
238  (left_waypoint->GetLaneId() * raw_waypoint->GetLaneId() > 0)) {
239 
240  SimpleWaypointPtr closest_simple_waypoint = GetWaypointInVicinity(left_waypoint->GetTransform().location);
241  if (closest_simple_waypoint == nullptr) {
242  closest_simple_waypoint = GetWaypoint(left_waypoint->GetTransform().location);
243  }
244  reference_waypoint->SetLeftWaypoint(closest_simple_waypoint);
245  }
246  }
247  } catch (const std::invalid_argument &e) {
248  cg::Location loc = reference_waypoint->GetLocation();
249  carla::log_info(
250  "Unable to link lane change connection at: "
251  + std::to_string(loc.x) + " "
252  + std::to_string(loc.y) + " "
253  + std::to_string(loc.z));
254  }
255  }
