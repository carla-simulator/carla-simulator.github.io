35  std::vector<SimpleWaypointPtr> InMemoryMap::GetSuccessors(const SegmentId segment_id,
36  const SegmentTopology &segment_topology, const SegmentMap &segment_map) {
37  std::vector<SimpleWaypointPtr> result;
38  if (segment_topology.find(segment_id) == segment_topology.end()) {
39  return result;
40  }
41 
42  for (const auto &successor_segment_id : segment_topology.at(segment_id).second) {
43  if (segment_map.find(successor_segment_id) == segment_map.end()) {
44  auto successors = GetSuccessors(successor_segment_id, segment_topology, segment_map);
45  result.insert(result.end(), successors.begin(), successors.end());
46  } else {
47  result.emplace_back(segment_map.at(successor_segment_id).front());
48  }
49  }
50  return result;
51  }

71  void InMemoryMap::SetUp() {
72 
73  // 1. Building segment topology (i.e., defining set of segment predecessors and successors)
74  auto waypoint_topology = _world_map->GetTopology();
75 
76  SegmentTopology segment_topology;
77  std::unordered_map<int64_t, std::pair<std::set<crd::RoadId>, std::set<crd::RoadId>>> std_road_connectivity;
78  std::unordered_map<crd::RoadId, bool> is_real_junction;
79 
80  for (auto &connection : waypoint_topology) {
81  auto &waypoint = connection.first;
82  auto &successor = connection.second;
83 
84  // Setting segment predecessors and successors.
85  SegmentId waypoint_segment_id = GetSegmentId(connection.first);
86  SegmentId successor_segment_id = GetSegmentId(connection.second);
87  segment_topology[waypoint_segment_id].second.push_back(successor_segment_id);
88  segment_topology[successor_segment_id].first.push_back(waypoint_segment_id);
89 
90  // From path to standard road.
91  if (waypoint->IsJunction() && !successor->IsJunction()) {
92  crd::RoadId path_id = waypoint->GetRoadId();
93  int64_t std_road_id = static_cast<int64_t>(successor->GetRoadId());
94  std_road_id = (successor->GetLaneId() < 0) ? -1 * std_road_id : std_road_id;
95 
96  std_road_connectivity[std_road_id].first.insert(path_id);
97  auto &in_paths = std_road_connectivity[std_road_id].first;
98  if (in_paths.size() >= 2) {
99  std::for_each(in_paths.begin(), in_paths.end(), [&is_real_junction](crd::RoadId id) {is_real_junction[id] = true;});
100  }
101  }
102 
103  // From standard road to path.
104  if (!waypoint->IsJunction() && successor->IsJunction()) {
105  crd::RoadId path_id = successor->GetRoadId();
106  int64_t std_road_id = static_cast<int64_t>(waypoint->GetRoadId());
107  std_road_id = (waypoint->GetLaneId() < 0) ? -1 * std_road_id : std_road_id;
108 
109  std_road_connectivity[std_road_id].second.insert(path_id);
110  auto &out_paths = std_road_connectivity[std_road_id].second;
111  if (out_paths.size() >= 2) {
112  std::for_each(out_paths.begin(), out_paths.end(), [&is_real_junction](crd::RoadId id) {is_real_junction[id] = true;});
113  }
114  }
115  }
116 
117  // 2. Consuming the raw dense topology from cc::Map into SimpleWaypoints.
118  SegmentMap segment_map;
119  auto raw_dense_topology = _world_map->GenerateWaypoints(TOPOLOGY_DISTANCE);
120  for (auto &waypoint_ptr: raw_dense_topology) {
121  segment_map[GetSegmentId(waypoint_ptr)].emplace_back(std::make_shared<SimpleWaypoint>(waypoint_ptr));
122  }
123 
124  // 3. Processing waypoints.
125  auto distance_squared =
126  [](cg::Location l1, cg::Location l2) {
127  return cg::Math::DistanceSquared(l1, l2);
128  };
129  auto square = [](float input) {return std::pow(input, 2);};
130  auto compare_s = [](const SimpleWaypointPtr &swp1, const SimpleWaypointPtr &swp2) {
131  return (swp1->GetWaypoint()->GetDistance() < swp2->GetWaypoint()->GetDistance());
132  };
133 
134  GeoGridId geodesic_grid_id_counter = -1;
135  for (auto &segment: segment_map) {
136  auto &segment_waypoints = segment.second;
137 
138  // Generating geodesic grid ids.
139  ++geodesic_grid_id_counter;
140 
141  // Ordering waypoints according to road direction.
142  std::sort(segment_waypoints.begin(), segment_waypoints.end(), compare_s);
143  auto lane_id = segment_waypoints.front()->GetWaypoint()->GetLaneId();
144  if (lane_id > 0) {
145  std::reverse(segment_waypoints.begin(), segment_waypoints.end());
146  }
147 
148  // Placing intra-segment connections.
149  cg::Location grid_edge_location = segment_waypoints.front()->GetLocation();
150  for (std::size_t i = 0; i < segment_waypoints.size() - 1; ++i) {
151 
152  // Assigning grid id.
153  if (distance_squared(grid_edge_location, segment_waypoints.at(i)->GetLocation()) >
154  square(MAX_GEODESIC_GRID_LENGTH)) {
155  ++geodesic_grid_id_counter;
156  grid_edge_location = segment_waypoints.at(i)->GetLocation();
157  }
158  segment_waypoints.at(i)->SetGeodesicGridId(geodesic_grid_id_counter);
159 
160  segment_waypoints.at(i)->SetNextWaypoint({segment_waypoints.at(i + 1)});
161  segment_waypoints.at(i + 1)->SetPreviousWaypoint({segment_waypoints.at(i)});
162  }
163  segment_waypoints.back()->SetGeodesicGridId(geodesic_grid_id_counter);
164 
165  // Adding simple waypoints to processed dense topology.
166  for (auto swp: segment_waypoints) {
167  // Checking whether the waypoint is a real junction.
168  auto road_id = swp->GetWaypoint()->GetRoadId();
169  if (swp->GetWaypoint()->IsJunction() && !is_real_junction.count(road_id)) {
170  swp->SetIsJunction(false);
171  } else {
172  swp->SetIsJunction(swp->GetWaypoint()->IsJunction());
173  }
174 
175  dense_topology.push_back(swp);
176  }
177  }
178 
179  // Localizing waypoints into grids.
180  for (auto &simple_waypoint: dense_topology) {
181  if (simple_waypoint != nullptr) {
182  const cg::Location loc = simple_waypoint->GetLocation();
183  const std::string grid_key = MakeGridKey(MakeGridId(loc.x, loc.y, true));
184  if (waypoint_grid.find(grid_key) == waypoint_grid.end()) {
185  waypoint_grid.insert({grid_key, {simple_waypoint}});
186  } else {
187  waypoint_grid.at(grid_key).insert(simple_waypoint);
188  }
189  const std::string ped_grid_key = MakeGridKey(MakeGridId(loc.x, loc.y, false));
190  if (ped_waypoint_grid.find(ped_grid_key) == ped_waypoint_grid.end()) {
191  ped_waypoint_grid.insert({ped_grid_key, {simple_waypoint}});
192  } else {
193  ped_waypoint_grid.at(ped_grid_key).insert(simple_waypoint);
194  }
195  }
196  }
197 
198  // Placing inter-segment connections.
199  for (auto &segment : segment_map) {
200  SegmentId segment_id = segment.first;
201  auto &segment_waypoints = segment.second;
202 
203  auto successors = GetSuccessors(segment_id, segment_topology, segment_map);
204  auto predecessors = GetPredecessors(segment_id, segment_topology, segment_map);
205 
206  segment_waypoints.front()->SetPreviousWaypoint(predecessors);
207  segment_waypoints.back()->SetNextWaypoint(successors);
208  }
209 
210  // Linking lane change connections.
211  for (auto &simple_waypoint:dense_topology) {
212  if (!simple_waypoint->CheckJunction()) {
213  FindAndLinkLaneChange(simple_waypoint);
214  }
215  }
216 
217  // Linking any unconnected segments.
218  for (auto &swp: dense_topology) {
219  if (swp->GetNextWaypoint().empty()) {
220  auto neighbour = swp->GetRightWaypoint();
221  if (!neighbour) {
222  neighbour = swp->GetLeftWaypoint();
223  }
224 
225  if (neighbour) {
226  swp->SetNextWaypoint(neighbour->GetNextWaypoint());
227  for (auto next_waypoint : neighbour->GetNextWaypoint()) {
228  next_waypoint->SetPreviousWaypoint({swp});
229  }
230  }
231  }
232  }
233 
234  MakeGeodesiGridCenters();
235  }

250  SimpleWaypointPtr InMemoryMap::GetWaypointInVicinity(cg::Location location) {
251 
252  const std::pair<int, int> grid_ids = MakeGridId(location.x, location.y, true);
253  SimpleWaypointPtr closest_waypoint = nullptr;
254  float closest_distance = INFINITE_DISTANCE;
255 
256  // Search all surrounding grids for closest waypoint.
257  for (int i = -1; i <= 1; ++i) {
258  for (int j = -1; j <= 1; ++j) {
259 
260  const std::string grid_key = MakeGridKey({grid_ids.first + i, grid_ids.second + j});
261  if (waypoint_grid.find(grid_key) != waypoint_grid.end()) {
262 
263  const auto &waypoint_set = waypoint_grid.at(grid_key);
264  if (closest_waypoint == nullptr) {
265  closest_waypoint = *waypoint_set.begin();
266  }
267 
268  for (auto &simple_waypoint: waypoint_set) {
269 
270  if (simple_waypoint->DistanceSquared(location) < std::pow(closest_distance, 2)) {
271  closest_waypoint = simple_waypoint;
272  closest_distance = simple_waypoint->DistanceSquared(location);
273  }
274  }
275  }
276  }
277  }
278 
279  // Return the closest waypoint in the surrounding grids
280  // only if it is in the same horizontal plane as the requested location.
281  if (closest_waypoint != nullptr && std::abs(closest_waypoint->GetLocation().z - location.z) > 1.0) {
282  closest_waypoint = nullptr;
283  }
284 
285  return closest_waypoint;
286  }

338  void InMemoryMap::FindAndLinkLaneChange(SimpleWaypointPtr reference_waypoint) {
339 
340  const WaypointPtr raw_waypoint = reference_waypoint->GetWaypoint();
341  const crd::element::LaneMarking::LaneChange lane_change = raw_waypoint->GetLaneChange();
342  const auto change_right = crd::element::LaneMarking::LaneChange::Right;
343  const auto change_left = crd::element::LaneMarking::LaneChange::Left;
344  const auto change_both = crd::element::LaneMarking::LaneChange::Both;
345 
346  try {
347  if (lane_change == change_right || lane_change == change_both) {
348 
349  const WaypointPtr right_waypoint = raw_waypoint->GetRight();
350  if (right_waypoint != nullptr &&
351  right_waypoint->GetType() == crd::Lane::LaneType::Driving &&
352  (right_waypoint->GetLaneId() * raw_waypoint->GetLaneId() > 0)) {
353 
354  SimpleWaypointPtr closest_simple_waypoint =
355  GetWaypointInVicinity(right_waypoint->GetTransform().location);
356  if (closest_simple_waypoint == nullptr) {
357  closest_simple_waypoint = GetWaypoint(right_waypoint->GetTransform().location);
358  }
359  reference_waypoint->SetRightWaypoint(closest_simple_waypoint);
360  }
361  }
362 
363  if (lane_change == change_left || lane_change == change_both) {
364 
365  const WaypointPtr left_waypoint = raw_waypoint->GetLeft();
366  if (left_waypoint != nullptr &&
367  left_waypoint->GetType() == crd::Lane::LaneType::Driving &&
368  (left_waypoint->GetLaneId() * raw_waypoint->GetLaneId() > 0)) {
369 
370  SimpleWaypointPtr closest_simple_waypoint =
371  GetWaypointInVicinity(left_waypoint->GetTransform().location);
372  if (closest_simple_waypoint == nullptr) {
373  closest_simple_waypoint = GetWaypoint(left_waypoint->GetTransform().location);
374  }
375  reference_waypoint->SetLeftWaypoint(closest_simple_waypoint);
376  }
377  }
378  } catch (const std::invalid_argument &e) {
379  cg::Location loc = reference_waypoint->GetLocation();
380  carla::log_info(
381  "Unable to link lane change connection at: " +
382  std::to_string(loc.x) + " " +
383  std::to_string(loc.y) + " " +
384  std::to_string(loc.z));
385  }
386  }
