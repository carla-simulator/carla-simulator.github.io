112  bool InMemoryMap::Load(const std::vector<uint8_t>& content) {
113  unsigned long pos = 0;
114  std::vector<CachedSimpleWaypoint> cached_waypoints;
115  std::unordered_map<uint64_t, uint32_t> id2index;
116 
117  // read total records
118  uint32_t total;
119  memcpy(&total, &content[pos], sizeof(total));
120  pos += sizeof(total);
121 
122  // read simple waypoints
123  for (uint32_t i=0; i < total; i++) {
124  CachedSimpleWaypoint cached_wp;
125  cached_wp.Read(content, pos);
126  cached_waypoints.push_back(cached_wp);
127  id2index.insert({cached_wp.waypoint_id, i});
128 
129  WaypointPtr waypoint_ptr = _world_map->GetWaypointXODR(cached_wp.road_id, cached_wp.lane_id, cached_wp.s);
130  SimpleWaypointPtr wp = std::make_shared<SimpleWaypoint>(waypoint_ptr);
131  dense_topology.push_back(wp);
132  }
133 
134  // connect waypoints
135  for (uint32_t i=0; i < dense_topology.size(); i++) {
136  auto wp = dense_topology.at(i);
137  auto cached_wp = cached_waypoints.at(i);
138 
139  std::vector<SimpleWaypointPtr> next_waypoints;
140  for (auto id : cached_wp.next_waypoints) {
141  next_waypoints.push_back(dense_topology.at(id2index.at(id)));
142  }
143  std::vector<SimpleWaypointPtr> previous_waypoints;
144  for (auto id : cached_wp.previous_waypoints) {
145  previous_waypoints.push_back(dense_topology.at(id2index.at(id)));
146  }
147  wp->SetNextWaypoint(next_waypoints);
148  wp->SetPreviousWaypoint(previous_waypoints);
149  if (cached_wp.next_left_waypoint > 0) {
150  wp->SetLeftWaypoint(dense_topology.at(id2index.at(cached_wp.next_left_waypoint)));
151  }
152  if (cached_wp.next_right_waypoint > 0) {
153  wp->SetRightWaypoint(dense_topology.at(id2index.at(cached_wp.next_right_waypoint)));
154  }
155  }
156 
157  // create spatial tree
158  SetUpSpatialTree();
159 
160  return true;

163  void InMemoryMap::SetUp() {
164 
165  // 1. Building segment topology (i.e., defining set of segment predecessors and successors)
166  assert(_world_map != nullptr && "No map reference found.");
167  auto waypoint_topology = _world_map->GetTopology();
168 
169  SegmentTopology segment_topology;
170  std::unordered_map<int64_t, std::pair<std::set<crd::RoadId>, std::set<crd::RoadId>>> std_road_connectivity;
171  std::unordered_map<crd::RoadId, bool> is_real_junction;
172 
173  for (auto &connection : waypoint_topology) {
174  auto &waypoint = connection.first;
175  auto &successor = connection.second;
176 
177  // Setting segment predecessors and successors.
178  SegmentId waypoint_segment_id = GetSegmentId(connection.first);
179  SegmentId successor_segment_id = GetSegmentId(connection.second);
180  using SegIdVectorPair = std::pair<std::vector<SegmentId>, std::vector<SegmentId>>;
181  SegIdVectorPair &connection_first = segment_topology[waypoint_segment_id];
182  SegIdVectorPair &connection_second = segment_topology[successor_segment_id];
183  connection_first.second.push_back(successor_segment_id);
184  connection_second.first.push_back(waypoint_segment_id);
185 
186  // From path to standard road.
187  bool waypoint_is_junction = waypoint->IsJunction();
188  bool successor_is_junction = successor->IsJunction();
189  if (waypoint_is_junction && !successor_is_junction) {
190  crd::RoadId path_id = waypoint->GetRoadId();
191  int64_t std_road_id = static_cast<int64_t>(successor->GetRoadId());
192  std_road_id = (successor->GetLaneId() < 0) ? -1 * std_road_id : std_road_id;
193 
194  std::set<crd::RoadId> &in_paths = std_road_connectivity[std_road_id].first;
195  in_paths.insert(path_id);
196 
197  if (in_paths.size() >= 2) {
198  for (auto &in_path_id: in_paths) {
199  is_real_junction[in_path_id] = true;
200  }
201  }
202  }
203 
204  // From standard road to path.
205  if (!waypoint_is_junction && successor_is_junction) {
206  crd::RoadId path_id = successor->GetRoadId();
207  int64_t std_road_id = static_cast<int64_t>(waypoint->GetRoadId());
208  std_road_id = (waypoint->GetLaneId() < 0) ? -1 * std_road_id : std_road_id;
209 
210  std::set<crd::RoadId> &out_paths = std_road_connectivity[std_road_id].second;
211  out_paths.insert(path_id);
212 
213  if (out_paths.size() >= 2) {
214  for (auto &out_path_id: out_paths) {
215  is_real_junction[out_path_id] = true;
216  }
217  }
218  }
219  }
220 
221  // 2. Consuming the raw dense topology from cc::Map into SimpleWaypoints.
222  SegmentMap segment_map;
223  assert(_world_map != nullptr && "No map reference found.");
224  auto raw_dense_topology = _world_map->GenerateWaypoints(MAP_RESOLUTION);
225  for (auto &waypoint_ptr: raw_dense_topology) {
226  segment_map[GetSegmentId(waypoint_ptr)].emplace_back(std::make_shared<SimpleWaypoint>(waypoint_ptr));
227  }
228 
229  // 3. Processing waypoints.
230  auto distance_squared =
231  [](cg::Location l1, cg::Location l2) {
232  return cg::Math::DistanceSquared(l1, l2);
233  };
234  auto square = [](float input) {return std::pow(input, 2);};
235  auto compare_s = [](const SimpleWaypointPtr &swp1, const SimpleWaypointPtr &swp2) {
236  return (swp1->GetWaypoint()->GetDistance() < swp2->GetWaypoint()->GetDistance());
237  };
238  auto wpt_angle = [](cg::Vector3D l1, cg::Vector3D l2) {
239  return cg::Math::GetVectorAngle(l1, l2);
240  };
241  auto max = [](int16_t x, int16_t y) {
242  return x ^ ((x ^ y) & -(x < y));
243  };
244 
245  GeoGridId geodesic_grid_id_counter = -1;
246  for (auto &segment: segment_map) {
247  auto &segment_waypoints = segment.second;
248 
249  // Generating geodesic grid ids.
250  ++geodesic_grid_id_counter;
251 
252  // Ordering waypoints according to road direction.
253  std::sort(segment_waypoints.begin(), segment_waypoints.end(), compare_s);
254  auto lane_id = segment_waypoints.front()->GetWaypoint()->GetLaneId();
255  if (lane_id > 0) {
256  std::reverse(segment_waypoints.begin(), segment_waypoints.end());
257  }
258 
259  // Adding more waypoints if the angle is too tight or if they are too distant.
260  for (std::size_t i = 0; i < segment_waypoints.size() - 1; ++i) {
261  double distance = std::abs(segment_waypoints.at(i)->GetWaypoint()->GetDistance() - segment_waypoints.at(i+1)->GetWaypoint()->GetDistance());
262  double angle = wpt_angle(segment_waypoints.at(i)->GetTransform().rotation.GetForwardVector(), segment_waypoints.at(i+1)->GetTransform().rotation.GetForwardVector());
263  int16_t angle_splits = static_cast<int16_t>(angle/MAX_WPT_RADIANS);
264  int16_t distance_splits = static_cast<int16_t>((distance*distance)/MAX_WPT_DISTANCE);
265  auto max_splits = max(angle_splits, distance_splits);
266  if (max_splits >= 1) {
267  // Compute how many waypoints do we need to generate.
268  for (uint16_t j = 0; j < max_splits; ++j) {
269  auto next_waypoints = segment_waypoints.at(i)->GetWaypoint()->GetNext(distance/(max_splits+1));
270  if (next_waypoints.size() != 0) {
271  auto new_waypoint = next_waypoints.front();
272  i++;
273  segment_waypoints.insert(segment_waypoints.begin()+static_cast<int64_t>(i), std::make_shared<SimpleWaypoint>(new_waypoint));
274  } else {
275  // Reached end of the road.
276  break;
277  }
278  }
279  }
280  }
281 
282  // Placing intra-segment connections.
283  cg::Location grid_edge_location = segment_waypoints.front()->GetLocation();
284  for (std::size_t i = 0; i < segment_waypoints.size() - 1; ++i) {
285 
286  // Assigning grid id.
287  if (distance_squared(grid_edge_location, segment_waypoints.at(i)->GetLocation()) >
288  square(MAX_GEODESIC_GRID_LENGTH)) {
289  ++geodesic_grid_id_counter;
290  grid_edge_location = segment_waypoints.at(i)->GetLocation();
291  }
292  segment_waypoints.at(i)->SetGeodesicGridId(geodesic_grid_id_counter);
293 
294  segment_waypoints.at(i)->SetNextWaypoint({segment_waypoints.at(i + 1)});
295  segment_waypoints.at(i + 1)->SetPreviousWaypoint({segment_waypoints.at(i)});
296  }
297  segment_waypoints.back()->SetGeodesicGridId(geodesic_grid_id_counter);
298 
299  // Adding simple waypoints to processed dense topology.
300  for (auto swp: segment_waypoints) {
301  // Checking whether the waypoint is in a real junction.
302  auto wpt = swp->GetWaypoint();
303  auto road_id = wpt->GetRoadId();
304  if (wpt->IsJunction() && !is_real_junction.count(road_id)) {
305  swp->SetIsJunction(false);
306  } else {
307  swp->SetIsJunction(swp->GetWaypoint()->IsJunction());
308  }
309 
310  dense_topology.push_back(swp);
311  }
312  }
313 
314  SetUpSpatialTree();
315 
316  // Placing inter-segment connections.
317  for (auto &segment : segment_map) {
318  SegmentId segment_id = segment.first;
319  auto &segment_waypoints = segment.second;
320 
321  auto successors = GetSuccessors(segment_id, segment_topology, segment_map);
322  auto predecessors = GetPredecessors(segment_id, segment_topology, segment_map);
323 
324  segment_waypoints.front()->SetPreviousWaypoint(predecessors);
325  segment_waypoints.back()->SetNextWaypoint(successors);
326  }
327 
328  // Linking lane change connections.
329  for (auto &simple_waypoint:dense_topology) {
330  if (!simple_waypoint->CheckJunction()) {
331  FindAndLinkLaneChange(simple_waypoint);
332  }
333  }
334 
335  // Linking any unconnected segments.
336  for (auto &swp: dense_topology) {
337  if (swp->GetNextWaypoint().empty()) {
338  auto neighbour = swp->GetRightWaypoint();
339  if (!neighbour) {
340  neighbour = swp->GetLeftWaypoint();
341  }
342 
343  if (neighbour) {
344  swp->SetNextWaypoint(neighbour->GetNextWaypoint());
345  for (auto next_waypoint : neighbour->GetNextWaypoint()) {
346  next_waypoint->SetPreviousWaypoint({swp});
347  }
348  }
349  }
350  }
351  }

376  NodeList InMemoryMap::GetWaypointsInDelta(const cg::Location loc, const uint16_t n_points, const float random_sample) const {
377  Point3D query_point(loc.x, loc.y, loc.z);
378 
379  Point3D lower_p1(loc.x + random_sample, loc.y + random_sample, loc.z + Z_DELTA);
380  Point3D lower_p2(loc.x - random_sample, loc.y - random_sample, loc.z - Z_DELTA);
381  Point3D upper_p1(loc.x + random_sample + DELTA, loc.y + random_sample + DELTA, loc.z + Z_DELTA);
382  Point3D upper_p2(loc.x - random_sample - DELTA, loc.y - random_sample - DELTA, loc.z - Z_DELTA);
383 
384  Box lower_query_box(lower_p2, lower_p1);
385  Box upper_query_box(upper_p2, upper_p1);
386 
387  NodeList result;
388  uint8_t x = 0;
389  for (Rtree::const_query_iterator
390  it = rtree.qbegin(bgi::within(upper_query_box)
391  && !bgi::within(lower_query_box)
392  && bgi::satisfies([&](SpatialTreeEntry const& v) { return !v.second->CheckJunction();}));
393  it != rtree.qend();
394  ++it) {
395  x++;
396  result.push_back(it->second);
397  if (x >= n_points)
398  break;
399  }
400 
401  return result;
402  }

408  void InMemoryMap::FindAndLinkLaneChange(SimpleWaypointPtr reference_waypoint) {
409 
410  const WaypointPtr raw_waypoint = reference_waypoint->GetWaypoint();
411  const crd::element::LaneMarking::LaneChange lane_change = raw_waypoint->GetLaneChange();
412 
413  /// Cheack for transits
414  switch(lane_change)
415  {
416  /// Left transit way point present only
417  case crd::element::LaneMarking::LaneChange::Left:
418  {
419  const WaypointPtr left_waypoint = raw_waypoint->GetLeft();
420  if (left_waypoint != nullptr &&
421  left_waypoint->GetType() == crd::Lane::LaneType::Driving &&
422  (left_waypoint->GetLaneId() * raw_waypoint->GetLaneId() > 0)) {
423 
424  SimpleWaypointPtr closest_simple_waypoint = GetWaypoint(left_waypoint->GetTransform().location);
425  reference_waypoint->SetLeftWaypoint(closest_simple_waypoint);
426  }
427  }
428  break;
429 
430  /// Right transit way point present only
431  case crd::element::LaneMarking::LaneChange::Right:
432  {
433  const WaypointPtr right_waypoint = raw_waypoint->GetRight();
434  if(right_waypoint != nullptr &&
435  right_waypoint->GetType() == crd::Lane::LaneType::Driving &&
436  (right_waypoint->GetLaneId() * raw_waypoint->GetLaneId() > 0)) {
437 
438  SimpleWaypointPtr closest_simple_waypoint = GetWaypoint(right_waypoint->GetTransform().location);
439  reference_waypoint->SetRightWaypoint(closest_simple_waypoint);
440  }
441  }
442  break;
443 
444  /// Both left and right transit present
445  case crd::element::LaneMarking::LaneChange::Both:
446  {
447  /// Right transit way point
448  const WaypointPtr right_waypoint = raw_waypoint->GetRight();
449  if (right_waypoint != nullptr &&
450  right_waypoint->GetType() == crd::Lane::LaneType::Driving &&
451  (right_waypoint->GetLaneId() * raw_waypoint->GetLaneId() > 0)) {
452 
453  SimpleWaypointPtr closest_simple_waypointR = GetWaypoint(right_waypoint->GetTransform().location);
454  reference_waypoint->SetRightWaypoint(closest_simple_waypointR);
455  }
456 
457  /// Left transit way point
458  const WaypointPtr left_waypoint = raw_waypoint->GetLeft();
459  if (left_waypoint != nullptr &&
460  left_waypoint->GetType() == crd::Lane::LaneType::Driving &&
461  (left_waypoint->GetLaneId() * raw_waypoint->GetLaneId() > 0)) {
462 
463  SimpleWaypointPtr closest_simple_waypointL = GetWaypoint(left_waypoint->GetTransform().location);
464  reference_waypoint->SetLeftWaypoint(closest_simple_waypointL);
465  }
466  }
467  break;
468 
469  /// For no transit waypoint (left or right)
470  default: break;
471  }
472  }
