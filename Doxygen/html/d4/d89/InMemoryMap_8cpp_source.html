28  void InMemoryMap::SetUp() {
29 
30  NodeList entry_node_list;
31  NodeList exit_node_list;
32 
33  auto distance_squared =
34  [](cg::Location l1, cg::Location l2) {
35  return cg::Math::DistanceSquared(l1, l2);
36  };
37  auto square = [](float input) {return std::pow(input, 2);};
38 
39  // Consuming the raw dense topology from cc::Map into SimpleWaypoints.
40  std::map<std::pair<crd::RoadId, crd::LaneId>, std::vector<SimpleWaypointPtr>> segment_map;
41  for (auto& waypoint_ptr: raw_dense_topology) {
42  auto road_id = waypoint_ptr->GetRoadId();
43  auto lane_id = waypoint_ptr->GetLaneId();
44  if (segment_map.find({road_id, lane_id}) != segment_map.end()) {
45  segment_map.at({road_id, lane_id}).push_back(std::make_shared<SimpleWaypoint>(waypoint_ptr));
46  } else {
47  segment_map.insert({{road_id, lane_id}, {std::make_shared<SimpleWaypoint>(waypoint_ptr)}});
48  }
49  }
50 
51  auto compare_s = [] (const SimpleWaypointPtr& swp1, const SimpleWaypointPtr& swp2) {
52  return (swp1->GetWaypoint()->GetDistance() < swp2->GetWaypoint()->GetDistance());
53  };
54 
55  GeoGridId geodesic_grid_id_counter = -1;
56  for (auto& segment: segment_map) {
57 
58  // Generating geodesic grid ids.
59  ++geodesic_grid_id_counter;
60 
61  // Ordering waypoints to be consecutive.
62  auto& segment_waypoints = segment.second;
63  std::sort(segment_waypoints.begin(), segment_waypoints.end(), compare_s);
64 
65  if (segment_waypoints.front()->DistanceSquared(segment_waypoints.back()) > square(0.1f)) {
66 
67  SimpleWaypointPtr first_point = segment_waypoints.at(0);
68  SimpleWaypointPtr second_point = segment_waypoints.at(1);
69  cg::Vector3D first_to_second = second_point->GetLocation() - first_point->GetLocation();
70  first_to_second = first_to_second.MakeUnitVector();
71  cg::Vector3D first_heading = first_point->GetForwardVector();
72 
73  if (cg::Math::Dot(first_heading, first_to_second) < 0.0f) {
74  std::reverse(segment_waypoints.begin(), segment_waypoints.end());
75  }
76 
77  // Registering segment end points.
78  entry_node_list.push_back(segment_waypoints.front());
79  exit_node_list.push_back(segment_waypoints.back());
80 
81  // Placing intra-segment connections.
82  cg::Location grid_edge_location = segment_waypoints.front()->GetLocation();
83  for (uint64_t i=0; i< segment_waypoints.size() -1; ++i) {
84 
85  // Assigning grid id.
86  if (distance_squared(grid_edge_location, segment_waypoints.at(i)->GetLocation()) >
87  square(MAX_GEODESIC_GRID_LENGTH)) {
88  ++geodesic_grid_id_counter;
89  grid_edge_location = segment_waypoints.at(i)->GetLocation();
90  }
91  segment_waypoints.at(i)->SetGeodesicGridId(geodesic_grid_id_counter);
92 
93  segment_waypoints.at(i)->SetNextWaypoint({segment_waypoints.at(i+1)});
94  segment_waypoints.at(i+1)->SetPreviousWaypoint({segment_waypoints.at(i)});
95  }
96  segment_waypoints.back()->SetGeodesicGridId(geodesic_grid_id_counter);
97 
98  // Adding to processed dense topology.
99  for (auto swp: segment_waypoints) {
100  dense_topology.push_back(swp);
101  }
102  }
103  }
104 
105  // Localizing waypoints into grids.
106  for (auto &simple_waypoint: dense_topology) {
107  if (simple_waypoint != nullptr) {
108  const cg::Location loc = simple_waypoint->GetLocation();
109  const std::string grid_key = MakeGridKey(MakeGridId(loc.x, loc.y, true));
110  if (waypoint_grid.find(grid_key) == waypoint_grid.end()) {
111  waypoint_grid.insert({grid_key, {simple_waypoint}});
112  } else {
113  waypoint_grid.at(grid_key).insert(simple_waypoint);
114  }
115  const std::string ped_grid_key = MakeGridKey(MakeGridId(loc.x, loc.y, false));
116  if (ped_waypoint_grid.find(ped_grid_key) == ped_waypoint_grid.end()) {
117  ped_waypoint_grid.insert({ped_grid_key, {simple_waypoint}});
118  } else {
119  ped_waypoint_grid.at(ped_grid_key).insert(simple_waypoint);
120  }
121  }
122  }
123 
124  // Linking segments.
125  uint64_t i = 0u, j = 0u;
126  for (SimpleWaypointPtr end_point : exit_node_list) {
127  j = 0u;
128  for (SimpleWaypointPtr begin_point: entry_node_list) {
129  if ((end_point->DistanceSquared(begin_point) < square(2.0f)) && (i != j)) {
130  end_point->SetNextWaypoint({begin_point});
131  begin_point->SetPreviousWaypoint({end_point});
132  }
133  ++j;
134  }
135  ++i;
136  }
137 
138  // Linking lane change connections.
139  for (auto &simple_waypoint:dense_topology) {
140  if (!simple_waypoint->CheckJunction()) {
141  FindAndLinkLaneChange(simple_waypoint);
142  }
143  }
144 
145  // Linking any unconnected segments.
146  for (auto& swp: dense_topology) {
147  if (swp->GetNextWaypoint().size() == 0) {
148  SimpleWaypointPtr nearest_sample = GetWaypointInVicinity(swp->GetLocation()
149  + cg::Location(swp->GetForwardVector() * 0.2f));
150  swp->SetNextWaypoint({nearest_sample});
151  nearest_sample->SetPreviousWaypoint({swp});
152  }
153  }
154 
155  MakeGeodesiGridCenters();
156  }

158  std::pair<int, int> InMemoryMap::MakeGridId (float x, float y, bool vehicle_or_pedestrian) {
159  if (vehicle_or_pedestrian) {
160  return {static_cast<int>(std::floor(x/GRID_SIZE)), static_cast<int>(std::floor(y/GRID_SIZE))};
161  } else {
162  return {static_cast<int>(std::floor(x/PED_GRID_SIZE)), static_cast<int>(std::floor(y/PED_GRID_SIZE))};
163  }

170  SimpleWaypointPtr InMemoryMap::GetWaypointInVicinity(cg::Location location) {
171 
172  const std::pair<int, int> grid_ids = MakeGridId(location.x, location.y, true);
173  SimpleWaypointPtr closest_waypoint = nullptr;
174  float closest_distance = INFINITE_DISTANCE;
175 
176  // Search all surrounding grids for closest waypoint.
177  for (int i = -1; i <= 1; ++i) {
178  for (int j = -1; j <= 1; ++j) {
179 
180  const std::string grid_key = MakeGridKey({grid_ids.first + i, grid_ids.second + j});
181  if (waypoint_grid.find(grid_key) != waypoint_grid.end()) {
182 
183  const auto& waypoint_set = waypoint_grid.at(grid_key);
184  if (closest_waypoint == nullptr) {
185  closest_waypoint = *waypoint_set.begin();
186  }
187 
188  for (auto &simple_waypoint: waypoint_set) {
189 
190  if (simple_waypoint->DistanceSquared(location) < std::pow(closest_distance, 2)) {
191  closest_waypoint = simple_waypoint;
192  closest_distance = simple_waypoint->DistanceSquared(location);
193  }
194  }
195  }
196  }
197  }
198 
199  // Return the closest waypoint in the surrounding grids
200  // only if it is in the same horizontal plane as the requested location.
201  if (closest_waypoint != nullptr &&
202  std::abs(closest_waypoint->GetLocation().z - location.z) > 1.0) { // Account for constant.
203 
204  closest_waypoint = nullptr;
205  }
206 
207  return closest_waypoint;
208  }

260  void InMemoryMap::FindAndLinkLaneChange(SimpleWaypointPtr reference_waypoint) {
261 
262  const WaypointPtr raw_waypoint = reference_waypoint->GetWaypoint();
263  const crd::element::LaneMarking::LaneChange lane_change = raw_waypoint->GetLaneChange();
264  const auto change_right = crd::element::LaneMarking::LaneChange::Right;
265  const auto change_left = crd::element::LaneMarking::LaneChange::Left;
266  const auto change_both = crd::element::LaneMarking::LaneChange::Both;
267 
268  try {
269  if (lane_change == change_right || lane_change == change_both) {
270 
271  const WaypointPtr right_waypoint = raw_waypoint->GetRight();
272  if (right_waypoint != nullptr &&
273  right_waypoint->GetType() == crd::Lane::LaneType::Driving &&
274  (right_waypoint->GetLaneId() * raw_waypoint->GetLaneId() > 0)) {
275 
276  SimpleWaypointPtr closest_simple_waypoint = GetWaypointInVicinity(right_waypoint->GetTransform().location);
277  if (closest_simple_waypoint == nullptr) {
278  closest_simple_waypoint = GetWaypoint(right_waypoint->GetTransform().location);
279  }
280  reference_waypoint->SetRightWaypoint(closest_simple_waypoint);
281  }
282  }
283 
284  if (lane_change == change_left || lane_change == change_both) {
285 
286  const WaypointPtr left_waypoint = raw_waypoint->GetLeft();
287  if (left_waypoint != nullptr &&
288  left_waypoint->GetType() == crd::Lane::LaneType::Driving &&
289  (left_waypoint->GetLaneId() * raw_waypoint->GetLaneId() > 0)) {
290 
291  SimpleWaypointPtr closest_simple_waypoint = GetWaypointInVicinity(left_waypoint->GetTransform().location);
292  if (closest_simple_waypoint == nullptr) {
293  closest_simple_waypoint = GetWaypoint(left_waypoint->GetTransform().location);
294  }
295  reference_waypoint->SetLeftWaypoint(closest_simple_waypoint);
296  }
297  }
298  } catch (const std::invalid_argument &e) {
299  cg::Location loc = reference_waypoint->GetLocation();
300  carla::log_info(
301  "Unable to link lane change connection at: "
302  + std::to_string(loc.x) + " "
303  + std::to_string(loc.y) + " "
304  + std::to_string(loc.z));
305  }
306  }

308  void InMemoryMap::MakeGeodesiGridCenters() {
309  for (auto& swp: dense_topology) {
310  GeoGridId ggid = swp->CheckJunction()? swp->GetJunctionId(): swp->GetGeodesicGridId();
311  if (geodesic_grid_center.find(ggid) == geodesic_grid_center.end()) {
312  geodesic_grid_center.insert({ggid, swp->GetLocation()});
313  } else {
314  cg::Location& grid_loc = geodesic_grid_center.at(ggid);
315  grid_loc = (grid_loc + swp->GetLocation())/2;
316  }
317  }
318  }
