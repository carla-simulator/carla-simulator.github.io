133 void AOpenDriveActor::BuildRoutes(FString MapName)
134 {
135  using Waypoint = carla::road::element::Waypoint;
136 
137  // As the OpenDrive file has the same name as level, build the path to the
138  // xodr file using the lavel name and the game content directory.
139  const FString XodrContent = UOpenDrive::LoadXODR(MapName);
140 
141  auto map = carla::opendrive::OpenDriveParser::Load(carla::rpc::FromFString(XodrContent));
142 
143  if (!map.has_value())
144  {
145  UE_LOG(LogCarla, Error, TEXT("Failed to parse OpenDrive file."));
146  return;
147  }
148 
149  // List with waypoints, each one at the end of each lane of the map
150  const std::vector<Waypoint> LaneWaypoints =
151  map->GenerateWaypointsOnRoadEntries();
152 
153  std::unordered_map<Waypoint, std::vector<Waypoint>> PredecessorMap;
154 
155  for (auto &Wp : LaneWaypoints)
156  {
157  const auto PredecessorsList = map->GetPredecessors(Wp);
158  if (PredecessorsList.empty())
159  {
160  continue;
161  }
162  const auto MinRoadId = *std::min_element(
163  PredecessorsList.begin(),
164  PredecessorsList.end(),
165  [](const auto &WaypointA, const auto &WaypointB) {
166  return WaypointA.road_id < WaypointB.road_id;
167  });
168  PredecessorMap[MinRoadId].emplace_back(Wp);
169  }
170 
171  for (auto &&PredecessorWp : PredecessorMap)
172  {
173  ARoutePlanner *RoutePlanner = nullptr;
174 
175  for (auto &&Wp : PredecessorWp.second)
176  {
177  std::vector<Waypoint> Waypoints;
178  auto CurrentWp = Wp;
179 
180  do
181  {
182  Waypoints.emplace_back(CurrentWp);
183  const auto Successors = map->GetNext(CurrentWp, RoadAccuracy);
184  if (Successors.empty())
185  {
186  break;
187  }
188  if (Successors.front().road_id != Wp.road_id)
189  {
190  break;
191  }
192  CurrentWp = Successors.front();
193  } while (CurrentWp.road_id == Wp.road_id);
194 
195  // connect the last wp of the current toad to the first wp of the following road
196  const auto FollowingWp = map->GetSuccessors(CurrentWp);
197  if (!FollowingWp.empty())
198  {
199  Waypoints.emplace_back(FollowingWp.front());
200  }
201 
202  if (Waypoints.size() >= 2)
203  {
204  TArray<FVector> Positions;
205  Positions.Reserve(Waypoints.size());
206  for (int i = 0; i < Waypoints.size(); ++i)
207  {
208  // Add the trigger height because the z position of the points does not
209  // influence on the driver AI and is easy to visualize in the editor
210  Positions.Add(map->ComputeTransform(Waypoints[i]).location +
211  FVector(0.f, 0.f, TriggersHeight));
212  }
213 
214  // If the route planner does not exist, create it
215  if (RoutePlanner == nullptr )
216  {
217  const auto WpTransform = map->ComputeTransform(Wp);
218  RoutePlanner = GetWorld()->SpawnActor<ARoutePlanner>();
219  RoutePlanner->bIsIntersection = map->IsJunction(Wp.road_id);
220  RoutePlanner->SetBoxExtent(FVector(70.f, 70.f, 50.f));
221  RoutePlanner->SetActorRotation(WpTransform.rotation);
222  RoutePlanner->SetActorLocation(WpTransform.location +
223  FVector(0.f, 0.f, TriggersHeight));
224  }
225 
226  if (RoutePlanner != nullptr)
227  {
228  RoutePlanner->AddRoute(1.f, Positions);
229  RoutePlanners.Add(RoutePlanner);
230  }
231  }
232  }
233  }
234 }

267 void AOpenDriveActor::AddSpawners()
268 {

273  if (!bOnIntersections && RoutePlanners[i]->bIsIntersection)
