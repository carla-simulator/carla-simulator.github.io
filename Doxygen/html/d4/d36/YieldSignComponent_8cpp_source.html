15 void UYieldSignComponent::InitializeSign(const carla::road::Map &Map)
16 {
17 
18  const double epsilon = 0.00001;
19 
20  auto References = GetAllReferencesToThisSignal(Map);
21 
22  for (auto& Reference : References)
23  {
24  auto RoadId = Reference.first;
25  const auto* SignalReference = Reference.second;
26  TSet<carla::road::RoadId> SignalPredecessors;
27  // Yield box
28  for(auto &validity : SignalReference->GetValidities())
29  {
30  for(auto lane : carla::geom::Math::GenerateRange(validity._from_lane, validity._to_lane))
31  {
32  if(lane == 0)
33  continue;
34 
35  auto signal_waypoint = Map.GetWaypoint(
36  RoadId, lane, SignalReference->GetS()).get();
37 
38  if(Map.GetLane(signal_waypoint).GetType() != cr::Lane::LaneType::Driving)
39  continue;
40 
41  // Get 90% of the half size of the width of the lane
42  float BoxSize = static_cast<float>(
43  0.9*Map.GetLaneWidth(signal_waypoint)/2.0);
44  // Get min and max
45  double LaneLength = Map.GetLane(signal_waypoint).GetLength();
46  double LaneDistance = Map.GetLane(signal_waypoint).GetDistance();
47  if(lane < 0)
48  {
49  signal_waypoint.s = FMath::Clamp(signal_waypoint.s - BoxSize,
50  LaneDistance + epsilon, LaneDistance + LaneLength - epsilon);
51  }
52  else
53  {
54  signal_waypoint.s = FMath::Clamp(signal_waypoint.s + BoxSize,
55  LaneDistance + epsilon, LaneDistance + LaneLength - epsilon);
56  }
57  float UnrealBoxSize = 100*BoxSize;
58  GenerateYieldBox(Map.ComputeTransform(signal_waypoint), UnrealBoxSize);
59 
60  auto Predecessors = Map.GetPredecessors(signal_waypoint);
61  for(auto &Prev : Predecessors)
62  {
63  if(!SignalPredecessors.Contains(Prev.road_id))
64  {
65  SignalPredecessors.Add(Prev.road_id);
66  }
67  }
68  }
69  }
70 
71  //Check boxes
72  if(Map.IsJunction(RoadId))
73  {
74  auto JuncId = Map.GetJunctionId(RoadId);
75  const auto * Junction = Map.GetJunction(JuncId);
76  if(Junction->RoadHasConflicts(RoadId))
77  {
78  const auto &ConflictingRoads = Junction->GetConflictsOfRoad(RoadId);
79  for(const auto &Conflict : ConflictingRoads)
80  {
81  auto Waypoints = Map.GenerateWaypointsInRoad(Conflict);
82  for(auto& Waypoint : Waypoints)
83  {
84  // Skip roads that share the same previous road
85  bool bHasSamePredecessor = false;
86  auto Predecessors = Map.GetPredecessors(Waypoint);
87  for(auto &Prev : Predecessors)
88  {
89  if(SignalPredecessors.Contains(Prev.road_id))
90  {
91  bHasSamePredecessor = true;
92  }
93  }
94  if(bHasSamePredecessor)
95  {
96  continue;
97  }
98 
99  if(Map.GetLane(Waypoint).GetType() != cr::Lane::LaneType::Driving)
100  continue;
101 
102  // Cover the road within the junction
103  auto CurrentWaypoint = Waypoint;
104  auto NextWaypoint = CurrentWaypoint;
105  float BoxSize = static_cast<float>(
106  0.9*Map.GetLaneWidth(NextWaypoint)/2.0);
107  float UEBoxSize = 100*BoxSize;
108  GenerateCheckBox(Map.ComputeTransform(NextWaypoint), UEBoxSize);
109  while (true)
110  {
111  auto Next = Map.GetNext(NextWaypoint, 2*BoxSize);
112  if (Next.size() != 1)
113  {
114  break;
115  }
116  NextWaypoint = Next.front();
117  if(NextWaypoint.road_id != Waypoint.road_id)
118  {
119  break;
120  }
121 
122  GenerateCheckBox(Map.ComputeTransform(NextWaypoint), UEBoxSize);
123  }
124  // Cover the road before the junction
125  // Hard coded anticipation time (boxes placed prior to the junction)
126  double AnticipationTime = 0.1;
127  auto Previous = Map.GetPrevious(Waypoint, 2*BoxSize);
128  std::queue<std::pair<float, carla::road::element::Waypoint>>
129  WaypointQueue;
130  for (auto & Prev : Previous)
131  {
132  WaypointQueue.push({AnticipationTime, Prev});
133  }
134  while (!WaypointQueue.empty())
135  {
136  auto CurrentElement = WaypointQueue.front();
137  WaypointQueue.pop();
138  GenerateCheckBox(Map.ComputeTransform(CurrentElement.second), UEBoxSize);
139 
140  float Speed = 40;
141  auto* InfoSpeed = Map.GetLane(CurrentElement.second).GetRoad()->GetInfo<carla::road::element::RoadInfoSpeed>(CurrentElement.second.s);
142  if(InfoSpeed)
143  {
144  Speed = InfoSpeed->GetSpeed();
145  }
146  float RemainingTime = CurrentElement.first - BoxSize/Speed;
147  if(RemainingTime > 0)
148  {
149  Previous = Map.GetPrevious(CurrentElement.second, 2*BoxSize);
150  for (auto & Prev : Previous)
151  {
152  WaypointQueue.push({RemainingTime, Prev});
153  }
154  }
155  }
156  }
157  }
158  }
159  }
160  }
161 }

179 void UYieldSignComponent::GiveWayIfPossible()
180 {
181  if (VehiclesToCheck.Num() == 0)

183  for (auto Vehicle : VehiclesInYield)

185  AWheeledVehicleAIController* VehicleController =
186  Cast<AWheeledVehicleAIController>(Vehicle->GetController());
187  VehicleController->SetTrafficLightState(ETrafficLightState::Green);

197  Cast<AWheeledVehicleAIController>(Vehicle->GetController());
